118 	public static class ItemHistogramBuilder {
119 
120 		private final AggregationGrid grid = new AggregationGrid();
121 		private final List<IColumn> columns = new ArrayList<>();
122 
123 		public void addCountColumn() {
124 			columns.add(new ColumnBuilder(Messages.COUNT_COLUMN_NAME, COUNT_COL_ID, AggregationGrid::getCount)
125 					.columnDrawer(COUNT_DRAWER).style(SWT.RIGHT).build());
126 		}
127 
128 		public void addColumn(String colId, IAggregator<?, ?> a) {
129 			int style = a.getValueType() instanceof LinearKindOfQuantity ? SWT.RIGHT : SWT.NONE;
130 			addColumn(colId, ic -> ic.getAggregate(a), a.getName(), a.getDescription(), style);
131 		}
132 
133 		public void addPercentageColumn(String colId, IAggregator<?, ?> a) {
134 			addPercentageColumn(colId, a, a.getName(), a.getDescription());
135 		}
136 
137 		public void addPercentageColumn(String colId, IAggregator<?, ?> a, String name, String description) {
138 			int style = a.getValueType() instanceof LinearKindOfQuantity ? SWT.RIGHT : SWT.NONE;
139 			addPercentageColumn(colId, (rowItems, allItems) -> {

140 				if (a.getValueType() instanceof LinearKindOfQuantity) {
141 					IQuantity rowResult = (IQuantity) rowItems.getAggregate(a);
142 					IQuantity allResult = (IQuantity) allItems.getAggregate(a);
143 					if (rowResult != null && allResult != null) {
144 						return UnitLookup.PERCENT.quantity(rowResult.ratioTo(allResult) * 100);
145 					}
146 				}
147 				return rowItems.getAggregate(a);
148 			}, name, description, style);


149 		}
150 
151 		public void addColumn(
152 			String colId, Function<IItemCollection, ?> valueFunction, String name, String description) {
153 			addColumn(colId, valueFunction, name, description, SWT.NONE);
154 		}
155 
156 		public void addColumn(
157 			String colId, Function<IItemCollection, ?> valueFunction, String name, String description, int style) {
158 			columns.add(new ColumnBuilder(name, colId, grid.addColumn(valueFunction)).description(description)
159 					.style(style).build());
160 		}
161 
162 		public void addPercentageColumn(
163 			String colId, BiFunction<IItemCollection, IItemCollection, ?> valueFunction, String name,
164 			String description, int style) {
165 			IMemberAccessor<?, Object> column = grid.addPercentageColumn(valueFunction);










166 			columns.add(new ColumnBuilder(name, colId, column).description(description).style(style)
167 					.columnDrawer(new BackgroundFractionDrawer() {
168 						@Override
169 						public void handleEvent(Event event) {
170 							Object row = event.item.getData();
171 							Object item = column.getMember(row);
172 							if (item instanceof Number) {
173 								draw(((Number) item).doubleValue() / 100, event);
174 							}
175 						}
176 					}).build());
177 		}
178 
179 		public <T> void addColumn(IAttribute<T> a) {
180 			// FIXME: Refactor/remove this method to avoid it being used instead of passing an IAggregator.
181 			// Accessing the thread-group is quite a special case as it is a property of the key (group by attribute).
182 			// The caller of this method should be responsible for passing a unique column id, as with aggregators.
183 			IMemberAccessor<Object, T> anyValueAccessor = row -> ItemCollectionToolkit
184 					.stream(AggregationGrid.getItems(row))
185 					.flatMap(is -> ItemIterableToolkit.stream(is).map(a.getAccessor(is.getType())::getMember))
186 					.filter(Objects::nonNull).findAny().orElse(null);
187 			columns.add(new ColumnBuilder(a.getName(), a.getIdentifier(), anyValueAccessor)
188 					.description(a.getDescription()).build());
189 		}

207 		public <T> ItemHistogram build(
208 			Composite container, String colLabel, ContentType<? super T> keyType, IAccessorFactory<T> classifier,
209 			TableSettings tableSettings) {
210 			return build(container, colLabel, keyType, classifier, tableSettings, SWT.BORDER);
211 		}
212 
213 		public <T> ItemHistogram build(
214 			Composite container, String colLabel, ContentType<? super T> keyType, IAccessorFactory<T> classifier,
215 			TableSettings tableSettings, int border) {
216 			List<IColumn> columns = new ArrayList<>();
217 			IMemberAccessor<?, Object> keyAccessor = AggregationGrid::getKey;
218 			ColumnLabelProvider keyLp = new DelegatingLabelProvider(new KeyLabelProvider(keyType), keyAccessor);
219 			columns.add(new ColumnBuilder(colLabel, KEY_COL_ID, keyAccessor).labelProvider(keyLp).build());
220 			columns.addAll(this.columns);
221 			return build(container, columns, classifier, tableSettings, border);
222 		}
223 
224 		private <T> ItemHistogram build(
225 			Composite container, List<IColumn> columns, IAccessorFactory<T> classifier, TableSettings tableSettings,
226 			int border) {
227 			TableViewer tableViewer = new TableViewer(container,
228 					SWT.MULTI | SWT.H_SCROLL | SWT.V_SCROLL | SWT.FULL_SELECTION | border);
229 			tableViewer.setContentProvider(ArrayContentProvider.getInstance());
230 			ColumnViewerToolTipSupport.enableFor(tableViewer);
231 			if (UIPlugin.getDefault().getAccessibilityMode()) {
232 				FocusTracker.enableFocusTracking(tableViewer.getTable());
233 			}
234 			return new ItemHistogram(ColumnManager.build(tableViewer, columns, tableSettings), classifier, grid);
235 		}

239 	/**
240 	 * Holds references to each selected row object. May be expensive to keep reference to as the
241 	 * row objects hold reference too all cell values.
242 	 */
243 	public static class HistogramSelection {
244 		private final List<?> selection;
245 
246 		private HistogramSelection(List<?> selection) {
247 			this.selection = selection;
248 		}
249 
250 		public int getRowCount() {
251 			return selection.size();
252 		}
253 
254 		public IItemCollection getItems() {
255 			List<IItemCollection> rows = selection.stream().map(AggregationGrid::getItems).collect(Collectors.toList());
256 			return ItemCollectionToolkit.merge(rows::stream);
257 		}
258 
259 		public <T> Stream<T> getSelectedRows(BiFunction<Object, IItemCollection, T> rowBuilder) {
260 			return selection.stream()
261 					.map(row -> rowBuilder.apply(AggregationGrid.getKey(row), AggregationGrid.getItems(row)));
262 		}

269 	private static final Listener COUNT_DRAWER = BackgroundFractionDrawer.unchecked(AggregationGrid::getCountFraction);
270 	private final AggregationGrid grid;
271 	private final ColumnManager columnManager;
272 	private final IAccessorFactory<?> classifier;
273 
274 	private ItemHistogram(ColumnManager columnManager, IAccessorFactory<?> classifier, AggregationGrid grid) {
275 		this.columnManager = columnManager;
276 		this.grid = grid;
277 		this.classifier = classifier;
278 	}
279 
280 	public ColumnManager getManager() {
281 		return columnManager;
282 	}
283 
284 	public HistogramSelection getAllRows() {
285 		return new HistogramSelection(Stream.of(((Table) columnManager.getViewer().getControl()).getItems())
286 				.map(ti -> ti.getData()).collect(Collectors.toList()));
287 	}
288 
289 	public HistogramSelection getSelection() {
290 		return new HistogramSelection(((IStructuredSelection) columnManager.getViewer().getSelection()).toList());
291 	}
292 
293 	public void show(IItemCollection items) {
294 		columnManager.getViewer().setInput(grid.buildRows(items, classifier));

297 	/*
298 	 * FIXME: Consider some sharing with ColumnBuilder.DEFAULT_LP.
299 	 * 
300 	 * But remember that the context is different and at least affects both getText() and
301 	 * getToolTipText(), which should be complementary (thus perhaps abstract).
302 	 */
303 	static class KeyLabelProvider extends ColumnLabelProvider {
304 
305 		private final Image image;
306 
307 		KeyLabelProvider(ContentType<?> ct) {
308 			image = TypeAppearance.getImage(ct.getIdentifier());
309 		}
310 
311 		@Override
312 		public Font getFont(Object key) {
313 			return JFaceResources.getFontRegistry().getBold(JFaceResources.DEFAULT_FONT);
314 		}
315 
316 		@Override
317 		public Image getImage(Object key) {
318 			return image;
319 		}
320 
321 		@Override
322 		public String getText(Object key) {
323 			if (key instanceof IDisplayable) {
324 				// Using EXACT here to make close key values user distinguishable. Could change to AUTO.
325 				return ((IDisplayable) key).displayUsing(IDisplayable.EXACT);
326 			}
327 			// Context-insensitive fallback
328 			return TypeHandling.getValueString(key);
329 		};
330 
331 		@Override
332 		public String getToolTipText(Object key) {
333 			if (key instanceof IDisplayable) {
334 				/*
335 				 * Since VERBOSE often gives the same result as EXACT, one could argue that no
336 				 * tooltip should be shown in this case. (If the text doesn't fit in the column,
337 				 * Windows provides a "tooltip" of its own, so at least on Windows we can ignore
338 				 * that. On other platforms, I am not sure.) If this is desirable, it should be
339 				 * implemented throughout the application by comparing with getText(). Otherwise, it
340 				 * may be considered a glitch by users.
341 				 */
342 				return ((IDisplayable) key).displayUsing(IDisplayable.VERBOSE);
343 			}
344 			return null;
345 		}
