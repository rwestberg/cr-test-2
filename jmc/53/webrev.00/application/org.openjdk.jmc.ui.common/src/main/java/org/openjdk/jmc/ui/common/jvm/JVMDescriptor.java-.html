 33 package org.openjdk.jmc.ui.common.jvm;
 34 
 35 /**
 36  * Metadata for a running JVM
 37  */
 38 public class JVMDescriptor {
 39 	private final String javaVersion;
 40 	private final JVMType jvmType;
 41 	private final String jvmVendor;
 42 	private final JVMArch jvmArch;
 43 	private final String javaCommand;
 44 	private final String jvmArguments;
 45 	private final Integer pid;
 46 	private final Boolean debug;
 47 	private final Connectable connectable;
 48 
 49 	public JVMDescriptor(String javaVersion, JVMType jvmType, JVMArch jvmArch, String javaCommand, String jvmArguments,
 50 			String jvmVendor, Integer pid, boolean debug, Connectable attachable) {
 51 		super();
 52 		this.javaVersion = javaVersion;
 53 		this.jvmType = jvmType;
 54 		this.jvmArch = jvmArch;
 55 		this.javaCommand = javaCommand;
 56 		this.jvmArguments = jvmArguments;
 57 		this.jvmVendor = jvmVendor;
 58 		this.pid = pid;
 59 		this.debug = debug;
 60 		connectable = attachable;
 61 	}
 62 
 63 	public String getJavaVersion() {
 64 		return javaVersion;
 65 	}
 66 
 67 	public JVMType getJvmType() {
 68 		return jvmType;
 69 	}
 70 
 71 	public JVMArch getJvmArch() {
 72 		return jvmArch;
 73 	}
 74 
 75 	public String getJavaCommand() {
 76 		return javaCommand;
 77 	}
 78 
 79 	public String getJVMArguments() {
 80 		return jvmArguments;
 81 	}
 82 
 83 	public String getJvmVendor() {
 84 		return jvmVendor;
 85 	}
 86 
 87 	public Integer getPid() {
 88 		return pid;
 89 	}
 90 
 91 	public Boolean isDebug() {
 92 		return debug;
 93 	}
 94 
 95 	public Boolean isUnconnectable() {
 96 		return connectable.isUnconnectable();
 97 	}
 98 
 99 	public Boolean isAttachable() {
100 		return connectable.isAttachable();
101 	}
102 
103 	@Override
104 	public String toString() {
105 		return "[JVMDescriptor] Java command: " + getJavaCommand() + " PID: " + getPid(); //$NON-NLS-1$ //$NON-NLS-2$
106 	}
107 }
