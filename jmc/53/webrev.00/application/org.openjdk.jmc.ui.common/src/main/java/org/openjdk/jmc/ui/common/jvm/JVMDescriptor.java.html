 33 package org.openjdk.jmc.ui.common.jvm;
 34 
 35 /**
 36  * Metadata for a running JVM
 37  */
 38 public class JVMDescriptor {
 39 	private final String javaVersion;
 40 	private final JVMType jvmType;
 41 	private final String jvmName;
 42 	private final String jvmVendor;
 43 	private final JVMArch jvmArch;
 44 	private final String javaCommand;
 45 	private final String jvmArguments;
 46 	private final Integer pid;
 47 	private final Boolean debug;
 48 	private final Connectable connectable;
 49 
 50 	public JVMDescriptor(String javaVersion, JVMType jvmType, JVMArch jvmArch, String javaCommand, String jvmArguments,
 51 			String jvmName, String jvmVendor, Integer pid, boolean debug, Connectable attachable) {
 52 		super();
 53 		this.javaVersion = javaVersion;
 54 		this.jvmType = jvmType;
 55 		this.jvmArch = jvmArch;
 56 		this.javaCommand = javaCommand;
 57 		this.jvmArguments = jvmArguments;
 58 		this.jvmName = jvmName;
 59 		this.jvmVendor = jvmVendor;
 60 		this.pid = pid;
 61 		this.debug = debug;
 62 		connectable = attachable;
 63 	}
 64 
 65 	public String getJavaVersion() {
 66 		return javaVersion;
 67 	}
 68 
 69 	public JVMType getJvmType() {
 70 		return jvmType;
 71 	}
 72 
 73 	public JVMArch getJvmArch() {
 74 		return jvmArch;
 75 	}
 76 
 77 	public String getJavaCommand() {
 78 		return javaCommand;
 79 	}
 80 
 81 	public String getJVMArguments() {
 82 		return jvmArguments;
 83 	}
 84 
 85 	public String getJvmName() {
 86 		return jvmName;
 87 	}
 88 
 89 	public String getJvmVendor() {
 90 		return jvmVendor;
 91 	}
 92 
 93 	public Integer getPid() {
 94 		return pid;
 95 	}
 96 
 97 	public Boolean isDebug() {
 98 		return debug;
 99 	}
100 
101 	public Boolean isUnconnectable() {
102 		return connectable.isUnconnectable();
103 	}
104 
105 	public Boolean isAttachable() {
106 		return connectable.isAttachable();
107 	}
108 
109 	@Override
110 	public String toString() {
111 		return "[JVMDescriptor] Java command: " + getJavaCommand() + " PID: " + getPid(); //$NON-NLS-1$ //$NON-NLS-2$
112 	}
113 }
