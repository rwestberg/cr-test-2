33 package org.openjdk.jmc.agent;
34 
35 import org.openjdk.jmc.agent.util.TypeUtils;
36 
37 /**
38  * Metadata for a return value to be logged by the agent.
39  */
40 public final class ReturnValue implements Attribute {
41 	private final String name;
42 	private final String fieldName;
43 	private final String description;
44 	private final String contentType;
45 	private final String relationKey;
46 	private final String converterClassName;
47 
48 	public ReturnValue(String name, String description, String contentType, String relationKey, String converterClassName) {
49 		this.name = name == null ? "Return Value" : name;
50 		this.description = description;
51 		this.contentType = contentType;
52 		this.relationKey = relationKey;
53 		this.converterClassName = converterClassName;
54 		this.fieldName = "field" + TypeUtils.deriveIdentifierPart(this.name); //$NON-NLS-1$
55 	}
56 
57 	public String getName() {
58 		return name;
59 	}
60 
61 	public String getDescription() {
62 		return description;
63 	}
64 
65 	public String getContentType() {
66 		return contentType;
67 	}
68 
69 	@Override
70 	public String getRelationKey() {
71 		return relationKey;
72 	}
73 
74 	@Override
75 	public String getConverterClassName() {
76 		return converterClassName;
77 	}
78 
79 	public String getFieldName() {
80 		return fieldName;
81 	}
82 
83 }
