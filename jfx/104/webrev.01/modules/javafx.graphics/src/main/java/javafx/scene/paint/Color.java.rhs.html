  39 /**
  40  * The Color class is used to encapsulate colors in the default sRGB color space.
  41  * Every color has an implicit alpha value of 1.0 or an explicit one provided
  42  * in the constructor. The alpha value defines the transparency of a color
  43  * and can be  represented by a double value in the range 0.0-1.0 or 0-255.
  44  * An alpha value of 1.0 or 255 means that the color is completely opaque
  45  * and an alpha value of 0 or 0.0 means that the color is completely transparent.
  46  * When constructing a {@code Color} with an explicit alpha or getting
  47  * the color/alpha components of a Color,
  48  * the color components are never premultiplied by the alpha component.
  49  *
  50  *
  51  * <p>{@code Color}s can be created with the constructor or with one of several
  52  * utility methods.  The following lines of code all create the same
  53  * blue color:</p>
  54  *
  55  * <pre><code>
  56  * {@literal
  57  * Color c = Color.BLUE;   //use the blue constant
  58  * Color c = new Color(0,0,1,1.0); // standard constructor, use 0->1.0 values, explicit alpha of 1.0
  59  *
  60  * Color c = Color.color(0,0,1.0); //use 0->1.0 values. implicit alpha of 1.0
  61  * Color c = Color.color(0,0,1.0,1.0); //use 0->1.0 values, explicit alpha of 1.0
  62  *
  63  * Color c = Color.rgb(0,0,255); //use 0->255 integers, implicit alpha of 1.0
  64  * Color c = Color.rgb(0,0,255,1.0); //use 0->255 integers, explicit alpha of 1.0
  65  *
  66  * Color c = Color.hsb(270,1.0,1.0); //hue = 270, saturation & value = 1.0. inplicit alpha of 1.0
  67  * Color c = Color.hsb(270,1.0,1.0,1.0); //hue = 270, saturation & value = 1.0, explicit alpha of 1.0
  68  *
  69  * Color c = Color.web("0x0000FF",1.0);// blue as a hex web value, explicit alpha
  70  * Color c = Color.web("0x0000FF");// blue as a hex web value, implicit alpha
  71  * Color c = Color.web("0x00F");// blue as a short hex web value, implicit alpha
  72  * Color c = Color.web("#0000FF",1.0);// blue as a hex web value, explicit alpha
  73  * Color c = Color.web("#0000FF");// blue as a hex web value, implicit alpha
  74  * Color c = Color.web("#00F");// blue as a short hex web value, implicit alpha
  75  * Color c = Color.web("0000FF",1.0);// blue as a hex web value, explicit alpha
  76  * Color c = Color.web("0000FF");// blue as a hex web value, implicit alpha
  77  * Color c = Color.web("00F");// blue as a short hex web value, implicit alpha
  78  * Color c = Color.web("rgba(0,0,255,1.0)");// blue as an rgb web value, explicit alpha
  79  * Color c = Color.web("rgb(0,0,255)");// blue as an rgb web value, implicit alpha
  80  * Color c = Color.web("rgba(0,0,100%,1.0)");// blue as an rgb percent web value, explicit alpha
  81  * Color c = Color.web("rgb(0,0,100%)");// blue as an rgb percent web value, implicit alpha
  82  * Color c = Color.web("hsla(270,100%,100%,1.0)");// blue as an hsl web value, explicit alpha
  83  * Color c = Color.web("hsl(270,100%,100%)");// blue as an hsl web value, implicit alpha
  84  * }
  85  * </code></pre>
  86  *
  87  * <p>
  88  * The creation of a {@code Color} will throw {@code IllegalArgumentException} if any
  89  * of the values are out of range.
  90  * </p>
  91  *
  92  * <p>
  93  * For example:
  94  * <pre>{@code
  95  * Rectangle rec1 = new Rectangle(5, 5, 50, 40);
  96  * rec1.setFill(Color.RED);
  97  * rec1.setStroke(Color.GREEN);
  98  * rec1.setStrokeWidth(3);
  99  *
 100  * Rectangle rec2 = new Rectangle(65, 5, 50, 40);
 101  * rec2.setFill(Color.rgb(91, 127, 255));
 102  * rec2.setStroke(Color.hsb(40, 0.7, 0.8));
 103  * rec2.setStrokeWidth(3);
 104  * }</pre>
 105  *
 106  * @since JavaFX 2.0
 107  */

 284     /**
 285      * Creates an RGB color specified with an HTML or CSS attribute string.
 286      *
 287      * <p>
 288      * This method supports the following formats:
 289      * <ul>
 290      * <li>Any standard HTML color name
 291      * <li>An HTML long or short format hex string with an optional hex alpha
 292      * channel.
 293      * Hexadecimal values may be preceded by either {@code "0x"} or {@code "#"}
 294      * and can either be 2 digits in the range {@code 00} to {@code 0xFF} or a
 295      * single digit in the range {@code 0} to {@code F}.
 296      * <li>An {@code rgb(r,g,b)} or {@code rgba(r,g,b,a)} format string.
 297      * Each of the {@code r}, {@code g}, or {@code b} values can be an integer
 298      * from 0 to 255 or a floating point percentage value from 0.0 to 100.0
 299      * followed by the percent ({@code %}) character.
 300      * The alpha component, if present, is a
 301      * floating point value from 0.0 to 1.0.  Spaces are allowed before or
 302      * after the numbers and between the percentage number and its percent
 303      * sign ({@code %}).
 304      * <li>An {@code hsl(h,s,l)} or {@code hsla(h,s,l,a)} format string.
 305      * The {@code h} value is a floating point number from 0.0 to 360.0
 306      * representing the hue angle on a color wheel in degrees with
 307      * {@code 0.0} or {@code 360.0} representing red, {@code 120.0}
 308      * representing green, and {@code 240.0} representing blue.  The
 309      * {@code s} value is the saturation of the desired color represented
 310      * as a floating point percentage from gray ({@code 0.0}) to
 311      * the fully saturated color ({@code 100.0}) and the {@code l} value
 312      * is the desired lightness or brightness of the desired color represented
 313      * as a floating point percentage from black ({@code 0.0}) to the full
 314      * brightness of the color ({@code 100.0}).
 315      * The alpha component, if present, is a floating
 316      * point value from 0.0 to 1.0.  Spaces are allowed before or
 317      * after the numbers and between the percentage number and its percent
 318      * sign ({@code %}).
 319      * </ul>
 320      *
 321      * <p>For formats without an alpha component and for named colors, opacity
 322      * is set according to the {@code opacity} argument. For colors specified
 323      * with an alpha component, the resulting opacity is a combination of the
 324      * parsed alpha component and the {@code opacity} argument, so a
 325      * transparent color becomes more transparent by specifying opacity.</p>

 384      * @param opacity the opacity component in range from 0.0 (transparent)
 385      *                to 1.0 (opaque)
 386      * @return the RGB color specified with the colorString

 391     public static Color web(String colorString, double opacity) {
 392         if (colorString == null) {
 393             throw new NullPointerException(
 394                     "The color components or name must be specified");
 395         }
 396         if (colorString.isEmpty()) {
 397             throw new IllegalArgumentException("Invalid color specification");
 398         }
 399 
 400         String color = colorString.toLowerCase(Locale.ROOT);
 401 
 402         if (color.startsWith("#")) {
 403             color = color.substring(1);
 404         } else if (color.startsWith("0x")) {
 405             color = color.substring(2);
 406         } else if (color.startsWith("rgb")) {
 407             if (color.startsWith("(", 3)) {
 408                 return parseRGBColor(color, 4, false, opacity);
 409             } else if (color.startsWith("a(", 3)) {
 410                 return parseRGBColor(color, 5, true, opacity);
 411             }
 412         } else if (color.startsWith("hsl")) {
 413             if (color.startsWith("(", 3)) {
 414                 return parseHSLColor(color, 4, false, opacity);
 415             } else if (color.startsWith("a(", 3)) {
 416                 return parseHSLColor(color, 5, true, opacity);
 417             }
 418         } else {
 419             Color col = NamedColors.get(color);
 420             if (col != null) {
 421                 if (opacity == 1.0) {
 422                     return col;
 423                 } else {
 424                     return Color.color(col.red, col.green, col.blue, opacity);
 425                 }
 426             }
 427         }
 428 
 429         int len = color.length();
 430 
 431         try {
 432             int r;
 433             int g;
 434             int b;
 435             int a;
 436 
 437             if (len == 3) {
 438                 r = Integer.parseInt(color.substring(0, 1), 16);
 439                 g = Integer.parseInt(color.substring(1, 2), 16);
 440                 b = Integer.parseInt(color.substring(2, 3), 16);
 441                 return Color.color(r / 15.0, g / 15.0, b / 15.0, opacity);
 442             } else if (len == 4) {
 443                 r = Integer.parseInt(color.substring(0, 1), 16);
 444                 g = Integer.parseInt(color.substring(1, 2), 16);
 445                 b = Integer.parseInt(color.substring(2, 3), 16);
 446                 a = Integer.parseInt(color.substring(3, 4), 16);
 447                 return Color.color(r / 15.0, g / 15.0, b / 15.0,
 448                         opacity * a / 15.0);
 449             } else if (len == 6) {
 450                 r = Integer.parseInt(color.substring(0, 2), 16);
 451                 g = Integer.parseInt(color.substring(2, 4), 16);
 452                 b = Integer.parseInt(color.substring(4, 6), 16);
 453                 return Color.rgb(r, g, b, opacity);
 454             } else if (len == 8) {
 455                 r = Integer.parseInt(color.substring(0, 2), 16);
 456                 g = Integer.parseInt(color.substring(2, 4), 16);
 457                 b = Integer.parseInt(color.substring(4, 6), 16);
 458                 a = Integer.parseInt(color.substring(6, 8), 16);
 459                 return Color.rgb(r, g, b, opacity * a / 255.0);
 460             }
 461         } catch (NumberFormatException nfe) {}
 462 
 463         throw new IllegalArgumentException("Invalid color specification");
 464     }

 466     private static Color parseRGBColor(String color, int roff,
 467                                        boolean hasAlpha, double a)
 468     {
 469         try {
 470             int rend = color.indexOf(',', roff);
 471             int gend = rend < 0 ? -1 : color.indexOf(',', rend+1);
 472             int bend = gend < 0 ? -1 : color.indexOf(hasAlpha ? ',' : ')', gend+1);
 473             int aend = hasAlpha ? (bend < 0 ? -1 : color.indexOf(')', bend+1)) : bend;
 474             if (aend >= 0) {
 475                 double r = parseComponent(color, roff, rend, PARSE_COMPONENT);
 476                 double g = parseComponent(color, rend+1, gend, PARSE_COMPONENT);
 477                 double b = parseComponent(color, gend+1, bend, PARSE_COMPONENT);
 478                 if (hasAlpha) {
 479                     a *= parseComponent(color, bend+1, aend, PARSE_ALPHA);
 480                 }
 481                 return new Color(r, g, b, a);
 482             }
 483         } catch (NumberFormatException nfe) {}
 484 
 485         throw new IllegalArgumentException("Invalid color specification");
 486     }

 488     private static Color parseHSLColor(String color, int hoff,
 489                                        boolean hasAlpha, double a)
 490     {
 491         try {
 492             int hend = color.indexOf(',', hoff);
 493             int send = hend < 0 ? -1 : color.indexOf(',', hend+1);
 494             int lend = send < 0 ? -1 : color.indexOf(hasAlpha ? ',' : ')', send+1);
 495             int aend = hasAlpha ? (lend < 0 ? -1 : color.indexOf(')', lend+1)) : lend;
 496             if (aend >= 0) {
 497                 double h = parseComponent(color, hoff, hend, PARSE_ANGLE);
 498                 double s = parseComponent(color, hend+1, send, PARSE_PERCENT);
 499                 double l = parseComponent(color, send+1, lend, PARSE_PERCENT);
 500                 if (hasAlpha) {
 501                     a *= parseComponent(color, lend+1, aend, PARSE_ALPHA);
 502                 }
 503                 return Color.hsb(h, s, l, a);
 504             }
 505         } catch (NumberFormatException nfe) {}
 506 
 507         throw new IllegalArgumentException("Invalid color specification");
 508     }

 510     private static final int PARSE_COMPONENT = 0; // percent, or clamped to [0,255] => [0,1]
 511     private static final int PARSE_PERCENT = 1; // clamped to [0,100]% => [0,1]
 512     private static final int PARSE_ANGLE = 2; // clamped to [0,360]
 513     private static final int PARSE_ALPHA = 3; // clamped to [0.0,1.0]
 514     private static double parseComponent(String color, int off, int end, int type) {
 515         color = color.substring(off, end).trim();
 516         if (color.endsWith("%")) {
 517             if (type > PARSE_PERCENT) {
 518                 throw new IllegalArgumentException("Invalid color specification");
 519             }
 520             type = PARSE_PERCENT;
 521             color = color.substring(0, color.length()-1).trim();
 522         } else if (type == PARSE_PERCENT) {
 523             throw new IllegalArgumentException("Invalid color specification");
 524         }
 525         double c = ((type == PARSE_COMPONENT)
 526                     ? Integer.parseInt(color)
 527                     : Double.parseDouble(color));
 528         switch (type) {
 529             case PARSE_ALPHA:
 530                 return (c < 0.0) ? 0.0 : ((c > 1.0) ? 1.0 : c);
 531             case PARSE_PERCENT:
 532                 return (c <= 0.0) ? 0.0 : ((c >= 100.0) ? 1.0 : (c / 100.0));
 533             case PARSE_COMPONENT:
 534                 return (c <= 0.0) ? 0.0 : ((c >= 255.0) ? 1.0 : (c / 255.0));
 535             case PARSE_ANGLE:
 536                 return ((c < 0.0)
 537                         ? ((c % 360.0) + 360.0)
 538                         : ((c > 360.0)
 539                             ? (c % 360.0)
 540                             : c));
 541         }
 542 
 543         throw new IllegalArgumentException("Invalid color specification");
 544     }

 705     /**
 706      * Creates a new {@code Color} based on this {@code Color} with hue,
 707      * saturation, brightness and opacity values altered. Hue is shifted
 708      * about the given value and normalized into its natural range, the
 709      * other components' values are multiplied by the given factors and
 710      * clipped into their ranges.
 711      *
 712      * Increasing brightness of black color is allowed by using an arbitrary,
 713      * very small source brightness instead of zero.
 714      * @param hueShift the hue shift
 715      * @param saturationFactor the saturation factor
 716      * @param brightnessFactor the brightness factor
 717      * @param opacityFactor the opacity factor
 718      * @return a {@code Color} based based on this {@code Color} with hue,
 719      * saturation, brightness and opacity values altered.
 720      */
 721     public Color deriveColor(double hueShift, double saturationFactor,
 722                              double brightnessFactor, double opacityFactor) {
 723 
 724         double[] hsb = Utils.RGBtoHSB(red, green, blue);
 725 
 726         /* Allow brightness increase of black color */
 727         double b = hsb[2];
 728         if (b == 0 && brightnessFactor > 1.0) {
 729             b = 0.05;
 730         }
 731 
 732         /* the tail "+ 360) % 360" solves shifts into negative numbers */
 733         double h = (((hsb[0] + hueShift) % 360) + 360) % 360;
 734         double s = Math.max(Math.min(hsb[1] * saturationFactor, 1.0), 0.0);
 735         b = Math.max(Math.min(b * brightnessFactor, 1.0), 0.0);
 736         double a = Math.max(Math.min(opacity * opacityFactor, 1.0), 0.0);
 737         return hsb(h, s, b, a);
 738     }
