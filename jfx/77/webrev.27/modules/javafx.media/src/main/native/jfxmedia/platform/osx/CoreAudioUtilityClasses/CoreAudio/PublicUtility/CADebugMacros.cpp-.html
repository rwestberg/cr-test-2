 9 #include "CADebugMacros.h"
10 #include <stdio.h>
11 #include <stdarg.h>
12 #if TARGET_API_MAC_OSX
13     #include <syslog.h>
14 #endif
15 
16 #if DEBUG
17 #include <stdio.h>
18 
19 void    DebugPrint(const char *fmt, ...)
20 {
21     va_list args;
22     va_start(args, fmt);
23     vprintf(fmt, args);
24     va_end(args);
25 }
26 #endif // DEBUG
27 
28 void    LogError(const char *fmt, ...)
29 {
30     va_list args;
31     va_start(args, fmt);
32 #if DEBUG
33     vprintf(fmt, args);
34 #endif
35 #if TARGET_API_MAC_OSX
36     vsyslog(LOG_ERR, fmt, args);
37 #endif
38     va_end(args);
39 }
40 
41 void    LogWarning(const char *fmt, ...)
42 {
43     va_list args;
44     va_start(args, fmt);
45 #if DEBUG
46     vprintf(fmt, args);
47 #endif
48 #if TARGET_API_MAC_OSX
49     vsyslog(LOG_WARNING, fmt, args);
50 #endif
51     va_end(args);
52 }
