26 package com.sun.glass.ui.monocle;
27 import com.sun.glass.utils.NativeLibLoader;
28 
29 class AndroidPlatform extends NativePlatform {
30 
31     AndroidPlatform() {
32         NativeLibLoader.loadLibrary("glass_monocle");
33     }
34 
35     @Override
36     protected InputDeviceRegistry createInputDeviceRegistry() {
37         return AndroidInputDeviceRegistry.getInstance();
38     }
39 
40     @Override
41     protected NativeCursor createCursor() {
42         return logSelectedCursor(new NullCursor());
43     }
44 
45     @Override
46     protected NativeScreen createScreen() {
47         return new AndroidScreen();
48     }
49 
50     /** Create the accelerated screen for this platform
51      *
52      * @param attributes a sequence of pairs (GLAttibute, value)
53      * @throws GLException
54      */
55     @Override
56     public synchronized AcceleratedScreen getAcceleratedScreen(
57             int[] attributes) throws GLException {
58         if (accScreen == null) {
59             accScreen = new AndroidAcceleratedScreen(attributes);
60         }
61         return accScreen;
62     }
63 
64 }
