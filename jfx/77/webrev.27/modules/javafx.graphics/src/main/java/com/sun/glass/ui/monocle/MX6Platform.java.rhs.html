26 package com.sun.glass.ui.monocle;
27 
28 class MX6Platform extends LinuxPlatform {
29 
30     @Override
31     protected NativeCursor createCursor() {
32         final NativeCursor c = useCursor ? new MX6Cursor() : new NullCursor();
33         return logSelectedCursor(c);
34     }
35 
36     @Override
37     public synchronized AcceleratedScreen getAcceleratedScreen(int[] attributes)
38             throws GLException {
39         if (accScreen == null) {
40             accScreen = new MX6AcceleratedScreen(attributes);
41         }
42         return accScreen;
43     }
44 
45 }
