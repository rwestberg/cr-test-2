26 // main fragment shader
27 
28 #ifdef GL_ES
29 
30 #ifndef EXTENSION_APPLIED
31 #define EXTENSION_APPLIED
32 #extension GL_OES_standard_derivatives : enable
33 #endif
34 
35 // Define default float precision for fragment shaders
36 #ifdef GL_FRAGMENT_PRECISION_HIGH
37 precision highp float;
38 precision highp int;
39 #else
40 precision mediump float;
41 precision mediump int;
42 #endif
43 
44 #else
45 
46 // Ignore GL_ES precision specifiers:
47 #define lowp
48 #define mediump
49 #define highp
50 
51 #endif
52 
53 vec4 apply_diffuse();
54 vec4 apply_selfIllum();
55 
56 uniform vec3 ambientColor;
57 
58 varying vec3 eyePos;
59 
60 void main()
61 {
62     gl_FragColor = vec4(0.0,0.0,0.0,1.0);
63     vec4 diffuse = apply_diffuse();
64 
65     if (diffuse.a == 0.0) discard;
66 
67     vec3 rez = (ambientColor) * diffuse.xyz;
68     rez += apply_selfIllum().xyz;
69 
70     gl_FragColor = vec4(clamp(rez, 0.0, 1.0) , diffuse.a);
71 }
