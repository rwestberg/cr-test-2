 26 package javafx.animation;
 27 
 28 import com.sun.scenario.animation.AbstractMasterTimer;
 29 import com.sun.scenario.animation.shared.ClipEnvelope;
 30 import com.sun.scenario.animation.shared.PulseReceiver;
 31 import javafx.util.Duration;
 32 
 33 public abstract class AnimationShim extends Animation {
 34 
 35     public AnimationShim() {
 36         super();
 37     }
 38 
 39     public AnimationShim(AbstractMasterTimer timer) {
 40         super(timer);
 41     }
 42 
 43     public AnimationShim(AbstractMasterTimer timer, ClipEnvelope clipEnvelope, int resolution) {
 44         super(timer, clipEnvelope, resolution);
 45     }
 46 
 47     public ClipEnvelope get_clipEnvelope() {
 48         return clipEnvelope;
 49     }
 50 
 51     public void setClipEnvelope(ClipEnvelope clipEnvelope) {
 52         this.clipEnvelope= clipEnvelope;
 53     }
 54 
 55     @Override
 56     public void doPause() {
 57         super.doPause();
 58     }
 59 
 60     @Override
 61     public void doStart(boolean forceSync) {
 62         super.doStart(forceSync);
 63     }
 64 
 65     @Override
 66     public void setCurrentRate(double currentRate) {
 67         super.setCurrentRate(currentRate);
 68     }
 69 
 70     @Override
 71     public void setCurrentTicks(long ticks) {
 72         super.setCurrentTicks(ticks);
 73     }
 74 
 75     @Override
 76     public boolean startable(boolean forceSync) {
 77         return super.startable(forceSync);
 78     }
 79 
 80     @Override
 81     public void doStop() {
 82         super.doStop();
 83     }
 84 
 85     @Override
 86     public void sync(boolean forceSync) {
 87         super.sync(forceSync);
 88     }
 89 
 90     @Override
 91     public void doTimePulse(long elapsedTime) {
 92         super.doTimePulse(elapsedTime);
 93     }
 94 
 95     @Override
 96     public void pauseReceiver() {
 97         super.pauseReceiver();
 98     }
 99 
100     @Override
101     public void resumeReceiver() {
102         super.resumeReceiver();
103     }
104 
105     public void shim_setCycleDuration(Duration value) {
106         setCycleDuration(value);
107     }
108 
109     @Override
110     public void startReceiver(long delay) {
111         super.startReceiver(delay);
112     }
113 
114     public PulseReceiver shim_pulseReceiver() {
115         return pulseReceiver;
116     }
117 
118     public void shim_finished() {
119         finished();
120     }
121 
122     @Override
123     abstract public void doPlayTo(long currentTicks, long cycleTicks);
124 
125     @Override
126     abstract public void doJumpTo(long currentTicks, long cycleTicks, boolean forceJump);
127 
128     //-------------------------------
129 
130     public static void finished(Animation a) {
131         a.finished();
132     }
133 
134     public static void doStart(Animation a, boolean forceSync) {
135         a.doStart(forceSync);
136     }
137 
138     public static boolean startable(Animation a, boolean forceSync) {
139         return a.startable(forceSync);
140     }
141 
142 }
