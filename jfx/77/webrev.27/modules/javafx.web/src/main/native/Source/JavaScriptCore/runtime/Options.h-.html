 42 // How do JSC VM options work?
 43 // ===========================
 44 // The JSC_OPTIONS() macro below defines a list of all JSC options in use,
 45 // along with their types and default values. The options values are actually
 46 // realized as an array of Options::Entry elements.
 47 //
 48 //     Options::initialize() will initialize the array of options values with
 49 // the defaults specified in JSC_OPTIONS() below. After that, the values can
 50 // be programmatically read and written to using an accessor method with the
 51 // same name as the option. For example, the option "useJIT" can be read and
 52 // set like so:
 53 //
 54 //     bool jitIsOn = Options::useJIT();  // Get the option value.
 55 //     Options::useJIT() = false;         // Sets the option value.
 56 //
 57 //     If you want to tweak any of these values programmatically for testing
 58 // purposes, you can do so in Options::initialize() after the default values
 59 // are set.
 60 //
 61 //     Alternatively, you can override the default values by specifying
 62 // environment variables of the form: JSC_<name of JSC option>.
 63 //
 64 // Note: Options::initialize() tries to ensure some sanity on the option values
 65 // which are set by doing some range checks, and value corrections. These
 66 // checks are done after the option values are set. If you alter the option
 67 // values after the sanity checks (for your own testing), then you're liable to
 68 // ensure that the new values set are sane and reasonable for your own run.
 69 
 70 class OptionRange {
 71 private:
 72     enum RangeState { Uninitialized, InitError, Normal, Inverted };
 73 public:
 74     OptionRange& operator= (const int& rhs)
 75     { // Only needed for initialization
 76         if (!rhs) {
 77             m_state = Uninitialized;
 78             m_rangeString = 0;
 79             m_lowLimit = 0;
 80             m_highLimit = 0;
 81         }
 82         return *this;
 83     }
 84 
 85     bool init(const char*);
 86     bool isInRange(unsigned);
 87     const char* rangeString() const { return (m_state > InitError) ? m_rangeString : s_nullRangeStr; }
 88 
 89     void dump(PrintStream& out) const;
 90 
 91 private:
 92     static const char* const s_nullRangeStr;
 93 
 94     RangeState m_state;
 95     const char* m_rangeString;
 96     unsigned m_lowLimit;
 97     unsigned m_highLimit;
 98 };

556 class Options {
557 public:
558     enum class DumpLevel {
559         None = 0,
560         Overridden,
561         All,
562         Verbose
563     };
564 
565     enum class Availability {
566         Normal = 0,
567         Restricted,
568         Configurable
569     };
570 
571     // This typedef is to allow us to eliminate the '_' in the field name in
572     // union inside Entry. This is needed to keep the style checker happy.
573     typedef int32_t int32;
574     typedef size_t size;
575 
576     // Declare the option IDs:
577     enum ID {
578 #define FOR_EACH_OPTION(type_, name_, defaultValue_, availability_, description_) \
579         name_##ID,
580         JSC_OPTIONS(FOR_EACH_OPTION)
581 #undef FOR_EACH_OPTION
582         numberOfOptions
583     };
584 
585     enum class Type {
586         boolType,
587         unsignedType,
588         doubleType,
589         int32Type,
590         sizeType,
591         optionRangeType,
592         optionStringType,
593         gcLogLevelType,
594     };
595 
596     JS_EXPORT_PRIVATE static void initialize();
597 
598     // Parses a string of options where each option is of the format "--<optionName>=<value>"
599     // and are separated by a space. The leading "--" is optional and will be ignored.
600     JS_EXPORT_PRIVATE static bool setOptions(const char* optionsList);
601 
602     // Parses a single command line option in the format "<optionName>=<value>"
603     // (no spaces allowed) and set the specified option if appropriate.
604     JS_EXPORT_PRIVATE static bool setOption(const char* arg);
605 
606     JS_EXPORT_PRIVATE static void dumpAllOptions(FILE*, DumpLevel, const char* title = nullptr);
607     JS_EXPORT_PRIVATE static void dumpAllOptionsInALine(StringBuilder&);
608 
609     JS_EXPORT_PRIVATE static void ensureOptionsAreCoherent();
610 
611     JS_EXPORT_PRIVATE static void enableRestrictedOptions(bool enableOrNot);
612 
613     // Declare accessors for each option:
614 #define FOR_EACH_OPTION(type_, name_, defaultValue_, availability_, description_) \
615     ALWAYS_INLINE static type_& name_() { return s_options[name_##ID].type_##Val; } \
616     ALWAYS_INLINE static type_& name_##Default() { return s_defaultOptions[name_##ID].type_##Val; }
617 
618     JSC_OPTIONS(FOR_EACH_OPTION)
619 #undef FOR_EACH_OPTION
620 
621     static bool isAvailable(ID, Availability);
622 
623 private:
624     // For storing for an option value:
625     union Entry {
626         bool boolVal;
627         unsigned unsignedVal;
628         double doubleVal;
629         int32 int32Val;
630         size sizeVal;
631         OptionRange optionRangeVal;
632         const char* optionStringVal;
633         GCLogging::Level gcLogLevelVal;
634     };
635 
636     // For storing constant meta data about each option:
637     struct EntryInfo {
638         const char* name;
639         const char* description;
640         Type type;
641         Availability availability;
642     };
643 
644     Options();
645 
646     enum DumpDefaultsOption {
647         DontDumpDefaults,
648         DumpDefaults
649     };
650     static void dumpOptionsIfNeeded();
651     static void dumpAllOptions(StringBuilder&, DumpLevel, const char* title,
652         const char* separator, const char* optionHeader, const char* optionFooter, DumpDefaultsOption);
653     static void dumpOption(StringBuilder&, DumpLevel, ID,
654         const char* optionHeader, const char* optionFooter, DumpDefaultsOption);
655 
656     static bool setOptionWithoutAlias(const char* arg);
657     static bool setAliasedOption(const char* arg);
658     static bool overrideAliasedOptionWithHeuristic(const char* name);
659 
660     // Declare the singleton instance of the options store:
661     JS_EXPORT_PRIVATE static Entry s_options[numberOfOptions];
662     static Entry s_defaultOptions[numberOfOptions];
663     static const EntryInfo s_optionsInfo[numberOfOptions];
664 
665     friend class Option;
666 };

677 
678     bool operator==(const Option& other) const;

682     const char* name() const;
683     const char* description() const;
684     Options::Type type() const;
685     Options::Availability availability() const;

695     GCLogging::Level& gcLogLevelVal();

700         : m_id(id)

703     }

707 };
