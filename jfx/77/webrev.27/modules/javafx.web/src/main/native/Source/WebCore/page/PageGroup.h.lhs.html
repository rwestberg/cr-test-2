26 #pragma once
27 
28 #include <wtf/HashSet.h>
29 #include <wtf/Noncopyable.h>
30 #include <wtf/text/WTFString.h>
31 
32 namespace WebCore {
33 
34 class Page;
35 #if ENABLE(VIDEO_TRACK)
36 class CaptionUserPreferences;
37 #endif
38 
39 class PageGroup {
40     WTF_MAKE_NONCOPYABLE(PageGroup); WTF_MAKE_FAST_ALLOCATED;
41 public:
42     WEBCORE_EXPORT explicit PageGroup(const String& name);
43     explicit PageGroup(Page&);
44     ~PageGroup();
45 
46     WEBCORE_EXPORT static PageGroup* pageGroup(const String& groupName);
47 
48     const HashSet<Page*>& pages() const { return m_pages; }
49 
50     void addPage(Page&);
51     void removePage(Page&);
52 
53     const String& name() { return m_name; }
54     unsigned identifier() { return m_identifier; }
55 
56 #if ENABLE(VIDEO_TRACK)
57     WEBCORE_EXPORT void captionPreferencesChanged();
58     WEBCORE_EXPORT CaptionUserPreferences& captionPreferences();
59 #endif
60 


61 private:
62     String m_name;
63     HashSet<Page*> m_pages;
64 
65     unsigned m_identifier;
66 
67 #if ENABLE(VIDEO_TRACK)
68     std::unique_ptr<CaptionUserPreferences> m_captionPreferences;
69 #endif


70 };
71 
72 } // namespace WebCore
