26 #pragma once
27 


28 #include "GraphicsContext3D.h"
29 #include <wtf/RefCounted.h>
30 #include <wtf/text/WTFString.h>
31 
32 namespace WebCore {
33 
34 class WebGLActiveInfo : public RefCounted<WebGLActiveInfo> {
35 public:
36     static Ref<WebGLActiveInfo> create(const String& name, GC3Denum type, GC3Dint size)
37     {
38         return adoptRef(*new WebGLActiveInfo(name, type, size));
39     }
40     String name() const { return m_name; }
41     GC3Denum type() const { return m_type; }
42     GC3Dint size() const { return m_size; }
43 
44 private:
45     WebGLActiveInfo(const String& name, GC3Denum type, GC3Dint size)
46         : m_name(name)
47         , m_type(type)
48         , m_size(size)
49     {
50         ASSERT(name.length());
51         ASSERT(type);
52         ASSERT(size);
53     }
54     String m_name;
55     GC3Denum m_type;
56     GC3Dint m_size;
57 };
58 
59 } // namespace WebCore
