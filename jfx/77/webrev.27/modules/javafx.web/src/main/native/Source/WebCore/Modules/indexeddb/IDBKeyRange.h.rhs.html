40 namespace WebCore {
41 
42 class IDBKey;
43 class ScriptExecutionContext;
44 
45 class IDBKeyRange final : public ScriptWrappable, public RefCounted<IDBKeyRange> {
46     WTF_MAKE_ISO_ALLOCATED(IDBKeyRange);
47 public:
48     static Ref<IDBKeyRange> create(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
49     static Ref<IDBKeyRange> create(RefPtr<IDBKey>&&);
50     ~IDBKeyRange();
51 
52     IDBKey* lower() const { return m_lower.get(); }
53     IDBKey* upper() const { return m_upper.get(); }
54     bool lowerOpen() const { return m_isLowerOpen; }
55     bool upperOpen() const { return m_isUpperOpen; }
56 
57     static ExceptionOr<Ref<IDBKeyRange>> only(RefPtr<IDBKey>&& value);
58     static ExceptionOr<Ref<IDBKeyRange>> only(JSC::ExecState&, JSC::JSValue key);
59 
60     static ExceptionOr<Ref<IDBKeyRange>> lowerBound(JSC::ExecState&, JSC::JSValue bound, bool open);
61     static ExceptionOr<Ref<IDBKeyRange>> upperBound(JSC::ExecState&, JSC::JSValue bound, bool open);
62 
63     static ExceptionOr<Ref<IDBKeyRange>> bound(JSC::ExecState&, JSC::JSValue lower, JSC::JSValue upper, bool lowerOpen, bool upperOpen);
64 
65     ExceptionOr<bool> includes(JSC::ExecState&, JSC::JSValue key);
66 
67     WEBCORE_EXPORT bool isOnlyKey() const;
68 
69 private:
70     IDBKeyRange(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
71 
72     RefPtr<IDBKey> m_lower;
73     RefPtr<IDBKey> m_upper;
74     bool m_isLowerOpen;
75     bool m_isUpperOpen;
76 };
