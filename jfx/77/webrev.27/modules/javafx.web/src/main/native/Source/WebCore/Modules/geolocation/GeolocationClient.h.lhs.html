26 #pragma once
27 
28 #include <wtf/Optional.h>
29 
30 namespace WebCore {
31 
32 class Geolocation;
33 class GeolocationPosition;
34 class Page;
35 
36 class GeolocationClient {
37 public:
38     virtual void geolocationDestroyed() = 0;
39 
40     virtual void startUpdating() = 0;
41     virtual void stopUpdating() = 0;
42     // FIXME: The V2 Geolocation specification proposes that this property is
43     // renamed. See http://www.w3.org/2008/geolocation/track/issues/6
44     // We should update WebKit to reflect this if and when the V2 specification
45     // is published.
46     virtual void setEnableHighAccuracy(bool) = 0;
47     virtual Optional<GeolocationPosition> lastPosition() = 0;
48 
49     virtual void requestPermission(Geolocation&) = 0;
50     virtual void cancelPermissionRequest(Geolocation&) = 0;
51 
52     void provideGeolocationTo(Page*, GeolocationClient&);
53 
54 protected:
55     virtual ~GeolocationClient() = default;
56 };
57 
58 WEBCORE_EXPORT void provideGeolocationTo(Page*, GeolocationClient&);
59 
60 } // namespace WebCore
