26 #pragma once
27 
28 #include "DeclarativeAnimation.h"
29 #include <wtf/Ref.h>
30 
31 namespace WebCore {
32 
33 class Animation;
34 class Element;
35 class RenderStyle;
36 
37 class CSSAnimation final : public DeclarativeAnimation {
38 public:
39     static Ref<CSSAnimation> create(Element&, const Animation&, const RenderStyle* oldStyle, const RenderStyle& newStyle);
40     ~CSSAnimation() = default;
41 
42     bool isCSSAnimation() const override { return true; }
43     const String& animationName() const { return m_animationName; }
44     const RenderStyle& unanimatedStyle() const { return *m_unanimatedStyle; }
45 
46     ExceptionOr<void> bindingsPlay() final;
47     ExceptionOr<void> bindingsPause() final;
48 
49 protected:
50     void syncPropertiesWithBackingAnimation() final;
51 
52 private:
53     CSSAnimation(Element&, const Animation&, const RenderStyle&);
54 
55     String m_animationName;
56     std::unique_ptr<RenderStyle> m_unanimatedStyle;
57     bool m_stickyPaused { false };
58 };
59 
60 } // namespace WebCore
61 
62 SPECIALIZE_TYPE_TRAITS_WEB_ANIMATION(CSSAnimation, isCSSAnimation())
