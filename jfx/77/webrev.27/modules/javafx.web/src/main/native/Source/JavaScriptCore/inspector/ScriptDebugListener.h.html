31 
32 #include "Debugger.h"
33 #include "SourceProvider.h"
34 #include <wtf/text/WTFString.h>
35 
36 namespace Inspector {
37 
38 struct ScriptBreakpointAction;
39 
40 class ScriptDebugListener {
41 public:
42     struct Script {
43         String url;
44         String source;
45         String sourceURL;
46         String sourceMappingURL;
47         RefPtr<JSC::SourceProvider> sourceProvider;
48         int startLine {0};
49         int startColumn {0};
50         int endLine {0};
51         int endColumn {0};
52         bool isContentScript {false};
53     };
54 
55     virtual ~ScriptDebugListener() { }
56 
57     virtual void didParseSource(JSC::SourceID, const Script&) = 0;
58     virtual void failedToParseSource(const String& url, const String& data, int firstLine, int errorLine, const String& errorMessage) = 0;
59 
60     virtual void willRunMicrotask() = 0;
61     virtual void didRunMicrotask() = 0;
62 
63     virtual void didPause(JSC::ExecState&, JSC::JSValue callFrames, JSC::JSValue exception) = 0;
64     virtual void didContinue() = 0;
65 
66     virtual void breakpointActionLog(JSC::ExecState&, const String&) = 0;
67     virtual void breakpointActionSound(int breakpointActionIdentifier) = 0;
68     virtual void breakpointActionProbe(JSC::ExecState&, const ScriptBreakpointAction&, unsigned batchId, unsigned sampleId, JSC::JSValue result) = 0;
69 };
70 
71 } // namespace Inspector
