27 #pragma once
28 
29 #include <wtf/text/WTFString.h>
30 
31 namespace WebCore {
32 
33 class ContentSecurityPolicyDirectiveList;
34 
35 class ContentSecurityPolicyDirective {
36     WTF_MAKE_FAST_ALLOCATED;
37 public:
38     ContentSecurityPolicyDirective(const ContentSecurityPolicyDirectiveList& directiveList, const String& name, const String& value)
39         : m_name(name)
40         , m_text(name + ' ' + value)
41         , m_directiveList(directiveList)
42     {
43     }
44 
45     const String& name() const { return m_name; }
46     const String& text() const { return m_text; }
47 
48     const ContentSecurityPolicyDirectiveList& directiveList() const { return m_directiveList; }
49 
50     bool isDefaultSrc() const;
51 
52 private:
53     String m_name;
54     String m_text;
55     const ContentSecurityPolicyDirectiveList& m_directiveList;
56 };
57 
58 } // namespace WebCore
