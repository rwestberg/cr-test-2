27 #pragma once
28 
29 #include "ProgressEvent.h"
30 
31 namespace WebCore {
32 
33 class XMLHttpRequestProgressEvent : public ProgressEvent {
34 public:
35     static Ref<XMLHttpRequestProgressEvent> create(const AtomString& type, bool lengthComputable = false, unsigned long long loaded = 0, unsigned long long total = 0)
36     {
37         return adoptRef(*new XMLHttpRequestProgressEvent(type, lengthComputable, loaded, total));
38     }
39     // Those 2 synonyms are included for compatibility with Firefox.
40     unsigned long long position() const { return loaded(); }
41     unsigned long long totalSize() const { return total(); }
42 
43     EventInterface eventInterface() const override { return XMLHttpRequestProgressEventInterfaceType; }
44 
45 private:
46     XMLHttpRequestProgressEvent(const AtomString& type, bool lengthComputable, unsigned long long loaded, unsigned long long total)
47         : ProgressEvent(type, lengthComputable, loaded, total)
48     {
49     }
50 };
51 
52 } // namespace WebCore
