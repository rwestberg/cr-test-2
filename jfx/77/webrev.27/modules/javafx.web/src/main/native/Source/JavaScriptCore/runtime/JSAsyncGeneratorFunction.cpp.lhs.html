42 JSAsyncGeneratorFunction::JSAsyncGeneratorFunction(VM& vm, FunctionExecutable* executable, JSScope* scope, Structure* structure)
43     : Base(vm, executable, scope, structure)
44 {
45 }
46 
47 JSAsyncGeneratorFunction* JSAsyncGeneratorFunction::createImpl(VM& vm, FunctionExecutable* executable, JSScope* scope, Structure* structure)
48 {
49     JSAsyncGeneratorFunction* asyncGenerator = new (NotNull, allocateCell<JSAsyncGeneratorFunction>(vm.heap)) JSAsyncGeneratorFunction(vm, executable, scope, structure);
50     ASSERT(asyncGenerator->structure(vm)->globalObject());
51     asyncGenerator->finishCreation(vm);
52     return asyncGenerator;
53 }
54 
55 JSAsyncGeneratorFunction* JSAsyncGeneratorFunction::create(VM& vm, FunctionExecutable* executable, JSScope* scope)
56 {
57     JSAsyncGeneratorFunction* asyncGenerator = createImpl(vm, executable, scope, scope->globalObject(vm)->asyncGeneratorFunctionStructure());
58     executable->notifyCreation(vm, asyncGenerator, "Allocating an async generator");
59     return asyncGenerator;
60 }
61 
62 JSAsyncGeneratorFunction* JSAsyncGeneratorFunction::create(VM& vm, FunctionExecutable* executable, JSScope* scope, Structure* structure)
63 {
64     JSAsyncGeneratorFunction* asyncGenerator = createImpl(vm, executable, scope, structure);
65     executable->notifyCreation(vm, asyncGenerator, "Allocating an async generator");
66     return asyncGenerator;
67 }
68 
69 JSAsyncGeneratorFunction* JSAsyncGeneratorFunction::createWithInvalidatedReallocationWatchpoint(VM& vm, FunctionExecutable* executable, JSScope* scope)
70 {
71     return createImpl(vm, executable, scope, scope->globalObject(vm)->asyncGeneratorFunctionStructure());
72 }
