42 unsigned UnnamedType::hash() const
43 {
44     switch (kind()) {
45     case Kind::TypeReference:
46         return downcast<TypeReference>(*this).hash();
47     case Kind::Pointer:
48         return downcast<PointerType>(*this).hash();
49     case Kind::ArrayReference:
50         return downcast<ArrayReferenceType>(*this).hash();
51     case Kind::Array:
52         return downcast<ArrayType>(*this).hash();
53     default:
54         RELEASE_ASSERT_NOT_REACHED();
55     }
56 }
57 
58 bool UnnamedType::operator==(const UnnamedType& other) const
59 {
60     if (other.kind() != kind())
61         return false;
62 
63     switch (kind()) {
64     case Kind::TypeReference:
65         return downcast<TypeReference>(*this) == downcast<TypeReference>(other);
66     case Kind::Pointer:
67         return downcast<PointerType>(*this) == downcast<PointerType>(other);
68     case Kind::ArrayReference:
69         return downcast<ArrayReferenceType>(*this) == downcast<ArrayReferenceType>(other);
70     case Kind::Array:
71         return downcast<ArrayType>(*this) == downcast<ArrayType>(other);
72     default:
73         RELEASE_ASSERT_NOT_REACHED();
74     }
75 }
