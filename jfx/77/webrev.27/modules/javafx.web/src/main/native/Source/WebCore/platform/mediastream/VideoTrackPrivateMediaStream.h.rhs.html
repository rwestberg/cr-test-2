27 
28 #if ENABLE(VIDEO_TRACK) && ENABLE(MEDIA_STREAM)
29 
30 #include "MediaStreamTrackPrivate.h"
31 #include "VideoTrackPrivate.h"
32 
33 namespace WebCore {
34 
35 class VideoTrackPrivateMediaStream final : public VideoTrackPrivate {
36     WTF_MAKE_NONCOPYABLE(VideoTrackPrivateMediaStream)
37 public:
38     static RefPtr<VideoTrackPrivateMediaStream> create(MediaStreamTrackPrivate& streamTrack)
39     {
40         return adoptRef(*new VideoTrackPrivateMediaStream(streamTrack));
41     }
42 
43     void setTrackIndex(int index) { m_index = index; }
44 
45     MediaStreamTrackPrivate& streamTrack() { return m_streamTrack.get(); }
46 
47     MediaTime timelineOffset() const { return m_timelineOffset; }
48     void setTimelineOffset(const MediaTime& offset) { m_timelineOffset = offset; }
49 
50 private:
51     VideoTrackPrivateMediaStream(MediaStreamTrackPrivate& track)
52         : m_streamTrack(track)
53         , m_id(track.id())
54         , m_label(track.label())
55         , m_timelineOffset(MediaTime::invalidTime())
56     {
57     }
58 
59     Kind kind() const final { return Kind::Main; }
60     AtomString id() const final { return m_id; }
61     AtomString label() const final { return m_label; }
62     AtomString language() const final { return emptyAtom(); }
63     int trackIndex() const final { return m_index; }
64 
65     Ref<MediaStreamTrackPrivate> m_streamTrack;
66     AtomString m_id;
67     AtomString m_label;
68     int m_index { 0 };
69     MediaTime m_timelineOffset;
70 };
71 
72 }
73 
74 #endif // ENABLE(VIDEO_TRACK) && ENABLE(MEDIA_STREAM)
