23 #include "config.h"
24 #include "MutationEvent.h"
25 
26 namespace WebCore {
27 
28 MutationEvent::MutationEvent(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, Node* relatedNode, const String& prevValue, const String& newValue)
29     : Event(type, canBubble, cancelable)
30     , m_relatedNode(relatedNode)
31     , m_prevValue(prevValue)
32     , m_newValue(newValue)
33 {
34 }
35 
36 void MutationEvent::initMutationEvent(const AtomicString& type, bool canBubble, bool cancelable, Node* relatedNode, const String& prevValue, const String& newValue, const String& attrName, unsigned short attrChange)
37 {
38     if (isBeingDispatched())
39         return;
40 
41     initEvent(type, canBubble, cancelable);
42 
43     m_relatedNode = relatedNode;
44     m_prevValue = prevValue;
45     m_newValue = newValue;
46     m_attrName = attrName;
47     m_attrChange = attrChange;
48 }
49 
50 EventInterface MutationEvent::eventInterface() const
51 {
52     return MutationEventInterfaceType;
53 }
