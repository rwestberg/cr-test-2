31 inline const UChar* ucharFrom(const wchar_t* characters)
32 {
33     static_assert(sizeof(wchar_t) == sizeof(UChar), "We assume wchar_t and UChar have the same size");
34     return reinterpret_cast<const UChar*>(characters);
35 }
