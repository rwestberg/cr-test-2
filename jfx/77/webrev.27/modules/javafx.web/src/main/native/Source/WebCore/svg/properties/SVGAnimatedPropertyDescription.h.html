22 #pragma once
23 
24 #include <wtf/HashMap.h>
25 #include <wtf/text/AtomString.h>
26 
27 namespace WebCore {
28 
29 class SVGElement;
30 
31 struct SVGAnimatedPropertyDescription {
32     // Empty value
33     SVGAnimatedPropertyDescription()
34         : m_element(nullptr)
35         , m_attributeName(nullptr)
36     {
37     }
38 
39     // Deleted value
40     SVGAnimatedPropertyDescription(WTF::HashTableDeletedValueType)
41         : m_element(reinterpret_cast<SVGElement*>(-1))
42     {
43     }
44 
45     bool isHashTableDeletedValue() const
46     {
47         return m_element == reinterpret_cast<SVGElement*>(-1);
48     }
49 
50     SVGAnimatedPropertyDescription(SVGElement* element, const AtomString& attributeName)
51         : m_element(element)
52         , m_attributeName(attributeName.impl())
53     {
54         ASSERT(m_element);
55         ASSERT(m_attributeName);
56     }
57 
58     bool operator==(const SVGAnimatedPropertyDescription& other) const
59     {
60         return m_element == other.m_element && m_attributeName == other.m_attributeName;
61     }
62 
63     SVGElement* m_element;
64     AtomStringImpl* m_attributeName;
65 };
66 
67 struct SVGAnimatedPropertyDescriptionHash {
68     static unsigned hash(const SVGAnimatedPropertyDescription& key)
69     {
70         return StringHasher::hashMemory<sizeof(SVGAnimatedPropertyDescription)>(&key);
71     }
72 
73     static bool equal(const SVGAnimatedPropertyDescription& a, const SVGAnimatedPropertyDescription& b)
74     {
75         return a == b;
76     }
77 
78     static const bool safeToCompareToEmptyOrDeleted = true;
79 };
80 
81 struct SVGAnimatedPropertyDescriptionHashTraits : WTF::SimpleClassHashTraits<SVGAnimatedPropertyDescription> { };
82 
83 } // namespace WebCore
