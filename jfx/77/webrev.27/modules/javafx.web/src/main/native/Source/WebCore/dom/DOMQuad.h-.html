26 #pragma once
27 
28 #include "DOMQuadInit.h"
29 #include "ScriptWrappable.h"
30 #include <wtf/Ref.h>
31 #include <wtf/RefCounted.h>
32 
33 namespace WebCore {
34 
35 class DOMPoint;
36 class DOMRect;
37 struct DOMPointInit;
38 struct DOMRectInit;
39 
40 class DOMQuad : public ScriptWrappable, public RefCounted<DOMQuad> {
41 public:
42     static Ref<DOMQuad> create(const DOMPointInit& p1, const DOMPointInit& p2, const DOMPointInit& p3, const DOMPointInit& p4) { return adoptRef(*new DOMQuad(p1, p2, p3, p4)); }
43     static Ref<DOMQuad> fromRect(const DOMRectInit& init) { return adoptRef(*new DOMQuad(init)); }
44     static Ref<DOMQuad> fromQuad(const DOMQuadInit& init) { return create(init.p1, init.p2, init.p3, init.p4); }
45 
46     const DOMPoint& p1() const { return m_p1; }
47     const DOMPoint& p2() const { return m_p2; }
48     const DOMPoint& p3() const { return m_p3; }
49     const DOMPoint& p4() const { return m_p4; }
50 
51     Ref<DOMRect> getBounds() const;
52 
53 private:
54     DOMQuad(const DOMPointInit&, const DOMPointInit&, const DOMPointInit&, const DOMPointInit&);
55     explicit DOMQuad(const DOMRectInit&);
56 
57     Ref<DOMPoint> m_p1;
58     Ref<DOMPoint> m_p2;
59     Ref<DOMPoint> m_p3;
60     Ref<DOMPoint> m_p4;
61 };
62 
63 }
