32 #include "config.h"
33 #include "DetailsMarkerControl.h"
34 
35 #include "HTMLNames.h"
36 #include "HTMLSummaryElement.h"
37 #include "RenderDetailsMarker.h"
38 #include <wtf/IsoMallocInlines.h>
39 
40 namespace WebCore {
41 
42 WTF_MAKE_ISO_ALLOCATED_IMPL(DetailsMarkerControl);
43 
44 Ref<DetailsMarkerControl> DetailsMarkerControl::create(Document& document)
45 {
46     return adoptRef(*new DetailsMarkerControl(document));
47 }
48 
49 DetailsMarkerControl::DetailsMarkerControl(Document& document)
50     : HTMLDivElement(HTMLNames::divTag, document)
51 {
52     setPseudo(AtomicString("-webkit-details-marker", AtomicString::ConstructFromLiteral));
53 }
54 
55 RenderPtr<RenderElement> DetailsMarkerControl::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
56 {
57     return createRenderer<RenderDetailsMarker>(*this, WTFMove(style));
58 }
59 
60 bool DetailsMarkerControl::rendererIsNeeded(const RenderStyle& style)
61 {
62     return downcast<HTMLSummaryElement>(shadowHost())->isActiveSummary() && HTMLDivElement::rendererIsNeeded(style);
63 }
