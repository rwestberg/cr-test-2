27 #pragma once
28 
29 #include "Event.h"
30 
31 namespace WebCore {
32 
33 class TransitionEvent final : public Event {
34 public:
35     static Ref<TransitionEvent> create(const AtomicString& type, const String& propertyName, double elapsedTime, const String& pseudoElement)
36     {
37         return adoptRef(*new TransitionEvent(type, propertyName, elapsedTime, pseudoElement));
38     }
39 
40     struct Init : EventInit {
41         String propertyName;
42         double elapsedTime { 0 };
43         String pseudoElement;
44     };
45 
46     static Ref<TransitionEvent> create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
47     {
48         return adoptRef(*new TransitionEvent(type, initializer, isTrusted));
49     }
50 
51     virtual ~TransitionEvent();
52 
53     const String& propertyName() const;
54     double elapsedTime() const;
55     const String& pseudoElement() const;
56 
57     EventInterface eventInterface() const override;
58 
59 private:
60     TransitionEvent(const AtomicString& type, const String& propertyName, double elapsedTime, const String& pseudoElement);
61     TransitionEvent(const AtomicString& type, const Init& initializer, IsTrusted);
62 
63     String m_propertyName;
64     double m_elapsedTime;
65     String m_pseudoElement;
66 };
67 
68 } // namespace WebCore
