25 #include "CachedHTMLCollection.h"

29 namespace WebCore {

31 class FormAssociatedElement;
32 class HTMLImageElement;

34 // This class is just a big hack to find form elements even in malformed HTML elements.
35 // The famous <table><tr><form><td> problem.
36 
37 class HTMLFormControlsCollection final : public CachedHTMLCollection<HTMLFormControlsCollection, CollectionTypeTraits<FormControls>::traversalType> {
38 public:
39     static Ref<HTMLFormControlsCollection> create(ContainerNode&, CollectionType);
40     virtual ~HTMLFormControlsCollection();
41 
42     HTMLElement* item(unsigned offset) const override;
43     Optional<Variant<RefPtr<RadioNodeList>, RefPtr<Element>>> namedItemOrItems(const String&) const;
44 
45     HTMLFormElement& ownerNode() const;
46 
47     // For CachedHTMLCollection.
48     HTMLElement* customElementAfter(Element*) const;
49 
50 private:
51     explicit HTMLFormControlsCollection(ContainerNode&);
52 
53     void invalidateCacheForDocument(Document&) override;
54     void updateNamedElementCache() const override;
55 
56     const Vector<FormAssociatedElement*>& unsafeFormControlElements() const;
57     Vector<Ref<FormAssociatedElement>> copyFormControlElementsVector() const;
58     const Vector<HTMLImageElement*>& formImageElements() const;
59 
60     mutable Element* m_cachedElement;
61     mutable unsigned m_cachedElementOffsetInArray;
62 };
63 
64 inline HTMLElement* HTMLFormControlsCollection::item(unsigned offset) const
65 {
66     return downcast<HTMLElement>(CachedHTMLCollection<HTMLFormControlsCollection, CollectionTypeTraits<FormControls>::traversalType>::item(offset));
67 }
68 
69 } // namespace WebCore

71 SPECIALIZE_TYPE_TRAITS_HTMLCOLLECTION(HTMLFormControlsCollection, FormControls)
