56 bool ignoreNextSetCursor = false;
57 
58 void Widget::setCursor(const Cursor& cursor)
59 {
60     // This is set by PluginViewWin so it can ignore the setCursor call made by
61     // EventHandler.cpp.
62     if (ignoreNextSetCursor) {
63         ignoreNextSetCursor = false;
64         return;
65     }
66 
67     FrameView* view = root();
68     if (!view)
69         return;
70     view->hostWindow()->setCursor(cursor);
71 }
