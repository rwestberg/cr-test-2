21 #pragma once
22 
23 #include "InternalFunction.h"
24 
25 namespace JSC {
26 
27 class FunctionPrototype final : public InternalFunction {
28 public:
29     typedef InternalFunction Base;
30 
31     static FunctionPrototype* create(VM& vm, Structure* structure)
32     {
33         FunctionPrototype* prototype = new (NotNull, allocateCell<FunctionPrototype>(vm.heap)) FunctionPrototype(vm, structure);
34         prototype->finishCreation(vm, String());
35         return prototype;
36     }
37 
38     void addFunctionProperties(VM&, JSGlobalObject*, JSFunction** callFunction, JSFunction** applyFunction, JSFunction** hasInstanceSymbolFunction);
39 
40     void initRestrictedProperties(VM&, JSGlobalObject*);
41 
42     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
43     {
44         return Structure::create(vm, globalObject, proto, TypeInfo(InternalFunctionType, StructureFlags), info());
45     }
46 
47     DECLARE_INFO;
48 
49 protected:
50     void finishCreation(VM&, const String& name);
51 
52 private:
53     FunctionPrototype(VM&, Structure*);
54 };
55 
56 } // namespace JSC
