26 #include "config.h"
27 #include "OverflowEvent.h"
28 
29 #include "EventNames.h"

31 namespace WebCore {
32 
33 OverflowEvent::OverflowEvent()
34     : m_orient(VERTICAL)
35     , m_horizontalOverflow(false)
36     , m_verticalOverflow(false)
37 {
38 }
39 
40 OverflowEvent::OverflowEvent(bool horizontalOverflowChanged, bool horizontalOverflow, bool verticalOverflowChanged, bool verticalOverflow)
41     : Event(eventNames().overflowchangedEvent, CanBubble::No, IsCancelable::No)
42     , m_horizontalOverflow(horizontalOverflow)
43     , m_verticalOverflow(verticalOverflow)
44 {
45     ASSERT(horizontalOverflowChanged || verticalOverflowChanged);
46 
47     if (horizontalOverflowChanged && verticalOverflowChanged)
48         m_orient = BOTH;
49     else if (horizontalOverflowChanged)
50         m_orient = HORIZONTAL;
51     else
52         m_orient = VERTICAL;
53 }
54 
55 OverflowEvent::OverflowEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
56     : Event(type, initializer, isTrusted)
57     , m_orient(initializer.orient)
58     , m_horizontalOverflow(initializer.horizontalOverflow)
59     , m_verticalOverflow(initializer.verticalOverflow)
60 {
61 }
62 
63 EventInterface OverflowEvent::eventInterface() const
64 {
65     return OverflowEventInterfaceType;
66 }
67 
68 void OverflowEvent::initOverflowEvent(unsigned short orient, bool horizontalOverflow, bool verticalOverflow)
69 {
70     if (isBeingDispatched())
71         return;
72 
73     initEvent(eventNames().overflowchangedEvent, false, false);
74 
75     m_orient = orient;
76     m_horizontalOverflow = horizontalOverflow;
77     m_verticalOverflow = verticalOverflow;
78 }
