28 
29 #include "HTMLElement.h"
30 #include "LiveNodeList.h"
31 #include <wtf/text/AtomicString.h>
32 
33 namespace WebCore {
34 
35 class RadioNodeList final : public CachedLiveNodeList<RadioNodeList> {
36 public:
37     static Ref<RadioNodeList> create(ContainerNode& rootNode, const AtomicString& name)
38     {
39         return adoptRef(*new RadioNodeList(rootNode, name));
40     }
41 
42     virtual ~RadioNodeList();
43 
44     HTMLElement* item(unsigned offset) const override;
45 
46     String value() const;
47     void setValue(const String&);
48 
49     bool elementMatches(Element&) const override;
50     bool isRootedAtDocument() const override { return m_isRootedAtDocument; }
51 
52 private:
53     RadioNodeList(ContainerNode&, const AtomicString& name);
54     bool checkElementMatchesRadioNodeListFilter(const Element&) const;
55 
56     AtomicString m_name;
57     bool m_isRootedAtDocument;
58 };
59 
60 inline HTMLElement* RadioNodeList::item(unsigned offset) const
61 {
62     return downcast<HTMLElement>(CachedLiveNodeList<RadioNodeList>::item(offset));
63 }
