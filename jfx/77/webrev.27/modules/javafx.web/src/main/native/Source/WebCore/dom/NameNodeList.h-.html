24 #pragma once
25 
26 #include "LiveNodeList.h"
27 #include <wtf/Forward.h>
28 #include <wtf/text/AtomicString.h>
29 
30 namespace WebCore {
31 
32 // NodeList which lists all Nodes in a Element with a given "name" attribute
33 class NameNodeList final : public CachedLiveNodeList<NameNodeList> {
34 public:
35     static Ref<NameNodeList> create(ContainerNode& rootNode, const AtomicString& name)
36     {
37         return adoptRef(*new NameNodeList(rootNode, name));
38     }
39 
40     virtual ~NameNodeList();
41 
42     bool elementMatches(Element&) const override;
43     bool isRootedAtDocument() const override { return false; }
44 
45 private:
46     NameNodeList(ContainerNode& rootNode, const AtomicString& name);
47 
48     AtomicString m_name;
49 };
50 
51 inline bool NameNodeList::elementMatches(Element& element) const
52 {
53     return element.getNameAttribute() == m_name;
54 }
55 
56 } // namespace WebCore
