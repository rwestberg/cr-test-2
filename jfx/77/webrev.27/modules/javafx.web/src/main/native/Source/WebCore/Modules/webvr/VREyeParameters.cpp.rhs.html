25 #include "config.h"
26 #include "VREyeParameters.h"
27 
28 #include "VRFieldOfView.h"
29 #include <JavaScriptCore/TypedArrayInlines.h>
30 
31 namespace WebCore {
32 
33 VREyeParameters::VREyeParameters(const FloatPoint3D& offset, const VRPlatformDisplayInfo::FieldOfView& fieldOfView, const RenderSize& renderSize)
34     : m_fieldOfView(VRFieldOfView::create(fieldOfView))
35     , m_offset(offset)
36     , m_renderSize(renderSize)
37 {
38 }
39 
40 Ref<Float32Array> VREyeParameters::offset() const
41 {
42     auto offset = Float32Array::create(3);
43     float* offsetData = offset->data();
44     offsetData[0] = m_offset.x();
45     offsetData[1] = m_offset.y();
46     offsetData[2] = m_offset.z();
47     return offset;
48 }
49 
50 const VRFieldOfView& VREyeParameters::fieldOfView() const
51 {
52     return m_fieldOfView;
53 }
54 
55 unsigned VREyeParameters::renderWidth() const
56 {
57     return m_renderSize.width;
58 }
59 
60 unsigned VREyeParameters::renderHeight() const
61 {
62     return m_renderSize.height;
63 }
