37 namespace WebCore {
38 
39 CSSCustomIdentValue::CSSCustomIdentValue(const String& str)
40     : CSSValue(CustomIdentClass)
41     , m_string(str)
42     , m_propertyId(CSSPropertyInvalid) { }
43 
44 CSSCustomIdentValue::CSSCustomIdentValue(CSSPropertyID id)
45     : CSSValue(CustomIdentClass)
46     , m_string()
47     , m_propertyId(id)
48 {
49     ASSERT(isKnownPropertyID());
50 }
51 
52 
53 String CSSCustomIdentValue::customCSSText() const
54 {
55     if (isKnownPropertyID())
56         return getPropertyNameAtomicString(m_propertyId);
57     StringBuilder builder;
58     serializeIdentifier(m_string, builder);
59     return builder.toString();
60 }
