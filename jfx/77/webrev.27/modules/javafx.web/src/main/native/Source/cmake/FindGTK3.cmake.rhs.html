35 find_package(PkgConfig)
36 pkg_check_modules(GTK3 gtk+-3.0)
37 
38 set(VERSION_OK TRUE)
39 if (GTK3_VERSION)
40     if (GTK3_FIND_VERSION_EXACT)
41         if (NOT("${GTK3_FIND_VERSION}" VERSION_EQUAL "${GTK3_VERSION}"))
42             set(VERSION_OK FALSE)
43         endif ()
44     else ()
45         if ("${GTK3_VERSION}" VERSION_LESS "${GTK3_FIND_VERSION}")
46             set(VERSION_OK FALSE)
47         endif ()
48     endif ()
49 endif ()
50 
51 if (GTK3_VERSION AND VERSION_OK)
52     pkg_check_modules(GTK3_QUARTZ gtk+-quartz-3.0)
53     if ("${GTK3_QUARTZ_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
54         set(GTK3_SUPPORTS_QUARTZ TRUE)
55     else ()
56         set(GTK3_SUPPORTS_QUARTZ FALSE)
57     endif ()
58 
59     pkg_check_modules(GTK3_X11 gtk+-x11-3.0)
60     if ("${GTK3_X11_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
61         set(GTK3_SUPPORTS_X11 TRUE)
62     else ()
63         set(GTK3_SUPPORTS_X11 FALSE)
64     endif ()
65 
66     pkg_check_modules(GTK3_WAYLAND gtk+-wayland-3.0)
67     if ("${GTK3_WAYLAND_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
68         set(GTK3_SUPPORTS_WAYLAND TRUE)
69     else ()
70         set(GTK3_SUPPORTS_WAYLAND FALSE)
71     endif ()






72 endif ()
73 
74 include(FindPackageHandleStandardArgs)
75 FIND_PACKAGE_HANDLE_STANDARD_ARGS(GTK3 DEFAULT_MSG GTK3_INCLUDE_DIRS GTK3_LIBRARIES VERSION_OK)
