38 ArrayBufferNeuteringWatchpointSet::ArrayBufferNeuteringWatchpointSet(VM& vm)
39     : Base(vm, vm.arrayBufferNeuteringWatchpointStructure.get())
40     , m_set(adoptRef(*new WatchpointSet(IsWatched)))
41 {
42 }

44 void ArrayBufferNeuteringWatchpointSet::destroy(JSCell* cell)
45 {
46     static_cast<ArrayBufferNeuteringWatchpointSet*>(cell)->ArrayBufferNeuteringWatchpointSet::~ArrayBufferNeuteringWatchpointSet();
47 }
48 
49 ArrayBufferNeuteringWatchpointSet* ArrayBufferNeuteringWatchpointSet::create(VM& vm)
50 {
51     ArrayBufferNeuteringWatchpointSet* result = new
52         (NotNull, allocateCell<ArrayBufferNeuteringWatchpointSet>(vm.heap))
53         ArrayBufferNeuteringWatchpointSet(vm);
54     result->finishCreation(vm);
55     return result;
56 }
57 
58 Structure* ArrayBufferNeuteringWatchpointSet::createStructure(VM& vm)
59 {
60     return Structure::create(vm, 0, jsNull(), TypeInfo(CellType, StructureFlags), info());
61 }
62 
63 void ArrayBufferNeuteringWatchpointSet::fireAll()
64 {
65     m_set->fireAll(vm(), "Array buffer was neutered");
66 }
