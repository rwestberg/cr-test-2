26 #pragma once
27 
28 #include "IDLTypes.h"
29 
30 namespace WebCore { namespace DOMJIT {
31 
32 template<typename IDLType>
33 struct DirectConverter;
34 
35 template<>
36 struct DirectConverter<IDLDOMString> {
37     static String directConvert(JSC::ExecState& state, JSC::JSString* string)
38     {
39         return string->value(&state);
40     }
41 };
42 
43 template<>
44 struct DirectConverter<IDLAtomStringAdaptor<IDLDOMString>> {
45     static String directConvert(JSC::ExecState& state, JSC::JSString* string)
46     {
47         return string->toAtomString(&state);
48     }
49 };
50 
51 template<>
52 struct DirectConverter<IDLRequiresExistingAtomStringAdaptor<IDLDOMString>> {
53     static String directConvert(JSC::ExecState& state, JSC::JSString* string)
54     {
55         return string->toExistingAtomString(&state);
56     }
57 };
58 
59 } }
