33 namespace WebCore {
34 
35 CSSDeferredParser::CSSDeferredParser(const CSSParserContext& context, const String& sheetText, StyleSheetContents& styleSheet)
36     : m_context(context)
37     , m_sheetText(sheetText)
38     , m_styleSheet(makeWeakPtr(styleSheet))
39 {
40 }
41 
42 Ref<ImmutableStyleProperties> CSSDeferredParser::parseDeclaration(const CSSParserTokenRange& range)
43 {
44     return CSSParserImpl::parseDeferredDeclaration(range, m_context, m_styleSheet.get());
45 }
46 
47 void CSSDeferredParser::parseRuleList(const CSSParserTokenRange& range, Vector<RefPtr<StyleRuleBase>>& childRules)
48 {
49     CSSParserImpl::parseDeferredRuleList(range, *this, childRules);
50 }
51 
52 void CSSDeferredParser::parseKeyframeList(const CSSParserTokenRange& range, StyleRuleKeyframes&keyframesRule)
53 {
54     CSSParserImpl::parseDeferredKeyframeList(range, *this, keyframesRule);
55 }
