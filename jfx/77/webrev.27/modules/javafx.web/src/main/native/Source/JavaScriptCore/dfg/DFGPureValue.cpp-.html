35 void PureValue::dump(PrintStream& out) const
36 {
37     out.print(Graph::opName(op()));
38     out.print("(");
39     CommaPrinter comma;
40     if (isVarargs()) {
41         for (unsigned i = 0; i < m_children.numChildren(); ++i)
42             out.print(comma, m_graph->m_varArgChildren[m_children.firstChild() + i].sanitized());
43     } else {
44         for (unsigned i = 0; i < AdjacencyList::Size; ++i) {
45             if (m_children.child(i))
46                 out.print(comma, m_children.child(i));
47         }
48     }
49     if (m_info)
50         out.print(comma, m_info);
51     out.print(")");
52 }
