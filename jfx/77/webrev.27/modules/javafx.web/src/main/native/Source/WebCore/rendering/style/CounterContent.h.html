25 #pragma once
26 
27 #include "RenderStyleConstants.h"
28 #include <wtf/text/AtomString.h>
29 
30 namespace WebCore {
31 
32 class CounterContent {
33     WTF_MAKE_FAST_ALLOCATED;
34 public:
35     CounterContent(const AtomString& identifier, ListStyleType style, const AtomString& separator)
36         : m_identifier(identifier)
37         , m_listStyle(style)
38         , m_separator(separator)
39     {
40     }
41 
42     const AtomString& identifier() const { return m_identifier; }
43     ListStyleType listStyle() const { return m_listStyle; }
44     const AtomString& separator() const { return m_separator; }
45 
46 private:
47     AtomString m_identifier;
48     ListStyleType m_listStyle;
49     AtomString m_separator;
50 };
51 
52 static inline bool operator==(const CounterContent& a, const CounterContent& b)
53 {
54     return a.identifier() == b.identifier()
55         && a.listStyle() == b.listStyle()
56         && a.separator() == b.separator();
57 }
58 
59 } // namespace WebCore
