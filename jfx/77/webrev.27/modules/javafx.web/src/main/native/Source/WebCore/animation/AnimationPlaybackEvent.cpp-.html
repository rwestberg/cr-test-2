31 namespace WebCore {
32 
33 AnimationPlaybackEvent::AnimationPlaybackEvent(const AtomicString& type, const AnimationPlaybackEventInit& initializer, IsTrusted isTrusted)
34     : Event(type, initializer, isTrusted)
35 {
36     if (initializer.currentTime)
37         m_currentTime = Seconds::fromMilliseconds(*initializer.currentTime);
38     else
39         m_currentTime = WTF::nullopt;
40 
41     if (initializer.timelineTime)
42         m_timelineTime = Seconds::fromMilliseconds(*initializer.timelineTime);
43     else
44         m_timelineTime = WTF::nullopt;
45 }
46 
47 AnimationPlaybackEvent::AnimationPlaybackEvent(const AtomicString& type, Optional<Seconds> currentTime, Optional<Seconds> timelineTime)
48     : Event(type, CanBubble::Yes, IsCancelable::No)
49     , m_currentTime(currentTime)
50     , m_timelineTime(timelineTime)
51 {
52 }
53 
54 AnimationPlaybackEvent::~AnimationPlaybackEvent() = default;
55 
56 Optional<double> AnimationPlaybackEvent::bindingsCurrentTime() const
57 {
58     if (!m_currentTime)
59         return WTF::nullopt;
60     return secondsToWebAnimationsAPITime(m_currentTime.value());
61 }
62 
63 Optional<double> AnimationPlaybackEvent::bindingsTimelineTime() const
64 {
65     if (!m_timelineTime)
66         return WTF::nullopt;
67     return secondsToWebAnimationsAPITime(m_timelineTime.value());
68 }
