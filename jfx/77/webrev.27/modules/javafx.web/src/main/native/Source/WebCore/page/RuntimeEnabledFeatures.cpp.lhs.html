39 namespace WebCore {
40 
41 RuntimeEnabledFeatures::RuntimeEnabledFeatures()
42 {
43 #if ENABLE(MEDIA_STREAM) && PLATFORM(COCOA)
44     m_isMediaDevicesEnabled = false;
45 #endif
46 #if PLATFORM(WATCHOS)
47     m_isWebSocketEnabled = false;
48 #endif
49 #if PLATFORM(GTK) && ENABLE(INPUT_TYPE_COLOR)
50     m_isInputTypeColorEnabled = true;
51 #endif
52 }
53 
54 RuntimeEnabledFeatures& RuntimeEnabledFeatures::sharedFeatures()
55 {
56     static NeverDestroyed<RuntimeEnabledFeatures> runtimeEnabledFeatures;
57 
58     return runtimeEnabledFeatures;
59 }
60 
61 bool RuntimeEnabledFeatures::spectreGadgetsEnabled() const
62 {
63     return JSC::Options::enableSpectreGadgets();
64 }
