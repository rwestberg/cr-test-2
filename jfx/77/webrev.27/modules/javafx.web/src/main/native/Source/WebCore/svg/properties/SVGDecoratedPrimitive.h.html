26 #pragma once
27 
28 #include "SVGDecoratedProperty.h"
29 
30 namespace WebCore {
31 
32 template<typename DecorationType, typename PropertyType>
33 class SVGDecoratedPrimitive : public SVGDecoratedProperty<DecorationType> {
34     using Base = SVGDecoratedProperty<DecorationType>;
35 
36 public:
37     SVGDecoratedPrimitive(const PropertyType& value)
38         : m_value(value)
39     {
40     }
41 
42 protected:
43     using Base::Base;
44 
45     void setValueInternal(const DecorationType& value) override { m_value = static_cast<PropertyType>(value); }
46     DecorationType valueInternal() const override { return static_cast<DecorationType>(m_value); }
47 
48     String valueAsString() const override { return SVGPropertyTraits<PropertyType>::toString(m_value); }
49 
50     PropertyType m_value;
51 };
52 
53 }
