26 #pragma once
27 
28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class PageTransitionEvent final : public Event {
33 public:
34     static Ref<PageTransitionEvent> create(const AtomString& type, bool persisted)
35     {
36         return adoptRef(*new PageTransitionEvent(type, persisted));
37     }
38 
39     struct Init : EventInit {
40         bool persisted { false };
41     };
42 
43     static Ref<PageTransitionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
44     {
45         return adoptRef(*new PageTransitionEvent(type, initializer, isTrusted));
46     }
47 
48     virtual ~PageTransitionEvent();
49 
50     EventInterface eventInterface() const override;
51 
52     bool persisted() const { return m_persisted; }
53 
54 private:
55     PageTransitionEvent(const AtomString& type, bool persisted);
56     PageTransitionEvent(const AtomString&, const Init&, IsTrusted);
57 
58     bool m_persisted;
59 };
60 
61 } // namespace WebCore
