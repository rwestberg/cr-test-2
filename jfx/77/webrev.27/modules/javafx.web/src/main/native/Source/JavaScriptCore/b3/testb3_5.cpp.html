  31 void testPatchpointManyWarmAnyImms()
  32 {
  33     Procedure proc;
  34     BasicBlock* root = proc.addBlock();
  35     Value* arg1 = root->appendNew<Const32Value>(proc, Origin(), 42);
  36     Value* arg2 = root->appendNew<Const64Value>(proc, Origin(), 43);
  37     Value* arg3 = root->appendNew<Const64Value>(proc, Origin(), 43000000000000ll);
  38     Value* arg4 = root->appendNew<ConstDoubleValue>(proc, Origin(), 42.5);
  39     Value* arg5 = root->appendNew<ConstFloatValue>(proc, Origin(), -42.5f);

  44     patchpoint->append(ConstrainedValue(arg4, ValueRep::WarmAny));
  45     patchpoint->append(ConstrainedValue(arg5, ValueRep::WarmAny));

  48             CHECK(params.size() == 5);
  49             CHECK(params[0] == ValueRep::constant(42));
  50             CHECK(params[1] == ValueRep::constant(43));
  51             CHECK(params[2] == ValueRep::constant(43000000000000ll));
  52             CHECK(params[3] == ValueRep::constantDouble(42.5));

  57         proc, Return, Origin(),

1593 template<typename InputType>
1594 InputType modelCompare(B3::Opcode opcode, InputType left, InputType right)
1595 {
1596     switch (opcode) {
1597     case Equal:
1598         return left == right;
1599     case NotEqual:
1600         return left != right;
1601     case LessThan:
1602         return left < right;
1603     case GreaterThan:
1604         return left > right;
1605     case LessEqual:
1606         return left <= right;
1607     case GreaterEqual:
1608         return left >= right;
1609     case Above:
1610         return static_cast<typename std::make_unsigned<InputType>::type>(left) >
1611             static_cast<typename std::make_unsigned<InputType>::type>(right);
1612     case Below:
1613         return static_cast<typename std::make_unsigned<InputType>::type>(left) <
1614             static_cast<typename std::make_unsigned<InputType>::type>(right);
1615     case AboveEqual:
1616         return static_cast<typename std::make_unsigned<InputType>::type>(left) >=
1617             static_cast<typename std::make_unsigned<InputType>::type>(right);
1618     case BelowEqual:
1619         return static_cast<typename std::make_unsigned<InputType>::type>(left) <=
1620             static_cast<typename std::make_unsigned<InputType>::type>(right);
1621     case BitAnd:
1622         return !!(left & right);
1623     default:
1624         RELEASE_ASSERT_NOT_REACHED();
1625         return 0;
1626     }
