31 namespace JSC {

33 std::unique_ptr<AccessCase> InstanceOfAccessCase::create(
34     VM& vm, JSCell* owner, AccessType accessType, Structure* structure,
35     const ObjectPropertyConditionSet& conditionSet, JSObject* prototype)
36 {
37     return std::unique_ptr<AccessCase>(new InstanceOfAccessCase(vm, owner, accessType, structure, conditionSet, prototype));
38 }
39 
40 void InstanceOfAccessCase::dumpImpl(PrintStream& out, CommaPrinter& comma) const
41 {
42     Base::dumpImpl(out, comma);
43     out.print(comma, "prototype = ", JSValue(prototype()));
44 }
45 
46 std::unique_ptr<AccessCase> InstanceOfAccessCase::clone() const
47 {
48     std::unique_ptr<InstanceOfAccessCase> result(new InstanceOfAccessCase(*this));
49     result->resetState();
50     return WTFMove(result);
51 }
52 
53 InstanceOfAccessCase::~InstanceOfAccessCase()
54 {
55 }

57 InstanceOfAccessCase::InstanceOfAccessCase(
58     VM& vm, JSCell* owner, AccessType accessType, Structure* structure,
59     const ObjectPropertyConditionSet& conditionSet, JSObject* prototype)
60     : Base(vm, owner, accessType, invalidOffset, structure, conditionSet, nullptr)
61     , m_prototype(vm, owner, prototype)
62 {
63 }
