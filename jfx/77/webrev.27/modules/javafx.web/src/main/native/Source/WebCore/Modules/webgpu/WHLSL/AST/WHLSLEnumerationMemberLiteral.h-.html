 33 namespace WebCore {
 34 
 35 namespace WHLSL {
 36 
 37 namespace AST {
 38 
 39 class EnumerationDefinition;
 40 class EnumerationMember;
 41 
 42 class EnumerationMemberLiteral : public Expression {
 43 public:
 44     EnumerationMemberLiteral(Lexer::Token&& origin, String&& left, String&& right)
 45         : Expression(WTFMove(origin))
 46         , m_left(WTFMove(left))
 47         , m_right(WTFMove(right))
 48     {
 49     }
 50 
 51     virtual ~EnumerationMemberLiteral() = default;
 52 
 53     explicit EnumerationMemberLiteral(const EnumerationMemberLiteral&) = delete;
 54     EnumerationMemberLiteral(EnumerationMemberLiteral&&) = default;
 55 
 56     EnumerationMemberLiteral& operator=(const EnumerationMemberLiteral&) = delete;
 57     EnumerationMemberLiteral& operator=(EnumerationMemberLiteral&&) = default;
 58 
 59     bool isEnumerationMemberLiteral() const override { return true; }
 60 
 61     static EnumerationMemberLiteral wrap(Lexer::Token&& origin, String&& left, String&& right, EnumerationDefinition& enumerationDefinition, EnumerationMember& enumerationMember)
 62     {
 63         EnumerationMemberLiteral result(WTFMove(origin), WTFMove(left), WTFMove(right));
 64         result.m_enumerationDefinition = &enumerationDefinition;
 65         result.m_enumerationMember = &enumerationMember;
 66         return result;
 67     }
 68 
 69     const String& left() const { return m_left; }
 70     const String& right() const { return m_right; }
 71 
 72     EnumerationMemberLiteral clone() const
 73     {
 74         auto result = EnumerationMemberLiteral(Lexer::Token(origin()), String(m_left), String(m_right));
 75         result.m_enumerationMember = m_enumerationMember;
 76         return result;
 77     }
 78 
 79     EnumerationDefinition* enumerationDefinition()
 80     {
 81         return m_enumerationDefinition;
 82     }
 83 
 84     EnumerationDefinition* enumerationDefinition() const
 85     {
 86         return m_enumerationDefinition;
 87     }
 88 
 89     EnumerationMember* enumerationMember()
 90     {
 91         return m_enumerationMember;
 92     }
 93 
 94     EnumerationMember* enumerationMember() const
 95     {
 96         return m_enumerationMember;
 97     }
 98 
 99     void setEnumerationMember(EnumerationDefinition& enumerationDefinition, EnumerationMember& enumerationMember)
100     {
101         m_enumerationDefinition = &enumerationDefinition;
102         m_enumerationMember = &enumerationMember;
103     }
104 
105 private:
106     String m_left;
107     String m_right;
108     EnumerationDefinition* m_enumerationDefinition { nullptr };
109     EnumerationMember* m_enumerationMember { nullptr };
110 };
111 
112 } // namespace AST
113 
114 }

116 }
