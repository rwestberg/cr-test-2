22 
23 #include <wtf/Forward.h>
24 #include <wtf/RefCounted.h>
25 #include <wtf/Vector.h>

26 
27 namespace WebCore {
28 
29 class Document;
30 class HTMLStyleElement;
31 class Node;
32 class ShadowRoot;
33 class StyleSheet;
34 class CSSStyleSheet;
35 
36 class StyleSheetList final : public RefCounted<StyleSheetList> {
37 public:
38     static Ref<StyleSheetList> create(Document& document) { return adoptRef(*new StyleSheetList(document)); }
39     static Ref<StyleSheetList> create(ShadowRoot& shadowRoot) { return adoptRef(*new StyleSheetList(shadowRoot)); }
40     WEBCORE_EXPORT ~StyleSheetList();
41 
42     WEBCORE_EXPORT unsigned length() const;
43     WEBCORE_EXPORT StyleSheet* item(unsigned index);
44 
45     CSSStyleSheet* namedItem(const AtomicString&) const;
46     Vector<AtomicString> supportedPropertyNames();
47 
48     Node* ownerNode() const;
49 
50     void detach();
51 
52 private:
53     StyleSheetList(Document&);
54     StyleSheetList(ShadowRoot&);
55     const Vector<RefPtr<StyleSheet>>& styleSheets() const;
56 
57     Document* m_document { nullptr };
58     ShadowRoot* m_shadowRoot { nullptr };
59     Vector<RefPtr<StyleSheet>> m_detachedStyleSheets;
60 };
