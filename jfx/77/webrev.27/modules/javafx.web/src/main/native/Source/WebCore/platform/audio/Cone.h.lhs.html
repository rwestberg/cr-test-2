29 #ifndef Cone_h
30 #define Cone_h
31 
32 #include "FloatPoint3D.h"
33 
34 namespace WebCore {
35 
36 // Cone gain is defined according to the OpenAL specification
37 
38 class ConeEffect {

39 public:
40     ConeEffect();
41 
42     // Returns scalar gain for the given source/listener positions/orientations
43     double gain(FloatPoint3D sourcePosition, FloatPoint3D sourceOrientation, FloatPoint3D listenerPosition);
44 
45     // Angles in degrees
46     void setInnerAngle(double innerAngle) { m_innerAngle = innerAngle; }
47     double innerAngle() const { return m_innerAngle; }
48 
49     void setOuterAngle(double outerAngle) { m_outerAngle = outerAngle; }
50     double outerAngle() const { return m_outerAngle; }
51 
52     void setOuterGain(double outerGain) { m_outerGain = outerGain; }
53     double outerGain() const { return m_outerGain; }
54 
55 protected:
56     double m_innerAngle;
57     double m_outerAngle;
58     double m_outerGain;
59 };
60 
61 } // namespace WebCore
62 
63 #endif // Cone_h
