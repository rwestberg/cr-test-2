31 #if ENABLE(WEB_AUDIO)
32 
33 #include "Panner.h"
34 
35 #include "EqualPowerPanner.h"
36 #include "HRTFPanner.h"
37 
38 namespace WebCore {
39 
40 std::unique_ptr<Panner> Panner::create(PanningModelType model, float sampleRate, HRTFDatabaseLoader* databaseLoader)
41 {
42     std::unique_ptr<Panner> panner;
43 
44     switch (model) {
45     case PanningModelType::Equalpower:
46         panner = makeUnique<EqualPowerPanner>(sampleRate);
47         break;
48 
49     case PanningModelType::HRTF:
50         panner = makeUnique<HRTFPanner>(sampleRate, databaseLoader);
51         break;
52     }
53 
54     return panner;
55 }
56 
57 } // namespace WebCore
58 
59 #endif // ENABLE(WEB_AUDIO)
