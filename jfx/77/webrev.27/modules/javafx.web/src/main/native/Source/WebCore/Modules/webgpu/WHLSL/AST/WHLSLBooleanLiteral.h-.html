33 namespace WebCore {

35 namespace WHLSL {
36 
37 namespace AST {
38 
39 class BooleanLiteral : public Expression {
40 public:
41     BooleanLiteral(Lexer::Token&& origin, bool value)
42         : Expression(WTFMove(origin))
43         , m_value(value)
44     {
45     }
46 
47     virtual ~BooleanLiteral() = default;
48 
49     BooleanLiteral(const BooleanLiteral&) = delete;
50     BooleanLiteral(BooleanLiteral&&) = default;
51 
52     BooleanLiteral& operator=(const BooleanLiteral&) = delete;
53     BooleanLiteral& operator=(BooleanLiteral&&) = default;
54 
55     bool value() const { return m_value; }
56 
57     bool isBooleanLiteral() const override { return true; }
58 
59     BooleanLiteral clone() const
60     {
61         return BooleanLiteral(Lexer::Token(origin()), m_value);
62     }
63 
64 private:
65     bool m_value;
66 };
67 
68 } // namespace AST
69 
70 }
71 
72 }
