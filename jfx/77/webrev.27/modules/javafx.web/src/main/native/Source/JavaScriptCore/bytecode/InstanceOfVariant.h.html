35 class InstanceOfVariant {

41     explicit operator bool() const { return !!m_structureSet.size(); }
42 
43     const StructureSet& structureSet() const { return m_structureSet; }
44     StructureSet& structureSet() { return m_structureSet; }
45 
46     const ObjectPropertyConditionSet& conditionSet() const { return m_conditionSet; }
47 
48     JSObject* prototype() const { return m_prototype; }
49 
50     bool isHit() const { return m_isHit; }
51 
52     bool attemptToMerge(const InstanceOfVariant& other);
53 
54     void dump(PrintStream&) const;
55     void dumpInContext(PrintStream&, DumpContext*) const;
56 
57 private:
58     friend class InstanceOfStatus;
59 
60     StructureSet m_structureSet;
61     ObjectPropertyConditionSet m_conditionSet;
62     JSObject* m_prototype { nullptr };
63     bool m_isHit { false };
64 };
