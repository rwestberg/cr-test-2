31 #pragma once
32 
33 #include "Event.h"
34 #include "EventNames.h"
35 
36 namespace WebCore {
37 
38 class CloseEvent : public Event {
39 public:
40     static Ref<CloseEvent> create(bool wasClean, unsigned short code, const String& reason)
41     {
42         return adoptRef(*new CloseEvent(wasClean, code, reason));
43     }
44 
45     struct Init : EventInit {
46         bool wasClean { false };
47         unsigned short code { 0 };
48         String reason;
49     };
50 
51     static Ref<CloseEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
52     {
53         return adoptRef(*new CloseEvent(type, initializer, isTrusted));
54     }
55 
56     bool wasClean() const { return m_wasClean; }
57     unsigned short code() const { return m_code; }
58     String reason() const { return m_reason; }
59 
60     // Event function.
61     EventInterface eventInterface() const override { return CloseEventInterfaceType; }
62 
63 private:
64     CloseEvent(bool wasClean, int code, const String& reason)
65         : Event(eventNames().closeEvent, CanBubble::No, IsCancelable::No)
66         , m_wasClean(wasClean)
67         , m_code(code)
68         , m_reason(reason)
69     {
70     }
71 
72     CloseEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
73         : Event(type, initializer, isTrusted)
74         , m_wasClean(initializer.wasClean)
75         , m_code(initializer.code)
76         , m_reason(initializer.reason)
77     {
78     }
79 
80     bool m_wasClean;
81     unsigned short m_code;
82     String m_reason;
83 };
84 
85 } // namespace WebCore
