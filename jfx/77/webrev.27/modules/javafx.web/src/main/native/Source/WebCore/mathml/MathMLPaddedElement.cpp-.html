 41 inline MathMLPaddedElement::MathMLPaddedElement(const QualifiedName& tagName, Document& document)

 46 Ref<MathMLPaddedElement> MathMLPaddedElement::create(const QualifiedName& tagName, Document& document)
 47 {

 49 }

 51 const MathMLElement::Length& MathMLPaddedElement::width()
 52 {
 53     return cachedMathMLLength(MathMLNames::widthAttr, m_width);
 54 }
 55 
 56 const MathMLElement::Length& MathMLPaddedElement::height()
 57 {
 58     return cachedMathMLLength(MathMLNames::heightAttr, m_height);
 59 }

 61 const MathMLElement::Length& MathMLPaddedElement::depth()

 63     return cachedMathMLLength(MathMLNames::depthAttr, m_depth);

 65 
 66 const MathMLElement::Length& MathMLPaddedElement::lspace()

 68     return cachedMathMLLength(MathMLNames::lspaceAttr, m_lspace);
 69 }
 70 
 71 const MathMLElement::Length& MathMLPaddedElement::voffset()
 72 {
 73     return cachedMathMLLength(MathMLNames::voffsetAttr, m_voffset);
 74 }
 75 
 76 void MathMLPaddedElement::parseAttribute(const QualifiedName& name, const AtomicString& value)
 77 {
 78     if (name == widthAttr)
 79         m_width = WTF::nullopt;
 80     else if (name == heightAttr)
 81         m_height = WTF::nullopt;
 82     else if (name == depthAttr)
 83         m_depth = WTF::nullopt;
 84     else if (name == lspaceAttr)
 85         m_lspace = WTF::nullopt;
 86     else if (name == voffsetAttr)
 87         m_voffset = WTF::nullopt;
 88 
 89     MathMLElement::parseAttribute(name, value);
 90 }
 91 
 92 RenderPtr<RenderElement> MathMLPaddedElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
 93 {
 94     ASSERT(hasTagName(MathMLNames::mpaddedTag));
 95     return createRenderer<RenderMathMLPadded>(*this, WTFMove(style));
 96 }
