30 // https://www.w3.org/TR/selection-api/#idl-def-Selection
31 [
32     GenerateIsReachable=ReachableFromDOMWindow,
33     ImplementationLacksVTable,
34     InterfaceName=Selection,
35 ] interface DOMSelection {
36     readonly attribute Node? anchorNode;
37     readonly attribute unsigned long anchorOffset;
38     readonly attribute Node? focusNode;
39     readonly attribute unsigned long focusOffset;
40 
41     readonly attribute boolean isCollapsed;
42     readonly attribute unsigned long rangeCount;
43 
44     void collapse(Node? node, optional unsigned long offset = 0);
45     [MayThrowException] void collapseToEnd();
46     [MayThrowException] void collapseToStart();
47 
48     [CEReactions] void deleteFromDocument();
49     boolean containsNode(Node node, optional boolean allowPartial = false);
50     void selectAllChildren(Node node);
51 
52     [MayThrowException] void extend(Node node, optional unsigned long offset = 0);
53 
54     [MayThrowException] Range getRangeAt(unsigned long index);
55     void removeAllRanges();
56     void addRange(Range range);
57 
58     [NotEnumerable] DOMString toString();
59 
60     readonly attribute DOMString type;
61 
62     void setBaseAndExtent(Node? baseNode, unsigned long baseOffset, Node? extentNode, unsigned long extentOffset);
63     void setPosition(Node? node, optional unsigned long offset = 0);
64 
65     void empty();
66 
67     // FIXME: The following operation should be implemented.
68     // void removeRange(Range range);
69 
70     // FIXME: Using "undefined" as default parameter value is wrong.
71     void modify(optional DOMString alter = "undefined", optional DOMString direction = "undefined", optional DOMString granularity = "undefined");
72 
73     readonly attribute Node? baseNode;
74     readonly attribute unsigned long baseOffset;
75     readonly attribute Node? extentNode;
76     readonly attribute unsigned long extentOffset;
77 };
