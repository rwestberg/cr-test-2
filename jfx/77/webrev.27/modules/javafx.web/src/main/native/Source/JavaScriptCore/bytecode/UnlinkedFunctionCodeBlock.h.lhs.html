26 #pragma once
27 
28 #include "UnlinkedCodeBlock.h"
29 
30 namespace JSC {
31 
32 class CachedFunctionCodeBlock;
33 
34 class UnlinkedFunctionCodeBlock final : public UnlinkedCodeBlock {
35 public:
36     typedef UnlinkedCodeBlock Base;
37     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38 
39     static UnlinkedFunctionCodeBlock* create(VM* vm, CodeType codeType, const ExecutableInfo& info, DebuggerMode debuggerMode)
40     {
41         UnlinkedFunctionCodeBlock* instance = new (NotNull, allocateCell<UnlinkedFunctionCodeBlock>(vm->heap)) UnlinkedFunctionCodeBlock(vm, vm->unlinkedFunctionCodeBlockStructure.get(), codeType, info, debuggerMode);
42         instance->finishCreation(*vm);
43         return instance;
44     }
45 
46     static void destroy(JSCell*);
47 
48 private:
49     friend CachedFunctionCodeBlock;
50 
51     UnlinkedFunctionCodeBlock(VM* vm, Structure* structure, CodeType codeType, const ExecutableInfo& info, DebuggerMode debuggerMode)
52         : Base(vm, structure, codeType, info, debuggerMode)
53     {
54     }
55 
56     UnlinkedFunctionCodeBlock(Decoder&, const CachedFunctionCodeBlock&);
57 
58 public:
59     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
60     {
61         return Structure::create(vm, globalObject, proto, TypeInfo(UnlinkedFunctionCodeBlockType, StructureFlags), info());
62     }
63 
64     DECLARE_INFO;
65 };
66 
67 }
