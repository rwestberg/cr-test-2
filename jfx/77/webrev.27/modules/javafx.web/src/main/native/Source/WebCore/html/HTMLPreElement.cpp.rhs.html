38 inline HTMLPreElement::HTMLPreElement(const QualifiedName& tagName, Document& document)
39     : HTMLElement(tagName, document)
40 {
41 }
42 
43 Ref<HTMLPreElement> HTMLPreElement::create(const QualifiedName& tagName, Document& document)
44 {
45     return adoptRef(*new HTMLPreElement(tagName, document));
46 }
47 
48 bool HTMLPreElement::isPresentationAttribute(const QualifiedName& name) const
49 {
50     if (name == wrapAttr)
51         return true;
52     return HTMLElement::isPresentationAttribute(name);
53 }
54 
55 void HTMLPreElement::collectStyleForPresentationAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
56 {
57     if (name == wrapAttr)
58         style.setProperty(CSSPropertyWhiteSpace, CSSValuePreWrap);
59     else
60         HTMLElement::collectStyleForPresentationAttribute(name, value, style);
61 }
