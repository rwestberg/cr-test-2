34 #include <JavaScriptCore/Weak.h>
35 #include <wtf/IsoMalloc.h>

37 namespace JSC {
38 class WeakHandleOwner;
39 }
40 
41 namespace WebCore {
42 
43 class JSDOMObject;
44 
45 class ScriptWrappable {
46 public:
47     JSDOMObject* wrapper() const;
48     void setWrapper(JSDOMObject*, JSC::WeakHandleOwner*, void*);
49     void clearWrapper(JSDOMObject*);
50 
51     template<typename Derived>
52     static ptrdiff_t offsetOfWrapper() { return CAST_OFFSET(Derived*, ScriptWrappable*) + OBJECT_OFFSETOF(ScriptWrappable, m_wrapper); }
53 
54 protected:
55     ~ScriptWrappable() = default;
56 
57 private:
58     JSC::Weak<JSDOMObject> m_wrapper;
59 };

61 } // namespace WebCore
