28 namespace WebCore {
29 
30 /*
31  *  The painting of a layer occurs in three distinct phases.  Each phase involves
32  *  a recursive descent into the layer's render objects. The first phase is the background phase.
33  *  The backgrounds and borders of all blocks are painted.  Inlines are not painted at all.
34  *  Floats must paint above block backgrounds but entirely below inline content that can overlap them.
35  *  In the foreground phase, all inlines are fully painted.  Inline replaced elements will get all
36  *  three phases invoked on them during this phase.
37  */
38 
39 enum class PaintPhase : uint8_t {
40     BlockBackground,
41     ChildBlockBackground,
42     ChildBlockBackgrounds,
43     Float,
44     Foreground,
45     Outline,
46     ChildOutlines,
47     SelfOutline,
48     Selection,
49     CollapsedTableBorders,
50     TextClip,
51     Mask,
52     ClippingMask,
53     EventRegion,
54 };
55 
56 enum class PaintBehavior : uint16_t {
57     Normal                      = 0,
58     SelectionOnly               = 1 << 0,
59     SkipSelectionHighlight      = 1 << 1,
60     ForceBlackText              = 1 << 2,
61     ForceWhiteText              = 1 << 3,
62     RenderingSVGMask            = 1 << 4,
63     SkipRootBackground          = 1 << 5,
64     RootBackgroundOnly          = 1 << 6,
65     SelectionAndBackgroundsOnly = 1 << 7,
66     ExcludeSelection            = 1 << 8,
67     FlattenCompositingLayers    = 1 << 9, // Paint doesn't stop at compositing layer boundaries.
68     Snapshotting                = 1 << 10,
69     TileFirstPaint              = 1 << 11,
70 };
71
