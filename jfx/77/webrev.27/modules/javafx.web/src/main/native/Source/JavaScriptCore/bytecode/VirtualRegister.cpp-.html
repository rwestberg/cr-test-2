33 void VirtualRegister::dump(PrintStream& out) const
34 {
35     if (!isValid()) {
36         out.print("<invalid>");
37         return;
38     }
39 
40     if (isHeader()) {
41         out.print("head", m_virtualRegister);
42         return;
43     }
44 
45     if (isConstant()) {
46         out.print("const", toConstantIndex());
47         return;
48     }
49 
50     if (isArgument()) {
51         if (!toArgument())
52             out.print("this");
53         else
54             out.print("arg", toArgument());
55         return;
56     }
57 
58     if (isLocal()) {
59         out.print("loc", toLocal());
60         return;
61     }
62 
63     RELEASE_ASSERT_NOT_REACHED();
64 }
