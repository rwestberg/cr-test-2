26 #include "config.h"
27 #include "SinkDocument.h"
28 

29 #include "RawDataDocumentParser.h"
30 #include <wtf/IsoMallocInlines.h>
31 
32 namespace WebCore {
33 
34 WTF_MAKE_ISO_ALLOCATED_IMPL(SinkDocument);
35 
36 class SinkDocumentParser final : public RawDataDocumentParser {
37 public:
38     static Ref<SinkDocumentParser> create(SinkDocument& document)
39     {
40         return adoptRef(*new SinkDocumentParser(document));
41     }
42 
43 private:
44     SinkDocumentParser(SinkDocument& document)
45         : RawDataDocumentParser(document)
46     {
47     }
48 
49     // Ignore all data.
50     void appendBytes(DocumentWriter&, const char*, size_t) override
51     {
52     }
53 };
54 
55 SinkDocument::SinkDocument(Frame* frame, const URL& url)
56     : HTMLDocument(frame, url)
57 {
58     setCompatibilityMode(DocumentCompatibilityMode::QuirksMode);
59     lockCompatibilityMode();
60 }
61 
62 Ref<DocumentParser> SinkDocument::createParser()
63 {
64     return SinkDocumentParser::create(*this);
65 }
