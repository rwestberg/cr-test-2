43 Ref<DataListButtonElement> DataListButtonElement::create(Document& document, DataListButtonOwner& owner)
44 {
45     return adoptRef(*new DataListButtonElement(document, owner));
46 }
47 
48 DataListButtonElement::DataListButtonElement(Document& document, DataListButtonOwner& owner)
49     : HTMLDivElement(divTag, document)
50     , m_owner(owner)
51 {
52     setPseudo(AtomicString("-webkit-list-button", AtomicString::ConstructFromLiteral));
53 }
54 
55 DataListButtonElement::~DataListButtonElement() { }
56 
57 void DataListButtonElement::defaultEventHandler(Event& event)
58 {
59     if (!is<MouseEvent>(event)) {
60         if (!event.defaultHandled())
61             HTMLDivElement::defaultEventHandler(event);
62         return;
63     }
64 
65     MouseEvent& mouseEvent = downcast<MouseEvent>(event);
66 
67     if (mouseEvent.type() == eventNames().clickEvent) {
68         m_owner.dataListButtonElementWasClicked();
69         event.setDefaultHandled();
70     }
71 
72     if (!event.defaultHandled())
73         HTMLDivElement::defaultEventHandler(event);
74 }
