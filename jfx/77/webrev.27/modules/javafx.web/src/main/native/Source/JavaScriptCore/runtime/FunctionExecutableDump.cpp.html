35 void FunctionExecutableDump::dump(PrintStream& out) const
36 {
37     out.print(m_executable->ecmaName().string(), "#");
38     if (m_executable->isGeneratedForCall())
39         out.print(m_executable->codeBlockForCall()->hashAsStringIfPossible());
40     else
41         out.print("<nogen>");
42     out.print("/");
43     if (m_executable->isGeneratedForConstruct())
44         out.print(m_executable->codeBlockForConstruct()->hashAsStringIfPossible());
45     else
46         out.print("<nogen>");
47     out.print(":[", RawPointer(m_executable), "]");
48 }
