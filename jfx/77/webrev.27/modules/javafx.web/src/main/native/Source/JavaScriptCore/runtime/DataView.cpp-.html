35 DataView::DataView(RefPtr<ArrayBuffer>&& buffer, unsigned byteOffset, unsigned byteLength)
36     : ArrayBufferView(WTFMove(buffer), byteOffset)
37     , m_byteLength(byteLength)
38 {
39 }
40 
41 Ref<DataView> DataView::create(
42     RefPtr<ArrayBuffer>&& buffer, unsigned byteOffset, unsigned byteLength)
43 {
44     return adoptRef(*new DataView(WTFMove(buffer), byteOffset, byteLength));
45 }
46 
47 Ref<DataView> DataView::create(RefPtr<ArrayBuffer>&& buffer)
48 {
49     unsigned byteLength = buffer->byteLength();
50     return create(WTFMove(buffer), 0, byteLength);
51 }
52 
53 JSArrayBufferView* DataView::wrap(ExecState* exec, JSGlobalObject* globalObject)
54 {
55     return JSDataView::create(
56         exec, globalObject->typedArrayStructure(TypeDataView), possiblySharedBuffer(), byteOffset(),
57         byteLength());
58 }
