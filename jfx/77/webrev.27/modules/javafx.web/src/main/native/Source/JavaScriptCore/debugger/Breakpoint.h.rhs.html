28 #include "DebuggerPrimitives.h"
29 #include <wtf/DoublyLinkedList.h>
30 #include <wtf/RefCounted.h>
31 #include <wtf/text/WTFString.h>
32 
33 namespace JSC {
34 
35 struct Breakpoint : public DoublyLinkedListNode<Breakpoint> {
36     Breakpoint()
37     {
38     }
39 
40     Breakpoint(SourceID sourceID, unsigned line, unsigned column, const String& condition, bool autoContinue, unsigned ignoreCount)
41         : sourceID(sourceID)
42         , line(line)
43         , column(column)
44         , condition(condition)
45         , autoContinue(autoContinue)
46         , ignoreCount(ignoreCount)
47     {
48     }
49 













50     BreakpointID id { noBreakpointID };
51     SourceID sourceID { noSourceID };
52     unsigned line { 0 };
53     unsigned column { 0 };
54     String condition;
55     bool autoContinue { false };
56     unsigned ignoreCount { 0 };
57     unsigned hitCount { 0 };
58     bool resolved { false };
59 
60     static const unsigned unspecifiedColumn = UINT_MAX;
61 
62 private:
63     Breakpoint* m_prev;
64     Breakpoint* m_next;
65 
66     friend class WTF::DoublyLinkedListNode<Breakpoint>;
67 };
68 
69 class BreakpointsList : public DoublyLinkedList<Breakpoint>,
70     public RefCounted<BreakpointsList> {
71 public:
72     ~BreakpointsList()
73     {
74         Breakpoint* breakpoint;
75         while ((breakpoint = removeHead()))
76             delete breakpoint;
77         ASSERT(isEmpty());
78     }
79 };
80 
81 } // namespace JSC
