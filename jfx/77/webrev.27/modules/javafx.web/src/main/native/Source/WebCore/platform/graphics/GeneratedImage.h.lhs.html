26 #pragma once
27 
28 #include "FloatSize.h"
29 #include "Image.h"
30 
31 namespace WebCore {
32 
33 class GeneratedImage : public Image {
34 public:
35     bool hasSingleSecurityOrigin() const override { return true; }
36 
37     void setContainerSize(const FloatSize& size) override { m_size = size; }
38     bool usesContainerSize() const override { return true; }
39     bool hasRelativeWidth() const override { return true; }
40     bool hasRelativeHeight() const override { return true; }
41     void computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio) override;
42 
43     FloatSize size() const override { return m_size; }
44 
45     // Assume that generated content has no decoded data we need to worry about
46     void destroyDecodedData(bool /*destroyAll*/ = true) override { }
47 
48 protected:
49     ImageDrawResult draw(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, CompositeOperator, BlendMode, DecodingMode, ImageOrientationDescription) override = 0;
50     void drawPattern(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform,
51         const FloatPoint& phase, const FloatSize& spacing, CompositeOperator, BlendMode) override = 0;
52 
53     // FIXME: Implement this to be less conservative.
54     bool currentFrameKnownToBeOpaque() const override { return false; }
55 
56     GeneratedImage() { }
57 
58 private:
59     bool isGeneratedImage() const override { return true; }
60 
61     FloatSize m_size;
62 };
63 
64 }
65 
66 SPECIALIZE_TYPE_TRAITS_IMAGE(GeneratedImage)
