29 namespace JSC {
30 
31 GigacageAlignedMemoryAllocator::GigacageAlignedMemoryAllocator(Gigacage::Kind kind)
32     : m_kind(kind)
33 {
34 }
35 
36 GigacageAlignedMemoryAllocator::~GigacageAlignedMemoryAllocator()
37 {
38 }
39 
40 void* GigacageAlignedMemoryAllocator::tryAllocateAlignedMemory(size_t alignment, size_t size)
41 {
42     return Gigacage::tryAlignedMalloc(m_kind, alignment, size);
43 }
44 
45 void GigacageAlignedMemoryAllocator::freeAlignedMemory(void* basePtr)
46 {
47     Gigacage::alignedFree(m_kind, basePtr);
48 }
49 
50 void GigacageAlignedMemoryAllocator::dump(PrintStream& out) const
51 {
52     out.print(Gigacage::name(m_kind), "Gigacage");
53 }
