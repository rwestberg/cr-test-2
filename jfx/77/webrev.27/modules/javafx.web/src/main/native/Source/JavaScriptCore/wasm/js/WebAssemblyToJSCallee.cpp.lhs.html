36 const ClassInfo WebAssemblyToJSCallee::s_info = { "WebAssemblyToJSCallee", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(WebAssemblyToJSCallee) };
37 
38 WebAssemblyToJSCallee* WebAssemblyToJSCallee::create(VM& vm, JSWebAssemblyModule* module)
39 {
40     Structure* structure = module->globalObject(vm)->webAssemblyToJSCalleeStructure();
41     WebAssemblyToJSCallee* callee = new (NotNull, allocateCell<WebAssemblyToJSCallee>(vm.heap)) WebAssemblyToJSCallee(vm, structure);
42     callee->finishCreation(vm, module);
43     return callee;
44 }
45 
46 Structure* WebAssemblyToJSCallee::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
47 {
48     return Structure::create(vm, globalObject, prototype, TypeInfo(WebAssemblyToJSCalleeType, StructureFlags), info());
49 }
50 
51 WebAssemblyToJSCallee::WebAssemblyToJSCallee(VM& vm, Structure* structure)
52     : Base(vm, structure)
53 {
54 }
55 
56 void WebAssemblyToJSCallee::finishCreation(VM& vm, JSWebAssemblyModule* module)
57 {
58     Base::finishCreation(vm);
59     m_module.set(vm, this, module);
60 }
61 
62 void WebAssemblyToJSCallee::visitChildren(JSCell* cell, SlotVisitor& visitor)
63 {
64     auto* thisObject = jsCast<WebAssemblyToJSCallee*>(cell);
65     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
66 
67     Base::visitChildren(thisObject, visitor);
68     visitor.append(thisObject->m_module);
69 }
