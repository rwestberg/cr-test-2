237 JNIEXPORT jlong JNICALL Java_com_sun_webkit_dom_HTMLElementImpl_insertAdjacentElementImpl(JNIEnv* env, jclass, jlong peer
238     , jstring where
239     , jlong element)
240 {
241     WebCore::JSMainThreadNullState state;
242     if (!element) {
243         raiseTypeErrorException(env);
244         return 0;
245     }
246     return JavaReturn<Element>(env, WTF::getPtr(raiseOnDOMError(env, IMPL->insertAdjacentElement(String(env, where), *static_cast<Element*>(jlong_to_ptr(element))))));
247 }
