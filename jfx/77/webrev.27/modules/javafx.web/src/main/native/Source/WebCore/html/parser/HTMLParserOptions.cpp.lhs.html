38 HTMLParserOptions::HTMLParserOptions()
39     : scriptEnabled(false)
40     , pluginsEnabled(false)
41     , usePreHTML5ParserQuirks(false)
42     , maximumDOMTreeDepth(Settings::defaultMaximumHTMLParserDOMTreeDepth)
43 {
44 }
45 
46 HTMLParserOptions::HTMLParserOptions(Document& document)
47 {
48     RefPtr<Frame> frame = document.frame();
49     scriptEnabled = frame && frame->script().canExecuteScripts(NotAboutToExecuteScript);
50     pluginsEnabled = frame && frame->loader().subframeLoader().allowPlugins();
51 
52     usePreHTML5ParserQuirks = document.settings().usePreHTML5ParserQuirks();
53     maximumDOMTreeDepth = document.settings().maximumHTMLParserDOMTreeDepth();
54 }
