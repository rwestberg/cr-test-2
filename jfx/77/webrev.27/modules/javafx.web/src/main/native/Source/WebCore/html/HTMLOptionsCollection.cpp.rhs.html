30 HTMLOptionsCollection::HTMLOptionsCollection(HTMLSelectElement& select)
31     : CachedHTMLCollection<HTMLOptionsCollection, CollectionTypeTraits<SelectOptions>::traversalType>(select, SelectOptions)
32 {
33 }
34 
35 Ref<HTMLOptionsCollection> HTMLOptionsCollection::create(HTMLSelectElement& select, CollectionType)
36 {
37     return adoptRef(*new HTMLOptionsCollection(select));
38 }
39 
40 ExceptionOr<void> HTMLOptionsCollection::add(const OptionOrOptGroupElement& element, const Optional<HTMLElementOrInt>& before)
41 {
42     return selectElement().add(element, before);
43 }
44 
45 void HTMLOptionsCollection::remove(int index)
46 {
47     selectElement().remove(index);
48 }
49 
50 int HTMLOptionsCollection::selectedIndex() const
51 {
52     return selectElement().selectedIndex();
53 }
54 
55 void HTMLOptionsCollection::setSelectedIndex(int index)
56 {
57     selectElement().setSelectedIndex(index);
58 }
59 
60 ExceptionOr<void> HTMLOptionsCollection::setLength(unsigned length)
61 {
62     return selectElement().setLength(length);
63 }
