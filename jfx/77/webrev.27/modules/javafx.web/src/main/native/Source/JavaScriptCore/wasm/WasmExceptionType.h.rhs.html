28 #if ENABLE(WEBASSEMBLY)
29 
30 namespace JSC {
31 
32 namespace Wasm {
33 
34 #define FOR_EACH_EXCEPTION(macro) \
35     macro(OutOfBoundsMemoryAccess,  "Out of bounds memory access") \
36     macro(OutOfBoundsTableAccess, "Out of bounds table access") \
37     macro(OutOfBoundsCallIndirect, "Out of bounds call_indirect") \
38     macro(NullTableEntry,  "call_indirect to a null table entry") \
39     macro(BadSignature, "call_indirect to a signature that does not match") \
40     macro(OutOfBoundsTrunc, "Out of bounds Trunc operation") \
41     macro(Unreachable, "Unreachable code should not be executed") \
42     macro(DivisionByZero, "Division by zero") \
43     macro(IntegerOverflow, "Integer overflow") \
44     macro(StackOverflow, "Stack overflow") \
45     macro(I64ArgumentType, "WebAssembly function with an i64 argument can't be called from JavaScript") \
46     macro(I64ReturnType, "WebAssembly function that returns i64 can't be called from JavaScript") \
47     macro(FuncrefNotWasm, "Funcref must be an exported wasm function")
48 
49 enum class ExceptionType : uint32_t {
50 #define MAKE_ENUM(enumName, error) enumName,
51     FOR_EACH_EXCEPTION(MAKE_ENUM)
52 #undef MAKE_ENUM
53 };
54 
55 ALWAYS_INLINE const char* errorMessageForExceptionType(ExceptionType type)
56 {
57     switch (type) {
58 #define SWITCH_CASE(enumName, error) \
59     case ExceptionType::enumName: return error;
60 
61     FOR_EACH_EXCEPTION(SWITCH_CASE)
62 #undef SWITCH_CASE
63     }
64     ASSERT_NOT_REACHED();
65     return "";
66 }
67 
68 } } // namespace JSC::Wasm
69 
70 #endif // ENABLE(WEBASSEMBLY)
