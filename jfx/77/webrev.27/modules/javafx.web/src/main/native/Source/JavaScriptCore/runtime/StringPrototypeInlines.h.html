30 namespace JSC {

32 template<typename NumberType>
33 ALWAYS_INLINE JSString* stringSlice(ExecState* exec, VM& vm, JSString* string, int32_t length, NumberType start, NumberType end)
34 {
35     NumberType from = start < 0 ? length + start : start;
36     NumberType to = end < 0 ? length + end : end;
37     if (to > from && to > 0 && from < length) {
38         if (from < 0)
39             from = 0;
40         if (to > length)
41             to = length;
42         return jsSubstring(vm, exec, string, static_cast<unsigned>(from), static_cast<unsigned>(to) - static_cast<unsigned>(from));
43     }
44     return jsEmptyString(vm);
45 }
