 25 #pragma once
 26 
 27 #include <string.h>
 28 #include <time.h>
 29 #include <wtf/Noncopyable.h>
 30 
 31 namespace WTF {
 32 
 33 class GregorianDateTime {

 34     WTF_MAKE_NONCOPYABLE(GregorianDateTime);
 35 public:
 36     GregorianDateTime()
 37         : m_year(0)
 38         , m_month(0)
 39         , m_yearDay(0)
 40         , m_monthDay(0)
 41         , m_weekDay(0)
 42         , m_hour(0)
 43         , m_minute(0)
 44         , m_second(0)
 45         , m_utcOffset(0)
 46         , m_isDST(0)
 47     {
 48     }
 49 
 50     inline int year() const { return m_year; }
 51     inline int month() const { return m_month; }
 52     inline int yearDay() const { return m_yearDay; }
 53     inline int monthDay() const { return m_monthDay; }
 54     inline int weekDay() const { return m_weekDay; }
 55     inline int hour() const { return m_hour; }
 56     inline int minute() const { return m_minute; }
 57     inline int second() const { return m_second; }
 58     inline int utcOffset() const { return m_utcOffset; }
 59     inline int isDST() const { return m_isDST; }
 60 
 61     inline void setYear(int year) { m_year = year; }
 62     inline void setMonth(int month) { m_month = month; }
 63     inline void setYearDay(int yearDay) { m_yearDay = yearDay; }
 64     inline void setMonthDay(int monthDay) { m_monthDay = monthDay; }
 65     inline void setWeekDay(int weekDay) { m_weekDay = weekDay; }
 66     inline void setHour(int hour) { m_hour = hour; }
 67     inline void setMinute(int minute) { m_minute = minute; }
 68     inline void setSecond(int second) { m_second = second; }
 69     inline void setUtcOffset(int utcOffset) { m_utcOffset = utcOffset; }
 70     inline void setIsDST(int isDST) { m_isDST = isDST; }
 71 
 72     WTF_EXPORT_PRIVATE void setToCurrentLocalTime();
 73 
 74     operator tm() const
 75     {
 76         tm ret;
 77         memset(&ret, 0, sizeof(ret));
 78 
 79         ret.tm_year = m_year - 1900;
 80         ret.tm_mon = m_month;
 81         ret.tm_yday = m_yearDay;
 82         ret.tm_mday = m_monthDay;
 83         ret.tm_wday = m_weekDay;
 84         ret.tm_hour = m_hour;
 85         ret.tm_min = m_minute;
 86         ret.tm_sec = m_second;
 87         ret.tm_isdst = m_isDST;
 88 
 89 #if HAVE(TM_GMTOFF)
 90         ret.tm_gmtoff = static_cast<long>(m_utcOffset);
 91 #endif
 92 
 93         return ret;
 94     }
 95 
 96     void copyFrom(const GregorianDateTime& other)
 97     {
 98         m_year = other.m_year;
 99         m_month = other.m_month;
100         m_yearDay = other.m_yearDay;
101         m_monthDay = other.m_monthDay;
102         m_weekDay = other.m_weekDay;
103         m_hour = other.m_hour;
104         m_minute = other.m_minute;
105         m_second = other.m_second;
106         m_utcOffset = other.m_utcOffset;
107         m_isDST = other.m_isDST;
108     }
109 
110 private:
111     int m_year;
112     int m_month;
113     int m_yearDay;
114     int m_monthDay;
115     int m_weekDay;
116     int m_hour;
117     int m_minute;
118     int m_second;
119     int m_utcOffset;
120     int m_isDST;
121 };
122 
123 } // namespace WTF
124 
125 using WTF::GregorianDateTime;
