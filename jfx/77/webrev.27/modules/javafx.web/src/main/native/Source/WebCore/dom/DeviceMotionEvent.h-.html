26 #pragma once
27 
28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class DeviceMotionData;
33 
34 class DeviceMotionEvent final : public Event {
35 public:
36     virtual ~DeviceMotionEvent();
37 
38     // FIXME: Merge this with DeviceMotionData::Acceleration
39     struct Acceleration {
40         Optional<double> x;
41         Optional<double> y;
42         Optional<double> z;
43     };
44 
45     // FIXME: Merge this with DeviceMotionData::RotationRate
46     struct RotationRate {
47         Optional<double> alpha;
48         Optional<double> beta;
49         Optional<double> gamma;
50     };
51 
52     static Ref<DeviceMotionEvent> create(const AtomicString& eventType, DeviceMotionData* deviceMotionData)
53     {
54         return adoptRef(*new DeviceMotionEvent(eventType, deviceMotionData));
55     }
56 
57     static Ref<DeviceMotionEvent> createForBindings()
58     {
59         return adoptRef(*new DeviceMotionEvent);
60     }
61 
62     Optional<Acceleration> acceleration() const;
63     Optional<Acceleration> accelerationIncludingGravity() const;
64     Optional<RotationRate> rotationRate() const;
65     Optional<double> interval() const;
66 
67     void initDeviceMotionEvent(const AtomicString& type, bool bubbles, bool cancelable, Optional<Acceleration>&&, Optional<Acceleration>&&, Optional<RotationRate>&&, Optional<double>);
68 
69 private:
70     DeviceMotionEvent();
71     DeviceMotionEvent(const AtomicString& eventType, DeviceMotionData*);
72 
73     EventInterface eventInterface() const override;
74 
75     RefPtr<DeviceMotionData> m_deviceMotionData;
76 };
77 
78 } // namespace WebCore
