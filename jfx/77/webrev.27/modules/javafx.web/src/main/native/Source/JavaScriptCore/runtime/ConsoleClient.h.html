31 namespace Inspector {
32 class ScriptArguments;

35 namespace JSC {
36 
37 class ExecState;
38 
39 class ConsoleClient {
40 public:
41     virtual ~ConsoleClient() { }
42 
43     JS_EXPORT_PRIVATE static void printConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, const String& url, unsigned lineNumber, unsigned columnNumber);
44     JS_EXPORT_PRIVATE static void printConsoleMessageWithArguments(MessageSource, MessageType, MessageLevel, JSC::ExecState*, Ref<Inspector::ScriptArguments>&&);
45 
46     void logWithLevel(ExecState*, Ref<Inspector::ScriptArguments>&&, MessageLevel);
47     void clear(ExecState*);
48     void dir(ExecState*, Ref<Inspector::ScriptArguments>&&);
49     void dirXML(ExecState*, Ref<Inspector::ScriptArguments>&&);
50     void table(ExecState*, Ref<Inspector::ScriptArguments>&&);
51     void trace(ExecState*, Ref<Inspector::ScriptArguments>&&);
52     void assertion(ExecState*, Ref<Inspector::ScriptArguments>&&);
53     void group(ExecState*, Ref<Inspector::ScriptArguments>&&);
54     void groupCollapsed(ExecState*, Ref<Inspector::ScriptArguments>&&);
55     void groupEnd(ExecState*, Ref<Inspector::ScriptArguments>&&);
56 
57     virtual void messageWithTypeAndLevel(MessageType, MessageLevel, JSC::ExecState*, Ref<Inspector::ScriptArguments>&&) = 0;
58     virtual void count(ExecState*, const String& label) = 0;
59     virtual void countReset(ExecState*, const String& label) = 0;
60     virtual void profile(ExecState*, const String& title) = 0;
61     virtual void profileEnd(ExecState*, const String& title) = 0;
62     virtual void takeHeapSnapshot(ExecState*, const String& title) = 0;
63     virtual void time(ExecState*, const String& label) = 0;
64     virtual void timeLog(ExecState*, const String& label, Ref<Inspector::ScriptArguments>&&) = 0;
65     virtual void timeEnd(ExecState*, const String& label) = 0;
66     virtual void timeStamp(ExecState*, Ref<Inspector::ScriptArguments>&&) = 0;
67     virtual void record(ExecState*, Ref<Inspector::ScriptArguments>&&) = 0;
68     virtual void recordEnd(ExecState*, Ref<Inspector::ScriptArguments>&&) = 0;
69     virtual void screenshot(ExecState*, Ref<Inspector::ScriptArguments>&&) = 0;
70 
71 private:
72     enum ArgumentRequirement { ArgumentRequired, ArgumentNotRequired };
73     void internalMessageWithTypeAndLevel(MessageType, MessageLevel, JSC::ExecState*, Ref<Inspector::ScriptArguments>&&, ArgumentRequirement);
74 };

76 } // namespace JSC
