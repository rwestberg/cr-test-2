26 #pragma once
27 
28 #include "XMLHttpRequest.h"
29 #include <wtf/Forward.h>
30 
31 namespace WebCore {
32 
33 class XMLHttpRequestUpload final : public XMLHttpRequestEventTarget {
34     WTF_MAKE_FAST_ALLOCATED;
35 public:
36     explicit XMLHttpRequestUpload(XMLHttpRequest&);
37 
38     void ref() { m_request.ref(); }
39     void deref() { m_request.deref(); }
40 
41     void dispatchThrottledProgressEvent(bool lengthComputable, unsigned long long loaded, unsigned long long total);
42     void dispatchProgressEvent(const AtomicString& type);
43 
44 private:
45     void refEventTarget() final { ref(); }
46     void derefEventTarget() final { deref(); }
47 
48     EventTargetInterface eventTargetInterface() const final { return XMLHttpRequestUploadEventTargetInterfaceType; }
49     ScriptExecutionContext* scriptExecutionContext() const final { return m_request.scriptExecutionContext(); }
50 
51     XMLHttpRequest& m_request;
52     bool m_lengthComputable { false };
53     unsigned long long m_loaded { 0 };
54     unsigned long long m_total { 0 };
55 };
56 
57 } // namespace WebCore
