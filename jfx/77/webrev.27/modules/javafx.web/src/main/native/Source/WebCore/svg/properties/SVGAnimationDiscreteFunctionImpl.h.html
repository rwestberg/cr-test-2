 26 #pragma once
 27 
 28 #include "SVGAnimationDiscreteFunction.h"
 29 #include "SVGPropertyTraits.h"
 30 
 31 namespace WebCore {
 32 
 33 class SVGAnimationBooleanFunction : public SVGAnimationDiscreteFunction<bool> {
 34 public:
 35     using Base = SVGAnimationDiscreteFunction<bool>;
 36     using Base::Base;
 37 
 38     void setFromAndToValues(SVGElement*, const String& from, const String& to) override
 39     {
 40         m_from = SVGPropertyTraits<bool>::fromString(from);
 41         m_to = SVGPropertyTraits<bool>::fromString(to);
 42     }
 43 };
 44 
 45 template<typename EnumType>
 46 class SVGAnimationEnumerationFunction : public SVGAnimationDiscreteFunction<EnumType> {
 47     using Base = SVGAnimationDiscreteFunction<EnumType>;
 48     using Base::m_from;
 49     using Base::m_to;
 50 
 51 public:
 52     using Base::Base;
 53 
 54     void setFromAndToValues(SVGElement*, const String& from, const String& to) override
 55     {
 56         m_from = SVGPropertyTraits<EnumType>::fromString(from);
 57         m_to = SVGPropertyTraits<EnumType>::fromString(to);
 58     }
 59 };
 60 
 61 class SVGAnimationOrientTypeFunction : public SVGAnimationDiscreteFunction<SVGMarkerOrientType> {
 62 public:
 63     using Base = SVGAnimationDiscreteFunction<SVGMarkerOrientType>;
 64     using Base::Base;
 65 
 66     void setFromAndToValues(SVGElement*, const String&, const String&) override
 67     {
 68         // Values will be set by SVGAnimatedAngleOrientAnimator.
 69         ASSERT_NOT_REACHED();
 70     }
 71 
 72 private:
 73     friend class SVGAnimatedAngleOrientAnimator;
 74 };
 75 
 76 class SVGAnimationPreserveAspectRatioFunction : public SVGAnimationDiscreteFunction<SVGPreserveAspectRatioValue> {
 77 public:
 78     using Base = SVGAnimationDiscreteFunction<SVGPreserveAspectRatioValue>;
 79     using Base::Base;
 80 
 81     void setFromAndToValues(SVGElement*, const String& from, const String& to) override
 82     {
 83         m_from = SVGPreserveAspectRatioValue(from);
 84         m_to = SVGPreserveAspectRatioValue(to);
 85     }
 86 };
 87 
 88 class SVGAnimationStringFunction : public SVGAnimationDiscreteFunction<String> {
 89 public:
 90     using Base = SVGAnimationDiscreteFunction<String>;
 91     using Base::Base;
 92 
 93     void setFromAndToValues(SVGElement*, const String& from, const String& to) override

 95         m_from = from;
 96         m_to = to;
 97     }
