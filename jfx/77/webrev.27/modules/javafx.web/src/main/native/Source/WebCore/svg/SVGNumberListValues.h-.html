21 #pragma once
22 
23 #include "SVGPropertyTraits.h"
24 #include <wtf/Vector.h>
25 
26 namespace WebCore {
27 
28 class SVGNumber;
29 class SVGNumberList;
30 
31 class SVGNumberListValues final : public Vector<float> {
32 public:
33     void parse(const String&);
34     String valueAsString() const;
35 };
36 
37 template<> struct SVGPropertyTraits<SVGNumberListValues> {
38     static SVGNumberListValues initialValue() { return { }; }
39     static SVGNumberListValues fromString(const String& string)
40     {
41         SVGNumberListValues list;
42         list.parse(string);
43         return list;
44     }
45     static Optional<SVGNumberListValues> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
46     static String toString(const SVGNumberListValues& list) { return list.valueAsString(); }
47 
48     using ListItemType = float;
49     using ListItemTearOff = SVGNumber;
50     using ListPropertyTearOff = SVGNumberList;
51 };
52 
53 } // namespace WebCore
