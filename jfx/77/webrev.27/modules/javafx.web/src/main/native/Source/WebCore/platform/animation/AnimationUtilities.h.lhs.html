26 #ifndef AnimationUtilities_h
27 #define AnimationUtilities_h
28 
29 #include "IntPoint.h"
30 #include "LayoutUnit.h"
31 
32 namespace WebCore {
33 
34 inline int blend(int from, int to, double progress)
35 {
36     return static_cast<int>(lround(static_cast<double>(from) + static_cast<double>(to - from) * progress));
37 }
38 
39 inline unsigned blend(unsigned from, unsigned to, double progress)
40 {
41     return static_cast<unsigned>(lround(to > from ? static_cast<double>(from) + static_cast<double>(to - from) * progress : static_cast<double>(from) - static_cast<double>(from - to) * progress));
42 }
43 
44 inline double blend(double from, double to, double progress)
45 {
46     return from + (to - from) * progress;
47 }
48 
49 inline float blend(float from, float to, double progress)
50 {
51     return static_cast<float>(from + (to - from) * progress);
52 }
53 
54 inline LayoutUnit blend(LayoutUnit from, LayoutUnit to, double progress)
55 {
56     return from + (to - from) * progress;
57 }
58 
59 inline IntPoint blend(const IntPoint& from, const IntPoint& to, double progress)
60 {
61     return IntPoint(blend(from.x(), to.x(), progress),
62                     blend(from.y(), to.y(), progress));
63 }
64 
65 } // namespace WebCore
66 
67 #endif // AnimationUtilities_h
