  35 typedef std::pair<UChar32, Form> Key;
  36 struct Entry {
  37     UChar32 character;
  38     unsigned form : 2;
  39     unsigned lspace : 3;
  40     unsigned rspace : 3;
  41     unsigned flags : 8;
  42 };
  43 static inline Key ExtractKey(const Entry* entry) { return Key(entry->character, static_cast<Form>(entry->form)); }
  44 static inline UChar32 ExtractChar(const Entry* entry) { return entry->character; }
  45 static inline Property ExtractProperty(const Entry& entry)
  46 {
  47     Property property;
  48     property.form = static_cast<Form>(entry.form);
  49     property.leadingSpaceInMathUnit = entry.lspace;
  50     property.trailingSpaceInMathUnit = entry.rspace;
  51     property.flags = entry.flags;
  52     return property;
  53 }

1110 Optional<Property> MathMLOperatorDictionary::search(UChar32 character, Form form, bool explicitForm)
1111 {
1112     if (!character)
1113         return WTF::nullopt;
1114 
1115     // We try and find the default values from the operator dictionary.
1116     if (auto* entry = tryBinarySearch<const Entry, Key>(dictionary, dictionarySize, Key(character, form), ExtractKey))
1117         return ExtractProperty(*entry);
1118 
1119     if (explicitForm)
1120         return WTF::nullopt;
1121 
1122     // If we did not find the desired operator form and if it was not set explicitely, we use the first one in the following order: Infix, Prefix, Postfix.
1123     // This is to handle bad MathML markup without explicit <mrow> delimiters like "<mo>(</mo><mi>a</mi><mo>)</mo><mo>(</mo><mi>b</mi><mo>)</mo>" where innerfences should not be considered infix.
1124     if (auto* entry = tryBinarySearch<const Entry, UChar32>(dictionary, dictionarySize, character, ExtractChar)) {
1125         // There are at most two other entries before the one found.
1126         if (entry != dictionary && (entry - 1)->character == character)
1127             entry--;
1128         if (entry != dictionary && (entry - 1)->character == character)
1129             entry--;
1130         return ExtractProperty(*entry);
1131     }
1132 
1133     return WTF::nullopt;
1134 }
