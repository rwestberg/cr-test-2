26 #pragma once
27 
28 #include <wtf/PrintStream.h>
29 
30 namespace WTF {
31 
32 class CommaPrinter {

33 public:
34     CommaPrinter(const char* comma = ", ", const char* start = "")
35         : m_comma(comma)
36         , m_start(start)
37         , m_didPrint(false)
38     {
39     }
40 
41     void dump(PrintStream& out) const
42     {
43         if (!m_didPrint) {
44             out.print(m_start);
45             m_didPrint = true;
46             return;
47         }
48 
49         out.print(m_comma);
50     }
51 
52     bool didPrint() const { return m_didPrint; }
53 
54 private:
55     const char* m_comma;
56     const char* m_start;
57     mutable bool m_didPrint;
58 };
59 
60 } // namespace WTF
61 
62 using WTF::CommaPrinter;
