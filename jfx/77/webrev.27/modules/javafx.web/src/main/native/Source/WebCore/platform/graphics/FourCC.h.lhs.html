26 #pragma once
27 
28 #include <wtf/text/WTFString.h>
29 
30 namespace WebCore {
31 
32 struct FourCC {

33     FourCC(uint32_t value) : value(value) { }
34 
35     template<std::size_t N>
36     constexpr FourCC(const char (&data)[N])
37     {
38         static_assert((N - 1) == 4, "FourCC literals must be exactly 4 characters long");
39         value = data[0] << 24 | data[1] << 16 | data[2] << 8 | data[3];
40     }
41 
42     String toString() const;
43     WEBCORE_EXPORT static Optional<FourCC> fromString(const String&);
44 
45     bool operator==(const FourCC& other) const { return value == other.value; }
46     bool operator!=(const FourCC& other) const { return value != other.value; }
47 
48     uint32_t value;
49 };
50 
51 }
