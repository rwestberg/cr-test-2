31 #pragma once
32 
33 #include <wtf/HashMap.h>
34 #include <wtf/text/StringHash.h>
35 #include <wtf/text/WTFString.h>
36 
37 namespace WebCore {
38 
39 class WebSocketExtensionParser {
40 public:
41     explicit WebSocketExtensionParser(const char* start, const char* end)
42         : m_current(start)
43         , m_end(end)
44     {
45     }
46     bool finished();
47     bool parsedSuccessfully();
48     bool parseExtension(String& extensionToken, HashMap<String, String>& parameters);
49 
50 private:
51     const String& currentToken() { return m_currentToken; }
52 
53     // The following member functions basically follow the grammar defined
54     // in Section 2.2 of RFC 2616.
55     bool consumeToken();
56     bool consumeQuotedString();
57     bool consumeQuotedStringOrToken();
58     bool consumeCharacter(char);
59 
60     void skipSpaces();
61 
62     const char* m_current;
63     const char* m_end;
64     String m_currentToken;
65 };
66 
67 } // namespace WebCore
