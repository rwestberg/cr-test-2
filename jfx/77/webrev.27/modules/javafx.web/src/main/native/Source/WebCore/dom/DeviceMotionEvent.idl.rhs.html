26 [
27     Conditional=DEVICE_ORIENTATION
28 ] interface DeviceMotionEvent : Event {
29     readonly attribute Acceleration? acceleration;
30     readonly attribute Acceleration? accelerationIncludingGravity;
31     readonly attribute RotationRate? rotationRate;
32     readonly attribute unrestricted double? interval;
33 
34     void initDeviceMotionEvent(optional DOMString type = "",
35                                optional boolean bubbles = false,
36                                optional boolean cancelable = false,
37                                optional Acceleration? acceleration = null,
38                                optional Acceleration? accelerationIncludingGravity = null,
39                                optional RotationRate? rotationRate = null,
40                                optional unrestricted double? interval = null);
41 };
42 
43 [
44     Conditional=DEVICE_ORIENTATION,
45     JSGenerateToJSObject
46 ] dictionary Acceleration {
47     double? x;
48     double? y;
49     double? z;
50 };
51 
52 [
53     Conditional=DEVICE_ORIENTATION,
54     JSGenerateToJSObject
55 ] dictionary RotationRate {
56     double? alpha;
57     double? beta;
58     double? gamma;
59 };
60 
61 DeviceMotionEvent implements DeviceOrientationOrMotionEvent;
