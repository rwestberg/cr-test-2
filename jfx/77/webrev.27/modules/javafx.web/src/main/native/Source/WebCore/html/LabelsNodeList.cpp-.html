36 LabelsNodeList::LabelsNodeList(LabelableElement& forNode)
37     : CachedLiveNodeList(forNode, InvalidateOnForTypeAttrChange)
38 {
39 }
40 
41 LabelsNodeList::~LabelsNodeList()
42 {
43     ownerNode().nodeLists()->removeCacheWithAtomicName(this, starAtom());
44 }
45 
46 bool LabelsNodeList::elementMatches(Element& testNode) const
47 {
48     return is<HTMLLabelElement>(testNode) && downcast<HTMLLabelElement>(testNode).control() == &ownerNode();
49 }
