35 AffineTransform ImageOrientation::transformFromDefault(const FloatSize& drawnSize) const
36 {
37     float w = drawnSize.width();
38     float h = drawnSize.height();
39 
40     switch (m_orientation) {
41     case OriginTopLeft:
42         return AffineTransform();
43     case OriginTopRight:
44         return AffineTransform(-1,  0,  0,  1,  w, 0);
45     case OriginBottomRight:
46         return AffineTransform(-1,  0,  0, -1,  w, h);
47     case OriginBottomLeft:
48         return AffineTransform( 1,  0,  0, -1,  0, h);
49     case OriginLeftTop:
50         return AffineTransform( 0,  1,  1,  0,  0, 0);
51     case OriginRightTop:
52         return AffineTransform( 0,  1, -1,  0,  w, 0);
53     case OriginRightBottom:
54         return AffineTransform( 0, -1, -1,  0,  w, h);
55     case OriginLeftBottom:
56         return AffineTransform( 0, -1,  1,  0,  0, h);
57     }
58 
59     ASSERT_NOT_REACHED();
60     return AffineTransform();
61 }
