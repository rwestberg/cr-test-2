31 namespace JSC {

33 class InstanceOfStatus;

35 class InstanceOfVariant {

36 public:
37     InstanceOfVariant() = default;
38     InstanceOfVariant(const StructureSet&, const ObjectPropertyConditionSet&, JSObject* prototype, bool isHit);
39 
40     explicit operator bool() const { return !!m_structureSet.size(); }
41 
42     const StructureSet& structureSet() const { return m_structureSet; }
43     StructureSet& structureSet() { return m_structureSet; }
44 
45     const ObjectPropertyConditionSet& conditionSet() const { return m_conditionSet; }
46 
47     JSObject* prototype() const { return m_prototype; }
48 
49     bool isHit() const { return m_isHit; }
50 
51     bool attemptToMerge(const InstanceOfVariant& other);
52 
53     void dump(PrintStream&) const;
54     void dumpInContext(PrintStream&, DumpContext*) const;
55 
56 private:
57     friend class InstanceOfStatus;
58 
59     StructureSet m_structureSet;
60     ObjectPropertyConditionSet m_conditionSet;
61     JSObject* m_prototype { nullptr };
62     bool m_isHit { false };
63 };
