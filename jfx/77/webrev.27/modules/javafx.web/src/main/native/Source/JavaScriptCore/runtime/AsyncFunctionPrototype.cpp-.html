42 AsyncFunctionPrototype::AsyncFunctionPrototype(VM& vm, Structure* structure)
43     : JSC::JSNonFinalObject(vm, structure)
44 {
45 }
46 
47 void AsyncFunctionPrototype::finishCreation(VM& vm)
48 {
49     Base::finishCreation(vm);
50     ASSERT(inherits(vm, info()));
51     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(0), PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum);
52     putDirectWithoutTransition(vm, vm.propertyNames->toStringTagSymbol, jsString(&vm, "AsyncFunction"), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
53     didBecomePrototype();
54 }
