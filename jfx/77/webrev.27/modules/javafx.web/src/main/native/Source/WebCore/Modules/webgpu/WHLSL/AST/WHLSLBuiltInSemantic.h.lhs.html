34 namespace WebCore {
35 
36 namespace WHLSL {
37 
38 namespace AST {
39 
40 class BuiltInSemantic : public BaseSemantic {

41 public:
42     enum class Variable : uint8_t {
43         SVInstanceID,
44         SVVertexID,
45         PSize,
46         SVPosition,
47         SVIsFrontFace,
48         SVSampleIndex,
49         SVInnerCoverage,
50         SVTarget,
51         SVDepth,
52         SVCoverage,
53         SVDispatchThreadID,
54         SVGroupID,
55         SVGroupIndex,
56         SVGroupThreadID
57     };
58 
59     BuiltInSemantic(Lexer::Token&& origin, Variable variable, Optional<unsigned>&& targetIndex = WTF::nullopt)
60         : BaseSemantic(WTFMove(origin))
61         , m_variable(variable)
62         , m_targetIndex(WTFMove(targetIndex))
63     {
64     }
65 
66     virtual ~BuiltInSemantic() = default;
67 
68     BuiltInSemantic(const BuiltInSemantic&) = delete;
69     BuiltInSemantic(BuiltInSemantic&&) = default;
70 
71     Variable variable() const { return m_variable; }
72     Optional<unsigned>& targetIndex() { return m_targetIndex; }
73 
74     bool operator==(const BuiltInSemantic& other) const
75     {
76         return m_variable == other.m_variable && m_targetIndex == other.m_targetIndex;
77     }
78 
79     bool operator!=(const BuiltInSemantic& other) const
80     {
81         return !(*this == other);
82     }
83 


































84     bool isAcceptableType(const UnnamedType&, const Intrinsics&) const override;
85     bool isAcceptableForShaderItemDirection(ShaderItemDirection, const Optional<EntryPointType>&) const override;
86 
87 private:
88     Variable m_variable;
89     Optional<unsigned> m_targetIndex;
90 };
91 
92 } // namespace AST
93 
94 }

96 }
