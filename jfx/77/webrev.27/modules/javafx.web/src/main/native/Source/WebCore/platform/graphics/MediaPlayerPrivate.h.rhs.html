254     virtual size_t extraMemoryCost() const
255     {
256         MediaTime duration = this->durationMediaTime();
257         if (!duration)
258             return 0;
259 
260         unsigned long long extra = totalBytes() * buffered()->totalDuration().toDouble() / duration.toDouble();

262     }
