29 namespace JSC {

31 #define INITIALIZE_PROPERTY_NAME(name) , name(Identifier::fromString(vm, #name))
32 #define INITIALIZE_KEYWORD(name) , name##Keyword(Identifier::fromString(vm, #name))
33 #define INITIALIZE_PRIVATE_NAME(name) , name##PrivateName(m_builtinNames->name##PrivateName())
34 #define INITIALIZE_SYMBOL(name) , name##Symbol(m_builtinNames->name##Symbol())
35 
36 CommonIdentifiers::CommonIdentifiers(VM* vm)
37     : nullIdentifier()
38     , emptyIdentifier(Identifier::EmptyIdentifier)
39     , underscoreProto(Identifier::fromString(vm, "__proto__"))
40     , thisIdentifier(Identifier::fromString(vm, "this"))
41     , useStrictIdentifier(Identifier::fromString(vm, "use strict"))
42     , timesIdentifier(Identifier::fromString(vm, "*"))
43     , m_builtinNames(new BuiltinNames(vm, this))
44     JSC_PARSER_PRIVATE_NAMES(INITIALIZE_PRIVATE_NAME)
45     JSC_COMMON_IDENTIFIERS_EACH_KEYWORD(INITIALIZE_KEYWORD)
46     JSC_COMMON_IDENTIFIERS_EACH_PROPERTY_NAME(INITIALIZE_PROPERTY_NAME)
47     JSC_COMMON_PRIVATE_IDENTIFIERS_EACH_WELL_KNOWN_SYMBOL(INITIALIZE_SYMBOL)
48 {
49 }
50 
51 CommonIdentifiers::~CommonIdentifiers()
52 {
53 }
54 
55 SymbolImpl* CommonIdentifiers::lookUpPrivateName(const Identifier& ident) const
56 {
57     return m_builtinNames->lookUpPrivateName(ident);
58 }
59 
60 Identifier CommonIdentifiers::getPublicName(VM& vm, SymbolImpl* symbol) const
61 {
62     return m_builtinNames->getPublicName(vm, symbol);
63 }
64 
65 void CommonIdentifiers::appendExternalName(const Identifier& publicName, const Identifier& privateName)
66 {
67     m_builtinNames->appendExternalName(publicName, privateName);
68 }
