23 #include "SVGPathSegWithContext.h"

25 namespace WebCore {
26 
27 class SVGPathSegClosePath final : public SVGPathSegWithContext {
28 public:
29     static Ref<SVGPathSegClosePath> create(const SVGPathElement& element, SVGPathSegRole role)
30     {
31         return adoptRef(*new SVGPathSegClosePath(element, role));
32     }
33 
34 private:
35     SVGPathSegClosePath(const SVGPathElement& element, SVGPathSegRole role)
36         : SVGPathSegWithContext(element, role)
37     {
38     }
39 
40     unsigned short pathSegType() const final { return PATHSEG_CLOSEPATH; }
41     String pathSegTypeAsLetter() const final { return "Z"; }
42 };

44 } // namespace WebCore
