 45 static double monotonicTimeToDOMHighResTimeStamp(MonotonicTime timeOrigin, MonotonicTime timeStamp)
 46 {
 47     ASSERT(timeStamp.secondsSinceEpoch().seconds() >= 0);
 48     if (!timeStamp || !timeOrigin)

 53 }
