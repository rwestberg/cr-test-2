20 #pragma once
21 
22 #include "SVGAnimatedListPropertyTearOff.h"
23 #include "SVGTransformList.h"
24 
25 namespace WebCore {
26 
27 class SVGAnimatedTransformListPropertyTearOff final : public SVGAnimatedListPropertyTearOff<SVGTransformListValues> {
28 public:
29     static Ref<SVGAnimatedTransformListPropertyTearOff> create(SVGElement* contextElement, const QualifiedName& attributeName, AnimatedPropertyType animatedPropertyType, SVGTransformListValues& values)
30     {
31         ASSERT(contextElement);
32         return adoptRef(*new SVGAnimatedTransformListPropertyTearOff(contextElement, attributeName, animatedPropertyType, values));
33     }
34 
35     Ref<ListPropertyTearOff> baseVal() final
36     {
37         if (m_baseVal)
38             return *static_cast<ListPropertyTearOff*>(m_baseVal.get());
39 
40         auto property = SVGTransformList::create(*this, BaseValRole, m_values, m_wrappers);
41         m_baseVal = makeWeakPtr(property.get());
42         return property;
43     }
44 
45     Ref<ListPropertyTearOff> animVal() final
46     {
47         if (m_animVal)
48             return *static_cast<ListPropertyTearOff*>(m_animVal.get());
49 
50         auto property = SVGTransformList::create(*this, AnimValRole, m_values, m_wrappers);
51         m_animVal = makeWeakPtr(property.get());
52         return property;
53     }
54 
55 private:
56     SVGAnimatedTransformListPropertyTearOff(SVGElement* contextElement, const QualifiedName& attributeName, AnimatedPropertyType animatedPropertyType, SVGTransformListValues& values)
57         : SVGAnimatedListPropertyTearOff<SVGTransformListValues>(contextElement, attributeName, animatedPropertyType, values)
58     {
59     }
60 };
61 
62 } // namespace WebCore
