31 #pragma once
32 
33 #if ENABLE(WEB_RTC)
34 
35 #include "Event.h"
36 #include <wtf/text/AtomicString.h>
37 
38 namespace WebCore {
39 
40 class MediaStream;
41 class MediaStreamTrack;
42 class RTCRtpReceiver;
43 class RTCRtpTransceiver;
44 
45 typedef Vector<RefPtr<MediaStream>> MediaStreamArray;
46 
47 class RTCTrackEvent : public Event {
48 public:
49     static Ref<RTCTrackEvent> create(const AtomicString& type, CanBubble, IsCancelable, RefPtr<RTCRtpReceiver>&&, RefPtr<MediaStreamTrack>&&, MediaStreamArray&&, RefPtr<RTCRtpTransceiver>&&);
50 
51     struct Init : EventInit {
52         RefPtr<RTCRtpReceiver> receiver;
53         RefPtr<MediaStreamTrack> track;
54         MediaStreamArray streams;
55         RefPtr<RTCRtpTransceiver> transceiver;
56     };
57     static Ref<RTCTrackEvent> create(const AtomicString& type, const Init&, IsTrusted = IsTrusted::No);
58 
59     RTCRtpReceiver* receiver() const { return m_receiver.get(); }
60     MediaStreamTrack* track() const  { return m_track.get(); }
61     const MediaStreamArray& streams() const  { return m_streams; }
62     RTCRtpTransceiver* transceiver() const  { return m_transceiver.get(); }
63 
64     virtual EventInterface eventInterface() const { return RTCTrackEventInterfaceType; }
65 
66 private:
67     RTCTrackEvent(const AtomicString& type, CanBubble, IsCancelable, RefPtr<RTCRtpReceiver>&&, RefPtr<MediaStreamTrack>&&, MediaStreamArray&&, RefPtr<RTCRtpTransceiver>&&);
68     RTCTrackEvent(const AtomicString& type, const Init&, IsTrusted);
69 
70     RefPtr<RTCRtpReceiver> m_receiver;
71     RefPtr<MediaStreamTrack> m_track;
72     MediaStreamArray m_streams;
73     RefPtr<RTCRtpTransceiver> m_transceiver;
74 };
75 
76 } // namespace WebCore
77 
78 #endif // ENABLE(WEB_RTC)
