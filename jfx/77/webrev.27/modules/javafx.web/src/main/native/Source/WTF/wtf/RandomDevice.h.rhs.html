27 #pragma once
28 
29 #include <wtf/Noncopyable.h>
30 #include <wtf/StdLibExtras.h>
31 
32 namespace WTF {
33 
34 class RandomDevice {
35     WTF_MAKE_NONCOPYABLE(RandomDevice);
36     WTF_MAKE_FAST_ALLOCATED;
37 public:
38 #if OS(DARWIN) || OS(FUCHSIA) || OS(WINDOWS)
39     RandomDevice() = default;
40 #else
41     RandomDevice();
42     ~RandomDevice();
43 #endif
44 
45     // This function attempts to fill buffer with randomness from the operating
46     // system. Rather than calling this function directly, consider calling
47     // cryptographicallyRandomNumber or cryptographicallyRandomValues.
48     void cryptographicallyRandomValues(unsigned char* buffer, size_t length);
49 
50 private:
51 #if OS(DARWIN) || OS(FUCHSIA) || OS(WINDOWS)
52 #elif OS(UNIX)
53     int m_fd { -1 };
54 #else
55 #error "This configuration doesn't have a strong source of randomness."
56 // WARNING: When adding new sources of OS randomness, the randomness must
57 //          be of cryptographic quality!
58 #endif
59 };
60 
61 }
