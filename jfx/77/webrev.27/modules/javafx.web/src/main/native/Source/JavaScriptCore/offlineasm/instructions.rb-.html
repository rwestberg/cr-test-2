320 def isBranch(instruction)
321     instruction =~ /^b/
322 end
323 
324 def hasFallThrough(instruction)
325     instruction != "ret" and instruction != "jmp"
326 end
327 
328 def isPowerOfTwo(value)
329     return false if value <= 0
330     (value & (value - 1)).zero?
331 end
