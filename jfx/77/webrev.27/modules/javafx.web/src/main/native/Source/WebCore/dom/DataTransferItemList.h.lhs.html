32 #pragma once
33 
34 #include "DataTransfer.h"
35 #include "ExceptionOr.h"
36 #include "ScriptWrappable.h"
37 #include <wtf/Forward.h>
38 #include <wtf/Ref.h>
39 #include <wtf/RefCounted.h>
40 #include <wtf/WeakPtr.h>
41 
42 namespace WebCore {
43 
44 class DataTransferItem;
45 class File;
46 
47 class DataTransferItemList : public ScriptWrappable, public CanMakeWeakPtr<DataTransferItemList> {
48     WTF_MAKE_NONCOPYABLE(DataTransferItemList); WTF_MAKE_FAST_ALLOCATED;

49 public:
50     DataTransferItemList(DataTransfer&);
51     ~DataTransferItemList();
52 
53     // DataTransfer owns DataTransferItemList, and DataTransfer is kept alive as long as DataTransferItemList is alive.
54     void ref() { m_dataTransfer.ref(); }
55     void deref() { m_dataTransfer.deref(); }
56     DataTransfer& dataTransfer() { return m_dataTransfer; }
57 
58     // DOM API
59     unsigned length() const;
60     RefPtr<DataTransferItem> item(unsigned index);
61     ExceptionOr<RefPtr<DataTransferItem>> add(const String& data, const String& type);
62     RefPtr<DataTransferItem> add(Ref<File>&&);
63     ExceptionOr<void> remove(unsigned index);
64     void clear();
65 
66     void didClearStringData(const String& type);
67     void didSetStringData(const String& type);
68     bool hasItems() const { return m_items.hasValue(); }
69     const Vector<Ref<DataTransferItem>>& items() const
70     {
71         ASSERT(m_items);
72         return *m_items;
73     }
74 
75 private:
76     Vector<Ref<DataTransferItem>>& ensureItems() const;
77 
78     DataTransfer& m_dataTransfer;
79     mutable Optional<Vector<Ref<DataTransferItem>>> m_items;
80 };
81 
82 } // namespace WebCore
