32 #include "config.h"
33 #include "PasswordInputType.h"
34 
35 #include "FormController.h"
36 #include "HTMLInputElement.h"
37 #include "InputTypeNames.h"
38 #include <wtf/Assertions.h>
39 
40 namespace WebCore {
41 
42 const AtomString& PasswordInputType::formControlType() const
43 {
44     return InputTypeNames::password();
45 }
46 
47 bool PasswordInputType::shouldSaveAndRestoreFormControlState() const
48 {
49     return false;
50 }
51 
52 FormControlState PasswordInputType::saveFormControlState() const
53 {
54     // Should never save/restore password fields.
55     ASSERT_NOT_REACHED();
56     return FormControlState();
57 }
58 
59 void PasswordInputType::restoreFormControlState(const FormControlState&)
60 {
61     // Should never save/restore password fields.
62     ASSERT_NOT_REACHED();
63 }
64 
65 bool PasswordInputType::shouldUseInputMethod() const
66 {
67     // Input methods are disabled for the password field because otherwise
68     // anyone can access the underlying password and display it in clear text.
69     return false;
70 }
71 
72 bool PasswordInputType::shouldResetOnDocumentActivation()
73 {
74     return true;
75 }
76 
77 bool PasswordInputType::shouldRespectListAttribute()
78 {
79     return false;
80 }
81 
82 bool PasswordInputType::isPasswordField() const
83 {
84     return true;
85 }

87 } // namespace WebCore
