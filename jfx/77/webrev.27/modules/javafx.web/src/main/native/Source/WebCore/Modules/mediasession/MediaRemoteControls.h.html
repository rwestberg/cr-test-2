26 #pragma once
27 
28 #if ENABLE(MEDIA_SESSION)
29 
30 #include "EventTarget.h"
31 #include <wtf/RefCounted.h>
32 
33 namespace WebCore {
34 
35 class MediaSession;
36 
37 class MediaRemoteControls final : public RefCounted<MediaRemoteControls>, public EventTargetWithInlineData {
38     WTF_MAKE_ISO_ALLOCATED(MediaRemoteControls);
39 public:
40     static Ref<MediaRemoteControls> create(ScriptExecutionContext& context, MediaSession* session = nullptr)
41     {
42         return adoptRef(*new MediaRemoteControls(context, session));
43     }
44 
45     bool previousTrackEnabled() const { return m_previousTrackEnabled; }
46     void setPreviousTrackEnabled(bool);
47 
48     bool nextTrackEnabled() const { return m_nextTrackEnabled; }
49     void setNextTrackEnabled(bool);
50 
51     using RefCounted<MediaRemoteControls>::ref;
52     using RefCounted<MediaRemoteControls>::deref;
53 
54     void clearSession();
55 
56     virtual ~MediaRemoteControls();
57 
58     EventTargetInterface eventTargetInterface() const override { return MediaRemoteControlsEventTargetInterfaceType; }
59     ScriptExecutionContext* scriptExecutionContext() const override { return &m_scriptExecutionContext; }
60 
61 private:
62     MediaRemoteControls(ScriptExecutionContext&, MediaSession*);
63 
64     ScriptExecutionContext& m_scriptExecutionContext;
65 
66     bool m_previousTrackEnabled { false };
67     bool m_nextTrackEnabled { false };
68 
69     MediaSession* m_session { nullptr };
70 
71     void refEventTarget() final { ref(); }
72     void derefEventTarget() final { deref(); }
73 };
74 
75 } // namespace WebCore
76 
77 #endif // ENABLE(MEDIA_SESSION)
