26 #pragma once
27 
28 #include "StaticRange.h"
29 #include "UIEvent.h"
30 
31 namespace WebCore {
32 
33 class DataTransfer;
34 class WindowProxy;
35 
36 class InputEvent final : public UIEvent {
37 public:
38     struct Init : UIEventInit {
39         String data;
40     };
41 
42     static Ref<InputEvent> create(const AtomicString& eventType, const String& inputType, IsCancelable, RefPtr<WindowProxy>&& view,
43         const String& data, RefPtr<DataTransfer>&&, const Vector<RefPtr<StaticRange>>& targetRanges, int detail);
44 
45     static Ref<InputEvent> create(const AtomicString& type, const Init& initializer)
46     {
47         return adoptRef(*new InputEvent(type, initializer));
48     }
49 
50     bool isInputEvent() const override { return true; }
51     EventInterface eventInterface() const final { return InputEventInterfaceType; }
52     const String& inputType() const { return m_inputType; }
53     const String& data() const { return m_data; }
54     RefPtr<DataTransfer> dataTransfer() const;
55     const Vector<RefPtr<StaticRange>>& getTargetRanges() { return m_targetRanges; }
56 
57 private:
58     InputEvent(const AtomicString& eventType, const String& inputType, IsCancelable, RefPtr<WindowProxy>&&, const String& data, RefPtr<DataTransfer>&&, const Vector<RefPtr<StaticRange>>& targetRanges, int detail);
59     InputEvent(const AtomicString& eventType, const Init&);
60 
61     String m_inputType;
62     String m_data;
63     RefPtr<DataTransfer> m_dataTransfer;
64     Vector<RefPtr<StaticRange>> m_targetRanges;
65 };
66 
67 } // namespace WebCore
68 
69 SPECIALIZE_TYPE_TRAITS_EVENT(InputEvent)
