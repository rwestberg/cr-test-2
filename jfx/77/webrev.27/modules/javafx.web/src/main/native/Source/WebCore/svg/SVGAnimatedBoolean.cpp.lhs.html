68 void SVGAnimatedBooleanAnimator::calculateAnimatedValue(float percentage, unsigned, SVGAnimatedType* from, SVGAnimatedType* to, SVGAnimatedType*, SVGAnimatedType* animated)
69 {
70     ASSERT(m_animationElement);
71     ASSERT(m_contextElement);
72 
73     const auto fromBoolean = (m_animationElement->animationMode() == ToAnimation ? animated : from)->as<bool>();
74     const auto toBoolean = to->as<bool>();
75     auto& animatedBoolean = animated->as<bool>();
76 
77     m_animationElement->animateDiscreteType<bool>(percentage, fromBoolean, toBoolean, animatedBoolean);
78 }

80 float SVGAnimatedBooleanAnimator::calculateDistance(const String&, const String&)
