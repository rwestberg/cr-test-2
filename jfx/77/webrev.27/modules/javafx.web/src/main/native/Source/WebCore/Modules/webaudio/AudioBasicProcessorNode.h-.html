36 // AudioBasicProcessorNode is an AudioNode with one input and one output where the input and output have the same number of channels.
37 class AudioBasicProcessorNode : public AudioNode {
38 public:
39     AudioBasicProcessorNode(AudioContext&, float sampleRate);
40 
41     // AudioNode
42     void process(size_t framesToProcess) override;
43     void pullInputs(size_t framesToProcess) override;
44     void reset() override;
45     void initialize() override;
46     void uninitialize() override;
47 
48     // Called in the main thread when the number of channels for the input may have changed.
49     void checkNumberOfChannelsForInput(AudioNodeInput*) override;
50 
51     // Returns the number of channels for both the input and the output.
52     unsigned numberOfChannels();
53 
54 protected:
55     double tailTime() const override;
56     double latencyTime() const override;
57 
58     AudioProcessor* processor() { return m_processor.get(); }
59     std::unique_ptr<AudioProcessor> m_processor;
60 };
