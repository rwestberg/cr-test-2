26 #include "config.h"
27 #include "UndoItem.h"
28 
29 #include "UndoManager.h"
30 #include <wtf/IsoMallocInlines.h>
31 
32 namespace WebCore {
33 
34 WTF_MAKE_ISO_ALLOCATED_IMPL(UndoItem);
35 
36 UndoManager* UndoItem::undoManager() const
37 {
38     return m_undoManager.get();
39 }
40 
41 void UndoItem::setUndoManager(UndoManager* undoManager)
42 {
43     m_undoManager = makeWeakPtr(undoManager);
44 }
45 
46 void UndoItem::invalidate()
47 {
48     if (auto* undoManager = m_undoManager.get()) {
49         undoManager->removeItem(*this);
50         m_undoManager = nullptr;
51     }
52 }
53 
54 bool UndoItem::isValid() const
55 {
56     return !!m_undoManager;
57 }
58 
59 Document* UndoItem::document() const
60 {
61     if (!isValid())
62         return nullptr;
63 
64     return &m_undoManager->document();
65 }
66 
67 } // namespace WebCore
