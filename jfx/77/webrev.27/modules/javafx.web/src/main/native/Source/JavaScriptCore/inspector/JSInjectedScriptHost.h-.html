26 #pragma once
27 
28 #include "JSDestructibleObject.h"
29 
30 namespace Inspector {
31 
32 class InjectedScriptHost;
33 
34 class JSInjectedScriptHost final : public JSC::JSDestructibleObject {
35 public:
36     typedef JSC::JSDestructibleObject Base;
37     static const unsigned StructureFlags = Base::StructureFlags;
38 
39     DECLARE_INFO;
40 
41     static JSC::Structure* createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
42     {
43         return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
44     }
45 
46     static JSInjectedScriptHost* create(JSC::VM& vm, JSC::Structure* structure, Ref<InjectedScriptHost>&& impl)
47     {
48         JSInjectedScriptHost* instance = new (NotNull, JSC::allocateCell<JSInjectedScriptHost>(vm.heap)) JSInjectedScriptHost(vm, structure, WTFMove(impl));
49         instance->finishCreation(vm);
50         return instance;
51     }
52 
53     static JSC::JSObject* createPrototype(JSC::VM&, JSC::JSGlobalObject*);
54     static void destroy(JSC::JSCell*);
55 
56     InjectedScriptHost& impl() const { return m_wrapped; }
57 
58     // Attributes.
59     JSC::JSValue evaluate(JSC::ExecState*) const;
60 
61     // Functions.
62     JSC::JSValue evaluateWithScopeExtension(JSC::ExecState*);
63     JSC::JSValue internalConstructorName(JSC::ExecState*);
64     JSC::JSValue isHTMLAllCollection(JSC::ExecState*);
65     JSC::JSValue subtype(JSC::ExecState*);
66     JSC::JSValue functionDetails(JSC::ExecState*);
67     JSC::JSValue getInternalProperties(JSC::ExecState*);
68     JSC::JSValue proxyTargetValue(JSC::ExecState*);
69     JSC::JSValue weakMapSize(JSC::ExecState*);
70     JSC::JSValue weakMapEntries(JSC::ExecState*);
71     JSC::JSValue weakSetSize(JSC::ExecState*);
72     JSC::JSValue weakSetEntries(JSC::ExecState*);
73     JSC::JSValue iteratorEntries(JSC::ExecState*);
74     JSC::JSValue queryObjects(JSC::ExecState*);
75 
76 protected:
77     void finishCreation(JSC::VM&);
78 
79 private:
80     JSInjectedScriptHost(JSC::VM&, JSC::Structure*, Ref<InjectedScriptHost>&&);
81 
82     Ref<InjectedScriptHost> m_wrapped;
83 };
84 
85 } // namespace Inspector
