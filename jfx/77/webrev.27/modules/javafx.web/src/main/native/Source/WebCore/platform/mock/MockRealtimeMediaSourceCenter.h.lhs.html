 31 #include "CaptureDeviceManager.h"
 32 #include "MockMediaDevice.h"
 33 #include "MockRealtimeAudioSource.h"
 34 #include "MockRealtimeVideoSource.h"
 35 #include "RealtimeMediaSourceCenter.h"
 36 
 37 namespace WebCore {
 38 
 39 class MockRealtimeMediaSourceCenter {
 40 public:
 41     WEBCORE_EXPORT static MockRealtimeMediaSourceCenter& singleton();
 42 
 43     WEBCORE_EXPORT static void setMockRealtimeMediaSourceCenterEnabled(bool);

 44 
 45     WEBCORE_EXPORT static void setDevices(Vector<MockMediaDevice>&&);
 46     WEBCORE_EXPORT static void addDevice(const MockMediaDevice&);
 47     WEBCORE_EXPORT static void removeDevice(const String& persistentId);
 48     WEBCORE_EXPORT static void resetDevices();
 49 
 50     void setMockAudioCaptureEnabled(bool isEnabled) { m_isMockAudioCaptureEnabled = isEnabled; }
 51     void setMockVideoCaptureEnabled(bool isEnabled) { m_isMockVideoCaptureEnabled = isEnabled; }
 52     void setMockDisplayCaptureEnabled(bool isEnabled) { m_isMockDisplayCaptureEnabled = isEnabled; }
 53 
 54     static Vector<CaptureDevice>& audioDevices();
 55     static Vector<CaptureDevice>& videoDevices();
 56     static Vector<CaptureDevice>& displayDevices();
 57 
 58     static Optional<MockMediaDevice> mockDeviceWithPersistentID(const String&);
 59     static Optional<CaptureDevice> captureDeviceWithPersistentID(CaptureDevice::DeviceType, const String&);
 60 
 61     CaptureDeviceManager& audioCaptureDeviceManager() { return m_audioCaptureDeviceManager; }
 62     CaptureDeviceManager& videoCaptureDeviceManager() { return m_videoCaptureDeviceManager; }
 63     CaptureDeviceManager& displayCaptureDeviceManager() { return m_displayCaptureDeviceManager; }
 64 
 65 private:
 66     MockRealtimeMediaSourceCenter() = default;
 67     friend NeverDestroyed<MockRealtimeMediaSourceCenter>;
 68 
 69     AudioCaptureFactory& audioCaptureFactory();
 70     VideoCaptureFactory& videoCaptureFactory();
 71     DisplayCaptureFactory& displayCaptureFactory();
 72 
 73     class MockAudioCaptureDeviceManager final : public CaptureDeviceManager {
 74     private:
 75         const Vector<CaptureDevice>& captureDevices() final { return MockRealtimeMediaSourceCenter::audioDevices(); }
 76         Optional<CaptureDevice> captureDeviceWithPersistentID(CaptureDevice::DeviceType type, const String& id) final { return MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(type, id); }
 77     };
 78     class MockVideoCaptureDeviceManager final : public CaptureDeviceManager {
 79     private:
 80         const Vector<CaptureDevice>& captureDevices() final { return MockRealtimeMediaSourceCenter::videoDevices(); }
 81         Optional<CaptureDevice> captureDeviceWithPersistentID(CaptureDevice::DeviceType type, const String& id) final { return MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(type, id); }
 82     };
 83     class MockDisplayCaptureDeviceManager final : public CaptureDeviceManager {
 84     private:
 85         const Vector<CaptureDevice>& captureDevices() final { return MockRealtimeMediaSourceCenter::displayDevices(); }
 86         Optional<CaptureDevice> captureDeviceWithPersistentID(CaptureDevice::DeviceType type, const String& id) final { return MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(type, id); }
 87     };
 88 
 89     MockAudioCaptureDeviceManager m_audioCaptureDeviceManager;
 90     MockVideoCaptureDeviceManager m_videoCaptureDeviceManager;
 91     MockDisplayCaptureDeviceManager m_displayCaptureDeviceManager;
 92 
 93     bool m_isMockAudioCaptureEnabled { true };
 94     bool m_isMockVideoCaptureEnabled { true };
 95     bool m_isMockDisplayCaptureEnabled { true };
 96 };

 98 }
