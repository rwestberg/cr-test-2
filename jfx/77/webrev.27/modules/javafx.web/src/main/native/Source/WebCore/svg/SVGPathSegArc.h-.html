 21 #pragma once
 22 
 23 #include "SVGPathSegWithContext.h"
 24 
 25 namespace WebCore {
 26 
 27 class SVGPathSegArc : public SVGPathSegWithContext {
 28 public:
 29     SVGPathSegArc(const SVGPathElement& element, SVGPathSegRole role, float x, float y, float r1, float r2, float angle, bool largeArcFlag, bool sweepFlag)
 30         : SVGPathSegWithContext(element, role)
 31         , m_x(x)
 32         , m_y(y)
 33         , m_r1(r1)
 34         , m_r2(r2)
 35         , m_angle(angle)
 36         , m_largeArcFlag(largeArcFlag)
 37         , m_sweepFlag(sweepFlag)
 38     {
 39     }
 40 
 41     float x() const { return m_x; }
 42     void setX(float x)
 43     {
 44         m_x = x;
 45         commitChange();
 46     }
 47 
 48     float y() const { return m_y; }
 49     void setY(float y)
 50     {
 51         m_y = y;
 52         commitChange();
 53     }
 54 
 55     float r1() const { return m_r1; }
 56     void setR1(float r1)
 57     {
 58         m_r1 = r1;
 59         commitChange();
 60     }
 61 
 62     float r2() const { return m_r2; }
 63     void setR2(float r2)
 64     {
 65         m_r2 = r2;
 66         commitChange();
 67     }
 68 
 69     float angle() const { return m_angle; }
 70     void setAngle(float angle)
 71     {
 72         m_angle = angle;
 73         commitChange();
 74     }
 75 
 76     bool largeArcFlag() const { return m_largeArcFlag; }
 77     void setLargeArcFlag(bool largeArcFlag)
 78     {
 79         m_largeArcFlag = largeArcFlag;
 80         commitChange();
 81     }
 82 
 83     bool sweepFlag() const { return m_sweepFlag; }
 84     void setSweepFlag(bool sweepFlag)
 85     {
 86         m_sweepFlag = sweepFlag;
 87         commitChange();
 88     }
 89 
 90 private:
 91     float m_x;
 92     float m_y;
 93     float m_r1;
 94     float m_r2;
 95     float m_angle;
 96 
 97     bool m_largeArcFlag : 1;
 98     bool m_sweepFlag : 1;
 99 };
100 
101 } // namespace WebCore
