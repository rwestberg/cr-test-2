40 private:
41     SVGClipPathElement(const QualifiedName&, Document&);
42 
43     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGClipPathElement, SVGGraphicsElement, SVGExternalResourcesRequired>;

56     PropertyRegistry m_propertyRegistry { *this };
