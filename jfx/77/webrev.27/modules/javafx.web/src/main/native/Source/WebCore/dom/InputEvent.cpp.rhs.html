36 Ref<InputEvent> InputEvent::create(const AtomString& eventType, const String& inputType, IsCancelable cancelable, RefPtr<WindowProxy>&& view, const String& data, RefPtr<DataTransfer>&& dataTransfer, const Vector<RefPtr<StaticRange>>& targetRanges, int detail)
37 {
38     return adoptRef(*new InputEvent(eventType, inputType, cancelable, WTFMove(view), data, WTFMove(dataTransfer), targetRanges, detail));
39 }

41 InputEvent::InputEvent(const AtomString& eventType, const String& inputType, IsCancelable cancelable, RefPtr<WindowProxy>&& view, const String& data, RefPtr<DataTransfer>&& dataTransfer, const Vector<RefPtr<StaticRange>>& targetRanges, int detail)
42     : UIEvent(eventType, CanBubble::Yes, cancelable, IsComposed::Yes, WTFMove(view), detail)
43     , m_inputType(inputType)
44     , m_data(data)
45     , m_dataTransfer(dataTransfer)
46     , m_targetRanges(targetRanges)
47 {
48 }

50 InputEvent::InputEvent(const AtomString& eventType, const Init& initializer)
51     : UIEvent(eventType, initializer)
52     , m_inputType(emptyString())
53     , m_data(initializer.data)
54 {
55 }

57 RefPtr<DataTransfer> InputEvent::dataTransfer() const
58 {
59     return m_dataTransfer;
60 }
