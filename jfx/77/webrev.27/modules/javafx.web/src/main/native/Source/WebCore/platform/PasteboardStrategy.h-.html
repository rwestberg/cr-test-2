 26 #ifndef PasteboardStrategy_h
 27 #define PasteboardStrategy_h
 28 
 29 #include <wtf/Forward.h>
 30 
 31 namespace WebCore {
 32 
 33 class Color;
 34 class SelectionData;
 35 class SharedBuffer;
 36 struct PasteboardImage;
 37 struct PasteboardItemInfo;
 38 struct PasteboardURL;
 39 struct PasteboardWebContent;
 40 struct PasteboardCustomData;
 41 
 42 class PasteboardStrategy {
 43 public:
 44 #if PLATFORM(IOS_FAMILY)
 45     virtual void writeToPasteboard(const PasteboardURL&, const String& pasteboardName) = 0;
 46     virtual void writeToPasteboard(const PasteboardWebContent&, const String& pasteboardName) = 0;
 47     virtual void writeToPasteboard(const PasteboardImage&, const String& pasteboardName) = 0;
 48     virtual void writeToPasteboard(const String& pasteboardType, const String&, const String& pasteboardName) = 0;
 49     virtual int getPasteboardItemsCount(const String& pasteboardName) = 0;
 50     virtual String readStringFromPasteboard(int index, const String& pasteboardType, const String& pasteboardName) = 0;
 51     virtual RefPtr<SharedBuffer> readBufferFromPasteboard(int index, const String& pasteboardType, const String& pasteboardName) = 0;
 52     virtual URL readURLFromPasteboard(int index, const String& pasteboardName, String& title) = 0;
 53     virtual Vector<PasteboardItemInfo> allPasteboardItemInfo(const String& pasteboardName) = 0;
 54     virtual PasteboardItemInfo informationForItemAtIndex(int index, const String& pasteboardName) = 0;
 55     virtual void updateSupportedTypeIdentifiers(const Vector<String>& identifiers, const String& pasteboardName) = 0;
 56     virtual void getTypesByFidelityForItemAtIndex(Vector<String>& types, uint64_t index, const String& pasteboardName) = 0;
 57 #endif // PLATFORM(IOS_FAMILY)
 58 #if PLATFORM(COCOA)
 59     virtual void getTypes(Vector<String>& types, const String& pasteboardName) = 0;
 60     virtual RefPtr<SharedBuffer> bufferForType(const String& pasteboardType, const String& pasteboardName) = 0;
 61     virtual void getPathnamesForType(Vector<String>& pathnames, const String& pasteboardType, const String& pasteboardName) = 0;
 62     virtual String stringForType(const String& pasteboardType, const String& pasteboardName) = 0;
 63     virtual Vector<String> allStringsForType(const String& pasteboardType, const String& pasteboardName) = 0;
 64     virtual long changeCount(const String& pasteboardName) = 0;
 65     virtual String uniqueName() = 0;
 66     virtual Color color(const String& pasteboardName) = 0;
 67     virtual URL url(const String& pasteboardName) = 0;
 68     virtual int getNumberOfFiles(const String& pasteboardName) = 0;
 69 
 70     virtual long addTypes(const Vector<String>& pasteboardTypes, const String& pasteboardName) = 0;
 71     virtual long setTypes(const Vector<String>& pasteboardTypes, const String& pasteboardName) = 0;
 72     virtual long setBufferForType(SharedBuffer*, const String& pasteboardType, const String& pasteboardName) = 0;
 73     virtual long setURL(const PasteboardURL&, const String& pasteboardName) = 0;
 74     virtual long setColor(const Color&, const String& pasteboardName) = 0;
 75     virtual long setStringForType(const String&, const String& pasteboardType, const String& pasteboardName) = 0;
 76 #endif
 77 
 78     virtual Vector<String> typesSafeForDOMToReadAndWrite(const String& pasteboardName, const String& origin) = 0;
 79     virtual long writeCustomData(const PasteboardCustomData&, const String& pasteboardName) = 0;
 80 
 81 #if PLATFORM(GTK)
 82     virtual void writeToClipboard(const String& pasteboardName, const SelectionData&) = 0;
 83     virtual Ref<SelectionData> readFromClipboard(const String& pasteboardName) = 0;
 84 #endif // PLATFORM(GTK)
 85 
 86 #if USE(LIBWPE)
 87     virtual void getTypes(Vector<String>& types) = 0;
 88     virtual String readStringFromPasteboard(int index, const String& pasteboardType) = 0;
 89     virtual void writeToPasteboard(const PasteboardWebContent&) = 0;
 90     virtual void writeToPasteboard(const String& pasteboardType, const String&) = 0;
 91 #endif
 92 
 93 protected:
 94     virtual ~PasteboardStrategy()
 95     {
 96     }
 97 };
 98 
 99 }
100 
101 #endif // !PasteboardStrategy_h
