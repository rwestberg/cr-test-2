26 #pragma once
27 
28 #include "ExceptionOr.h"
29 #include <wtf/Ref.h>
30 #include <wtf/RefCounted.h>
31 #include <wtf/RefPtr.h>
32 
33 namespace WebCore {
34 
35 class Node;
36 class Range;
37 
38 class StaticRange : public RefCounted<StaticRange> {
39 public:
40     ~StaticRange();
41 
42     static Ref<StaticRange> createFromRange(const Range&);
43     static Ref<StaticRange> create(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
44 
45     struct Init {
46         RefPtr<Node> startContainer;
47         unsigned long startOffset { 0 };
48         RefPtr<Node> endContainer;
49         unsigned long endOffset { 0 };
50     };
51 
52     static ExceptionOr<Ref<StaticRange>> create(Init&&);
53 
54     unsigned startOffset() const { return m_startOffset; }
55     unsigned endOffset() const { return m_endOffset; }
56     Node* startContainer() const;
57     Node* endContainer() const;
58     bool collapsed() const;
59 
60 private:
61     StaticRange(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
62 
63     Ref<Node> m_startContainer;
64     unsigned m_startOffset;
65     Ref<Node> m_endContainer;
66     unsigned m_endOffset;
67 };
68 
69 }
