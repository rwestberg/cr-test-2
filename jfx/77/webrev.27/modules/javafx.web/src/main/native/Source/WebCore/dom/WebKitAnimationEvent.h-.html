26 #pragma once
27 
28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class WebKitAnimationEvent final : public Event {
33 public:
34     static Ref<WebKitAnimationEvent> create(const AtomicString& type, const String& animationName, double elapsedTime)
35     {
36         return adoptRef(*new WebKitAnimationEvent(type, animationName, elapsedTime));
37     }
38 
39     struct Init : EventInit {
40         String animationName;
41         double elapsedTime { 0.0 };
42     };
43 
44     static Ref<WebKitAnimationEvent> create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
45     {
46         return adoptRef(*new WebKitAnimationEvent(type, initializer, isTrusted));
47     }
48 
49     virtual ~WebKitAnimationEvent();
50 
51     const String& animationName() const;
52     double elapsedTime() const;
53 
54     EventInterface eventInterface() const override;
55 
56 private:
57     WebKitAnimationEvent(const AtomicString& type, const String& animationName, double elapsedTime);
58     WebKitAnimationEvent(const AtomicString&, const Init&, IsTrusted);
59 
60     String m_animationName;
61     double m_elapsedTime;
62 };
63 
64 } // namespace WebCore
