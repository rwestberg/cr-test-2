 5 #  GTK3_INCLUDE_DIRS - the GTK+ 3. include directories
 6 #  GTK3_LIBRARIES - link these to use GTK+ 3.
 7 #  GTK3_SUPPORTS_GESTURES - GTK+ supports gestures (GTK+ >= 3.14)
 8 #  GTK3_SUPPORTS_QUARTZ - GTK+ supports Quartz backend
 9 #  GTK3_SUPPORTS_X11 - GTK+ supports X11 backend
10 #  GTK3_SUPPORTS_WAYLAND - GTK+ supports Wayland backend

36 find_package(PkgConfig)
37 pkg_check_modules(GTK3 gtk+-3.0)
38 
39 set(VERSION_OK TRUE)
40 if (GTK3_VERSION)
41     if (GTK3_FIND_VERSION_EXACT)
42         if (NOT("${GTK3_FIND_VERSION}" VERSION_EQUAL "${GTK3_VERSION}"))
43             set(VERSION_OK FALSE)
44         endif ()
45     else ()
46         if ("${GTK3_VERSION}" VERSION_LESS "${GTK3_FIND_VERSION}")
47             set(VERSION_OK FALSE)
48         endif ()
49     endif ()
50 endif ()
51 
52 if (GTK3_VERSION AND VERSION_OK)
53     pkg_check_modules(GTK3_QUARTZ gtk+-quartz-3.0)
54     if ("${GTK3_QUARTZ_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
55         set(GTK3_SUPPORTS_QUARTZ TRUE)
56     else ()
57         set(GTK3_SUPPORTS_QUARTZ FALSE)
58     endif ()
59 
60     pkg_check_modules(GTK3_X11 gtk+-x11-3.0)
61     if ("${GTK3_X11_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
62         set(GTK3_SUPPORTS_X11 TRUE)
63     else ()
64         set(GTK3_SUPPORTS_X11 FALSE)
65     endif ()
66 
67     pkg_check_modules(GTK3_WAYLAND gtk+-wayland-3.0)
68     if ("${GTK3_WAYLAND_VERSION}" VERSION_EQUAL "${GTK3_VERSION}")
69         set(GTK3_SUPPORTS_WAYLAND TRUE)
70     else ()
71         set(GTK3_SUPPORTS_WAYLAND FALSE)
72     endif ()
73 
74     if (NOT("${GTK3_VERSION}" VERSION_LESS "3.14.0"))
75         set(GTK3_SUPPORTS_GESTURES ON)
76     else ()
77         set(GTK3_SUPPORTS_GESTURES OFF)
78     endif ()
79 endif ()
80 
81 include(FindPackageHandleStandardArgs)
82 FIND_PACKAGE_HANDLE_STANDARD_ARGS(GTK3 DEFAULT_MSG GTK3_INCLUDE_DIRS GTK3_LIBRARIES VERSION_OK)
