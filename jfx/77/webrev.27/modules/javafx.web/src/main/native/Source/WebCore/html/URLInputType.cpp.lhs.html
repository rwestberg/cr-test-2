41 namespace WebCore {
42 
43 const AtomicString& URLInputType::formControlType() const
44 {
45     return InputTypeNames::url();
46 }
47 
48 bool URLInputType::typeMismatchFor(const String& value) const
49 {
50     return !value.isEmpty() && !URL(URL(), value).isValid();
51 }
52 
53 bool URLInputType::typeMismatch() const
54 {
55     ASSERT(element());
56     return typeMismatchFor(element()->value());
57 }
58 
59 String URLInputType::typeMismatchText() const
60 {
61     return validationMessageTypeMismatchForURLText();
62 }
63 
64 bool URLInputType::isURLField() const
65 {
66     return true;
67 }
68 
69 String URLInputType::sanitizeValue(const String& proposedValue) const
70 {
71     return stripLeadingAndTrailingHTMLSpaces(BaseTextInputType::sanitizeValue(proposedValue));
72 }
