26 #pragma once
27 
28 #include "DeviceOrientationOrMotionEvent.h"
29 #include "Event.h"
30 
31 namespace WebCore {
32 
33 class DeviceMotionData;
34 
35 class DeviceMotionEvent final : public Event, public DeviceOrientationOrMotionEvent {
36 public:
37     virtual ~DeviceMotionEvent();
38 
39     // FIXME: Merge this with DeviceMotionData::Acceleration
40     struct Acceleration {
41         Optional<double> x;
42         Optional<double> y;
43         Optional<double> z;
44     };
45 
46     // FIXME: Merge this with DeviceMotionData::RotationRate
47     struct RotationRate {
48         Optional<double> alpha;
49         Optional<double> beta;
50         Optional<double> gamma;
51     };
52 
53     static Ref<DeviceMotionEvent> create(const AtomString& eventType, DeviceMotionData* deviceMotionData)
54     {
55         return adoptRef(*new DeviceMotionEvent(eventType, deviceMotionData));
56     }
57 
58     static Ref<DeviceMotionEvent> createForBindings()
59     {
60         return adoptRef(*new DeviceMotionEvent);
61     }
62 
63     Optional<Acceleration> acceleration() const;
64     Optional<Acceleration> accelerationIncludingGravity() const;
65     Optional<RotationRate> rotationRate() const;
66     Optional<double> interval() const;
67 
68     void initDeviceMotionEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<Acceleration>&&, Optional<Acceleration>&&, Optional<RotationRate>&&, Optional<double>);
69 
70 private:
71     DeviceMotionEvent();
72     DeviceMotionEvent(const AtomString& eventType, DeviceMotionData*);
73 
74     EventInterface eventInterface() const override;
75 
76     RefPtr<DeviceMotionData> m_deviceMotionData;
77 };
78 
79 } // namespace WebCore
