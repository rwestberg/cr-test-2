29 namespace WebCore {

31 WebKitTransitionEvent::WebKitTransitionEvent(const AtomicString& type, const String& propertyName, double elapsedTime, const String& pseudoElement)
32     : Event(type, CanBubble::Yes, IsCancelable::Yes)
33     , m_propertyName(propertyName)
34     , m_elapsedTime(elapsedTime)
35     , m_pseudoElement(pseudoElement)
36 {
37 }
38 
39 WebKitTransitionEvent::WebKitTransitionEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
40     : Event(type, initializer, isTrusted)
41     , m_propertyName(initializer.propertyName)
42     , m_elapsedTime(initializer.elapsedTime)
43     , m_pseudoElement(initializer.pseudoElement)
44 {
45 }
46 
47 WebKitTransitionEvent::~WebKitTransitionEvent() = default;
48 
49 const String& WebKitTransitionEvent::propertyName() const
50 {
51     return m_propertyName;
52 }
53 
54 double WebKitTransitionEvent::elapsedTime() const
55 {
56     return m_elapsedTime;
57 }
58 
59 const String& WebKitTransitionEvent::pseudoElement() const
60 {
61     return m_pseudoElement;
62 }
63 
64 EventInterface WebKitTransitionEvent::eventInterface() const
65 {
66     return WebKitTransitionEventInterfaceType;
67 }
