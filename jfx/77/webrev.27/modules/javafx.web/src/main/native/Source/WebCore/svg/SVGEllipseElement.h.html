24 #include "SVGExternalResourcesRequired.h"

28 namespace WebCore {
29 
30 class SVGEllipseElement final : public SVGGeometryElement, public SVGExternalResourcesRequired {
31     WTF_MAKE_ISO_ALLOCATED(SVGEllipseElement);
32 public:
33     static Ref<SVGEllipseElement> create(const QualifiedName&, Document&);
34 
35     const SVGLengthValue& cx() const { return m_cx->currentValue(); }
36     const SVGLengthValue& cy() const { return m_cy->currentValue(); }
37     const SVGLengthValue& rx() const { return m_rx->currentValue(); }
38     const SVGLengthValue& ry() const { return m_ry->currentValue(); }
39 
40     SVGAnimatedLength& cxAnimated() { return m_cx; }
41     SVGAnimatedLength& cyAnimated() { return m_cy; }
42     SVGAnimatedLength& rxAnimated() { return m_rx; }
43     SVGAnimatedLength& ryAnimated() { return m_ry; }
44 
45 private:
46     SVGEllipseElement(const QualifiedName&, Document&);
47 
48     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGEllipseElement, SVGGeometryElement, SVGExternalResourcesRequired>;
49     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
50 
51     void parseAttribute(const QualifiedName&, const AtomString&) final;
52     void svgAttributeChanged(const QualifiedName&) final;
53 
54     bool isValid() const final { return SVGTests::isValid(); }
55     bool selfHasRelativeLengths() const final { return true; }
56 
57     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
58 
59     PropertyRegistry m_propertyRegistry { *this };
60     Ref<SVGAnimatedLength> m_cx { SVGAnimatedLength::create(this, LengthModeWidth) };
61     Ref<SVGAnimatedLength> m_cy { SVGAnimatedLength::create(this, LengthModeHeight) };
62     Ref<SVGAnimatedLength> m_rx { SVGAnimatedLength::create(this, LengthModeWidth) };
63     Ref<SVGAnimatedLength> m_ry { SVGAnimatedLength::create(this, LengthModeHeight) };
64 };

66 } // namespace WebCore
