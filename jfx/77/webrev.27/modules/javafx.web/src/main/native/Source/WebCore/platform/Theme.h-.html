26 #pragma once
27 
28 #include "ThemeTypes.h"
29 #include <wtf/Forward.h>
30 #include <wtf/Optional.h>
31 
32 namespace WebCore {
33 
34 class ControlStates;
35 class FloatRect;
36 class FontCascade;
37 class FontCascadeDescription;
38 class GraphicsContext;
39 class LengthBox;
40 class ScrollView;
41 
42 struct LengthSize;
43 
44 class Theme {
45 public:
46     static Theme& singleton();
47 
48     // A function to obtain the baseline position adjustment for a "leaf" control. This will be used if a baseline
49     // position cannot be determined by examining child content. Checkboxes and radio buttons are examples of
50     // controls that need to do this. The adjustment is an offset that adds to the baseline, e.g., marginTop() + height() + |offset|.
51     // The offset is not zoomed.
52     virtual int baselinePositionAdjustment(ControlPart) const;
53 
54     // The font description result should have a zoomed font size.
55     virtual Optional<FontCascadeDescription> controlFont(ControlPart, const FontCascade&, float zoomFactor) const;
56 
57     // The size here is in zoomed coordinates already. If a new size is returned, it also needs to be in zoomed coordinates.
58     virtual LengthSize controlSize(ControlPart, const FontCascade&, const LengthSize& zoomedSize, float zoomFactor) const;
59 
60     // Returns the minimum size for a control in zoomed coordinates.
61     virtual LengthSize minimumControlSize(ControlPart, const FontCascade&, float zoomFactor) const;
62 
63     // Allows the theme to modify the existing padding/border.
64     virtual LengthBox controlPadding(ControlPart, const FontCascade&, const LengthBox& zoomedBox, float zoomFactor) const;
65     virtual LengthBox controlBorder(ControlPart, const FontCascade&, const LengthBox& zoomedBox, float zoomFactor) const;
66 
67     // Whether or not whitespace: pre should be forced on always.
68     virtual bool controlRequiresPreWhiteSpace(ControlPart) const;
69 
70     // Method for painting a control. The rect is in zoomed coordinates.
71     virtual void paint(ControlPart, ControlStates&, GraphicsContext&, const FloatRect& zoomedRect, float zoomFactor, ScrollView*, float deviceScaleFactor, float pageScaleFactor, bool useSystemAppearance, bool useDarkAppearance);
72 
73     // Some controls may spill out of their containers (e.g., the check on an OS X checkbox).  When these controls repaint,
74     // the theme needs to communicate this inflated rect to the engine so that it can invalidate the whole control.
75     // The rect passed in is in zoomed coordinates, so the inflation should take that into account and make sure the inflation
76     // amount is also scaled by the zoomFactor.
77     virtual void inflateControlPaintRect(ControlPart, const ControlStates&, FloatRect& zoomedRect, float zoomFactor) const;
78 
79     virtual void drawNamedImage(const String&, GraphicsContext&, const FloatRect&) const;
80 
81     virtual bool userPrefersReducedMotion() const;
82 
83 protected:
84     Theme() = default;
85     virtual ~Theme() = default;
86 
87 private:
88     Theme(const Theme&) = delete;
89     void operator=(const Theme&) = delete;
90 };
91 
92 } // namespace WebCore
