27 // https://html.spec.whatwg.org/multipage/webappapis.html#windoweventhandlers
28 [
29     NoInterfaceObject,
30 ] interface WindowEventHandlers {
31     // Commented out event handlers are defined by the HTML5 specification but
32     // are not yet implemented.
33 
34     // [WindowEventHandler] attribute EventHandler onafterprint;.
35     // [WindowEventHandler] attribute EventHandler onbeforeprint;.
36     [WindowEventHandler] attribute EventHandler onbeforeunload;
37     [WindowEventHandler] attribute EventHandler onhashchange;
38     [WindowEventHandler] attribute EventHandler onlanguagechange;
39     [WindowEventHandler] attribute EventHandler onmessage;
40     [WindowEventHandler] attribute EventHandler onoffline;
41     [WindowEventHandler] attribute EventHandler ononline;
42     [WindowEventHandler] attribute EventHandler onpagehide;
43     [WindowEventHandler] attribute EventHandler onpageshow;
44     [WindowEventHandler] attribute EventHandler onpopstate;
45     [WindowEventHandler] attribute EventHandler onstorage;
46     [WindowEventHandler] attribute EventHandler onunload;
47 
48 
49     // Additions that are not yet part of the standard.
50 
51     [NotEnumerable, WindowEventHandler, Conditional=ORIENTATION_EVENTS] attribute EventHandler onorientationchange;
52 };
