24 #pragma once
25 
26 #include "ScriptWrappable.h"
27 #include <wtf/Forward.h>
28 #include <wtf/IsoMalloc.h>
29 #include <wtf/RefCounted.h>
30 
31 namespace WebCore {
32 
33 class Node;
34 
35 class NodeList : public ScriptWrappable, public RefCounted<NodeList> {
36     WTF_MAKE_ISO_ALLOCATED_EXPORT(NodeList, WEBCORE_EXPORT);
37 public:
38     virtual ~NodeList() = default;
39 
40     // DOM methods & attributes for NodeList
41     virtual unsigned length() const = 0;
42     virtual Node* item(unsigned index) const = 0;
43 
44     class Iterator {
45     public:
46         explicit Iterator(NodeList& list) : m_list(list) { }
47         Node* next() { return m_list->item(m_index++); }
48 
49     private:
50         size_t m_index { 0 };
51         Ref<NodeList> m_list;
52     };
53     Iterator createIterator() { return Iterator(*this); }
54 
55     // Other methods (not part of DOM)
56     virtual bool isLiveNodeList() const { return false; }
57     virtual bool isChildNodeList() const { return false; }
58     virtual bool isEmptyNodeList() const { return false; }
59     virtual size_t memoryCost() const { return 0; }
60 };
61 
62 } // namespace WebCore
