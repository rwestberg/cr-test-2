26 #pragma once
27 
28 #include "FontRanges.h"
29 #include <wtf/Forward.h>
30 #include <wtf/RefCounted.h>
31 
32 namespace WebCore {
33 
34 class FontCascadeDescription;
35 class FontDescription;
36 class FontSelectorClient;
37 
38 class FontAccessor : public RefCounted<FontAccessor> {
39 public:
40     virtual ~FontAccessor() = default;
41 
42     virtual const Font* font(ExternalResourceDownloadPolicy) const = 0;
43     virtual bool isLoading() const = 0;
44 };
45 
46 class FontSelector : public RefCounted<FontSelector> {
47 public:
48     virtual ~FontSelector() = default;
49 
50     virtual FontRanges fontRangesForFamily(const FontDescription&, const AtomString&) = 0;
51     virtual RefPtr<Font> fallbackFontAt(const FontDescription&, size_t) = 0;
52 
53     virtual size_t fallbackFontCount() = 0;
54 
55     virtual void opportunisticallyStartFontDataURLLoading(const FontCascadeDescription&, const AtomString& family) = 0;
56 
57     virtual void fontCacheInvalidated() { }
58 
59     virtual void registerForInvalidationCallbacks(FontSelectorClient&) = 0;
60     virtual void unregisterForInvalidationCallbacks(FontSelectorClient&) = 0;
61 
62     virtual unsigned uniqueId() const = 0;
63     virtual unsigned version() const = 0;
64 };
65 
66 }
