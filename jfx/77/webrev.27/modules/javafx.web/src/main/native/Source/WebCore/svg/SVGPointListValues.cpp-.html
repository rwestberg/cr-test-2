29 String SVGPointListValues::valueAsString() const
30 {
31     StringBuilder builder;
32 
33     unsigned size = this->size();
34     for (unsigned i = 0; i < size; ++i) {
35         if (i > 0)
36             builder.append(' '); // FIXME: Shouldn't we use commas to seperate?
37 
38         const auto& point = at(i);
39         builder.appendNumber(point.x());
40         builder.append(' ');
41         builder.appendNumber(point.y());
42     }
43 
44     return builder.toString();
45 }
