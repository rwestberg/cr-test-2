30 #pragma once
31 
32 #include "CSSPropertyNames.h"
33 #include "CSSValue.h"
34 
35 namespace WebCore {
36 
37 class CSSCustomIdentValue : public CSSValue {
38 public:
39     static Ref<CSSCustomIdentValue> create(const String& str)
40     {
41         return adoptRef(*new CSSCustomIdentValue(str));
42     }
43 
44     // FIXME: Remove this and lazily parse the CSSPropertyID in isKnownPropertyID().
45     static Ref<CSSCustomIdentValue> create(CSSPropertyID id)
46     {
47         return adoptRef(*new CSSCustomIdentValue(id));
48     }
49 
50     String value() const { ASSERT(!isKnownPropertyID()); return m_string; }
51     bool isKnownPropertyID() const { return m_propertyId != CSSPropertyInvalid; }
52     CSSPropertyID valueAsPropertyID() const { ASSERT(isKnownPropertyID()); return m_propertyId; }
53 
54     String customCSSText() const;
55 
56     bool equals(const CSSCustomIdentValue& other) const
57     {
58         return isKnownPropertyID() ? m_propertyId == other.m_propertyId : m_string == other.m_string;
59     }
60 
61 private:
62     CSSCustomIdentValue(const String&);
63     CSSCustomIdentValue(CSSPropertyID);
64 
65     // FIXME: Change this to an AtomicString.
66     String m_string;
67     CSSPropertyID m_propertyId;
68 };
69 
70 } // namespace WebCore
71 
72 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSCustomIdentValue, isCustomIdentValue())
