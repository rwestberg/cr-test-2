27 
28 #include "ActiveDOMObject.h"
29 #include "CanvasRenderingContext.h"
30 #include <wtf/IsoMalloc.h>
31 
32 namespace WebCore {
33 
34 class HTMLCanvasElement;
35 
36 class GPUBasedCanvasRenderingContext : public CanvasRenderingContext, public ActiveDOMObject {
37     WTF_MAKE_ISO_NONALLOCATABLE(GPUBasedCanvasRenderingContext);
38 public:
39 
40     bool isGPUBased() const override { return true; }
41 
42     bool isAccelerated() const override
43     {
44 #if PLATFORM(WIN)
45         // FIXME: Implement accelerated canvas on Windows.
46         return false;
47 #else
48         return true;
49 #endif
50     }
51 
52     virtual void reshape(int width, int height) = 0;
53     virtual void markLayerComposited() = 0;
54 
55 protected:
56     explicit GPUBasedCanvasRenderingContext(CanvasBase&);
57 
58     HTMLCanvasElement* htmlCanvas() const;
59     void notifyCanvasContentChanged();

60 };
61 
62 } // namespace WebCore
63 
64 SPECIALIZE_TYPE_TRAITS_CANVASRENDERINGCONTEXT(WebCore::GPUBasedCanvasRenderingContext, isGPUBased())
