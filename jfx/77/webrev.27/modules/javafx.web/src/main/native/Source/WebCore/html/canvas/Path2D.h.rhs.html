28 #pragma once
29 
30 #include "CanvasPath.h"
31 #include "SVGPathUtilities.h"

32 #include <wtf/RefCounted.h>
33 
34 namespace WebCore {
35 
36 struct DOMMatrix2DInit;
37 
38 class WEBCORE_EXPORT Path2D final : public RefCounted<Path2D>, public CanvasPath {
39     WTF_MAKE_FAST_ALLOCATED;
40 public:
41     virtual ~Path2D();
42 
43     static Ref<Path2D> create()
44     {
45         return adoptRef(*new Path2D);
46     }
47 
48     static Ref<Path2D> create(const Path& path)
49     {
50         return adoptRef(*new Path2D(path));
51     }
52 
53     static Ref<Path2D> create(const Path2D& path)
54     {
55         return create(path.path());
56     }
57 
58     static Ref<Path2D> create(const String& pathData)
59     {
60         return create(buildPathFromString(pathData));
61     }
62 
63     ExceptionOr<void> addPath(Path2D&, DOMMatrix2DInit&&);
64 
65     const Path& path() const { return m_path; }
66 
67 private:
68     Path2D() = default;
69     Path2D(const Path& path)
70         : CanvasPath(path)
71     {
72     }
73 };
74 
75 } // namespace WebCore
