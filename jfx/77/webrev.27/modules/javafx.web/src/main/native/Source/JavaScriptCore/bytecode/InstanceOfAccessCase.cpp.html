35 std::unique_ptr<AccessCase> InstanceOfAccessCase::create(
36     VM& vm, JSCell* owner, AccessType accessType, Structure* structure,
37     const ObjectPropertyConditionSet& conditionSet, JSObject* prototype)
38 {
39     return std::unique_ptr<AccessCase>(new InstanceOfAccessCase(vm, owner, accessType, structure, conditionSet, prototype));
40 }
41 
42 void InstanceOfAccessCase::dumpImpl(PrintStream& out, CommaPrinter& comma) const
43 {
44     Base::dumpImpl(out, comma);
45     out.print(comma, "prototype = ", JSValue(prototype()));
46 }
47 
48 std::unique_ptr<AccessCase> InstanceOfAccessCase::clone() const
49 {
50     std::unique_ptr<InstanceOfAccessCase> result(new InstanceOfAccessCase(*this));
51     result->resetState();
52     return result;
53 }

61     const ObjectPropertyConditionSet& conditionSet, JSObject* prototype)
62     : Base(vm, owner, accessType, invalidOffset, structure, conditionSet, nullptr)
