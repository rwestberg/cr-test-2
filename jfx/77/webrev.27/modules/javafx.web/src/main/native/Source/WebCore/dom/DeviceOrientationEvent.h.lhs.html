26 #pragma once
27 

28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class DeviceOrientationData;
33 
34 class DeviceOrientationEvent final : public Event {
35 public:
36     static Ref<DeviceOrientationEvent> create(const AtomicString& eventType, DeviceOrientationData* orientation)
37     {
38         return adoptRef(*new DeviceOrientationEvent(eventType, orientation));
39     }
40 
41     static Ref<DeviceOrientationEvent> createForBindings()
42     {
43         return adoptRef(*new DeviceOrientationEvent);
44     }
45 
46     virtual ~DeviceOrientationEvent();
47 
48     Optional<double> alpha() const;
49     Optional<double> beta() const;
50     Optional<double> gamma() const;
51 
52 #if PLATFORM(IOS_FAMILY)
53     Optional<double> compassHeading() const;
54     Optional<double> compassAccuracy() const;
55 
56     void initDeviceOrientationEvent(const AtomicString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<double> compassHeading, Optional<double> compassAccuracy);
57 #else
58     Optional<bool> absolute() const;
59 
60     void initDeviceOrientationEvent(const AtomicString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<bool> absolute);
61 #endif
62 
63 private:
64     DeviceOrientationEvent();
65     DeviceOrientationEvent(const AtomicString& eventType, DeviceOrientationData*);
66 
67     EventInterface eventInterface() const override;
68 
69     RefPtr<DeviceOrientationData> m_orientation;
70 };
71 
72 } // namespace WebCore
