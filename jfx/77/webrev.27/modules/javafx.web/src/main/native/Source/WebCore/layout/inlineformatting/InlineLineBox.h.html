26 #pragma once
27 
28 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
29 
30 #include "DisplayBox.h"
31 
32 namespace WebCore {
33 namespace Layout {
34 
35 class LineBox {
36 public:
37     struct Baseline {
38         LayoutUnit height() const { return ascent + descent; }
39 
40         LayoutUnit ascent;
41         LayoutUnit descent;
42     };
43     LineBox(Display::Rect, const Baseline&, LayoutUnit baselineOffset);
44 
45     LayoutPoint logicalTopLeft() const { return m_rect.topLeft(); }
46 
47     LayoutUnit logicalLeft() const { return m_rect.left(); }
48     LayoutUnit logicalRight() const { return m_rect.right(); }
49     LayoutUnit logicalTop() const { return m_rect.top(); }
50     LayoutUnit logicalBottom() const { return m_rect.bottom(); }
51 
52     LayoutUnit logicalWidth() const { return m_rect.width(); }
53     LayoutUnit logicalHeight() const { return m_rect.height(); }
54 
55     Baseline baseline() const { return m_baseline; }
56     // Baseline offset from line logical top. Note that offset does not necessarily equal to ascent.
57     //
58     // ------------------- line logical top
59     //             ^
60     //             |
61     //   ^         | baseline offset
62     //   |         |
63     //   | ascent  |
64     //   |         |
65     //   v         v
66     //   ----------------- baseline
67     //   ^
68     //   | descent
69     //   v
70     // ------------------- line logical bottom
71     LayoutUnit baselineOffset() const { return m_baselineOffset; }
72 
73 private:
74     Display::Rect m_rect;
75     Baseline m_baseline;
76     LayoutUnit m_baselineOffset;
77 };
78 
79 inline LineBox::LineBox(Display::Rect rect, const Baseline& baseline, LayoutUnit baselineOffset)
80     : m_rect(rect)
81     , m_baseline(baseline)
82     , m_baselineOffset(baselineOffset)
83 {
84 }
85 
86 }
87 }
88 
89 #endif
