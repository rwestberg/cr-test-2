40 namespace WebCore {
41 
42 class IDBKey;
43 class ScriptExecutionContext;
44 
45 class IDBKeyRange : public ScriptWrappable, public RefCounted<IDBKeyRange> {
46 public:
47     static Ref<IDBKeyRange> create(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
48     static Ref<IDBKeyRange> create(RefPtr<IDBKey>&&);
49     ~IDBKeyRange();
50 
51     IDBKey* lower() const { return m_lower.get(); }
52     IDBKey* upper() const { return m_upper.get(); }
53     bool lowerOpen() const { return m_isLowerOpen; }
54     bool upperOpen() const { return m_isUpperOpen; }
55 
56     static ExceptionOr<Ref<IDBKeyRange>> only(RefPtr<IDBKey>&& value);
57     static ExceptionOr<Ref<IDBKeyRange>> only(JSC::ExecState&, JSC::JSValue key);
58 
59     static ExceptionOr<Ref<IDBKeyRange>> lowerBound(JSC::ExecState&, JSC::JSValue bound, bool open);
60     static ExceptionOr<Ref<IDBKeyRange>> upperBound(JSC::ExecState&, JSC::JSValue bound, bool open);
61 
62     static ExceptionOr<Ref<IDBKeyRange>> bound(JSC::ExecState&, JSC::JSValue lower, JSC::JSValue upper, bool lowerOpen, bool upperOpen);
63 
64     ExceptionOr<bool> includes(JSC::ExecState&, JSC::JSValue key);
65 
66     WEBCORE_EXPORT bool isOnlyKey() const;
67 
68 private:
69     IDBKeyRange(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
70 
71     RefPtr<IDBKey> m_lower;
72     RefPtr<IDBKey> m_upper;
73     bool m_isLowerOpen;
74     bool m_isUpperOpen;
75 };
