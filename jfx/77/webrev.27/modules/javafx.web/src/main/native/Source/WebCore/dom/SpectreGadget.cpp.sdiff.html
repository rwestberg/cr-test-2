33 inline SpectreGadget::SpectreGadget(const String& text)
34 {
35     if (RuntimeEnabledFeatures::sharedFeatures().spectreGadgetsEnabled()) {
36         m_data.resize(text.length());
37         setReadLength(text.length());
38         m_data.fill(0);
39         m_dataPtr = m_data.data();
40 
41         for (size_t i = 0; i < m_readLength; i++)
42             m_data[i] = text.characterAt(i);
43     } else {
44         setReadLength(0);
45         m_dataPtr = nullptr;
46     }
47 }
