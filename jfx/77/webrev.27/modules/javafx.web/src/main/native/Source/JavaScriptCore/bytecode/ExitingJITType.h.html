30 namespace JSC {
31 
32 enum ExitingJITType : uint8_t {
33     ExitFromAnything,
34     ExitFromDFG,
35     ExitFromFTL
36 };
37 
38 inline ExitingJITType exitingJITTypeFor(JITType type)
39 {
40     switch (type) {
41     case JITType::DFGJIT:
42         return ExitFromDFG;
43     case JITType::FTLJIT:
44         return ExitFromFTL;
45     default:
46         RELEASE_ASSERT_NOT_REACHED();
47         return ExitFromAnything;
48     }
49 }
50 
51 } // namespace JSC
