35 Ref<RTCDataChannelEvent> RTCDataChannelEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<RTCDataChannel>&& channel)
36 {
37     return adoptRef(*new RTCDataChannelEvent(type, canBubble, cancelable, WTFMove(channel)));
38 }
39 
40 Ref<RTCDataChannelEvent> RTCDataChannelEvent::create(const AtomString& type, Init&& initializer, IsTrusted isTrusted)
41 {
42     return adoptRef(*new RTCDataChannelEvent(type, WTFMove(initializer), isTrusted));
43 }
44 
45 RTCDataChannelEvent::RTCDataChannelEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<RTCDataChannel>&& channel)
46     : Event(type, canBubble, cancelable)
47     , m_channel(WTFMove(channel))
48 {
49 }
50 
51 RTCDataChannelEvent::RTCDataChannelEvent(const AtomString& type, Init&& initializer, IsTrusted isTrusted)
52     : Event(type, initializer, isTrusted)
53     , m_channel(initializer.channel.releaseNonNull())
54 {
55 }
56 
57 RTCDataChannel& RTCDataChannelEvent::channel()
58 {
59     return m_channel.get();
60 }
61 
62 EventInterface RTCDataChannelEvent::eventInterface() const
63 {
64     return RTCDataChannelEventInterfaceType;
65 }
