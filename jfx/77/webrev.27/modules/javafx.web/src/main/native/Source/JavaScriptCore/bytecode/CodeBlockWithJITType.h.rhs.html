30 namespace JSC {
31 
32 // We sometimes what to print the CodeBlock's ID before setting its JITCode. At that
33 // point the CodeBlock will claim a bogus JITType. This helper class lets us do that.
34 
35 class CodeBlockWithJITType {
36 public:
37     CodeBlockWithJITType(CodeBlock* codeBlock, JITType jitType)
38         : m_codeBlock(codeBlock)
39         , m_jitType(jitType)
40     {
41     }
42 
43     void dump(PrintStream& out) const
44     {
45         m_codeBlock->dumpAssumingJITType(out, m_jitType);
46     }
47 private:
48     CodeBlock* m_codeBlock;
49     JITType m_jitType;
50 };
51 
52 } // namespace JSC
