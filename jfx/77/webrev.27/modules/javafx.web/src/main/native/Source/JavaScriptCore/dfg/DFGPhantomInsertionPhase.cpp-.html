 46 namespace DFGPhantomInsertionPhaseInternal {
 47 static const bool verbose = false;
 48 }
 49 
 50 class PhantomInsertionPhase : public Phase {
 51 public:
 52     PhantomInsertionPhase(Graph& graph)
 53         : Phase(graph, "phantom insertion")
 54         , m_insertionSet(graph)
 55         , m_values(OperandsLike, graph.block(0)->variablesAtHead)
 56     {
 57     }
 58 
 59     bool run()
 60     {
 61         // We assume that DCE has already run. If we run before DCE then we think that all
 62         // SetLocals execute, which is inaccurate. That causes us to insert too few Phantoms.
 63         DFG_ASSERT(m_graph, nullptr, m_graph.m_refCountState == ExactRefCount);
 64 
 65         if (DFGPhantomInsertionPhaseInternal::verbose) {
 66             dataLog("Graph before Phantom insertion:\n");
 67             m_graph.dump();
 68         }
 69 
 70         m_graph.clearEpochs();
 71 
 72         for (BasicBlock* block : m_graph.blocksInNaturalOrder())
 73             handleBlock(block);
 74 
 75         if (DFGPhantomInsertionPhaseInternal::verbose) {
 76             dataLog("Graph after Phantom insertion:\n");
 77             m_graph.dump();
 78         }
 79 
 80         return true;
 81     }
 82 
 83 private:
 84     void handleBlock(BasicBlock* block)
 85     {
 86         // FIXME: For blocks that have low register pressure, it would make the most sense to
 87         // simply insert Phantoms at the last point possible since that would obviate the need to
 88         // query bytecode liveness:
 89         //
 90         // - If we MovHint @x into loc42 then put a Phantom on the last MovHinted value in loc42.
 91         // - At the end of the block put Phantoms for each MovHinted value.
 92         //
 93         // This will definitely not work if there are any phantom allocations. For those blocks
 94         // where this would be legal, it remains to be seen how profitable it would be even if there
 95         // was high register pressure. After all, a Phantom would cause a spill but it wouldn't
 96         // cause a fill.
 97         //
 98         // https://bugs.webkit.org/show_bug.cgi?id=144524
 99 
100         m_values.fill(nullptr);
101 
102         Epoch currentEpoch = Epoch::first();
103         unsigned lastExitingIndex = 0;
104         for (unsigned nodeIndex = 0; nodeIndex < block->size(); ++nodeIndex) {
105             Node* node = block->at(nodeIndex);
106             if (DFGPhantomInsertionPhaseInternal::verbose)
107                 dataLog("Considering ", node, "\n");
108 
109             switch (node->op()) {
110             case MovHint:
111                 m_values.operand(node->unlinkedLocal()) = node->child1().node();
112                 break;
113 
114             case ZombieHint:
115                 m_values.operand(node->unlinkedLocal()) = nullptr;
116                 break;
117 
118             case GetLocal:
119             case SetArgument:
120                 m_values.operand(node->local()) = nullptr;
121                 break;
122 
123             default:
124                 break;
125             }
126 
127             bool nodeMayExit = mayExit(m_graph, node) != DoesNotExit;
128             if (nodeMayExit) {
129                 currentEpoch.bump();
130                 lastExitingIndex = nodeIndex;
131             }
132 
133             m_graph.doToChildren(
134                 node,
135                 [&] (Edge edge) {
136                     edge->setEpoch(currentEpoch);
137                 });
138 
139             node->setEpoch(currentEpoch);
140 
141             VirtualRegister alreadyKilled;
142 
143             auto processKilledOperand = [&] (VirtualRegister reg) {
144                 if (DFGPhantomInsertionPhaseInternal::verbose)
145                     dataLog("    Killed operand: ", reg, "\n");
146 
147                 // Already handled from SetLocal.
148                 if (reg == alreadyKilled)
149                     return;
150 
151                 Node* killedNode = m_values.operand(reg);
152                 if (!killedNode)
153                     return;
154 
155                 // We only need to insert a Phantom if the node hasn't been used since the last
156                 // exit, and was born before the last exit.
157                 if (killedNode->epoch() == currentEpoch)
158                     return;
159 
160                 if (DFGPhantomInsertionPhaseInternal::verbose) {
161                     dataLog(
162                         "    Inserting Phantom on ", killedNode, " after ",
163                         block->at(lastExitingIndex), "\n");
164                 }
165 
166                 // We have exact ref counts, so creating a new use means that we have to
167                 // increment the ref count.
168                 killedNode->postfixRef();
169 
170                 Node* lastExitingNode = block->at(lastExitingIndex);
171 
172                 m_insertionSet.insertNode(
173                     lastExitingIndex + 1, SpecNone, Phantom,
174                     lastExitingNode->origin.forInsertingAfter(m_graph, lastExitingNode),
175                     killedNode->defaultEdge());
176             };
177 
178             if (node->op() == SetLocal) {
179                 VirtualRegister local = node->local();
180                 if (nodeMayExit) {
181                     // If the SetLocal does exit, we need the MovHint of its local
182                     // to be live until the SetLocal is done.
183                     processKilledOperand(local);
184                     alreadyKilled = local;
185                 }
186                 m_values.operand(local) = nullptr;
187             }
188 
189             forAllKilledOperands(m_graph, node, block->tryAt(nodeIndex + 1), processKilledOperand);
190         }
191 
192         m_insertionSet.execute(block);
193     }
194 
195     InsertionSet m_insertionSet;
196     Operands<Node*> m_values;
197 };
198 
199 } // anonymous namespace
200 
201 bool performPhantomInsertion(Graph& graph)
202 {
203     return runPhase<PhantomInsertionPhase>(graph);
204 }
