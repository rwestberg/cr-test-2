35 namespace WebCore {
36 
37 namespace WHLSL {
38 
39 namespace AST {
40 
41 class WhileLoop final : public Statement {
42     WTF_MAKE_FAST_ALLOCATED;
43 public:
44     WhileLoop(CodeLocation location, UniqueRef<Expression>&& conditional, UniqueRef<Statement>&& body)
45         : Statement(location, Kind::WhileLoop)
46         , m_conditional(WTFMove(conditional))
47         , m_body(WTFMove(body))
48     {
49     }
50 
51     ~WhileLoop() = default;
52 
53     WhileLoop(const WhileLoop&) = delete;
54     WhileLoop(WhileLoop&&) = default;
55 


56     Expression& conditional() { return m_conditional; }
57     Statement& body() { return m_body; }
58 
59 private:
60     UniqueRef<Expression> m_conditional;
61     UniqueRef<Statement> m_body;
62 };
63 
64 } // namespace AST
65 
66 }
67 
68 }
69 
70 DEFINE_DEFAULT_DELETE(WhileLoop)
71 
72 SPECIALIZE_TYPE_TRAITS_WHLSL_STATEMENT(WhileLoop, isWhileLoop())
