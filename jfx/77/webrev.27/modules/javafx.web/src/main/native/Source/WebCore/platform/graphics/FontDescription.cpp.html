38 FontDescription::FontDescription()
39     : m_fontSelectionRequest { FontCascadeDescription::initialWeight(), FontCascadeDescription::initialStretch(), FontCascadeDescription::initialItalic() }
40     , m_orientation(static_cast<unsigned>(FontOrientation::Horizontal))
41     , m_nonCJKGlyphOrientation(static_cast<unsigned>(NonCJKGlyphOrientation::Mixed))
42     , m_widthVariant(static_cast<unsigned>(FontWidthVariant::RegularWidth))
43     , m_renderingMode(static_cast<unsigned>(FontRenderingMode::Normal))
44     , m_textRendering(static_cast<unsigned>(TextRenderingMode::AutoTextRendering))
45     , m_script(USCRIPT_COMMON)
46     , m_fontSynthesis(FontSynthesisWeight | FontSynthesisStyle | FontSynthesisSmallCaps)
47     , m_variantCommonLigatures(static_cast<unsigned>(FontVariantLigatures::Normal))
48     , m_variantDiscretionaryLigatures(static_cast<unsigned>(FontVariantLigatures::Normal))
49     , m_variantHistoricalLigatures(static_cast<unsigned>(FontVariantLigatures::Normal))
50     , m_variantContextualAlternates(static_cast<unsigned>(FontVariantLigatures::Normal))
51     , m_variantPosition(static_cast<unsigned>(FontVariantPosition::Normal))
52     , m_variantCaps(static_cast<unsigned>(FontVariantCaps::Normal))
53     , m_variantNumericFigure(static_cast<unsigned>(FontVariantNumericFigure::Normal))
54     , m_variantNumericSpacing(static_cast<unsigned>(FontVariantNumericSpacing::Normal))
55     , m_variantNumericFraction(static_cast<unsigned>(FontVariantNumericFraction::Normal))
56     , m_variantNumericOrdinal(static_cast<unsigned>(FontVariantNumericOrdinal::Normal))
57     , m_variantNumericSlashedZero(static_cast<unsigned>(FontVariantNumericSlashedZero::Normal))
58     , m_variantAlternates(static_cast<unsigned>(FontVariantAlternates::Normal))
59     , m_variantEastAsianVariant(static_cast<unsigned>(FontVariantEastAsianVariant::Normal))
60     , m_variantEastAsianWidth(static_cast<unsigned>(FontVariantEastAsianWidth::Normal))
61     , m_variantEastAsianRuby(static_cast<unsigned>(FontVariantEastAsianRuby::Normal))
62     , m_opticalSizing(static_cast<unsigned>(FontOpticalSizing::Enabled))
63     , m_fontStyleAxis(FontCascadeDescription::initialFontStyleAxis() == FontStyleAxis::ital)
64     , m_shouldAllowUserInstalledFonts(static_cast<unsigned>(AllowUserInstalledFonts::No))
65     , m_shouldAllowDesignSystemUIFonts(false)
66 {
67 }

69 void FontDescription::setLocale(const AtomString& locale)
70 {
71     m_locale = locale;
72     m_script = localeToScriptCodeForFontSelection(m_locale);
73 }
