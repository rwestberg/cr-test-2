181     virtual void visit(AST::VariableReference&);
182 
183     bool hasError() const { return !m_expectedError; }
184     Expected<void, Error> result() { return m_expectedError; }
185 
186     template<typename T> void checkErrorAndVisit(T& x)
187     {
188         if (!hasError())
189             visit(x);
190     }
191 
192 protected:
193     void setError(Error error)
194     {
195         ASSERT(!hasError());
196         m_expectedError = makeUnexpected(error);
197     }
198 
199 private:
200     Expected<void, Error> m_expectedError;
201 };
