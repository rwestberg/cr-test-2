29 #pragma once
30 
31 #include "DOMStringList.h"
32 #include "DOMWindowProperty.h"
33 #include "ExceptionOr.h"
34 #include "ScriptWrappable.h"
35 
36 namespace WebCore {
37 
38 class DOMWindow;
39 
40 class Location : public ScriptWrappable, public RefCounted<Location>, public DOMWindowProperty {

41 public:
42     static Ref<Location> create(DOMWindow& window) { return adoptRef(*new Location(window)); }
43 
44     ExceptionOr<void> setHref(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
45     String href() const;
46 
47     ExceptionOr<void> assign(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
48     void replace(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
49     void reload(DOMWindow& activeWindow);
50 
51     ExceptionOr<void> setProtocol(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
52     String protocol() const;
53     ExceptionOr<void> setHost(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
54     String host() const;
55     ExceptionOr<void> setHostname(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
56     String hostname() const;
57     ExceptionOr<void> setPort(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
58     String port() const;
59     ExceptionOr<void> setPathname(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
60     String pathname() const;
61     ExceptionOr<void> setSearch(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
62     String search() const;
63     ExceptionOr<void> setHash(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
64     String hash() const;
65     String origin() const;
66 
67     String toString() const { return href(); }
68 
69     Ref<DOMStringList> ancestorOrigins() const;
70 
71 private:
72     explicit Location(DOMWindow&);
73 
74     ExceptionOr<void> setLocation(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
75 
76     const URL& url() const;
77 };
78 
79 } // namespace WebCore
