 62 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<SVGPathSeg>&& object)
 63 {
 64     switch (object->pathSegType()) {
 65     case SVGPathSeg::PATHSEG_CLOSEPATH:
 66         return createWrapper<SVGPathSegClosePath>(globalObject, WTFMove(object));
 67     case SVGPathSeg::PATHSEG_MOVETO_ABS:
 68         return createWrapper<SVGPathSegMovetoAbs>(globalObject, WTFMove(object));
 69     case SVGPathSeg::PATHSEG_MOVETO_REL:
 70         return createWrapper<SVGPathSegMovetoRel>(globalObject, WTFMove(object));
 71     case SVGPathSeg::PATHSEG_LINETO_ABS:
 72         return createWrapper<SVGPathSegLinetoAbs>(globalObject, WTFMove(object));
 73     case SVGPathSeg::PATHSEG_LINETO_REL:
 74         return createWrapper<SVGPathSegLinetoRel>(globalObject, WTFMove(object));
 75     case SVGPathSeg::PATHSEG_CURVETO_CUBIC_ABS:
 76         return createWrapper<SVGPathSegCurvetoCubicAbs>(globalObject, WTFMove(object));
 77     case SVGPathSeg::PATHSEG_CURVETO_CUBIC_REL:
 78         return createWrapper<SVGPathSegCurvetoCubicRel>(globalObject, WTFMove(object));
 79     case SVGPathSeg::PATHSEG_CURVETO_QUADRATIC_ABS:
 80         return createWrapper<SVGPathSegCurvetoQuadraticAbs>(globalObject, WTFMove(object));
 81     case SVGPathSeg::PATHSEG_CURVETO_QUADRATIC_REL:
 82         return createWrapper<SVGPathSegCurvetoQuadraticRel>(globalObject, WTFMove(object));
 83     case SVGPathSeg::PATHSEG_ARC_ABS:
 84         return createWrapper<SVGPathSegArcAbs>(globalObject, WTFMove(object));
 85     case SVGPathSeg::PATHSEG_ARC_REL:
 86         return createWrapper<SVGPathSegArcRel>(globalObject, WTFMove(object));
 87     case SVGPathSeg::PATHSEG_LINETO_HORIZONTAL_ABS:
 88         return createWrapper<SVGPathSegLinetoHorizontalAbs>(globalObject, WTFMove(object));
 89     case SVGPathSeg::PATHSEG_LINETO_HORIZONTAL_REL:
 90         return createWrapper<SVGPathSegLinetoHorizontalRel>(globalObject, WTFMove(object));
 91     case SVGPathSeg::PATHSEG_LINETO_VERTICAL_ABS:
 92         return createWrapper<SVGPathSegLinetoVerticalAbs>(globalObject, WTFMove(object));
 93     case SVGPathSeg::PATHSEG_LINETO_VERTICAL_REL:
 94         return createWrapper<SVGPathSegLinetoVerticalRel>(globalObject, WTFMove(object));
 95     case SVGPathSeg::PATHSEG_CURVETO_CUBIC_SMOOTH_ABS:
 96         return createWrapper<SVGPathSegCurvetoCubicSmoothAbs>(globalObject, WTFMove(object));
 97     case SVGPathSeg::PATHSEG_CURVETO_CUBIC_SMOOTH_REL:
 98         return createWrapper<SVGPathSegCurvetoCubicSmoothRel>(globalObject, WTFMove(object));
 99     case SVGPathSeg::PATHSEG_CURVETO_QUADRATIC_SMOOTH_ABS:
100         return createWrapper<SVGPathSegCurvetoQuadraticSmoothAbs>(globalObject, WTFMove(object));
101     case SVGPathSeg::PATHSEG_CURVETO_QUADRATIC_SMOOTH_REL:
102         return createWrapper<SVGPathSegCurvetoQuadraticSmoothRel>(globalObject, WTFMove(object));
103     case SVGPathSeg::PATHSEG_UNKNOWN:
104     default:
105         return createWrapper<SVGPathSeg>(globalObject, WTFMove(object));
106     }
107 }
108 
109 JSValue toJS(ExecState* state, JSDOMGlobalObject* globalObject, SVGPathSeg& object)
110 {
111     return wrap(state, globalObject, object);
112 }
