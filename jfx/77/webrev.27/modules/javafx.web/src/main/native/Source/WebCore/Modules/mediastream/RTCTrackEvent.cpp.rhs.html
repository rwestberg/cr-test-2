42 Ref<RTCTrackEvent> RTCTrackEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<RTCRtpReceiver>&& receiver, RefPtr<MediaStreamTrack>&& track, Vector<RefPtr<MediaStream>>&& streams, RefPtr<RTCRtpTransceiver>&& transceiver)
43 {
44     return adoptRef(*new RTCTrackEvent(type, canBubble, cancelable, WTFMove(receiver), WTFMove(track), WTFMove(streams), WTFMove(transceiver)));
45 }
46 
47 Ref<RTCTrackEvent> RTCTrackEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
48 {
49     return adoptRef(*new RTCTrackEvent(type, initializer, isTrusted));
50 }
51 
52 RTCTrackEvent::RTCTrackEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<RTCRtpReceiver>&& receiver, RefPtr<MediaStreamTrack>&& track, Vector<RefPtr<MediaStream>>&& streams, RefPtr<RTCRtpTransceiver>&& transceiver)
53     : Event(type, canBubble, cancelable)
54     , m_receiver(WTFMove(receiver))
55     , m_track(WTFMove(track))
56     , m_streams(WTFMove(streams))
57     , m_transceiver(WTFMove(transceiver))
58 {
59 }
60 
61 RTCTrackEvent::RTCTrackEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
62     : Event(type, initializer, isTrusted)
63     , m_receiver(initializer.receiver)
64     , m_track(initializer.track)
65     , m_streams(initializer.streams)
66     , m_transceiver(initializer.transceiver)
67 {
68 }
