33 LiveNodeList::LiveNodeList(ContainerNode& ownerNode, NodeListInvalidationType invalidationType)
34     : m_ownerNode(ownerNode)
35     , m_invalidationType(invalidationType)
36     , m_isRegisteredForInvalidationAtDocument(false)
37 {
38     ASSERT(m_invalidationType == static_cast<unsigned>(invalidationType));
39 }
40 
41 LiveNodeList::~LiveNodeList() = default;
42 
43 ContainerNode& LiveNodeList::rootNode() const
44 {
45     if (isRootedAtDocument() && ownerNode().isConnected())
46         return ownerNode().document();
47 
48     return ownerNode();
49 }
