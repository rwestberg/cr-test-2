27 
28 #include <wtf/Ref.h>
29 #include <wtf/RefCounted.h>
30 
31 namespace WebCore {
32 
33 class Node;
34 class Range;
35 
36 class StaticRange : public RefCounted<StaticRange> {
37 public:
38     ~StaticRange();
39 
40     static Ref<StaticRange> createFromRange(const Range&);
41     static Ref<StaticRange> create(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
42 
43     unsigned startOffset() const { return m_startOffset; }
44     unsigned endOffset() const { return m_endOffset; }
45     Node* startContainer() const;
46     Node* endContainer() const;
47     bool collapsed() const;
48 
49 private:
50     StaticRange(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
51 
52     Ref<Node> m_startContainer;
53     unsigned m_startOffset;
54     Ref<Node> m_endContainer;
55     unsigned m_endOffset;
56 };

58 }
