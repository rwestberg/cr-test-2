34 Ref<MediaStreamTrackEvent> MediaStreamTrackEvent::create(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<MediaStreamTrack>&& track)
35 {
36     return adoptRef(*new MediaStreamTrackEvent(type, canBubble, cancelable, WTFMove(track)));
37 }
38 
39 Ref<MediaStreamTrackEvent> MediaStreamTrackEvent::create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
40 {
41     return adoptRef(*new MediaStreamTrackEvent(type, initializer, isTrusted));
42 }
43 
44 MediaStreamTrackEvent::MediaStreamTrackEvent(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<MediaStreamTrack>&& track)
45     : Event(type, canBubble, cancelable)
46     , m_track(WTFMove(track))
47 {
48 }
49 
50 MediaStreamTrackEvent::MediaStreamTrackEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
51     : Event(type, initializer, isTrusted)
52     , m_track(initializer.track)
53 {
54 }
55 
56 MediaStreamTrackEvent::~MediaStreamTrackEvent() = default;
57 
58 MediaStreamTrack* MediaStreamTrackEvent::track() const
59 {
60     return m_track.get();
61 }
62 
63 EventInterface MediaStreamTrackEvent::eventInterface() const
64 {
65     return MediaStreamTrackEventInterfaceType;
66 }
