28 #if ENABLE(WEBASSEMBLY)
29 
30 namespace JSC {
31 
32 namespace Wasm {
33 
34 #define FOR_EACH_EXCEPTION(macro) \
35     macro(OutOfBoundsMemoryAccess,  "Out of bounds memory access") \

36     macro(OutOfBoundsCallIndirect, "Out of bounds call_indirect") \
37     macro(NullTableEntry,  "call_indirect to a null table entry") \
38     macro(BadSignature, "call_indirect to a signature that does not match") \
39     macro(OutOfBoundsTrunc, "Out of bounds Trunc operation") \
40     macro(Unreachable, "Unreachable code should not be executed") \
41     macro(DivisionByZero, "Division by zero") \
42     macro(IntegerOverflow, "Integer overflow") \
43     macro(StackOverflow, "Stack overflow") \
44     macro(I64ArgumentType, "WebAssembly function with an i64 argument can't be called from JavaScript") \
45     macro(I64ReturnType, "WebAssembly function that returns i64 can't be called from JavaScript")

46 
47 enum class ExceptionType : uint32_t {
48 #define MAKE_ENUM(enumName, error) enumName,
49     FOR_EACH_EXCEPTION(MAKE_ENUM)
50 #undef MAKE_ENUM
51 };
52 
53 ALWAYS_INLINE const char* errorMessageForExceptionType(ExceptionType type)
54 {
55     switch (type) {
56 #define SWITCH_CASE(enumName, error) \
57     case ExceptionType::enumName: return error;
58 
59     FOR_EACH_EXCEPTION(SWITCH_CASE)
60 #undef SWITCH_CASE
61     }
62     ASSERT_NOT_REACHED();
63     return "";
64 }
65 
66 } } // namespace JSC::Wasm
67 
68 #endif // ENABLE(WEBASSEMBLY)
