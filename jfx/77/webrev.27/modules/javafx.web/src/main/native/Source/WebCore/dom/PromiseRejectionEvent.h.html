26 #pragma once
27 
28 #include "Event.h"
29 #include "JSValueInWrappedObject.h"
30 
31 namespace WebCore {
32 
33 class DOMPromise;
34 
35 class PromiseRejectionEvent final : public Event {
36 public:
37     struct Init : EventInit {
38         RefPtr<DOMPromise> promise;
39         JSC::JSValue reason;
40     };
41 
42     static Ref<PromiseRejectionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
43     {
44         return adoptRef(*new PromiseRejectionEvent(type, initializer, isTrusted));
45     }
46 
47     virtual ~PromiseRejectionEvent();
48 
49     DOMPromise& promise() const { return m_promise.get(); }
50     const JSValueInWrappedObject& reason() const { return m_reason; }
51 
52     EventInterface eventInterface() const override { return PromiseRejectionEventInterfaceType; }
53 
54 private:
55     PromiseRejectionEvent(const AtomString&, const Init&, IsTrusted);
56 
57     Ref<DOMPromise> m_promise;
58     JSValueInWrappedObject m_reason;
59 };
60 
61 } // namespace WebCore
