21 #include "config.h"
22 #include "HTMLOptionsCollection.h"

24 namespace WebCore {
25 
26 HTMLOptionsCollection::HTMLOptionsCollection(HTMLSelectElement& select)
27     : CachedHTMLCollection<HTMLOptionsCollection, CollectionTypeTraits<SelectOptions>::traversalType>(select, SelectOptions)
28 {
29 }
30 
31 Ref<HTMLOptionsCollection> HTMLOptionsCollection::create(HTMLSelectElement& select, CollectionType)
32 {
33     return adoptRef(*new HTMLOptionsCollection(select));
34 }
35 
36 ExceptionOr<void> HTMLOptionsCollection::add(const OptionOrOptGroupElement& element, const Optional<HTMLElementOrInt>& before)
37 {
38     return selectElement().add(element, before);
39 }
40 
41 void HTMLOptionsCollection::remove(int index)
42 {
43     selectElement().remove(index);
44 }
45 
46 int HTMLOptionsCollection::selectedIndex() const
47 {
48     return selectElement().selectedIndex();
49 }
50 
51 void HTMLOptionsCollection::setSelectedIndex(int index)
52 {
53     selectElement().setSelectedIndex(index);
54 }
55 
56 ExceptionOr<void> HTMLOptionsCollection::setLength(unsigned length)
57 {
58     return selectElement().setLength(length);
59 }
