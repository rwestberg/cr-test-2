31 #include "Element.h"
32 #include "NodeList.h"
33 #include <wtf/Vector.h>
34 
35 namespace WebCore {
36 
37 class WEBCORE_EXPORT StaticNodeList final : public NodeList {
38     WTF_MAKE_ISO_ALLOCATED(StaticNodeList);
39 public:
40     static Ref<StaticNodeList> create(Vector<Ref<Node>>&& nodes = { })
41     {
42         return adoptRef(*new StaticNodeList(WTFMove(nodes)));
43     }
44 
45     unsigned length() const override;
46     Node* item(unsigned index) const override;
47 
48 private:
49     StaticNodeList(Vector<Ref<Node>>&& nodes)
50         : m_nodes(WTFMove(nodes))
51     { }
52 
53     Vector<Ref<Node>> m_nodes;
54 };
55 
56 class StaticElementList final : public NodeList {
57     WTF_MAKE_ISO_ALLOCATED(StaticElementList);
58 public:
59     static Ref<StaticElementList> create(Vector<Ref<Element>>&& elements = { })
60     {
61         return adoptRef(*new StaticElementList(WTFMove(elements)));
62     }
63 
64     unsigned length() const override;
65     Element* item(unsigned index) const override;
66 
67 private:
68     StaticElementList(Vector<Ref<Element>>&& elements)
69         : m_elements(WTFMove(elements))
70     { }
71 
72     Vector<Ref<Element>> m_elements;
73 };

75 } // namespace WebCore
