25 #include "config.h"
26 #include "VREyeParameters.h"
27 
28 #include "VRFieldOfView.h"

29 
30 namespace WebCore {
31 
32 VREyeParameters::VREyeParameters(const FloatPoint3D& offset, const VRPlatformDisplayInfo::FieldOfView& fieldOfView, const RenderSize& renderSize)
33     : m_fieldOfView(VRFieldOfView::create(fieldOfView))
34     , m_offset(offset)
35     , m_renderSize(renderSize)
36 {
37 }
38 
39 Ref<Float32Array> VREyeParameters::offset() const
40 {
41     auto offset = Float32Array::create(3);
42     float* offsetData = offset->data();
43     offsetData[0] = m_offset.x();
44     offsetData[1] = m_offset.y();
45     offsetData[2] = m_offset.z();
46     return offset;
47 }
48 
49 const VRFieldOfView& VREyeParameters::fieldOfView() const
50 {
51     return m_fieldOfView;
52 }
53 
54 unsigned VREyeParameters::renderWidth() const
55 {
56     return m_renderSize.width;
57 }
58 
59 unsigned VREyeParameters::renderHeight() const
60 {
61     return m_renderSize.height;
62 }
