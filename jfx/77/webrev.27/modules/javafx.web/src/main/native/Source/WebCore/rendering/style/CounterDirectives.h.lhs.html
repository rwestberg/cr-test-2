21 #pragma once
22 
23 #include <wtf/HashMap.h>
24 #include <wtf/text/AtomicStringHash.h>
25 
26 namespace WebCore {
27 
28 struct CounterDirectives {
29     Optional<int> resetValue;
30     Optional<int> incrementValue;
31 };
32 
33 constexpr bool operator==(const CounterDirectives& a, const CounterDirectives& b)
34 {
35     return a.incrementValue == b.incrementValue && a.resetValue == b.resetValue;
36 }
37 
38 constexpr bool operator!=(const CounterDirectives& a, const CounterDirectives& b)
39 {
40     return !(a == b);
41 }
42 
43 using CounterDirectiveMap = HashMap<AtomicString, CounterDirectives>;
