26 #pragma once
27 
28 #include "CSSPropertyNames.h"
29 #include "DeclarativeAnimation.h"
30 #include <wtf/Ref.h>
31 
32 namespace WebCore {
33 
34 class Animation;
35 class Element;
36 class RenderStyle;
37 
38 class CSSTransition final : public DeclarativeAnimation {
39     WTF_MAKE_ISO_ALLOCATED(CSSTransition);
40 public:
41     static Ref<CSSTransition> create(Element&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle* oldStyle, const RenderStyle& newStyle, Seconds delay, Seconds duration, const RenderStyle& reversingAdjustedStartStyle, double);
42     ~CSSTransition() = default;
43 
44     bool isCSSTransition() const override { return true; }
45     String transitionProperty() const { return getPropertyNameString(m_property); }
46     CSSPropertyID property() const { return m_property; }
47     MonotonicTime generationTime() const { return m_generationTime; }
48     const RenderStyle& targetStyle() const { return *m_targetStyle; }
49     const RenderStyle& currentStyle() const { return *m_currentStyle; }
50     const RenderStyle& reversingAdjustedStartStyle() const { return *m_reversingAdjustedStartStyle; }
51     double reversingShorteningFactor() const { return m_reversingShorteningFactor; }
52 
53     void resolve(RenderStyle&) final;
54 
55 private:
56     CSSTransition(Element&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle& targetStyle, const RenderStyle& reversingAdjustedStartStyle, double);
57     void setTimingProperties(Seconds delay, Seconds duration);
58 
59     CSSPropertyID m_property;
60     MonotonicTime m_generationTime;
61     std::unique_ptr<RenderStyle> m_targetStyle;
62     std::unique_ptr<RenderStyle> m_currentStyle;
63     std::unique_ptr<RenderStyle> m_reversingAdjustedStartStyle;
64     double m_reversingShorteningFactor;
65 
66 };
67 
68 } // namespace WebCore
69
