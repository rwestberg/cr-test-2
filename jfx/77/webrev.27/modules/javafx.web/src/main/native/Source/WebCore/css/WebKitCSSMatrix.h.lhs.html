 26 #pragma once
 27 
 28 #include "ExceptionOr.h"
 29 #include "ScriptWrappable.h"
 30 #include "TransformationMatrix.h"
 31 
 32 namespace WebCore {
 33 
 34 // FIXME: Remove this class once WebKitCSSMatrix becomes an alias to DOMMatrix.
 35 class WebKitCSSMatrix final : public ScriptWrappable, public RefCounted<WebKitCSSMatrix> {

 36 public:
 37     static Ref<WebKitCSSMatrix> create(const TransformationMatrix&);
 38     static ExceptionOr<Ref<WebKitCSSMatrix>> create(const String&);
 39 
 40     ~WebKitCSSMatrix();
 41 
 42     double a() const { return m_matrix.a(); }
 43     double b() const { return m_matrix.b(); }
 44     double c() const { return m_matrix.c(); }
 45     double d() const { return m_matrix.d(); }
 46     double e() const { return m_matrix.e(); }
 47     double f() const { return m_matrix.f(); }
 48 
 49     void setA(double f) { m_matrix.setA(f); }
 50     void setB(double f) { m_matrix.setB(f); }
 51     void setC(double f) { m_matrix.setC(f); }
 52     void setD(double f) { m_matrix.setD(f); }
 53     void setE(double f) { m_matrix.setE(f); }
 54     void setF(double f) { m_matrix.setF(f); }
 55 
 56     double m11() const { return m_matrix.m11(); }
 57     double m12() const { return m_matrix.m12(); }
 58     double m13() const { return m_matrix.m13(); }
 59     double m14() const { return m_matrix.m14(); }
 60     double m21() const { return m_matrix.m21(); }
 61     double m22() const { return m_matrix.m22(); }
 62     double m23() const { return m_matrix.m23(); }
 63     double m24() const { return m_matrix.m24(); }
 64     double m31() const { return m_matrix.m31(); }
 65     double m32() const { return m_matrix.m32(); }
 66     double m33() const { return m_matrix.m33(); }
 67     double m34() const { return m_matrix.m34(); }
 68     double m41() const { return m_matrix.m41(); }
 69     double m42() const { return m_matrix.m42(); }
 70     double m43() const { return m_matrix.m43(); }
 71     double m44() const { return m_matrix.m44(); }
 72 
 73     void setM11(double f) { m_matrix.setM11(f); }
 74     void setM12(double f) { m_matrix.setM12(f); }
 75     void setM13(double f) { m_matrix.setM13(f); }
 76     void setM14(double f) { m_matrix.setM14(f); }
 77     void setM21(double f) { m_matrix.setM21(f); }
 78     void setM22(double f) { m_matrix.setM22(f); }
 79     void setM23(double f) { m_matrix.setM23(f); }
 80     void setM24(double f) { m_matrix.setM24(f); }
 81     void setM31(double f) { m_matrix.setM31(f); }
 82     void setM32(double f) { m_matrix.setM32(f); }
 83     void setM33(double f) { m_matrix.setM33(f); }
 84     void setM34(double f) { m_matrix.setM34(f); }
 85     void setM41(double f) { m_matrix.setM41(f); }
 86     void setM42(double f) { m_matrix.setM42(f); }
 87     void setM43(double f) { m_matrix.setM43(f); }
 88     void setM44(double f) { m_matrix.setM44(f); }
 89 
 90     ExceptionOr<void> setMatrixValue(const String&);
 91 
 92     // The following math function return a new matrix with the
 93     // specified operation applied. The this value is not modified.
 94 
 95     // Multiply this matrix by secondMatrix, on the right (result = this * secondMatrix)
 96     RefPtr<WebKitCSSMatrix> multiply(WebKitCSSMatrix* secondMatrix) const;
 97 
 98     // Return the inverse of this matrix. Throw an exception if the matrix is not invertible
 99     ExceptionOr<Ref<WebKitCSSMatrix>> inverse() const;
100 
101     // Return this matrix translated by the passed values.
102     // Passing a NaN will use a value of 0. This allows the 3D form to used for 2D operations
103     // Operation is performed as though the this matrix is multiplied by a matrix with
104     // the translation values on the left (result = translation(x,y,z) * this)
105     Ref<WebKitCSSMatrix> translate(double x, double y, double z) const;
106 
107     // Returns this matrix scaled by the passed values.
108     // Passing scaleX or scaleZ as NaN uses a value of 1, but passing scaleY of NaN
109     // makes it the same as scaleX. This allows the 3D form to used for 2D operations
110     // Operation is performed as though the this matrix is multiplied by a matrix with
111     // the scale values on the left (result = scale(x,y,z) * this)
112     Ref<WebKitCSSMatrix> scale(double scaleX, double scaleY, double scaleZ) const;
113 
114     // Returns this matrix rotated by the passed values.
115     // If rotY and rotZ are NaN, rotate about Z (rotX=0, rotateY=0, rotateZ=rotX).
116     // Otherwise use a rotation value of 0 for any passed NaN.
117     // Operation is performed as though the this matrix is multiplied by a matrix with
118     // the rotation values on the left (result = rotation(x,y,z) * this)
119     Ref<WebKitCSSMatrix> rotate(double rotX, double rotY, double rotZ) const;
120 
121     // Returns this matrix rotated about the passed axis by the passed angle.
122     // Passing a NaN will use a value of 0. If the axis is (0,0,0) use a value
123     // Operation is performed as though the this matrix is multiplied by a matrix with
124     // the rotation values on the left (result = rotation(x,y,z,angle) * this)
125     Ref<WebKitCSSMatrix> rotateAxisAngle(double x, double y, double z, double angle) const;
126 
127     // Return this matrix skewed along the X axis by the passed values.
128     // Passing a NaN will use a value of 0.
129     // Operation is performed as though the this matrix is multiplied by a matrix with
130     // the skew values on the left (result = skewX(angle) * this)
131     Ref<WebKitCSSMatrix> skewX(double angle) const;
132 
133     // Return this matrix skewed along the Y axis by the passed values.
134     // Passing a NaN will use a value of 0.
135     // Operation is performed as though the this matrix is multiplied by a matrix with
136     // the skew values on the left (result = skewY(angle) * this)
137     Ref<WebKitCSSMatrix> skewY(double angle) const;
138 
139     const TransformationMatrix& transform() const { return m_matrix; }
140 
141     ExceptionOr<String> toString() const;
142 
143 private:
144     WebKitCSSMatrix() = default;
145     WebKitCSSMatrix(const TransformationMatrix&);
146 
147     TransformationMatrix m_matrix;
148 };
149 
150 } // namespace WebCore
