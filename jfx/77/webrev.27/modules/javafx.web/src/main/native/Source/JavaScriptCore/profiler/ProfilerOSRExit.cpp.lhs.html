35 OSRExit::OSRExit(unsigned id, const OriginStack& origin, ExitKind kind, bool isWatchpoint)
36     : m_id(id)
37     , m_origin(origin)
38     , m_exitKind(kind)
39     , m_isWatchpoint(isWatchpoint)
40     , m_counter(0)
41 {
42 }
43 
44 OSRExit::~OSRExit()
45 {
46 }
47 
48 JSValue OSRExit::toJS(ExecState* exec) const
49 {
50     VM& vm = exec->vm();
51     JSObject* result = constructEmptyObject(exec);
52     result->putDirect(vm, vm.propertyNames->id, jsNumber(m_id));
53     result->putDirect(vm, vm.propertyNames->origin, m_origin.toJS(exec));
54     result->putDirect(vm, vm.propertyNames->exitKind, jsString(exec, exitKindToString(m_exitKind)));
55     result->putDirect(vm, vm.propertyNames->isWatchpoint, jsBoolean(m_isWatchpoint));
56     result->putDirect(vm, vm.propertyNames->count, jsNumber(m_counter));
57     return result;
58 }
