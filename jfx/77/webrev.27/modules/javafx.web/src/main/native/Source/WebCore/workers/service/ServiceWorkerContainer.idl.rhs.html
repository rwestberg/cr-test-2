26 // FIXME: The spec IDL exposes this on Worker in addition to Window.
27 // We don't currently support nested workers.
28 
29 [
30     SecureContext,
31     Exposed=(Window,ServiceWorker),
32     Conditional=SERVICE_WORKER,
33     EnabledAtRuntime=ServiceWorker,
34     GenerateIsReachable=ReachableFromNavigator
35 ] interface ServiceWorkerContainer : EventTarget {
36     readonly attribute ServiceWorker? controller;
37     readonly attribute Promise<ServiceWorkerRegistration> ready;
38 
39     [NewObject, ImplementedAs=addRegistration] Promise<ServiceWorkerRegistration> register(USVString scriptURL, optional RegistrationOptions options);
40     [NewObject] Promise<any> getRegistration(optional USVString clientURL = "");
41     [NewObject] Promise<sequence<ServiceWorkerRegistration>> getRegistrations();
42 
43     void startMessages();
44 
45     // events
46     attribute EventHandler oncontrollerchange;
47     attribute EventHandler onmessage;
48     attribute EventHandler onmessageerror;
49 };
50 
51 dictionary RegistrationOptions {
52     USVString scope;
53     WorkerType type = "classic";
54     ServiceWorkerUpdateViaCache updateViaCache = "imports";
55 };
