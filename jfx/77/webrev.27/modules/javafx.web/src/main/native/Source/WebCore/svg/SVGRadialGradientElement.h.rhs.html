22 #pragma once
23 

24 #include "SVGGradientElement.h"
25 #include "SVGNames.h"
26 
27 namespace WebCore {
28 
29 struct RadialGradientAttributes;
30 
31 class SVGRadialGradientElement final : public SVGGradientElement {
32     WTF_MAKE_ISO_ALLOCATED(SVGRadialGradientElement);
33 public:
34     static Ref<SVGRadialGradientElement> create(const QualifiedName&, Document&);
35 
36     bool collectGradientAttributes(RadialGradientAttributes&);
37 
38     const SVGLengthValue& cx() const { return m_cx->currentValue(); }
39     const SVGLengthValue& cy() const { return m_cy->currentValue(); }
40     const SVGLengthValue& r() const { return m_r->currentValue(); }
41     const SVGLengthValue& fx() const { return m_fx->currentValue(); }
42     const SVGLengthValue& fy() const { return m_fy->currentValue(); }
43     const SVGLengthValue& fr() const { return m_fr->currentValue(); }
44 
45     SVGAnimatedLength& cxAnimated() { return m_cx; }
46     SVGAnimatedLength& cyAnimated() { return m_cy; }
47     SVGAnimatedLength& rAnimated() { return m_r; }
48     SVGAnimatedLength& fxAnimated() { return m_fx; }
49     SVGAnimatedLength& fyAnimated() { return m_fy; }
50     SVGAnimatedLength& frAnimated() { return m_fr; }
51 
52 private:
53     SVGRadialGradientElement(const QualifiedName&, Document&);
54 
55     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGRadialGradientElement, SVGGradientElement>;
56     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }

57 
58     void parseAttribute(const QualifiedName&, const AtomString&) override;

59     void svgAttributeChanged(const QualifiedName&) override;
60 
61     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;
62 
63     bool selfHasRelativeLengths() const override;
64 
65     PropertyRegistry m_propertyRegistry { *this };
66     Ref<SVGAnimatedLength> m_cx { SVGAnimatedLength::create(this, LengthModeWidth, "50%") };
67     Ref<SVGAnimatedLength> m_cy { SVGAnimatedLength::create(this, LengthModeHeight, "50%") };
68     Ref<SVGAnimatedLength> m_r { SVGAnimatedLength::create(this, LengthModeOther, "50%") };
69     Ref<SVGAnimatedLength> m_fx { SVGAnimatedLength::create(this, LengthModeWidth) };
70     Ref<SVGAnimatedLength> m_fy { SVGAnimatedLength::create(this, LengthModeHeight) };
71     Ref<SVGAnimatedLength> m_fr { SVGAnimatedLength::create(this, LengthModeOther, "0%") };
72 };
73 
74 } // namespace WebCore
