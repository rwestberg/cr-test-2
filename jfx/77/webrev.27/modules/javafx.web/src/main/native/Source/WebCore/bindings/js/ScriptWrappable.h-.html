36 namespace JSC {
37 class WeakHandleOwner;

40 namespace WebCore {
41 
42 class JSDOMObject;
43 
44 class ScriptWrappable {
45 public:
46     JSDOMObject* wrapper() const;
47     void setWrapper(JSDOMObject*, JSC::WeakHandleOwner*, void*);
48     void clearWrapper(JSDOMObject*);
49 
50     template<typename Derived>
51     static ptrdiff_t offsetOfWrapper() { return CAST_OFFSET(Derived*, ScriptWrappable*) + OBJECT_OFFSETOF(ScriptWrappable, m_wrapper); }
52 
53 protected:
54     ~ScriptWrappable() = default;
55 
56 private:
57     JSC::Weak<JSDOMObject> m_wrapper;
58 };
