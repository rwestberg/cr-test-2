28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class WebGLContextEvent final : public Event {
33 public:
34     static Ref<WebGLContextEvent> create(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, const String& statusMessage)
35     {
36         return adoptRef(*new WebGLContextEvent(type, canBubble, cancelable, statusMessage));
37     }
38 
39     struct Init : EventInit {
40         String statusMessage;
41     };
42 
43     static Ref<WebGLContextEvent> create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
44     {
45         return adoptRef(*new WebGLContextEvent(type, initializer, isTrusted));
46     }
47     virtual ~WebGLContextEvent();
48 
49     const String& statusMessage() const { return m_statusMessage; }
50 
51     EventInterface eventInterface() const override;
52 
53 private:
54     WebGLContextEvent(const AtomicString& type, CanBubble, IsCancelable, const String& statusMessage);
55     WebGLContextEvent(const AtomicString&, const Init&, IsTrusted);
56 
57     String m_statusMessage;
58 };
59 
60 } // namespace WebCore
