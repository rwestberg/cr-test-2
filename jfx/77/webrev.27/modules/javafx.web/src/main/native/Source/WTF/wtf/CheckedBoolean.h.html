26 #pragma once
27 
28 #include <wtf/Assertions.h>
29 
30 class CheckedBoolean {
31     WTF_MAKE_FAST_ALLOCATED;
32 public:
33 #if !ASSERT_DISABLED
34     CheckedBoolean(const CheckedBoolean& other)
35         : m_value(other.m_value)
36         , m_checked(false)
37     {
38         other.m_checked = true;
39     }
40 #endif
41 
42     CheckedBoolean(bool value)
43         : m_value(value)
44 #if !ASSERT_DISABLED
45         , m_checked(false)
46 #endif
47     {
48     }
49 
50     ~CheckedBoolean()
51     {
52         ASSERT(m_checked);
53     }
54 
55     operator bool()
56     {
57 #if !ASSERT_DISABLED
58         m_checked = true;
59 #endif
60         return m_value;
61     }
62 
63 private:
64     bool m_value;
65 #if !ASSERT_DISABLED
66     mutable bool m_checked;
67 #endif
68 };
