33 namespace WebCore {

35 class TextDecoder : public RefCounted<TextDecoder> {
36 public:
37     struct Options {
38         bool fatal { false };
39         bool ignoreBOM { false };
40     };
41     struct DecodeOptions {
42         bool stream { false };
43     };
44 
45     static ExceptionOr<Ref<TextDecoder>> create(const String& label, Options);
46 
47     String encoding() const;
48     bool fatal() const { return m_options.fatal; }
49     bool ignoreBOM() const { return m_options.ignoreBOM; }
50     ExceptionOr<String> decode(Optional<BufferSource::VariantType>, DecodeOptions);
51 
52 private:
53     String prependBOMIfNecessary(const String&);
54     void ignoreBOMIfNecessary(const uint8_t*& data, size_t& length);
55     TextDecoder(const char*, Options);
56     TextEncoding m_textEncoding;
57     Options m_options;
58     bool m_hasDecoded { false };
59     Vector<uint8_t> m_buffer;
60 };
