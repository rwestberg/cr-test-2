33 AutocapitalizeType autocapitalizeTypeForAttributeValue(const AtomicString& attributeValue)
34 {
35     // Omitted / missing values are the Default state.
36     if (attributeValue.isEmpty())
37         return AutocapitalizeTypeDefault;
38 
39     if (equalLettersIgnoringASCIICase(attributeValue, "on") || equalLettersIgnoringASCIICase(attributeValue, "sentences"))
40         return AutocapitalizeTypeSentences;
41     if (equalLettersIgnoringASCIICase(attributeValue, "off") || equalLettersIgnoringASCIICase(attributeValue, "none"))
42         return AutocapitalizeTypeNone;
43     if (equalLettersIgnoringASCIICase(attributeValue, "words"))
44         return AutocapitalizeTypeWords;
45     if (equalLettersIgnoringASCIICase(attributeValue, "characters"))
46         return AutocapitalizeTypeAllCharacters;
47 
48     // Unrecognized values fall back to "on".
49     return AutocapitalizeTypeSentences;
50 }
51 
52 const AtomicString& stringForAutocapitalizeType(AutocapitalizeType type)
53 {
54     switch (type) {
55     case AutocapitalizeTypeDefault:
56         return nullAtom();
57     case AutocapitalizeTypeNone: {
58         static NeverDestroyed<const AtomicString> valueNone("none", AtomicString::ConstructFromLiteral);
59         return valueNone;
60     }
61     case AutocapitalizeTypeSentences: {
62         static NeverDestroyed<const AtomicString> valueSentences("sentences", AtomicString::ConstructFromLiteral);
63         return valueSentences;
64     }
65     case AutocapitalizeTypeWords: {
66         static NeverDestroyed<const AtomicString> valueWords("words", AtomicString::ConstructFromLiteral);
67         return valueWords;
68     }
69     case AutocapitalizeTypeAllCharacters: {
70         static NeverDestroyed<const AtomicString> valueAllCharacters("characters", AtomicString::ConstructFromLiteral);
71         return valueAllCharacters;
72     }
73     }
74 
75     ASSERT_NOT_REACHED();
76     return nullAtom();
77 }
