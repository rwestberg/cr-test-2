44         Right,
45         Left,
46         Bottom,
47     };

51     WEBCORE_EXPORT virtual void windowObjectCleared() = 0;

54     virtual void startWindowDrag() = 0;

69     WEBCORE_EXPORT virtual void requestSetDockSide(DockSide) = 0;

73     WEBCORE_EXPORT virtual void openInNewTab(const String& url) = 0;
74 
75     virtual bool canSave() = 0;

79     virtual void inspectedURLChanged(const String&) = 0;

82     virtual void pagePaused() { }

85     WEBCORE_EXPORT virtual void sendMessageToBackend(const String&) = 0;

87     WEBCORE_EXPORT virtual bool isUnderTest() = 0;
88 };
