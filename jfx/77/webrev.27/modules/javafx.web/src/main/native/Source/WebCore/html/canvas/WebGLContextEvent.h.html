34 class WebGLContextEvent final : public Event {
35 public:
36     static Ref<WebGLContextEvent> create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, const String& statusMessage)
37     {
38         return adoptRef(*new WebGLContextEvent(type, canBubble, cancelable, statusMessage));
39     }
40 
41     struct Init : EventInit {
42         String statusMessage;
43     };
44 
45     static Ref<WebGLContextEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
46     {
47         return adoptRef(*new WebGLContextEvent(type, initializer, isTrusted));
48     }
49     virtual ~WebGLContextEvent();
50 
51     const String& statusMessage() const { return m_statusMessage; }
52 
53     EventInterface eventInterface() const override;
54 
55 private:
56     WebGLContextEvent(const AtomString& type, CanBubble, IsCancelable, const String& statusMessage);
57     WebGLContextEvent(const AtomString&, const Init&, IsTrusted);
58 
59     String m_statusMessage;
60 };
