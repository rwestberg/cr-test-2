31 #if ENABLE(MEDIA_STREAM)
32 
33 #include "Event.h"
34 #include "OverconstrainedError.h"
35 #include <wtf/text/AtomicString.h>
36 
37 namespace WebCore {
38 
39 class OverconstrainedErrorEvent : public Event {
40 public:
41     virtual ~OverconstrainedErrorEvent() = default;
42 
43     static Ref<OverconstrainedErrorEvent> create(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, OverconstrainedError* error)
44     {
45         return adoptRef(*new OverconstrainedErrorEvent(type, canBubble, cancelable, error));
46     }
47 
48     struct Init : EventInit {
49         RefPtr<OverconstrainedError> error;
50     };
51 
52     static Ref<OverconstrainedErrorEvent> create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
53     {
54         return adoptRef(*new OverconstrainedErrorEvent(type, initializer, isTrusted));
55     }
56 
57     OverconstrainedError* error() const { return m_error.get(); }
58     EventInterface eventInterface() const override { return OverconstrainedErrorEventInterfaceType; }
59 
60 private:
61     explicit OverconstrainedErrorEvent(const AtomicString& type, CanBubble canBubble, IsCancelable cancelable, OverconstrainedError* error)
62         : Event(type, canBubble, cancelable)
63         , m_error(error)
64     {
65     }
66     OverconstrainedErrorEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
67         : Event(type, initializer, isTrusted)
68         , m_error(initializer.error)
69     {
70     }
71 
72     RefPtr<OverconstrainedError> m_error;
73 };
74 
75 } // namespace WebCore
76 
77 #endif // ENABLE(MEDIA_STREAM)
