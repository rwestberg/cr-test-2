28 #include "DOMWindowProperty.h"
29 #include "ExceptionOr.h"
30 #include "ScriptWrappable.h"

32 namespace WebCore {
33 
34 class StorageArea;
35 
36 class Storage final : public ScriptWrappable, public RefCounted<Storage>, public DOMWindowProperty {
37     WTF_MAKE_ISO_ALLOCATED(Storage);
38 public:
39     static Ref<Storage> create(DOMWindow&, Ref<StorageArea>&&);
40     ~Storage();
41 
42     unsigned length() const;
43     String key(unsigned index) const;
44     String getItem(const String& key) const;
45     ExceptionOr<void> setItem(const String& key, const String& value);
46     ExceptionOr<void> removeItem(const String& key);
47     ExceptionOr<void> clear();
48     bool contains(const String& key) const;
49 
50     // Bindings support functions.
51     bool isSupportedPropertyName(const String&) const;
52     Vector<AtomString> supportedPropertyNames() const;
53 
54     StorageArea& area() const { return m_storageArea.get(); }
55 
56 private:
57     Storage(DOMWindow&, Ref<StorageArea>&&);
58 
59     const Ref<StorageArea> m_storageArea;
60 };

62 } // namespace WebCore
