24 #pragma once
25 
26 #include "LiveNodeList.h"
27 #include <wtf/Forward.h>
28 #include <wtf/text/AtomString.h>
29 
30 namespace WebCore {
31 
32 // NodeList which lists all Nodes in a Element with a given "name" attribute
33 class NameNodeList final : public CachedLiveNodeList<NameNodeList> {
34     WTF_MAKE_ISO_ALLOCATED(NameNodeList);
35 public:
36     static Ref<NameNodeList> create(ContainerNode& rootNode, const AtomString& name)
37     {
38         return adoptRef(*new NameNodeList(rootNode, name));
39     }
40 
41     virtual ~NameNodeList();
42 
43     bool elementMatches(Element&) const override;
44     bool isRootedAtDocument() const override { return false; }
45 
46 private:
47     NameNodeList(ContainerNode& rootNode, const AtomString& name);
48 
49     AtomString m_name;
50 };
51 
52 inline bool NameNodeList::elementMatches(Element& element) const
53 {
54     return element.getNameAttribute() == m_name;
55 }
56 
57 } // namespace WebCore
