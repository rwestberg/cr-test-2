31 #pragma once
32 
33 #include <array>
34 #include <wtf/Vector.h>
35 
36 #if PLATFORM(COCOA)
37 #include <CommonCrypto/CommonDigest.h>
38 #endif
39 
40 namespace WTF {
41 
42 class MD5 {
43     WTF_MAKE_FAST_ALLOCATED;
44 public:
45     WTF_EXPORT_PRIVATE MD5();
46 
47     void addBytes(const Vector<uint8_t>& input)
48     {
49         addBytes(input.data(), input.size());
50     }
51     WTF_EXPORT_PRIVATE void addBytes(const uint8_t* input, size_t length);
52 
53     // Size of the SHA1 hash
54     WTF_EXPORT_PRIVATE static const size_t hashSize = 16;
55 
56     // type for computing MD5 hash
57     typedef std::array<uint8_t, hashSize> Digest;
58 
59     // checksum has a side effect of resetting the state of the object.
60     WTF_EXPORT_PRIVATE void checksum(Digest&);
61 
62 private:
63 #if PLATFORM(COCOA)
64     CC_MD5_CTX m_context;
65 #else
66     uint32_t m_buf[4];
67     uint32_t m_bits[2];
68     uint8_t m_in[64];
69 #endif
70 };
71 
72 } // namespace WTF
73 
74 using WTF::MD5;
