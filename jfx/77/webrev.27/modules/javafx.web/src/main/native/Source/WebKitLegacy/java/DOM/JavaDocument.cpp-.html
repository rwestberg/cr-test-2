 274 JNIEXPORT jstring JNICALL Java_com_sun_webkit_dom_DocumentImpl_getReadyStateImpl(JNIEnv* env, jclass, jlong peer)
 275 {
 276     WebCore::JSMainThreadNullState state;
 277     auto readyState = IMPL->readyState();
 278     const char* readyStateStr { };
 279     switch (readyState) {
 280     case WebCore::Document::Loading:
 281         readyStateStr = "loading";
 282         break;
 283     case WebCore::Document::Interactive:
 284         readyStateStr = "interactive";
 285         break;
 286     case WebCore::Document::Complete:
 287         readyStateStr = "complete";
 288         break;
 289     default:
 290         ASSERT_NOT_REACHED();
 291     }
 292     return JavaReturn<String>(env, String(readyStateStr));
 293 }
