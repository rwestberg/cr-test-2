41 class EffectfulExpressionStatement : public Statement {
42 public:
43     EffectfulExpressionStatement(UniqueRef<Expression>&& effectfulExpression)
44         : Statement(Lexer::Token(effectfulExpression->origin()))
45         , m_effectfulExpression(WTFMove(effectfulExpression))
46     {
47     }
48 
49     virtual ~EffectfulExpressionStatement() = default;
50 
51     EffectfulExpressionStatement(const EffectfulExpressionStatement&) = delete;
52     EffectfulExpressionStatement(EffectfulExpressionStatement&&) = default;
53 
54     bool isEffectfulExpressionStatement() const override { return true; }
55 
56     Expression& effectfulExpression() { return m_effectfulExpression; }
57 
58 private:
59     UniqueRef<Expression> m_effectfulExpression;
60 };
