34 WTF_MAKE_ISO_ALLOCATED_IMPL(LiveNodeList);
35 
36 LiveNodeList::LiveNodeList(ContainerNode& ownerNode, NodeListInvalidationType invalidationType)
37     : m_ownerNode(ownerNode)
38     , m_invalidationType(invalidationType)
39     , m_isRegisteredForInvalidationAtDocument(false)
40 {
41     ASSERT(m_invalidationType == static_cast<unsigned>(invalidationType));
42 }
43 
44 LiveNodeList::~LiveNodeList() = default;
45 
46 ContainerNode& LiveNodeList::rootNode() const
47 {
48     if (isRootedAtDocument() && ownerNode().isConnected())
49         return ownerNode().document();
50 
51     return ownerNode();
52 }
