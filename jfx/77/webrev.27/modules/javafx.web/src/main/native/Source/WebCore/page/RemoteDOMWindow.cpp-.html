 35 RemoteDOMWindow::RemoteDOMWindow(Ref<RemoteFrame>&& frame, GlobalWindowIdentifier&& identifier)
 36     : AbstractDOMWindow(WTFMove(identifier))
 37     , m_frame(WTFMove(frame))
 38 {
 39     m_frame->setWindow(this);
 40 }
 41 
 42 RemoteDOMWindow::~RemoteDOMWindow()
 43 {
 44     if (m_frame)
 45         m_frame->setWindow(nullptr);
 46 }
 47 
 48 WindowProxy* RemoteDOMWindow::self() const
 49 {
 50     if (!m_frame)
 51         return nullptr;
 52     return &m_frame->windowProxy();
 53 }

 97 WindowProxy* RemoteDOMWindow::opener() const
 98 {
 99     if (!m_frame)
100         return nullptr;
101 
102     auto* openerFrame = m_frame->opener();
103     if (!openerFrame)
104         return nullptr;
105 
106     return &openerFrame->windowProxy();
107 }
