35 namespace WebCore {
36 
37 class TypedOMCSSUnitValue final : public TypedOMCSSNumericValue {
38     WTF_MAKE_ISO_ALLOCATED(TypedOMCSSUnitValue);
39 public:
40     static Ref<TypedOMCSSUnitValue> create(double value, const String& unit)
41     {
42         return adoptRef(*new TypedOMCSSUnitValue(value, unit));
43     }
44 
45     // FIXME: not correct.
46     String toString() final { return makeString((int) m_value, m_unit); }
47 
48     double value() const { return m_value; }
49     void setValue(double value) { m_value = value; }
50     const String& unit() const { return m_unit; }
51     void setUnit(const String& unit) { m_unit = unit; }
52 
53 private:
54     TypedOMCSSUnitValue(double value, const String& unit)
55         : m_value(value)
56         , m_unit(unit)
57     {
58     }
59 
60     bool isUnitValue() final { return true; }
61 
62     double m_value;
63     String m_unit;
64 };

66 } // namespace WebCore
