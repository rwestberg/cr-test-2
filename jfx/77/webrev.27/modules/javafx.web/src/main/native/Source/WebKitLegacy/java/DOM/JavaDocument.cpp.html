 275 JNIEXPORT jstring JNICALL Java_com_sun_webkit_dom_DocumentImpl_getReadyStateImpl(JNIEnv* env, jclass, jlong peer)
 276 {
 277     WebCore::JSMainThreadNullState state;
 278     auto readyState = IMPL->readyState();
 279     const char* readyStateStr { };
 280     switch (readyState) {
 281     case WebCore::Document::Loading:
 282         readyStateStr = "loading";
 283         break;
 284     case WebCore::Document::Interactive:
 285         readyStateStr = "interactive";
 286         break;
 287     case WebCore::Document::Complete:
 288         readyStateStr = "complete";
 289         break;
 290     default:
 291         ASSERT_NOT_REACHED();
 292     }
 293     return JavaReturn<String>(env, String(readyStateStr));
 294 }
