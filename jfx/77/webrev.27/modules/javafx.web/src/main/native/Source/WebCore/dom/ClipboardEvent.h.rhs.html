24 #pragma once
25 
26 #include "Event.h"
27 
28 namespace WebCore {
29 
30 class DataTransfer;
31 
32 class ClipboardEvent final : public Event {
33 public:
34     virtual ~ClipboardEvent();
35 
36     struct Init : EventInit {
37         RefPtr<DataTransfer> clipboardData;
38     };
39 
40     static Ref<ClipboardEvent> create(const AtomString& type, Ref<DataTransfer>&& dataTransfer)
41     {
42         return adoptRef(*new ClipboardEvent(type, WTFMove(dataTransfer)));
43     }
44 
45     static Ref<ClipboardEvent> create(const AtomString& type, const Init& init)
46     {
47         return adoptRef(*new ClipboardEvent(type, init));
48     }
49 
50     DataTransfer* clipboardData() const { return m_clipboardData.get(); }
51 
52 private:
53     ClipboardEvent(const AtomString& type, Ref<DataTransfer>&&);
54     ClipboardEvent(const AtomString& type, const Init&);
55 
56     EventInterface eventInterface() const final;
57     bool isClipboardEvent() const final;
58 
59     RefPtr<DataTransfer> m_clipboardData;
60 };
61 
62 } // namespace WebCore
