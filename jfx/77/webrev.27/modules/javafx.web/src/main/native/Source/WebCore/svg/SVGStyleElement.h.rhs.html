33     virtual ~SVGStyleElement();
34 
35     CSSStyleSheet* sheet() const { return m_styleSheetOwner.sheet(); }
36 
37     bool disabled() const;
38     void setDisabled(bool);
39 
40     const AtomString& type() const;
41     void setType(const AtomString&);
42 
43     const AtomString& media() const;

45 
46 private:
47     SVGStyleElement(const QualifiedName&, Document&, bool createdByParser);
48 
49     void parseAttribute(const QualifiedName&, const AtomString&) final;
50     InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) final;
51     void removedFromAncestor(RemovalType, ContainerNode&) final;
52     void childrenChanged(const ChildChange&) final;
53 
54     bool rendererIsNeeded(const RenderStyle&) final { return false; }
55 
56     void finishParsingChildren() final;
57 
58     virtual bool isLoading() const { return m_styleSheetOwner.isLoading(); }

61     Timer* svgLoadEventTimer() final { return &m_svgLoadEventTimer; }
62 
63     String title() const final;
64 
65     InlineStyleSheetOwner m_styleSheetOwner;
66     Timer m_svgLoadEventTimer;
67 };
