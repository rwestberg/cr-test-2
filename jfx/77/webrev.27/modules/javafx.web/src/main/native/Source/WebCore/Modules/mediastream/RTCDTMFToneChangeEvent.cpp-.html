33 namespace WebCore {
34 
35 Ref<RTCDTMFToneChangeEvent> RTCDTMFToneChangeEvent::create(const String& tone)
36 {
37     return adoptRef(*new RTCDTMFToneChangeEvent(tone));
38 }
39 
40 Ref<RTCDTMFToneChangeEvent> RTCDTMFToneChangeEvent::create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
41 {
42     return adoptRef(*new RTCDTMFToneChangeEvent(type, initializer, isTrusted));
43 }
44 
45 RTCDTMFToneChangeEvent::RTCDTMFToneChangeEvent(const String& tone)
46     : Event(eventNames().tonechangeEvent, CanBubble::No, IsCancelable::No)
47     , m_tone(tone)
48 {
49 }
50 
51 RTCDTMFToneChangeEvent::RTCDTMFToneChangeEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
52     : Event(type, initializer, isTrusted)
53     , m_tone(initializer.tone)
54 {
55 }
56 
57 RTCDTMFToneChangeEvent::~RTCDTMFToneChangeEvent() = default;
58 
59 const String& RTCDTMFToneChangeEvent::tone() const
60 {
61     return m_tone;
62 }
63 
64 EventInterface RTCDTMFToneChangeEvent::eventInterface() const
65 {
66     return RTCDTMFToneChangeEventInterfaceType;
67 }
