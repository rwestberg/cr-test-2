30 #pragma once
31 
32 #include "CodeBlock.h"
33 
34 namespace JSC {
35 
36 // Program code is not marked by any function, so we make the global object
37 // responsible for marking it.
38 
39 class GlobalCodeBlock : public CodeBlock {
40     typedef CodeBlock Base;
41 
42 protected:
43     GlobalCodeBlock(VM& vm, Structure* structure, CopyParsedBlockTag, GlobalCodeBlock& other)
44         : CodeBlock(vm, structure, CopyParsedBlock, other)
45     {
46     }
47 
48     GlobalCodeBlock(VM& vm, Structure* structure, ScriptExecutable* ownerExecutable, UnlinkedCodeBlock* unlinkedCodeBlock, JSScope* scope)
49         : CodeBlock(vm, structure, ownerExecutable, unlinkedCodeBlock, scope)
50     {
51     }
52 };
53 static_assert(sizeof(GlobalCodeBlock) == sizeof(CodeBlock), "Subclasses of CodeBlock should be the same size to share IsoSubspace");
54 
55 } // namespace JSC
