61 // helper operators specific to dumping the render tree. these are used in various classes to dump the render tree
62 // these could be defined in separate namespace to avoid matching these generic signatures unintentionally.
63 
64 template<typename Item>
65 WTF::TextStream& operator<<(WTF::TextStream& ts, const Vector<Item*>& v)
66 {
67     ts << "[";
68 
69     for (unsigned i = 0; i < v.size(); i++) {
70         ts << *v[i];
71         if (i < v.size() - 1)
72             ts << ", ";
73     }
74 
75     ts << "]";
76     return ts;
77 }
78 
79 template<typename Pointer>
80 WTF::TextStream& operator<<(WTF::TextStream& ts, Pointer* t)
81 {
82     ts << reinterpret_cast<intptr_t>(t);
83     return ts;
84 }
