32 ] interface StaticRange {
33     readonly attribute unsigned long startOffset;
34     readonly attribute unsigned long endOffset;
35     readonly attribute Node startContainer;
36     readonly attribute Node endContainer;
37     readonly attribute boolean collapsed;
38 };

45 };
