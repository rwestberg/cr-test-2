26 #pragma once
27 
28 #include <wtf/PrintStream.h>
29 
30 namespace WTF {
31 
32 class CommaPrinter final {
33     WTF_MAKE_FAST_ALLOCATED;
34 public:
35     CommaPrinter(const char* comma = ", ", const char* start = "")
36         : m_comma(comma)
37         , m_start(start)
38         , m_didPrint(false)
39     {
40     }
41 
42     void dump(PrintStream& out) const
43     {
44         if (!m_didPrint) {
45             out.print(m_start);
46             m_didPrint = true;
47             return;
48         }
49 
50         out.print(m_comma);
51     }
52 
53     bool didPrint() const { return m_didPrint; }
54 
55 private:
56     const char* m_comma;
57     const char* m_start;
58     mutable bool m_didPrint;
59 };
60 
61 } // namespace WTF
62 
63 using WTF::CommaPrinter;
