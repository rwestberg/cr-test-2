28 #include <JavaScriptCore/TypedArrayInlines.h>

30 namespace WebCore {
31 
32 static RefPtr<Float32Array> optionalFloat3ToJSCArray(const Optional<VRPlatformTrackingInfo::Float3>& data)
33 {
34     if (!data)
35         return nullptr;
36 
37     return Float32Array::create(data->data, 3);
38 }
39 
40 RefPtr<Float32Array> VRPose::position() const
41 {
42     if (!m_trackingInfo.position)
43         return nullptr;
44 
45     auto& position = *m_trackingInfo.position;
46     float positionData[3] = { position.x(), position.y(), position.z() };
47     return Float32Array::create(positionData, 3);
48 }
49 
50 RefPtr<Float32Array> VRPose::linearVelocity() const
51 {
52     return optionalFloat3ToJSCArray(m_trackingInfo.linearVelocity);
53 }
54 
55 RefPtr<Float32Array> VRPose::linearAcceleration() const
56 {
57     return optionalFloat3ToJSCArray(m_trackingInfo.linearAcceleration);
58 }
59 
60 RefPtr<Float32Array> VRPose::orientation() const
61 {
62     if (!m_trackingInfo.orientation)
63         return nullptr;
64 
65     auto& orientation = *m_trackingInfo.orientation;
66     float orientationData[4] = { orientation.x, orientation.y, orientation.z, orientation.w };
67     return Float32Array::create(orientationData, 4);
68 }
69 
70 RefPtr<Float32Array> VRPose::angularVelocity() const
71 {
72     return optionalFloat3ToJSCArray(m_trackingInfo.angularVelocity);
73 }
74 
75 RefPtr<Float32Array> VRPose::angularAcceleration() const
76 {
77     return optionalFloat3ToJSCArray(m_trackingInfo.angularAcceleration);
78 }
