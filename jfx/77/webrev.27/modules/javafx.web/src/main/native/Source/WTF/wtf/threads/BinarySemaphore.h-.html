26 #pragma once
27 
28 #include <wtf/Condition.h>
29 #include <wtf/Lock.h>
30 #include <wtf/Noncopyable.h>
31 #include <wtf/TimeWithDynamicClockType.h>
32 
33 namespace WTF {
34 
35 class BinarySemaphore {
36     WTF_MAKE_NONCOPYABLE(BinarySemaphore);
37     WTF_MAKE_FAST_ALLOCATED;
38 public:
39     constexpr BinarySemaphore() = default;
40 
41     WTF_EXPORT_PRIVATE void signal();
42     WTF_EXPORT_PRIVATE bool waitUntil(const TimeWithDynamicClockType&);
43 
44     bool waitFor(Seconds relativeTimeout)
45     {
46         return waitUntil(MonotonicTime::now() + relativeTimeout);
47     }
48 
49     void wait()
50     {
51         waitUntil(ParkingLot::Time::infinity());
52     }
53 
54 private:
55     bool m_isSet { false };
56     Lock m_lock;
57     Condition m_condition;
58 };
59 
60 } // namespace WTF
61 
62 using WTF::BinarySemaphore;
