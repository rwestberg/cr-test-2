28 #if ENABLE(WEBGPU)
29 
30 #include "WHLSLExpression.h"

31 #include "WHLSLUnsignedIntegerLiteralType.h"
32 #include <wtf/FastMalloc.h>
33 
34 namespace WebCore {
35 
36 namespace WHLSL {
37 
38 namespace AST {
39 
40 class UnsignedIntegerLiteral final : public Expression {
41     WTF_MAKE_FAST_ALLOCATED;
42 public:
43     UnsignedIntegerLiteral(CodeLocation location, unsigned value)
44         : Expression(location, Kind::UnsignedIntegerLiteral)
45         , m_type(location, value)
46         , m_value(value)
47     {
48     }
49 
50     ~UnsignedIntegerLiteral() = default;
51 
52     UnsignedIntegerLiteral(const UnsignedIntegerLiteral&) = delete;
53     UnsignedIntegerLiteral(UnsignedIntegerLiteral&&) = default;
54 
55     UnsignedIntegerLiteral& operator=(const UnsignedIntegerLiteral&) = delete;
56     UnsignedIntegerLiteral& operator=(UnsignedIntegerLiteral&&) = default;
57 
58     UnsignedIntegerLiteralType& type() { return m_type; }
59     unsigned value() const { return m_value; }
60 


61     UnsignedIntegerLiteral clone() const
62     {
63         UnsignedIntegerLiteral result(codeLocation(), m_value);
64         result.m_type = m_type.clone();
65         if (auto* resolvedType = m_type.maybeResolvedType())
66             result.m_type.resolve(const_cast<AST::UnnamedType&>(*resolvedType));
67         copyTypeTo(result);
68         return result;
69     }
70 
71     int64_t valueForSelectedType() const;
72 
73 private:
74     UnsignedIntegerLiteralType m_type;
75     unsigned m_value;
76 };
77 
78 } // namespace AST
79 
80 }
81 
82 }
83 
84 DEFINE_DEFAULT_DELETE(UnsignedIntegerLiteral)
85 
86 SPECIALIZE_TYPE_TRAITS_WHLSL_EXPRESSION(UnsignedIntegerLiteral, isUnsignedIntegerLiteral())
