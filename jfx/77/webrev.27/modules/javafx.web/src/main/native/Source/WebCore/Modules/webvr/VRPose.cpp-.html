25 #include "config.h"
26 #include "VRPose.h"
27 
28 namespace WebCore {
29 
30 static RefPtr<Float32Array> optionalFloat3ToJSCArray(const Optional<VRPlatformTrackingInfo::Float3>& data)
31 {
32     if (!data)
33         return nullptr;
34 
35     return Float32Array::create(data->data, 3);
36 }
37 
38 RefPtr<Float32Array> VRPose::position() const
39 {
40     if (!m_trackingInfo.position)
41         return nullptr;
42 
43     auto& position = *m_trackingInfo.position;
44     float positionData[3] = { position.x(), position.y(), position.z() };
45     return Float32Array::create(positionData, 3);
46 }
47 
48 RefPtr<Float32Array> VRPose::linearVelocity() const
49 {
50     return optionalFloat3ToJSCArray(m_trackingInfo.linearVelocity);
51 }
52 
53 RefPtr<Float32Array> VRPose::linearAcceleration() const
54 {
55     return optionalFloat3ToJSCArray(m_trackingInfo.linearAcceleration);
56 }
57 
58 RefPtr<Float32Array> VRPose::orientation() const
59 {
60     if (!m_trackingInfo.orientation)
61         return nullptr;
62 
63     auto& orientation = *m_trackingInfo.orientation;
64     float orientationData[4] = { orientation.x, orientation.y, orientation.z, orientation.w };
65     return Float32Array::create(orientationData, 4);
66 }
67 
68 RefPtr<Float32Array> VRPose::angularVelocity() const
69 {
70     return optionalFloat3ToJSCArray(m_trackingInfo.angularVelocity);
71 }
72 
73 RefPtr<Float32Array> VRPose::angularAcceleration() const
74 {
75     return optionalFloat3ToJSCArray(m_trackingInfo.angularAcceleration);
76 }
