30 namespace WebCore {

32 class Gradient;
33 class Image;

35 class GradientImage final : public GeneratedImage {
36 public:
37     static Ref<GradientImage> create(Gradient& generator, const FloatSize& size)
38     {
39         return adoptRef(*new GradientImage(generator, size));
40     }
41 
42     virtual ~GradientImage();
43 
44     const Gradient& gradient() const { return m_gradient.get(); }
45 
46 private:
47     GradientImage(Gradient&, const FloatSize&);
48 
49     ImageDrawResult draw(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& = { }) final;
50     void drawPattern(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& = { }) final;
51     bool isGradientImage() const final { return true; }
52     void dump(WTF::TextStream&) const final;
53 
54     Ref<Gradient> m_gradient;
55     RefPtr<Image> m_cachedImage;
56     FloatSize m_cachedAdjustedSize;
57     unsigned m_cachedGeneratorHash;
58     FloatSize m_cachedScaleFactor;
59 };

61 }
