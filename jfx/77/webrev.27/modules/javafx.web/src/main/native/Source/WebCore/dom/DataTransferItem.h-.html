41 namespace WebCore {
42 
43 class DOMFileSystem;
44 class DataTransferListItem;
45 class File;
46 class FileSystemEntry;
47 class ScriptExecutionContext;
48 class StringCallback;
49 
50 class DataTransferItem : public RefCounted<DataTransferItem> {
51 public:
52     static Ref<DataTransferItem> create(WeakPtr<DataTransferItemList>&&, const String&);
53     static Ref<DataTransferItem> create(WeakPtr<DataTransferItemList>&&, const String&, Ref<File>&&);
54 
55     ~DataTransferItem();
56 
57     RefPtr<File> file() { return m_file; }
58     void clearListAndPutIntoDisabledMode();
59 
60     bool isFile() const { return m_file; }
61     String kind() const;
62     String type() const;
63     void getAsString(Document&, RefPtr<StringCallback>&&) const;
64     RefPtr<File> getAsFile() const;
65     RefPtr<FileSystemEntry> getAsEntry(ScriptExecutionContext&) const;
66 
67 private:
68     DataTransferItem(WeakPtr<DataTransferItemList>&&, const String&);
69     DataTransferItem(WeakPtr<DataTransferItemList>&&, const String&, Ref<File>&&);
70 
71     bool isInDisabledMode() const { return !m_list; }
72 
73     WeakPtr<DataTransferItemList> m_list;
74     const String m_type;
75     RefPtr<File> m_file;
76 };
