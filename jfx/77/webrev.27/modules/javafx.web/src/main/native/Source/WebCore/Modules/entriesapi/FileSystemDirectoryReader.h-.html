34 namespace WebCore {
35 
36 class ErrorCallback;
37 class FileSystemDirectoryEntry;
38 class FileSystemEntriesCallback;
39 class ScriptExecutionContext;
40 
41 class FileSystemDirectoryReader final : public ScriptWrappable, public ActiveDOMObject, public RefCounted<FileSystemDirectoryReader> {
42 public:
43     static Ref<FileSystemDirectoryReader> create(ScriptExecutionContext& context, FileSystemDirectoryEntry& directory)
44     {
45         return adoptRef(*new FileSystemDirectoryReader(context, directory));
46     }
47 
48     ~FileSystemDirectoryReader();
49 
50     void readEntries(ScriptExecutionContext&, Ref<FileSystemEntriesCallback>&&, RefPtr<ErrorCallback>&&);
51 
52 private:
53     FileSystemDirectoryReader(ScriptExecutionContext&, FileSystemDirectoryEntry&);
54 
55     const char* activeDOMObjectName() const final;
56     bool canSuspendForDocumentSuspension() const final;
57 
58     Ref<FileSystemDirectoryEntry> m_directory;
59     Optional<Exception> m_error;
60     bool m_isReading { false };
61     bool m_isDone { false };
62 };
