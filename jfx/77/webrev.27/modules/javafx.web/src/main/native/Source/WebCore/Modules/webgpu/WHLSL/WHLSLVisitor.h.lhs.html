179     bool error() const { return m_error; }

180 
181     template<typename T> void checkErrorAndVisit(T& x)
182     {
183         if (!m_error)
184             visit(x);
185     }
186 
187 protected:
188     void setError()
189     {
190         ASSERT(!m_error);
191         m_error = true;
192     }
193
