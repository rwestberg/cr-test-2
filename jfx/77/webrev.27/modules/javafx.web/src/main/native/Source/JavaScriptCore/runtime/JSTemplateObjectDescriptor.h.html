31 namespace JSC {

33 class JSTemplateObjectDescriptor final : public JSCell {
34 public:
35     using Base = JSCell;
36 
37     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38     static const bool needsDestruction = true;
39     DECLARE_INFO;
40 
41     static JSTemplateObjectDescriptor* create(VM&, Ref<TemplateObjectDescriptor>&&, int);
42 
43     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
44     {
45         return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
46     }
47 
48     const TemplateObjectDescriptor& descriptor() const { return m_descriptor.get(); }
49 
50     JSArray* createTemplateObject(ExecState*);
51 
52     int endOffset() const { return m_endOffset; }
53 
54 protected:
55     static void destroy(JSCell*);
56 
57 private:
58     JSTemplateObjectDescriptor(VM&, Ref<TemplateObjectDescriptor>&&, int);
59 
60     Ref<TemplateObjectDescriptor> m_descriptor;
61     int m_endOffset { 0 };
62 };

64 } // namespace JSC
