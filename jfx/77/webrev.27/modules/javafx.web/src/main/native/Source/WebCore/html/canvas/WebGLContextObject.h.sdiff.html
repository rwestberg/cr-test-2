33 class WebGLRenderingContextBase;

35 // WebGLContextObject the base class for objects that are owned by a specific
36 // WebGLRenderingContext.
37 class WebGLContextObject : public WebGLObject {
38 public:
39     virtual ~WebGLContextObject();
40 
41     WebGLRenderingContextBase* context() const { return m_context; }
42 
43     bool validate(const WebGLContextGroup*, const WebGLRenderingContextBase& context) const override
44     {
45         return &context == m_context;
46     }
47 
48     void detachContext();
49 
50 protected:
51     WebGLContextObject(WebGLRenderingContextBase&);
52 
53     bool hasGroupOrContext() const override
54     {
55         return m_context;
56     }
57 
58     GraphicsContext3D* getAGraphicsContext3D() const override;
59 
60 private:
61     WebGLRenderingContextBase* m_context;
62 };

64 } // namespace WebCore
