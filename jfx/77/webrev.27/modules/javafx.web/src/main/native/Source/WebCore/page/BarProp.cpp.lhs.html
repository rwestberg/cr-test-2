32 #include "Chrome.h"

35 
36 namespace WebCore {
37 


38 BarProp::BarProp(DOMWindow& window, Type type)
39     : DOMWindowProperty(&window)
40     , m_type(type)
41 {
42 }
43 
44 BarProp::Type BarProp::type() const
45 {
46     return m_type;
47 }
48 
49 bool BarProp::visible() const
50 {
51     auto* frame = this->frame();
52     if (!frame)
53         return false;
54     auto* page = frame->page();
55     if (!page)
56         return false;
57 
58     switch (m_type) {
59     case Locationbar:
60     case Personalbar:
61     case Toolbar:
62         return page->chrome().toolbarsVisible();
63     case Menubar:
64         return page->chrome().menubarVisible();
65     case Scrollbars:
66         return page->chrome().scrollbarsVisible();
67     case Statusbar:
68         return page->chrome().statusbarVisible();
69     }
70 
71     ASSERT_NOT_REACHED();
72     return false;
73 }
