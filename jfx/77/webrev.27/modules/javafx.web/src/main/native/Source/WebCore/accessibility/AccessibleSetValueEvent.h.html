32 class AccessibleSetValueEvent : public Event {
33 public:
34     static Ref<AccessibleSetValueEvent> create(const AtomString& type, const AtomString& value)
35     {
36         return adoptRef(*new AccessibleSetValueEvent(type, value));
37     }
38 
39     virtual ~AccessibleSetValueEvent();
40 
41     const AtomString& value() const { return m_value; }
42 
43 protected:
44     AccessibleSetValueEvent(const AtomString& type, const AtomString& value);
45 
46     // Event.
47     EventInterface eventInterface() const override { return AccessibleSetValueEventInterfaceType; }
48 
49 private:
50     const AtomString m_value;
51 };
