 28 #if ENABLE(B3_JIT)

 30 #include "B3Type.h"
 31 #include <wtf/PrintStream.h>
 32 
 33 #if ASSERT_DISABLED
 34 IGNORE_RETURN_TYPE_WARNINGS_BEGIN
 35 #endif
 36 
 37 namespace JSC { namespace B3 {
 38 
 39 template<typename T>
 40 class TypeMap {
 41 public:
 42     TypeMap()
 43         : m_void()
 44         , m_int32()
 45         , m_int64()
 46         , m_float()
 47         , m_double()
 48     {
 49     }
 50 
 51     T& at(Type type)
 52     {
 53         switch (type.kind()) {
 54         case Void:
 55             return m_void;
 56         case Int32:
 57             return m_int32;
 58         case Int64:
 59             return m_int64;
 60         case Float:
 61             return m_float;
 62         case Double:
 63             return m_double;
 64         case Tuple:
 65             return m_tuple;
 66         }
 67         ASSERT_NOT_REACHED();
 68     }
 69 
 70     const T& at(Type type) const
 71     {
 72         return bitwise_cast<TypeMap*>(this)->at(type);
 73     }
 74 
 75     T& operator[](Type type)
 76     {
 77         return at(type);
 78     }
 79 
 80     const T& operator[](Type type) const
 81     {
 82         return at(type);
 83     }
 84 
 85     void dump(PrintStream& out) const
 86     {
 87         out.print(
 88             "{void = ", m_void,
 89             ", int32 = ", m_int32,
 90             ", int64 = ", m_int64,
 91             ", float = ", m_float,
 92             ", double = ", m_double, "}");
 93     }
 94 
 95 private:
 96     T m_void;
 97     T m_int32;
 98     T m_int64;
 99     T m_float;
100     T m_double;
101     T m_tuple;
102 };
103 
104 } } // namespace JSC::B3
