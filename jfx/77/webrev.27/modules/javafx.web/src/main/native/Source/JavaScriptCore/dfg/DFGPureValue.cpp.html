33 void PureValue::dump(PrintStream& out) const
34 {
35     out.print(Graph::opName(op()));
36     out.print("(");
37     CommaPrinter comma;
38     if (isVarargs()) {
39         for (unsigned i = 0; i < m_children.numChildren(); ++i)
40             out.print(comma, m_graph->m_varArgChildren[m_children.firstChild() + i].sanitized());
41     } else {
42         for (unsigned i = 0; i < AdjacencyList::Size; ++i) {
43             if (m_children.child(i))
44                 out.print(comma, m_children.child(i));
45         }
46     }
47     if (m_info)
48         out.print(comma, m_info);
49     out.print(")");
50 }
