38 ExceptionOr<Ref<DOMPoint>> DOMPointReadOnly::matrixTransform(DOMMatrixInit&& matrixInit) const
39 {
40     auto matrixOrException = DOMMatrixReadOnly::fromMatrix(WTFMove(matrixInit));
41     if (matrixOrException.hasException())
42         return matrixOrException.releaseException();
43 
44     auto matrix = matrixOrException.releaseReturnValue();
45 
46     double x = this->x();
47     double y = this->y();
48     double z = this->z();
49     double w = this->w();
50     matrix->transformationMatrix().map4ComponentPoint(x, y, z, w);
51 
52     return { DOMPoint::create(x, y, z, w) };
53 }
