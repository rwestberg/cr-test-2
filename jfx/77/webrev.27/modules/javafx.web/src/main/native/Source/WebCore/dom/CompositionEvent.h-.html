27 #pragma once
28 
29 #include "UIEvent.h"
30 
31 namespace WebCore {
32 
33 class CompositionEvent final : public UIEvent {
34 public:
35     static Ref<CompositionEvent> create(const AtomicString& type, RefPtr<WindowProxy>&& view, const String& data)
36     {
37         return adoptRef(*new CompositionEvent(type, WTFMove(view), data));
38     }
39 
40     static Ref<CompositionEvent> createForBindings()
41     {
42         return adoptRef(*new CompositionEvent);
43     }
44 
45     struct Init : UIEventInit {
46         String data;
47     };
48 
49     static Ref<CompositionEvent> create(const AtomicString& type, const Init& initializer)
50     {
51         return adoptRef(*new CompositionEvent(type, initializer));
52     }
53 
54     virtual ~CompositionEvent();
55 
56     void initCompositionEvent(const AtomicString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&&, const String& data);
57 
58     String data() const { return m_data; }
59 
60     EventInterface eventInterface() const override;
61 
62 private:
63     CompositionEvent();
64     CompositionEvent(const AtomicString& type, RefPtr<WindowProxy>&&, const String&);
65     CompositionEvent(const AtomicString& type, const Init&);
66 
67     bool isCompositionEvent() const override;
68 
69     String m_data;
70 };
71 
72 } // namespace WebCore
73 
74 SPECIALIZE_TYPE_TRAITS_EVENT(CompositionEvent)
