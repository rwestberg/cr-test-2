26 #pragma once
27 
28 #include <wtf/PrintStream.h>
29 #include <wtf/text/WTFString.h>
30 
31 namespace WTF {
32 
33 class Indenter {
34 public:
35     Indenter(unsigned count = 0, String string = "  "_s)
36         : m_count(count)
37         , m_string(string)
38     { }
39 
40     Indenter(const Indenter& other)
41         : m_count(other.m_count)
42         , m_string(other.m_string)
43     { }
44 
45     void dump(PrintStream& out) const
46     {
47         unsigned levels = m_count;
48         while (levels--)
49             out.print(m_string);
50     }
51 
52     unsigned operator++() { return ++m_count; }
53     unsigned operator++(int) { return m_count++; }
54     unsigned operator--() { return --m_count; }
55     unsigned operator--(int) { return m_count--; }
56 
57 private:
58     unsigned m_count;
59     String m_string;
60 };
61 
62 } // namespace WTF
63 
64 using WTF::Indenter;
