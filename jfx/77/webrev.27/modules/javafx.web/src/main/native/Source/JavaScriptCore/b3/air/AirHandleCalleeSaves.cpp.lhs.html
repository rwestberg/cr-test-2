36 void handleCalleeSaves(Code& code)
37 {
38     RegisterSet usedCalleeSaves;
39 
40     for (BasicBlock* block : code) {
41         for (Inst& inst : *block) {
42             inst.forEachTmpFast(
43                 [&] (Tmp& tmp) {
44                     // At first we just record all used regs.
45                     usedCalleeSaves.set(tmp.reg());
46                 });
47 
48             if (inst.kind.opcode == Patch)
49                 usedCalleeSaves.merge(inst.extraClobberedRegs());
50         }
51     }
52 
53     handleCalleeSaves(code, WTFMove(usedCalleeSaves));
54 }
55 
56 void handleCalleeSaves(Code& code, RegisterSet usedCalleeSaves)
57 {
58     // We filter to really get the callee saves.
59     usedCalleeSaves.filter(RegisterSet::calleeSaveRegisters());
60     usedCalleeSaves.filter(code.mutableRegs());
61     usedCalleeSaves.exclude(RegisterSet::stackRegisters()); // We don't need to save FP here.
62 
63     if (!usedCalleeSaves.numberOfSetRegisters())
64         return;
65 
66     RegisterAtOffsetList calleeSaveRegisters = RegisterAtOffsetList(usedCalleeSaves);
67 
68     size_t byteSize = 0;
69     for (const RegisterAtOffset& entry : calleeSaveRegisters)
70         byteSize = std::max(static_cast<size_t>(-entry.offset()), byteSize);
71 
72     code.setCalleeSaveRegisterAtOffsetList(
73         WTFMove(calleeSaveRegisters),
74         code.addStackSlot(byteSize, StackSlotKind::Locked));
75 }
