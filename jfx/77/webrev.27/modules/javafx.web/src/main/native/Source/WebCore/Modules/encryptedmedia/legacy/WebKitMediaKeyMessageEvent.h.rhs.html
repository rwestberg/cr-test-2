31 #include "Event.h"

34 namespace WebCore {
35 
36 class WebKitMediaKeyMessageEvent : public Event {
37 public:
38     virtual ~WebKitMediaKeyMessageEvent();
39 
40     static Ref<WebKitMediaKeyMessageEvent> create(const AtomString& type, Uint8Array* message, const String& destinationURL)
41     {
42         return adoptRef(*new WebKitMediaKeyMessageEvent(type, message, destinationURL));
43     }
44 
45     struct Init : EventInit {
46         RefPtr<Uint8Array> message;
47         String destinationURL;
48     };
49 
50     static Ref<WebKitMediaKeyMessageEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
51     {
52         return adoptRef(*new WebKitMediaKeyMessageEvent(type, initializer, isTrusted));
53     }
54 
55     EventInterface eventInterface() const override;
56 
57     Uint8Array* message() const { return m_message.get(); }
58     String destinationURL() const { return m_destinationURL; }
59 
60 private:
61     WebKitMediaKeyMessageEvent(const AtomString& type, Uint8Array* message, const String& destinationURL);
62     WebKitMediaKeyMessageEvent(const AtomString& type, const Init&, IsTrusted);
63 
64     RefPtr<Uint8Array> m_message;
65     String m_destinationURL;
66 };

68 } // namespace WebCore
