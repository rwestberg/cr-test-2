25 #pragma once
26 
27 #include "AudioBasicProcessorNode.h"
28 #include "BiquadProcessor.h"
29 
30 namespace WebCore {
31 
32 class AudioParam;
33 
34 class BiquadFilterNode final : public AudioBasicProcessorNode {
35     WTF_MAKE_ISO_ALLOCATED(BiquadFilterNode);
36 public:
37     static Ref<BiquadFilterNode> create(AudioContext& context, float sampleRate)
38     {
39         return adoptRef(*new BiquadFilterNode(context, sampleRate));
40     }
41 
42     BiquadFilterType type() const;
43     void setType(BiquadFilterType);
44 
45     AudioParam* frequency() { return biquadProcessor()->parameter1(); }
46     AudioParam* q() { return biquadProcessor()->parameter2(); }
47     AudioParam* gain() { return biquadProcessor()->parameter3(); }
48     AudioParam* detune() { return biquadProcessor()->parameter4(); }
49 
50     // Get the magnitude and phase response of the filter at the given
51     // set of frequencies (in Hz). The phase response is in radians.
52     void getFrequencyResponse(const RefPtr<Float32Array>& frequencyHz, const RefPtr<Float32Array>& magResponse, const RefPtr<Float32Array>& phaseResponse);
53 
54 private:
55     BiquadFilterNode(AudioContext&, float sampleRate);
56 
57     BiquadProcessor* biquadProcessor() { return static_cast<BiquadProcessor*>(processor()); }
58 };
59 
60 } // namespace WebCore
