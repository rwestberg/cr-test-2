26 
27 #include "AudioDestinationNode.h"
28 
29 namespace WebCore {
30 
31 class AudioDestination;
32 
33 class DefaultAudioDestinationNode final : public AudioDestinationNode {
34     WTF_MAKE_ISO_ALLOCATED(DefaultAudioDestinationNode);
35 public:
36     static Ref<DefaultAudioDestinationNode> create(AudioContext& context)
37     {
38         return adoptRef(*new DefaultAudioDestinationNode(context));
39     }
40 
41     virtual ~DefaultAudioDestinationNode();
42 
43 private:
44     explicit DefaultAudioDestinationNode(AudioContext&);
45     void createDestination();
46 
47     void initialize() final;
48     void uninitialize() final;
49     ExceptionOr<void> setChannelCount(unsigned) final;
50 
51     void enableInput(const String& inputDeviceId) final;
52     void startRendering() final;
53     void resume(Function<void ()>&&) final;
54     void suspend(Function<void ()>&&) final;
55     void close(Function<void ()>&&) final;
56     unsigned maxChannelCount() const final;
57     bool isPlaying() final;
58 
59     std::unique_ptr<AudioDestination> m_destination;
60     String m_inputDeviceId;
61     unsigned m_numberOfInputChannels { 0 };
62 };

64 } // namespace WebCore
