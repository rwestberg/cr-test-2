27 #include <WebCore/PageCache.h>

30 #include "com_sun_webkit_PageCache.h"

32 extern "C" {
33 
34 JNIEXPORT jint JNICALL Java_com_sun_webkit_PageCache_twkGetCapacity
35   (JNIEnv *, jclass)
36 {
37     return WebCore::PageCache::singleton().maxSize();
38 }
39 
40 JNIEXPORT void JNICALL Java_com_sun_webkit_PageCache_twkSetCapacity
41   (JNIEnv *, jclass, jint capacity)
42 {
43     ASSERT(capacity >= 0);
44     WebCore::PageCache::singleton().setMaxSize(capacity);
45 }
