32 #include <wtf/text/StringBuilder.h>

34 namespace WebCore {
35 
36 CSSFontVariationValue::CSSFontVariationValue(FontTag tag, float value)
37     : CSSValue(FontVariationClass)
38     , m_tag(tag)
39     , m_value(value)
40 {
41 }
42 
43 String CSSFontVariationValue::customCSSText() const
44 {
45     StringBuilder builder;
46     builder.append('"');
47     for (char c : m_tag)
48         builder.append(c);
49     builder.appendLiteral("\" ");
50     builder.appendNumber(m_value);
51     return builder.toString();
52 }
53 
54 bool CSSFontVariationValue::equals(const CSSFontVariationValue& other) const
55 {
56     return m_tag == other.m_tag && m_value == other.m_value;
57 }

59 }
