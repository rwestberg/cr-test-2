31 #include "AccessibilityMockObject.h"
32 #include "AccessibilityRenderObject.h"

34 namespace WebCore {
35 
36 class HTMLInputElement;
37 
38 class AccessibilitySlider : public AccessibilityRenderObject {
39 public:
40     static Ref<AccessibilitySlider> create(RenderObject*);
41     virtual ~AccessibilitySlider() = default;
42 
43 protected:
44     explicit AccessibilitySlider(RenderObject*);
45 
46 private:
47     HTMLInputElement* inputElement() const;
48     AccessibilityObject* elementAccessibilityHitTest(const IntPoint&) const override;
49 
50     AccessibilityRole roleValue() const override { return AccessibilityRole::Slider; }
51     bool isSlider() const final { return true; }
52     bool isInputSlider() const override { return true; }
53     bool isControl() const override { return true; }
54 
55     void addChildren() override;
56 
57     bool canSetValueAttribute() const override { return true; }
58     const AtomString& getAttribute(const QualifiedName& attribute) const;
59 
60     void setValue(const String&) override;
61     float valueForRange() const override;
62     float maxValueForRange() const override;
63     float minValueForRange() const override;
64     AccessibilityOrientation orientation() const override;
65 };
66 
67 class AccessibilitySliderThumb final : public AccessibilityMockObject {
68 public:
69     static Ref<AccessibilitySliderThumb> create();
70     virtual ~AccessibilitySliderThumb() = default;
71 
72     AccessibilityRole roleValue() const override { return AccessibilityRole::SliderThumb; }
73     LayoutRect elementRect() const override;
74 
75 private:
76     AccessibilitySliderThumb();
77 
78     bool isSliderThumb() const override { return true; }
79     bool computeAccessibilityIsIgnored() const override;
80 };
81 
82 } // namespace WebCore
