25 #pragma once
26 
27 #include "AudioBasicProcessorNode.h"
28 #include "BiquadProcessor.h"
29 
30 namespace WebCore {
31 
32 class AudioParam;
33 
34 class BiquadFilterNode final : public AudioBasicProcessorNode {

35 public:
36     static Ref<BiquadFilterNode> create(AudioContext& context, float sampleRate)
37     {
38         return adoptRef(*new BiquadFilterNode(context, sampleRate));
39     }
40 
41     BiquadFilterType type() const;
42     void setType(BiquadFilterType);
43 
44     AudioParam* frequency() { return biquadProcessor()->parameter1(); }
45     AudioParam* q() { return biquadProcessor()->parameter2(); }
46     AudioParam* gain() { return biquadProcessor()->parameter3(); }
47     AudioParam* detune() { return biquadProcessor()->parameter4(); }
48 
49     // Get the magnitude and phase response of the filter at the given
50     // set of frequencies (in Hz). The phase response is in radians.
51     void getFrequencyResponse(const RefPtr<Float32Array>& frequencyHz, const RefPtr<Float32Array>& magResponse, const RefPtr<Float32Array>& phaseResponse);
52 
53 private:
54     BiquadFilterNode(AudioContext&, float sampleRate);
55 
56     BiquadProcessor* biquadProcessor() { return static_cast<BiquadProcessor*>(processor()); }
57 };
58 
59 } // namespace WebCore
