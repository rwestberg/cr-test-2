59 bool HTMLHtmlElement::isURLAttribute(const Attribute& attribute) const
60 {
61     return attribute.name() == manifestAttr || HTMLElement::isURLAttribute(attribute);
62 }
63 
64 void HTMLHtmlElement::insertedByParser()
65 {
66     // When parsing a fragment, its dummy document has a null parser.
67     if (!document().parser() || !document().parser()->documentWasLoadedAsPartOfNavigation())
68         return;
69 
70     if (!document().frame())
71         return;
72 
73     auto documentLoader = makeRefPtr(document().frame()->loader().documentLoader());
74     if (!documentLoader)
75         return;
76 
77     auto& manifest = attributeWithoutSynchronization(manifestAttr);
78     if (manifest.isEmpty())
79         documentLoader->applicationCacheHost().selectCacheWithoutManifest();
80     else {
81         document().addConsoleMessage(MessageSource::AppCache, MessageLevel::Warning, "ApplicationCache is deprecated. Please use ServiceWorkers instead."_s);
82         documentLoader->applicationCacheHost().selectCacheWithManifest(document().completeURL(manifest));
83     }
84 }
