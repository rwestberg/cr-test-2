35 #include "MediaPlayer.h"
36 #include "PlatformScreen.h"

38 #include <wtf/NeverDestroyed.h>
39 
40 namespace WebCore {
41 
42 RuntimeEnabledFeatures::RuntimeEnabledFeatures()
43 {
44 #if ENABLE(MEDIA_STREAM) && PLATFORM(COCOA)
45     m_isMediaDevicesEnabled = false;
46 #endif
47 #if PLATFORM(WATCHOS)
48     m_isWebSocketEnabled = false;
49 #endif
50 #if PLATFORM(GTK) && ENABLE(INPUT_TYPE_COLOR)
51     m_isInputTypeColorEnabled = true;
52 #endif
53 }
54 
55 RuntimeEnabledFeatures& RuntimeEnabledFeatures::sharedFeatures()
56 {
57     static NeverDestroyed<RuntimeEnabledFeatures> runtimeEnabledFeatures;
58 
59     return runtimeEnabledFeatures;
60 }
61 
62 bool RuntimeEnabledFeatures::spectreGadgetsEnabled() const
63 {
64     return JSC::Options::enableSpectreGadgets();
65 }
66 
67 #if ENABLE(TOUCH_EVENTS)
68 bool RuntimeEnabledFeatures::touchEventsEnabled() const
69 {
70     return m_touchEventsEnabled.valueOr(screenHasTouchDevice());
71 }
72 #endif
73 
74 } // namespace WebCore
