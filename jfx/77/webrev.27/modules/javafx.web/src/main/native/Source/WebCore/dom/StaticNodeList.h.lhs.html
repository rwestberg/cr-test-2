31 #include "Element.h"
32 #include "NodeList.h"
33 #include <wtf/Vector.h>
34 
35 namespace WebCore {
36 
37 class WEBCORE_EXPORT StaticNodeList final : public NodeList {

38 public:
39     static Ref<StaticNodeList> create(Vector<Ref<Node>>&& nodes = { })
40     {
41         return adoptRef(*new StaticNodeList(WTFMove(nodes)));
42     }
43 
44     unsigned length() const override;
45     Node* item(unsigned index) const override;
46 
47 private:
48     StaticNodeList(Vector<Ref<Node>>&& nodes)
49         : m_nodes(WTFMove(nodes))
50     { }
51 
52     Vector<Ref<Node>> m_nodes;
53 };
54 
55 class StaticElementList final : public NodeList {

56 public:
57     static Ref<StaticElementList> create(Vector<Ref<Element>>&& elements = { })
58     {
59         return adoptRef(*new StaticElementList(WTFMove(elements)));
60     }
61 
62     unsigned length() const override;
63     Element* item(unsigned index) const override;
64 
65 private:
66     StaticElementList(Vector<Ref<Element>>&& elements)
67         : m_elements(WTFMove(elements))
68     { }
69 
70     Vector<Ref<Element>> m_elements;
71 };
