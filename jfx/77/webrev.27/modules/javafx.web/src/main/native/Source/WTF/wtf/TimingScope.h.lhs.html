26 #pragma once
27 
28 #include <wtf/MonotonicTime.h>
29 #include <wtf/Noncopyable.h>
30 
31 namespace WTF {
32 
33 // TimingScope is intended for microbenchmarking sections of code.
34 //
35 // Do not leave in shipping code.
36 //
37 // Mean scope durations are printed using WTFLogAlways every Nth call, where N is specified by logIterationInterval.
38 
39 class TimingScope {

40     WTF_MAKE_NONCOPYABLE(TimingScope);
41 public:
42     TimingScope(const char* name, unsigned logIterationInterval = 1)
43         : m_startTime(MonotonicTime::now())
44         , m_name(name)
45         , m_logIterationInterval(logIterationInterval)
46     {
47     }
48 
49     ~TimingScope()
50     {
51         scopeDidEnd();
52     }
53 
54 private:
55     WTF_EXPORT_PRIVATE void scopeDidEnd();
56 
57     MonotonicTime m_startTime;
58     const char* m_name;
59     unsigned m_logIterationInterval;
60 };
61 
62 } // namespace WTF
63 
64 using WTF::TimingScope;
