35 RuntimeType runtimeTypeForValue(VM& vm, JSValue value)
36 {
37     if (UNLIKELY(!value))
38         return TypeNothing;
39 
40     if (value.isUndefined())
41         return TypeUndefined;
42     if (value.isNull())
43         return TypeNull;
44     if (value.isAnyInt())
45         return TypeAnyInt;
46     if (value.isNumber())
47         return TypeNumber;
48     if (value.isString())
49         return TypeString;
50     if (value.isBoolean())
51         return TypeBoolean;
52     if (value.isObject())
53         return TypeObject;
54     if (value.isFunction(vm))
55         return TypeFunction;
56     if (value.isSymbol())
57         return TypeSymbol;
58     if (value.isBigInt())
59         return TypeBigInt;
60 
61     return TypeNothing;
62 }
63 
64 String runtimeTypeAsString(RuntimeType type)
65 {
66     if (type == TypeUndefined)
67         return "Undefined"_s;
68     if (type == TypeNull)
69         return "Null"_s;
70     if (type == TypeAnyInt)
71         return "Integer"_s;
72     if (type == TypeNumber)
73         return "Number"_s;
74     if (type == TypeString)
75         return "String"_s;
76     if (type == TypeObject)
77         return "Object"_s;
78     if (type == TypeBoolean)
79         return "Boolean"_s;
80     if (type == TypeFunction)
81         return "Function"_s;
82     if (type == TypeSymbol)
83         return "Symbol"_s;
84     if (type == TypeBigInt)
85         return "BigInt"_s;
86     if (type == TypeNothing)
87         return "(Nothing)"_s;
88 
89     RELEASE_ASSERT_NOT_REACHED();
90     return emptyString();
91 }
