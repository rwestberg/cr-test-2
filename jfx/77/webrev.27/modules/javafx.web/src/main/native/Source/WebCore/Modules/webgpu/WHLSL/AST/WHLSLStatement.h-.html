40 class Statement : public Value {
41 public:
42     Statement(Lexer::Token&& origin)
43         : m_origin(WTFMove(origin))
44     {
45     }
46 
47     virtual ~Statement() = default;
48 
49     Statement(const Statement&) = delete;
50     Statement(Statement&&) = default;
51 
52     virtual bool isBlock() const { return false; }
53     virtual bool isBreak() const { return false; }
54     virtual bool isContinue() const { return false; }
55     virtual bool isDoWhileLoop() const { return false; }
56     virtual bool isEffectfulExpressionStatement() const { return false; }
57     virtual bool isFallthrough() const { return false; }
58     virtual bool isForLoop() const { return false; }
59     virtual bool isIfStatement() const { return false; }
60     virtual bool isReturn() const { return false; }
61     virtual bool isSwitchCase() const { return false; }
62     virtual bool isSwitchStatement() const { return false; }
63     virtual bool isTrap() const { return false; }
64     virtual bool isVariableDeclarationsStatement() const { return false; }
65     virtual bool isWhileLoop() const { return false; }
66 
67 private:
68     Lexer::Token m_origin;
69 };
70 
71 using Statements = Vector<UniqueRef<Statement>>;
