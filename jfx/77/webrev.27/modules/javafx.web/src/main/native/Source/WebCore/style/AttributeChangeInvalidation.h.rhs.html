26 #pragma once
27 
28 #include "Element.h"
29 
30 namespace WebCore {
31 
32 struct InvalidationRuleSet;
33 
34 namespace Style {
35 
36 class AttributeChangeInvalidation {
37 public:
38     AttributeChangeInvalidation(Element&, const QualifiedName&, const AtomString& oldValue, const AtomString& newValue);
39     ~AttributeChangeInvalidation();
40 
41 private:
42     void invalidateStyle(const QualifiedName&, const AtomString& oldValue, const AtomString& newValue);
43     void invalidateStyleWithRuleSets();
44 
45     const bool m_isEnabled;
46     Element& m_element;
47 
48     Vector<const InvalidationRuleSet*, 4> m_invalidationRuleSets;
49 };
50 
51 inline AttributeChangeInvalidation::AttributeChangeInvalidation(Element& element, const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
52     : m_isEnabled(element.needsStyleInvalidation())
53     , m_element(element)
54 {
55     if (!m_isEnabled)
56         return;
57     invalidateStyle(attributeName, oldValue, newValue);
58     invalidateStyleWithRuleSets();
59 }
60 
61 inline AttributeChangeInvalidation::~AttributeChangeInvalidation()
62 {
63     if (!m_isEnabled)
64         return;
65     invalidateStyleWithRuleSets();
66 }
67 
68 }
69 }
