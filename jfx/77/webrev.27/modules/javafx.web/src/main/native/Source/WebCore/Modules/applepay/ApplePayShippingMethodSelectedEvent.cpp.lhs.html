35 static inline ApplePayShippingMethod convert(const ApplePaySessionPaymentRequest::ShippingMethod& shippingMethod)
36 {
37     ApplePayShippingMethod convertedMethod;
38     convertedMethod.label = shippingMethod.label;
39     convertedMethod.detail = shippingMethod.detail;
40     convertedMethod.identifier = shippingMethod.identifier;
41     convertedMethod.amount = shippingMethod.amount;
42 
43     return convertedMethod;
44 }
45 
46 ApplePayShippingMethodSelectedEvent::ApplePayShippingMethodSelectedEvent(const AtomicString& type, const ApplePaySessionPaymentRequest::ShippingMethod& shippingMethod)
47     : Event(type, CanBubble::No, IsCancelable::No)
48     , m_shippingMethod(convert(shippingMethod))
49 {
50 }
51 
52 ApplePayShippingMethodSelectedEvent::~ApplePayShippingMethodSelectedEvent() = default;
53 
54 EventInterface ApplePayShippingMethodSelectedEvent::eventInterface() const
55 {
56     return ApplePayShippingMethodSelectedEventInterfaceType;
57 }
