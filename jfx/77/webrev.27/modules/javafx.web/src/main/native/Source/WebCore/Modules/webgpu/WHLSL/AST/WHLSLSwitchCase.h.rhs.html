30 #include "WHLSLBlock.h"
31 #include "WHLSLCodeLocation.h"
32 #include "WHLSLConstantExpression.h"

35 #include <wtf/Optional.h>

37 namespace WebCore {
38 
39 namespace WHLSL {
40 
41 namespace AST {
42 
43 class SwitchCase final : public Statement {
44     WTF_MAKE_FAST_ALLOCATED;
45 public:
46     SwitchCase(CodeLocation location, Optional<ConstantExpression>&& value, Block&& block)
47         : Statement(location, Kind::SwitchCase)
48         , m_value(WTFMove(value))
49         , m_block(WTFMove(block))
50     {
51     }
52 
53     ~SwitchCase() = default;
54 
55     SwitchCase(const SwitchCase&) = delete;
56     SwitchCase(SwitchCase&&) = default;
57 


58     Optional<ConstantExpression>& value() { return m_value; }
59     Block& block() { return m_block; }
60 
61 private:
62     Optional<ConstantExpression> m_value;
63     Block m_block;
64 };
65 
66 } // namespace AST
67 
68 }
69 
70 }
71 
72 DEFINE_DEFAULT_DELETE(SwitchCase)
73 
74 SPECIALIZE_TYPE_TRAITS_WHLSL_STATEMENT(SwitchCase, isSwitchCase())
