31 namespace WebCore {
32 
33 class Storage;
34 
35 class StorageEvent : public Event {
36 public:
37     static Ref<StorageEvent> create(const AtomString& type, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
38     static Ref<StorageEvent> createForBindings();
39 
40     struct Init : EventInit {
41         String key;
42         String oldValue;
43         String newValue;
44         String url;
45         RefPtr<Storage> storageArea;
46     };
47 
48     static Ref<StorageEvent> create(const AtomString&, const Init&, IsTrusted = IsTrusted::No);
49     virtual ~StorageEvent();
50 
51     const String& key() const { return m_key; }
52     const String& oldValue() const { return m_oldValue; }
53     const String& newValue() const { return m_newValue; }
54     const String& url() const { return m_url; }
55     Storage* storageArea() const { return m_storageArea.get(); }
56 
57     void initStorageEvent(const AtomString& type, bool canBubble, bool cancelable, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
58 
59     // Needed once we support init<blank>EventNS
60     // void initStorageEventNS(in DOMString namespaceURI, in DOMString typeArg, in boolean canBubbleArg, in boolean cancelableArg, in DOMString keyArg, in DOMString oldValueArg, in DOMString newValueArg, in DOMString urlArg, Storage storageAreaArg);
61 
62     EventInterface eventInterface() const override;
63 
64 private:
65     StorageEvent();
66     StorageEvent(const AtomString& type, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
67     StorageEvent(const AtomString&, const Init&, IsTrusted);
68 
69     String m_key;
70     String m_oldValue;
71     String m_newValue;
72     String m_url;
73     RefPtr<Storage> m_storageArea;
74 };
