30 namespace WebCore {

32 class WebGLExtension {
33     WTF_MAKE_FAST_ALLOCATED;
34 public:
35     // Extension names are needed to properly wrap instances in JavaScript objects.
36     enum ExtensionName {
37         WebGLLoseContextName,
38         EXTBlendMinMaxName,
39         EXTFragDepthName,
40         EXTShaderTextureLODName,
41         EXTTextureFilterAnisotropicName,
42         EXTsRGBName,
43         OESTextureFloatName,
44         OESTextureFloatLinearName,
45         OESTextureHalfFloatName,
46         OESTextureHalfFloatLinearName,
47         OESStandardDerivativesName,
48         OESVertexArrayObjectName,
49         WebGLDebugRendererInfoName,
50         WebGLDebugShadersName,
51         WebGLCompressedTextureS3TCName,
52         WebGLDepthTextureName,
53         WebGLDrawBuffersName,
54         OESElementIndexUintName,
55         WebGLCompressedTextureATCName,
56         WebGLCompressedTexturePVRTCName,
57         WebGLCompressedTextureASTCName,
58         ANGLEInstancedArraysName,
59     };
60 
61     void ref() { m_context.ref(); }
62     void deref() { m_context.deref(); }
63     WebGLRenderingContextBase& context() { return m_context; }
64 
65     virtual ~WebGLExtension();
66     virtual ExtensionName getName() const = 0;
67 
68 protected:
69     WebGLExtension(WebGLRenderingContextBase&);
70     WebGLRenderingContextBase& m_context;
71 };

73 } // namespace WebCore
