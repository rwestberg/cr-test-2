33 BiquadFilterNode::BiquadFilterNode(AudioContext& context, float sampleRate)
34     : AudioBasicProcessorNode(context, sampleRate)
35 {
36     // Initially setup as lowpass filter.
37     m_processor = std::make_unique<BiquadProcessor>(context, sampleRate, 1, false);
38     setNodeType(NodeTypeBiquadFilter);
39 }
40 
41 BiquadFilterType BiquadFilterNode::type() const
42 {
43     return const_cast<BiquadFilterNode*>(this)->biquadProcessor()->type();
44 }
45 
46 void BiquadFilterNode::setType(BiquadFilterType type)
47 {
48     biquadProcessor()->setType(type);
49 }
50 
51 void BiquadFilterNode::getFrequencyResponse(const RefPtr<Float32Array>& frequencyHz, const RefPtr<Float32Array>& magResponse, const RefPtr<Float32Array>& phaseResponse)
52 {
53     if (!frequencyHz || !magResponse || !phaseResponse)
54         return;
55 
56     int n = std::min(frequencyHz->length(), std::min(magResponse->length(), phaseResponse->length()));
57 
58     if (n)
59         biquadProcessor()->getFrequencyResponse(n, frequencyHz->data(), magResponse->data(), phaseResponse->data());
60 }
