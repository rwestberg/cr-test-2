31 namespace WebCore {

33 Ref<WebGPUTextureView> WebGPUTextureView::create(RefPtr<GPUTexture>&& view)
34 {
35     return adoptRef(*new WebGPUTextureView(WTFMove(view)));
36 }
37 
38 WebGPUTextureView::WebGPUTextureView(RefPtr<GPUTexture>&& view)
39     : m_texture(WTFMove(view))
40 {
41 }
42 
43 void WebGPUTextureView::destroy()
44 {
45     if (!m_texture)
46         return;
47 
48     m_texture->destroy();
49     m_texture = nullptr;
50 }
51 
52 } // namespace WebCore
