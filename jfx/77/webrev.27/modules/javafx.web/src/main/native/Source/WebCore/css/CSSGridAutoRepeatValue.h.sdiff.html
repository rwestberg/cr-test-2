38 // CSSGridAutoRepeatValue stores the track sizes and line numbers when the auto-repeat
39 // syntax is used
40 //
41 // Right now the auto-repeat syntax is as follows:
42 // <auto-repeat>  = repeat( [ auto-fill | auto-fit ], <line-names>? <fixed-size> <line-names>? )
43 //
44 // meaning that only one fixed size track is allowed. It could be argued that a different
45 // class storing two CSSGridLineNamesValue and one CSSValue (for the track size) fits
46 // better but the CSSWG has left the door open to allow more than one track in the
47 // future. That's why we're using a list, it's prepared for future changes and it also
48 // allows us to keep the parsing algorithm almost intact.
49 class CSSGridAutoRepeatValue final : public CSSValueList {
50 public:
51     static Ref<CSSGridAutoRepeatValue> create(CSSValueID id)
52     {
53         return adoptRef(*new CSSGridAutoRepeatValue(id));
54     }
55 
56     String customCSSText() const;


57     CSSValueID autoRepeatID() const { return m_autoRepeatID; }
58 
59 private:
60     CSSGridAutoRepeatValue(CSSValueID id)
61         : CSSValueList(GridAutoRepeatClass, SpaceSeparator)
62         , m_autoRepeatID(id)
63     {
64         ASSERT(id == CSSValueAutoFill || id == CSSValueAutoFit);
65     }
66 
67     const CSSValueID m_autoRepeatID;
68 };
69 
70 } // namespace WebCore
71 
72 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSGridAutoRepeatValue, isGridAutoRepeatValue());
