149 function(add_library target type)
150     _add_library(${ARGV})
151     if ("${type}" STREQUAL "SHARED")
152         sign(${target})
153     endif ()
154 endfunction()

156 macro(WEBKIT_EXECUTABLE _target)
157     _WEBKIT_EXECUTABLE(${_target})
158     playstation_setup_libc(${_target})
159     playstation_setup_fp(${_target})
160     if (NOT ${_target} MATCHES "^LLInt")
161         sign(${_target})
162     endif ()
163     if (PLAYSTATION_${_target}_WRAP)
164         foreach (WRAP ${PLAYSTATION_${_target}_WRAP})
165             target_link_options(${_target} PRIVATE -Wl,--wrap=${WRAP})
166         endforeach ()
167     endif ()
168 endmacro()
