26 #ifndef PEImage_h
27 #define PEImage_h
28 
29 #include <windows.h>
30 
31 namespace WebCore {
32 
33 class PEImage {
34 public:
35     explicit PEImage(HMODULE);
36 
37     bool isValid() const { return m_ntHeaders; }
38 
39     // A relative virtual address (RVA) is an address relative to the base of the image. This
40     // function will convert an RVA to an non-relative address, which can then be dereferenced.
41     const void* convertRVAToAddress(DWORD) const;
42 
43     // Returns the address of the data corresponding to the given entry in the image's
44     // DataDirectory.
45     const void* dataDirectoryEntryAddress(DWORD entryIndex) const;
46 
47 private:
48     HMODULE m_module;
49     const IMAGE_NT_HEADERS* m_ntHeaders;
50 };
51 
52 } // namespace WebCore
53 
54 #endif // PEImage_h
