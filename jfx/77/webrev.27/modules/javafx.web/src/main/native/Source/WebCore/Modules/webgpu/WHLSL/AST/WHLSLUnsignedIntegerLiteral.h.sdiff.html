43     UnsignedIntegerLiteral(CodeLocation location, unsigned value)
44         : Expression(location, Kind::UnsignedIntegerLiteral)
45         , m_type(location, value)
46         , m_value(value)
47     {
48     }

57 
58     UnsignedIntegerLiteralType& type() { return m_type; }
59     unsigned value() const { return m_value; }
60 


61     UnsignedIntegerLiteral clone() const
62     {
63         UnsignedIntegerLiteral result(codeLocation(), m_value);
64         result.m_type = m_type.clone();
65         if (auto* resolvedType = m_type.maybeResolvedType())
66             result.m_type.resolve(const_cast<AST::UnnamedType&>(*resolvedType));
67         copyTypeTo(result);
68         return result;
69     }
70 
71     int64_t valueForSelectedType() const;
72 
73 private:
74     UnsignedIntegerLiteralType m_type;
75     unsigned m_value;
