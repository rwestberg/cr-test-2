36 JSValue Bytecode::toJS(ExecState* exec) const
37 {
38     VM& vm = exec->vm();
39     JSObject* result = constructEmptyObject(exec);
40     result->putDirect(vm, vm.propertyNames->bytecodeIndex, jsNumber(m_bytecodeIndex));
41     result->putDirect(vm, vm.propertyNames->opcode, jsString(vm, String::fromUTF8(opcodeNames[m_opcodeID])));
42     result->putDirect(vm, vm.propertyNames->description, jsString(vm, String::fromUTF8(m_description)));
43     return result;
44 }
