35 JSWebAssemblyLinkError* JSWebAssemblyLinkError::create(ExecState* exec, VM& vm, Structure* structure, const String& message)
36 {
37     auto* instance = new (NotNull, allocateCell<JSWebAssemblyLinkError>(vm.heap)) JSWebAssemblyLinkError(vm, structure);
38     instance->m_sourceAppender = defaultSourceAppender;
39     bool useCurrentFrame = true;
40     instance->finishCreation(exec, vm, message, useCurrentFrame);
41     return instance;
42 }
43 
44 JSWebAssemblyLinkError::JSWebAssemblyLinkError(VM& vm, Structure* structure)
45     : Base(vm, structure)
46 {
47 }
48 
49 const ClassInfo JSWebAssemblyLinkError::s_info = { "WebAssembly.LinkError", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(JSWebAssemblyLinkError) };
50 
51 
52 JSObject* createJSWebAssemblyLinkError(ExecState* exec, VM& vm, const String& message)
53 {
54     ASSERT(!message.isEmpty());
55     JSGlobalObject* globalObject = exec->lexicalGlobalObject();
56     return JSWebAssemblyLinkError::create(exec, vm, globalObject->webAssemblyLinkErrorStructure(), message);
57 }
