26 #include "config.h"
27 #include "SessionID.h"
28 
29 #include <wtf/MainThread.h>
30 
31 namespace PAL {
32 
33 static uint64_t currentPersistentID = SessionID::DefaultSessionID;
34 static uint64_t currentEphemeralID = SessionID::LegacyPrivateSessionID;
35 static bool generationProtectionEnabled;
36 
37 SessionID SessionID::generatePersistentSessionID()
38 {
39     ASSERT(isMainThread());
40     RELEASE_ASSERT(!generationProtectionEnabled);
41 
42     return SessionID(++currentPersistentID);
43 }
44 
45 SessionID SessionID::generateEphemeralSessionID()
46 {
47     ASSERT(isMainThread());
48     RELEASE_ASSERT(!generationProtectionEnabled);
49 
50     return SessionID(++currentEphemeralID);
51 }
52 
53 void SessionID::enableGenerationProtection()
54 {
55     generationProtectionEnabled = true;
56 }
