34 HashMap<GlobalWindowIdentifier, AbstractDOMWindow*>& AbstractDOMWindow::allWindows()
35 {
36     ASSERT(isMainThread());
37     static NeverDestroyed<HashMap<GlobalWindowIdentifier, AbstractDOMWindow*>> map;
38     return map;

41 AbstractDOMWindow::AbstractDOMWindow(GlobalWindowIdentifier&& identifier)
42     : m_identifier(WTFMove(identifier))
