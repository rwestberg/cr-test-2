53 void HTMLDivElement::collectStyleForPresentationAttribute(const QualifiedName& name, const AtomicString& value, MutableStyleProperties& style)
54 {
55     if (name == alignAttr) {
56         if (equalLettersIgnoringASCIICase(value, "middle") || equalLettersIgnoringASCIICase(value, "center"))
57             addPropertyToPresentationAttributeStyle(style, CSSPropertyTextAlign, CSSValueWebkitCenter);
58         else if (equalLettersIgnoringASCIICase(value, "left"))
59             addPropertyToPresentationAttributeStyle(style, CSSPropertyTextAlign, CSSValueWebkitLeft);
60         else if (equalLettersIgnoringASCIICase(value, "right"))
61             addPropertyToPresentationAttributeStyle(style, CSSPropertyTextAlign, CSSValueWebkitRight);
62         else
63             addPropertyToPresentationAttributeStyle(style, CSSPropertyTextAlign, value);
64     } else
65         HTMLElement::collectStyleForPresentationAttribute(name, value, style);
66 }
