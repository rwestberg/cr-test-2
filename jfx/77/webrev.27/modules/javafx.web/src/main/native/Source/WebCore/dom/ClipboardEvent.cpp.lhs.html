23 #include "config.h"
24 #include "ClipboardEvent.h"

26 #include "DataTransfer.h"

28 namespace WebCore {
29 
30 ClipboardEvent::ClipboardEvent(const AtomicString& type, Ref<DataTransfer>&& dataTransfer)
31     : Event(type, CanBubble::Yes, IsCancelable::Yes, IsComposed::Yes)
32     , m_clipboardData(WTFMove(dataTransfer))
33 {
34 }
35 
36 ClipboardEvent::ClipboardEvent(const AtomicString& type, const Init& init)
37     : Event(type, init, IsTrusted::No)
38     , m_clipboardData(init.clipboardData)
39 {
40 }
41 
42 ClipboardEvent::~ClipboardEvent() = default;
43 
44 EventInterface ClipboardEvent::eventInterface() const
45 {
46     return ClipboardEventInterfaceType;
47 }
48 
49 bool ClipboardEvent::isClipboardEvent() const
50 {
51     return true;
52 }
