29 
30 #include "FrameLoaderTypes.h"
31 #include "ScrollTypes.h"
32 #include <wtf/Forward.h>
33 #include <wtf/Noncopyable.h>
34 
35 namespace WebCore {
36 
37 class Frame;
38 class FrameLoadRequest;
39 class Page;
40 class PlatformKeyboardEvent;
41 class PlatformMouseEvent;
42 class PlatformWheelEvent;
43 
44 // Real user inputs come from WebKit or WebKit2.
45 // Synthetic inputs come from within WebCore (i.e., from web replay or fake mouse moves).
46 enum class InputSource {
47     User,
48     Synthetic
49 };
50 
51 class UserInputBridge {
52     WTF_MAKE_FAST_ALLOCATED;
53     WTF_MAKE_NONCOPYABLE(UserInputBridge);
54 public:
55     UserInputBridge(Page&);
56 
57     // User input APIs.
58 #if ENABLE(CONTEXT_MENUS)
59     WEBCORE_EXPORT bool handleContextMenuEvent(const PlatformMouseEvent&, Frame&, InputSource = InputSource::User);
60 #endif
61     WEBCORE_EXPORT bool handleMousePressEvent(const PlatformMouseEvent&, InputSource source = InputSource::User);
62     WEBCORE_EXPORT bool handleMouseReleaseEvent(const PlatformMouseEvent&, InputSource source = InputSource::User);
63     WEBCORE_EXPORT bool handleMouseMoveEvent(const PlatformMouseEvent&, InputSource source = InputSource::User);
64     WEBCORE_EXPORT bool handleMouseMoveOnScrollbarEvent(const PlatformMouseEvent&, InputSource source = InputSource::User);
65     WEBCORE_EXPORT bool handleMouseForceEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
66     WEBCORE_EXPORT bool handleWheelEvent(const PlatformWheelEvent&, InputSource source = InputSource::User);
67     WEBCORE_EXPORT bool handleKeyEvent(const PlatformKeyboardEvent&, InputSource source = InputSource::User);
68     WEBCORE_EXPORT bool handleAccessKeyEvent(const PlatformKeyboardEvent&, InputSource source = InputSource::User);
69     void focusSetActive(bool active, InputSource source = InputSource::User);
70     void focusSetFocused(bool focused, InputSource source = InputSource::User);
71     WEBCORE_EXPORT bool scrollRecursively(ScrollDirection, ScrollGranularity, InputSource source = InputSource::User);
72     WEBCORE_EXPORT bool logicalScrollRecursively(ScrollLogicalDirection, ScrollGranularity, InputSource source = InputSource::User);
73 
74     // Navigation APIs.
75     WEBCORE_EXPORT void loadRequest(FrameLoadRequest&&, InputSource = InputSource::User);
76     WEBCORE_EXPORT void reloadFrame(Frame*, OptionSet<ReloadOption>, InputSource = InputSource::User);
77     WEBCORE_EXPORT void stopLoadingFrame(Frame*, InputSource source = InputSource::User);
78     WEBCORE_EXPORT bool tryClosePage(InputSource source = InputSource::User);
79 
80 private:
81     Page& m_page;
82 };
83 
84 } // namespace WebCore
