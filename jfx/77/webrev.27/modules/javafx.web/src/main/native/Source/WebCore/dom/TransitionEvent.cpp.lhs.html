28 #include "TransitionEvent.h"

30 namespace WebCore {
31 
32 TransitionEvent::TransitionEvent(const AtomicString& type, const String& propertyName, double elapsedTime, const String& pseudoElement)
33     : Event(type, CanBubble::Yes, IsCancelable::Yes)
34     , m_propertyName(propertyName)
35     , m_elapsedTime(elapsedTime)
36     , m_pseudoElement(pseudoElement)
37 {
38 }
39 
40 TransitionEvent::TransitionEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
41     : Event(type, initializer, isTrusted)
42     , m_propertyName(initializer.propertyName)
43     , m_elapsedTime(initializer.elapsedTime)
44     , m_pseudoElement(initializer.pseudoElement)
45 {
46 }
47 
48 TransitionEvent::~TransitionEvent() = default;
49 
50 const String& TransitionEvent::propertyName() const
51 {
52     return m_propertyName;
53 }
54 
55 double TransitionEvent::elapsedTime() const
56 {
57     return m_elapsedTime;
58 }
59 
60 const String& TransitionEvent::pseudoElement() const
61 {
62     return m_pseudoElement;
63 }
64 
65 EventInterface TransitionEvent::eventInterface() const
66 {
67     return TransitionEventInterfaceType;
68 }
