31 #pragma once
32 
33 #include <wtf/Forward.h>
34 
35 namespace WebCore {
36 
37 class WebSocketChannelClient {
38 public:
39     virtual ~WebSocketChannelClient() = default;
40     virtual void didConnect() = 0;
41     virtual void didReceiveMessage(const String&) = 0;
42     virtual void didReceiveBinaryData(Vector<uint8_t>&&) = 0;
43     virtual void didReceiveMessageError() = 0;
44     virtual void didUpdateBufferedAmount(unsigned bufferedAmount) = 0;
45     virtual void didStartClosingHandshake() = 0;
46     enum ClosingHandshakeCompletionStatus {
47         ClosingHandshakeIncomplete,
48         ClosingHandshakeComplete
49     };
50     virtual void didClose(unsigned unhandledBufferedAmount, ClosingHandshakeCompletionStatus, unsigned short code, const String& reason) = 0;
51     virtual void didUpgradeURL() = 0;
52 
53 protected:
54     WebSocketChannelClient() = default;
55 };
56 
57 } // namespace WebCore
