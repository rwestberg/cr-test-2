305     struct InspectorEventListener {
306         int identifier { 1 };
307         RefPtr<EventTarget> eventTarget;
308         RefPtr<EventListener> eventListener;
309         AtomicString eventType;
310         bool useCapture { false };
311         bool disabled { false };
312         bool hasBreakpoint { false };
313 
314         InspectorEventListener() { }
315 
316         InspectorEventListener(int identifier, EventTarget& target, const AtomicString& type, EventListener& listener, bool capture)
317             : identifier(identifier)
318             , eventTarget(&target)
319             , eventListener(&listener)
320             , eventType(type)
321             , useCapture(capture)
322         {
323         }
324 
325         bool matches(EventTarget& target, const AtomicString& type, EventListener& listener, bool capture)
326         {
327             if (eventTarget.get() != &target)
328                 return false;
329             if (eventListener.get() != &listener)
330                 return false;
331             if (eventType != type)
332                 return false;
333             if (useCapture != capture)
334                 return false;
335             return true;
336         }
337     };

281     struct InspectorEventListener {
282         int identifier { 1 };
283         RefPtr<EventTarget> eventTarget;
284         RefPtr<EventListener> eventListener;
285         AtomString eventType;
286         bool useCapture { false };
287         bool disabled { false };
288         bool hasBreakpoint { false };
289 
290         InspectorEventListener() { }
291 
292         InspectorEventListener(int identifier, EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
293             : identifier(identifier)
294             , eventTarget(&target)
295             , eventListener(&listener)
296             , eventType(type)
297             , useCapture(capture)
298         {
299         }
300 
301         bool matches(EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
302         {
303             if (eventTarget.get() != &target)
304                 return false;
305             if (eventListener.get() != &listener)
306                 return false;
307             if (eventType != type)
308                 return false;
309             if (useCapture != capture)
310                 return false;
311             return true;
312         }
313     };
