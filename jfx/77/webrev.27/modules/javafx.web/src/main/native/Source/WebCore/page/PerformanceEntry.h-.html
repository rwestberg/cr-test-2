32 #pragma once
33 
34 #include "Performance.h"
35 #include <wtf/Optional.h>
36 #include <wtf/RefCounted.h>
37 #include <wtf/text/WTFString.h>
38 
39 namespace WebCore {
40 
41 class PerformanceEntry : public RefCounted<PerformanceEntry> {
42 public:
43     virtual ~PerformanceEntry();
44 
45     String name() const { return m_name; }
46     String entryType() const { return m_entryType; }
47     double startTime() const { return m_startTime; }
48     double duration() const { return m_duration; }
49 
50     enum class Type {
51         Navigation = 1 << 0,
52         Mark = 1 << 1,
53         Measure = 1 << 2,
54         Resource = 1 << 3,
55     };
56 
57     Type type() const { return m_type; }
58 
59     static Optional<Type> parseEntryTypeString(const String& entryType);
60 
61     bool isResource() const { return m_type == Type::Resource; }
62     bool isMark() const { return m_type == Type::Mark; }
63     bool isMeasure() const { return m_type == Type::Measure; }
64 
65     static bool startTimeCompareLessThan(const RefPtr<PerformanceEntry>& a, const RefPtr<PerformanceEntry>& b)
66     {
67         return a->startTime() < b->startTime();
68     }
69 
70 protected:
71     PerformanceEntry(Type, const String& name, const String& entryType, double startTime, double finishTime);
72 
73 private:
74     const String m_name;
75     const String m_entryType;
76     const double m_startTime;
77     const double m_duration;
78     const Type m_type;
79 };
80 
81 } // namespace WebCore
