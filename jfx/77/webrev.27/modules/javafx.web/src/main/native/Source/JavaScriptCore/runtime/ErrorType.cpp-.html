26 #include "config.h"
27 #include "ErrorType.h"

29 namespace JSC {
30 
31 ASCIILiteral errorTypeName(ErrorType errorType)
32 {
33     static const ASCIILiteral errorTypeNames[] = {
34         "Error"_s,
35         "EvalError"_s,
36         "RangeError"_s,
37         "ReferenceError"_s,
38         "SyntaxError"_s,
39         "TypeError"_s,
40         "URIError"_s,
41     };
42     return errorTypeNames[static_cast<unsigned>(errorType)];
43 }
44 
45 } // namespace JSC
46 
47 namespace WTF {
48 
49 void printInternal(PrintStream& out, JSC::ErrorType errorType)
50 {
51     out.print(JSC::errorTypeName(errorType));
52 }

54 } // namespace WTF
