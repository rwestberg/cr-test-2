 34 namespace WebCore {

 36 using namespace HTMLNames;
 37 
 38 AccessibilityMenuListOption::AccessibilityMenuListOption()
 39 {
 40 }
 41 
 42 void AccessibilityMenuListOption::setElement(HTMLElement* element)
 43 {
 44     ASSERT_ARG(element, is<HTMLOptionElement>(element));
 45     m_element = element;
 46 }
 47 
 48 Element* AccessibilityMenuListOption::actionElement() const
 49 {
 50     return m_element.get();
 51 }
 52 
 53 bool AccessibilityMenuListOption::isEnabled() const
 54 {
 55     // isDisabledFormControl() returns true if the parent <select> element is disabled,
 56     // which we don't want.
 57     return !downcast<HTMLOptionElement>(*m_element).ownElementDisabled();
 58 }
 59 
 60 bool AccessibilityMenuListOption::isVisible() const
 61 {
 62     if (!m_parent)
 63         return false;
 64 
 65     // In a single-option select with the popup collapsed, only the selected
 66     // item is considered visible.
 67     return !m_parent->isOffScreen() || isSelected();
 68 }
 69 
 70 bool AccessibilityMenuListOption::isOffScreen() const
 71 {
 72     // Invisible list options are considered to be offscreen.
 73     return !isVisible();
 74 }
 75 
 76 bool AccessibilityMenuListOption::isSelected() const
 77 {
 78     return downcast<HTMLOptionElement>(*m_element).selected();
 79 }
 80 
 81 void AccessibilityMenuListOption::setSelected(bool selected)
 82 {
 83     if (!canSetSelectedAttribute())
 84         return;
 85 
 86     if (selected && dispatchAccessibilityEventWithType(AccessibilityEventType::Select))
 87         return;
 88 
 89     downcast<HTMLOptionElement>(*m_element).setSelected(selected);
 90 }
 91 
 92 String AccessibilityMenuListOption::nameForMSAA() const
 93 {
 94     return stringValue();
 95 }
 96 
 97 bool AccessibilityMenuListOption::canSetSelectedAttribute() const
 98 {
 99     return isEnabled();
100 }
101 
102 bool AccessibilityMenuListOption::computeAccessibilityIsIgnored() const
103 {
104     return accessibilityIsIgnoredByDefault();
105 }
106 
107 LayoutRect AccessibilityMenuListOption::elementRect() const
108 {
109     AccessibilityObject* parent = parentObject();
110     if (!parent)
111         return boundingBoxRect();
112     ASSERT(parent->isMenuListPopup());
113 
114     AccessibilityObject* grandparent = parent->parentObject();
115     if (!grandparent)
116         return boundingBoxRect();
117     ASSERT(grandparent->isMenuList());
118 
119     return grandparent->elementRect();
120 }
121 
122 String AccessibilityMenuListOption::stringValue() const
123 {
124     return downcast<HTMLOptionElement>(*m_element).label();
125 }
