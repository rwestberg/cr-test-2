38 inline HTMLTableCaptionElement::HTMLTableCaptionElement(const QualifiedName& tagName, Document& document)
39     : HTMLElement(tagName, document)
40 {
41     ASSERT(hasTagName(captionTag));
42 }
43 
44 Ref<HTMLTableCaptionElement> HTMLTableCaptionElement::create(const QualifiedName& tagName, Document& document)
45 {
46     return adoptRef(*new HTMLTableCaptionElement(tagName, document));
47 }
48 
49 void HTMLTableCaptionElement::collectStyleForPresentationAttribute(const QualifiedName& name, const AtomicString& value, MutableStyleProperties& style)
50 {
51     if (name == alignAttr) {
52         if (!value.isEmpty())
53             addPropertyToPresentationAttributeStyle(style, CSSPropertyCaptionSide, value);
54     } else
55         HTMLElement::collectStyleForPresentationAttribute(name, value, style);
56 }
