26 #ifndef HostWindow_h
27 #define HostWindow_h
28 
29 #include "Widget.h"
30 
31 namespace WebCore {
32 
33 class Cursor;
34 
35 class HostWindow {
36     WTF_MAKE_NONCOPYABLE(HostWindow); WTF_MAKE_FAST_ALLOCATED;
37 public:
38     HostWindow() = default;
39     virtual ~HostWindow() = default;
40 
41     // Requests the host invalidate the root view, not the contents.
42     virtual void invalidateRootView(const IntRect& updateRect) = 0;
43 
44     // Requests the host invalidate the contents and the root view.
45     virtual void invalidateContentsAndRootView(const IntRect& updateRect) = 0;
46 
47     // Requests the host scroll backingstore by the specified delta, rect to scroll, and clip rect.
48     virtual void scroll(const IntSize& scrollDelta, const IntRect& rectToScroll, const IntRect& clipRect) = 0;
49 
50     // Requests the host invalidate the contents, not the root view. This is the slow path for scrolling.
51     virtual void invalidateContentsForSlowScroll(const IntRect& updateRect) = 0;
52 
53     // Methods for doing coordinate conversions to and from screen coordinates.
54     virtual IntPoint screenToRootView(const IntPoint&) const = 0;
55     virtual IntRect rootViewToScreen(const IntRect&) const = 0;
56     virtual IntPoint accessibilityScreenToRootView(const IntPoint&) const = 0;
57     virtual IntRect rootViewToAccessibilityScreen(const IntRect&) const = 0;
58 
59     // Method for retrieving the native client of the page.
60     virtual PlatformPageClient platformPageClient() const = 0;
61 
62     // Request that the cursor change.
63     virtual void setCursor(const Cursor&) = 0;
64 
65     virtual void setCursorHiddenUntilMouseMoves(bool) = 0;
66 
67     virtual void scheduleAnimation() = 0;
68 
69     virtual PlatformDisplayID displayID() const = 0;
70     virtual void windowScreenDidChange(PlatformDisplayID) = 0;
71 
72     virtual FloatSize screenSize() const = 0;
73     virtual FloatSize availableScreenSize() const = 0;
74     virtual FloatSize overrideScreenSize() const = 0;
75 };
76 
77 } // namespace WebCore
78 
79 #endif // HostWindow_h
