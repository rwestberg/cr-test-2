38 JSToWasmICCallee* JSToWasmICCallee::create(VM& vm, JSGlobalObject* globalObject, WebAssemblyFunction* function)
39 {
40     auto* structure = globalObject->jsToWasmICCalleeStructure();
41     JSToWasmICCallee* result = new (NotNull, allocateCell<JSToWasmICCallee>(vm.heap)) JSToWasmICCallee(vm, globalObject, structure);
42     result->finishCreation(vm);
43     result->m_function.set(vm, result, function);
44     return result;
45 }
46 
47 Structure* JSToWasmICCallee::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
48 {
49     return Structure::create(vm, globalObject, prototype, TypeInfo(JSCalleeType, StructureFlags), info());
50 }
51 
52 void JSToWasmICCallee::visitChildren(JSCell* cell, SlotVisitor& visitor)
53 {
54     JSToWasmICCallee* thisObject = jsCast<JSToWasmICCallee*>(cell);
55     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
56 
57     Base::visitChildren(thisObject, visitor);
58     visitor.append(thisObject->m_function);
59 }
