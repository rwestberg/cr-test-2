26 #pragma once
27 
28 #include <type_traits>
29 
30 namespace WTF {
31 
32 template<typename KeyTypeArg, typename ValueTypeArg>
33 struct KeyValuePair {
34     using KeyType = KeyTypeArg;
35     using ValueType = ValueTypeArg;
36 
37     KeyValuePair()
38     {
39     }
40 
41     template<typename K, typename V>
42     KeyValuePair(K&& key, V&& value)
43         : key(std::forward<K>(key))
44         , value(std::forward<V>(value))
45     {
46     }
47 
48     template <typename K, typename V>
49     KeyValuePair(KeyValuePair<K, V>&& other)
50         : key(std::forward<K>(other.key))
51         , value(std::forward<V>(other.value))
52     {
53     }
54 
55     KeyType key;
56     ValueType value { };
57 };
58 
59 template<typename K, typename V>
60 inline KeyValuePair<typename std::decay<K>::type, typename std::decay<V>::type> makeKeyValuePair(K&& key, V&& value)
61 {
62     return KeyValuePair<typename std::decay<K>::type, typename std::decay<V>::type> { std::forward<K>(key), std::forward<V>(value) };
63 }
64 
65 }
66 
67 using WTF::KeyValuePair;
