26 #pragma once
27 
28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class OverflowEvent final : public Event {
33 public:
34     enum orientType {
35         HORIZONTAL = 0,
36         VERTICAL   = 1,
37         BOTH       = 2
38     };
39 
40     static Ref<OverflowEvent> create(bool horizontalOverflowChanged, bool horizontalOverflow, bool verticalOverflowChanged, bool verticalOverflow)
41     {
42         return adoptRef(*new OverflowEvent(horizontalOverflowChanged, horizontalOverflow, verticalOverflowChanged, verticalOverflow));
43     }
44 
45     static Ref<OverflowEvent> createForBindings()
46     {
47         return adoptRef(*new OverflowEvent);
48     }
49 
50     struct Init : EventInit {
51         unsigned short orient { 0 };
52         bool horizontalOverflow { false };
53         bool verticalOverflow { false };
54     };
55 
56     static Ref<OverflowEvent> create(const AtomicString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
57     {
58         return adoptRef(*new OverflowEvent(type, initializer, isTrusted));
59     }
60 
61     WEBCORE_EXPORT void initOverflowEvent(unsigned short orient, bool horizontalOverflow, bool verticalOverflow);
62 
63     unsigned short orient() const { return m_orient; }
64     bool horizontalOverflow() const { return m_horizontalOverflow; }
65     bool verticalOverflow() const { return m_verticalOverflow; }
66 
67     EventInterface eventInterface() const override;
68 
69 private:
70     OverflowEvent();
71     OverflowEvent(bool horizontalOverflowChanged, bool horizontalOverflow, bool verticalOverflowChanged, bool verticalOverflow);
72     OverflowEvent(const AtomicString&, const Init&, IsTrusted);
73 
74     unsigned short m_orient;
75     bool m_horizontalOverflow;
76     bool m_verticalOverflow;
77 };
78 
79 } // namespace WebCore
