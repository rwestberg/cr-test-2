  9 # Libtool library version, not to be confused with API version.
 10 # See http://www.gnu.org/software/libtool/manual/html_node/Libtool-versioning.html
 11 macro(CALCULATE_LIBRARY_VERSIONS_FROM_LIBTOOL_TRIPLE library_name current revision age)
 12     math(EXPR ${library_name}_VERSION_MAJOR "${current} - ${age}")
 13     set(${library_name}_VERSION_MINOR ${age})
 14     set(${library_name}_VERSION_MICRO ${revision})
 15     set(${library_name}_VERSION ${${library_name}_VERSION_MAJOR}.${age}.${revision})
 16 endmacro()

477 # We add --whole-archive to all libraries manually to prevent the linker from trimming
478 # symbols that we actually need later. With ld64 on darwin, we use -all_load instead.

483         foreach (library IN LISTS ${_list_name})

485         endforeach ()
486         set(${_list_name} "${${_list_name}_TMP}")
487     endif ()
488 endmacro()

412 # CMake does not automatically add --whole-archive when building shared objects from
413 # a list of convenience libraries. This can lead to missing symbols in the final output.
414 # We add --whole-archive to all libraries manually to prevent the linker from trimming
415 # symbols that we actually need later. With ld64 on darwin, we use -all_load instead.
416 macro(ADD_WHOLE_ARCHIVE_TO_LIBRARIES _list_name)
417     if (CMAKE_SYSTEM_NAME MATCHES "Darwin")
418         list(APPEND ${_list_name} -Wl,-all_load)
419     else ()
420         set(_tmp)
421         foreach (item IN LISTS ${_list_name})
422             if ("${item}" STREQUAL "PRIVATE" OR "${item}" STREQUAL "PUBLIC")
423                 list(APPEND _tmp "${item}")
424             else ()
425                 list(APPEND _tmp -Wl,--whole-archive "${item}" -Wl,--no-whole-archive)
426             endif ()
427         endforeach ()
428         set(${_list_name} ${_tmp})
429     endif ()
430 endmacro()
