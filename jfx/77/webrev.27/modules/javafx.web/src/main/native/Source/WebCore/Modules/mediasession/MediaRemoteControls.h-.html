26 #pragma once
27 
28 #if ENABLE(MEDIA_SESSION)
29 
30 #include "EventTarget.h"
31 #include <wtf/RefCounted.h>
32 
33 namespace WebCore {
34 
35 class MediaSession;
36 
37 class MediaRemoteControls : public RefCounted<MediaRemoteControls>, public EventTargetWithInlineData {
38 public:
39     static Ref<MediaRemoteControls> create(ScriptExecutionContext& context, MediaSession* session = nullptr)
40     {
41         return adoptRef(*new MediaRemoteControls(context, session));
42     }
43 
44     bool previousTrackEnabled() const { return m_previousTrackEnabled; }
45     void setPreviousTrackEnabled(bool);
46 
47     bool nextTrackEnabled() const { return m_nextTrackEnabled; }
48     void setNextTrackEnabled(bool);
49 
50     using RefCounted<MediaRemoteControls>::ref;
51     using RefCounted<MediaRemoteControls>::deref;
52 
53     void clearSession();
54 
55     virtual ~MediaRemoteControls();
56 
57     EventTargetInterface eventTargetInterface() const override { return MediaRemoteControlsEventTargetInterfaceType; }
58     ScriptExecutionContext* scriptExecutionContext() const override { return &m_scriptExecutionContext; }
59 
60 private:
61     MediaRemoteControls(ScriptExecutionContext&, MediaSession*);
62 
63     ScriptExecutionContext& m_scriptExecutionContext;
64 
65     bool m_previousTrackEnabled { false };
66     bool m_nextTrackEnabled { false };
67 
68     MediaSession* m_session { nullptr };
69 
70     void refEventTarget() final { ref(); }
71     void derefEventTarget() final { deref(); }
72 };
73 
74 } // namespace WebCore
75 
76 #endif // ENABLE(MEDIA_SESSION)
