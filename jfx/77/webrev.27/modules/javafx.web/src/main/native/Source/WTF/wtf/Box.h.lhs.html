26 #pragma once
27 
28 #include <wtf/RefPtr.h>
29 #include <wtf/ThreadSafeRefCounted.h>
30 
31 namespace WTF {
32 
33 // Box<T> is a reference-counted pointer to T that allocates T using FastMalloc and prepends a reference
34 // count to it.
35 template<typename T>
36 class Box {

37 public:
38     Box() = default;
39     Box(Box&&) = default;
40     Box(const Box&) = default;
41 
42     Box(std::nullptr_t)
43     {
44     }
45 
46     Box& operator=(Box&&) = default;
47     Box& operator=(const Box&) = default;
48 
49     template<typename... Arguments>
50     static Box create(Arguments&&... arguments)
51     {
52         Box result;
53         result.m_data = adoptRef(new Data(std::forward<Arguments>(arguments)...));
54         return result;
55     }
56 
57     T* get() const { return &m_data->value; }
58 
59     T& operator*() const { return m_data->value; }
60     T* operator->() const { return &m_data->value; }
61 
62     explicit operator bool() const { return static_cast<bool>(m_data); }
63 
64 private:
65     struct Data : ThreadSafeRefCounted<Data> {
66         template<typename... Arguments>
67         Data(Arguments&&... arguments)
68             : value(std::forward<Arguments>(arguments)...)
69         {
70         }
71 
72         T value;
73     };
74 
75     RefPtr<Data> m_data;
76 };
77 
78 } // namespace WTF
79 
80 using WTF::Box;
