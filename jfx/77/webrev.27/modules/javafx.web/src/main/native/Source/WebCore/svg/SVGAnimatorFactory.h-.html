40 namespace WebCore {
41 
42 class SVGAnimationElement;
43 
44 class SVGAnimatorFactory {
45 public:
46     static std::unique_ptr<SVGAnimatedTypeAnimator> create(SVGAnimationElement* animationElement, SVGElement* contextElement, AnimatedPropertyType attributeType)
47     {
48         ASSERT(animationElement);
49         ASSERT(contextElement);
50 
51         switch (attributeType) {
52         case AnimatedAngle:
53             return std::make_unique<SVGAnimatedAngleAnimator>(animationElement, contextElement);
54         case AnimatedBoolean:
55             return std::make_unique<SVGAnimatedBooleanAnimator>(animationElement, contextElement);
56         case AnimatedColor:
57             return std::make_unique<SVGAnimatedColorAnimator>(*animationElement, *contextElement);
58         case AnimatedEnumeration:
59             return std::make_unique<SVGAnimatedEnumerationAnimator>(animationElement, contextElement);
60         case AnimatedInteger:
61             return std::make_unique<SVGAnimatedIntegerAnimator>(animationElement, contextElement);
62         case AnimatedIntegerOptionalInteger:
63             return std::make_unique<SVGAnimatedIntegerOptionalIntegerAnimator>(animationElement, contextElement);
64         case AnimatedLength:
65             return std::make_unique<SVGAnimatedLengthAnimator>(animationElement, contextElement);
66         case AnimatedLengthList:
67             return std::make_unique<SVGAnimatedLengthListAnimator>(animationElement, contextElement);
68         case AnimatedNumber:
69             return std::make_unique<SVGAnimatedNumberAnimator>(animationElement, contextElement);
70         case AnimatedNumberList:
71             return std::make_unique<SVGAnimatedNumberListAnimator>(animationElement, contextElement);
72         case AnimatedNumberOptionalNumber:
73             return std::make_unique<SVGAnimatedNumberOptionalNumberAnimator>(animationElement, contextElement);
74         case AnimatedPath:
75             return std::make_unique<SVGAnimatedPathAnimator>(animationElement, contextElement);
76         case AnimatedPoints:
77             return std::make_unique<SVGAnimatedPointListAnimator>(animationElement, contextElement);
78         case AnimatedPreserveAspectRatio:
79             return std::make_unique<SVGAnimatedPreserveAspectRatioAnimator>(animationElement, contextElement);
80         case AnimatedRect:
81             return std::make_unique<SVGAnimatedRectAnimator>(animationElement, contextElement);
82         case AnimatedString:
83             return std::make_unique<SVGAnimatedStringAnimator>(animationElement, contextElement);
84         case AnimatedTransformList:
85             return std::make_unique<SVGAnimatedTransformListAnimator>(animationElement, contextElement);
86         case AnimatedUnknown:
87             break;
88         }
89 
90         ASSERT_NOT_REACHED();
91         return nullptr;
92     }
93 
94 private:
95     SVGAnimatorFactory() { }
96 };
97 
98 } // namespace WebCore
