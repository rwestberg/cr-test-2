37 bool RealtimeMediaSourceSupportedConstraints::supportsConstraint(MediaConstraintType constraint) const
38 {
39     switch (constraint) {
40     case MediaConstraintType::Unknown:
41         return false;
42     case MediaConstraintType::Width:
43         return supportsWidth();
44     case MediaConstraintType::Height:
45         return supportsHeight();
46     case MediaConstraintType::AspectRatio:
47         return supportsAspectRatio();
48     case MediaConstraintType::FrameRate:
49         return supportsFrameRate();
50     case MediaConstraintType::FacingMode:
51         return supportsFacingMode();
52     case MediaConstraintType::Volume:
53         return supportsVolume();
54     case MediaConstraintType::SampleRate:
55         return supportsSampleRate();
56     case MediaConstraintType::SampleSize:
57         return supportsSampleSize();
58     case MediaConstraintType::EchoCancellation:
59         return supportsEchoCancellation();
60     case MediaConstraintType::DeviceId:
61         return supportsDeviceId();
62     case MediaConstraintType::GroupId:
63         return supportsGroupId();
64     case MediaConstraintType::DisplaySurface:
65         return supportsDisplaySurface();
66     case MediaConstraintType::LogicalSurface:
67         return supportsLogicalSurface();
68     }
69 
70     ASSERT_NOT_REACHED();
71     return false;
72 }
73 
74 } // namespace WebCore
