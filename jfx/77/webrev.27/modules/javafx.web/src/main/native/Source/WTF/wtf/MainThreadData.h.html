21 #pragma once
22 
23 #include <wtf/RunLoop.h>
24 #include <wtf/StdLibExtras.h>
25 
26 namespace WTF {
27 
28 template<typename T>
29 class MainThreadData {
30     WTF_MAKE_FAST_ALLOCATED;
31 public:
32     MainThreadData() { }
33     explicit MainThreadData(T&& data)
34         : m_data(WTFMove(data))
35     { }
36 
37     T* operator->()
38     {
39         ASSERT(isMainThread());
40         return &m_data;
41     }
42 
43     T& operator*()
44     {
45         ASSERT(isMainThread());
46         return m_data;
47     }
48 
49 private:
50     T m_data;
51 };
52 
53 } // namespace WTF
