26 #pragma once
27 
28 #include <wtf/Assertions.h>
29 
30 class CheckedBoolean {

31 public:
32 #if !ASSERT_DISABLED
33     CheckedBoolean(const CheckedBoolean& other)
34         : m_value(other.m_value)
35         , m_checked(false)
36     {
37         other.m_checked = true;
38     }
39 #endif
40 
41     CheckedBoolean(bool value)
42         : m_value(value)
43 #if !ASSERT_DISABLED
44         , m_checked(false)
45 #endif
46     {
47     }
48 
49     ~CheckedBoolean()
50     {
51         ASSERT(m_checked);
52     }
53 
54     operator bool()
55     {
56 #if !ASSERT_DISABLED
57         m_checked = true;
58 #endif
59         return m_value;
60     }
61 
62 private:
63     bool m_value;
64 #if !ASSERT_DISABLED
65     mutable bool m_checked;
66 #endif
67 };
