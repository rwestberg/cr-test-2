28 /*

36 #define WM_GESTURENOTIFY 0x011A

38 // Gesture Information Flags
39 #define GF_BEGIN 0x00000001

41 #define GF_END 0x00000004

56 // Pan Gesture Configuration Flags

58 #define GC_PAN_WITH_SINGLE_FINGER_VERTICALLY 0x00000002
59 #define GC_PAN_WITH_SINGLE_FINGER_HORIZONTALLY 0x00000004
60 #define GC_PAN_WITH_GUTTER 0x00000008
61 #define GC_PAN_WITH_INERTIA 0x00000010
62 
63 // Rotate Gesture Configuration Flags
64 #define GC_ROTATE 0x00000001
65 
66 // Two finger tap configuration flags
67 #define GC_TWOFINGERTAP 0x00000001
68

70 #define GC_PRESSANDTAP 0x00000001
