58 struct ScrollableAreaParameters {
59     ScrollElasticity horizontalScrollElasticity { ScrollElasticityNone };
60     ScrollElasticity verticalScrollElasticity { ScrollElasticityNone };
61 
62     ScrollbarMode horizontalScrollbarMode { ScrollbarAuto };
63     ScrollbarMode verticalScrollbarMode { ScrollbarAuto };
64 
65     bool hasEnabledHorizontalScrollbar { false };
66     bool hasEnabledVerticalScrollbar { false };
67 



68     bool useDarkAppearanceForScrollbars { false };
69 
70     bool operator==(const ScrollableAreaParameters& other) const
71     {
72         return horizontalScrollElasticity == other.horizontalScrollElasticity
73             && verticalScrollElasticity == other.verticalScrollElasticity
74             && horizontalScrollbarMode == other.horizontalScrollbarMode
75             && verticalScrollbarMode == other.verticalScrollbarMode
76             && hasEnabledHorizontalScrollbar == other.hasEnabledHorizontalScrollbar
77             && hasEnabledVerticalScrollbar == other.hasEnabledVerticalScrollbar
78             && useDarkAppearanceForScrollbars == other.useDarkAppearanceForScrollbars;
79     }
80 };
