26 #pragma once

28 #if ENABLE(WEBGL)
29 
30 #include "WebGLObject.h"
31 
32 namespace WebCore {
33 
34 class GraphicsContext3D;
35 class WebGLRenderingContextBase;
36 
37 // WebGLContextObject the base class for objects that are owned by a specific
38 // WebGLRenderingContext.
39 class WebGLContextObject : public WebGLObject {
40 public:
41     virtual ~WebGLContextObject();
42 
43     WebGLRenderingContextBase* context() const { return m_context; }
44 
45     bool validate(const WebGLContextGroup*, const WebGLRenderingContextBase& context) const override
46     {
47         return &context == m_context;
48     }
49 
50     void detachContext();
51 
52 protected:
53     WebGLContextObject(WebGLRenderingContextBase&);
54 
55     bool hasGroupOrContext() const override
56     {
57         return m_context;
58     }
59 
60     GraphicsContext3D* getAGraphicsContext3D() const override;
61 
62 private:
63     WebGLRenderingContextBase* m_context;
64 };
65 
66 } // namespace WebCore
67 
68 #endif
