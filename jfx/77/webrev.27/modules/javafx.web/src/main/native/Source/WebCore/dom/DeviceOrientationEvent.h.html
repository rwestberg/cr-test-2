26 #pragma once
27 
28 #include "DeviceOrientationOrMotionEvent.h"
29 #include "Event.h"
30 
31 namespace WebCore {
32 
33 class DeviceOrientationData;
34 
35 class DeviceOrientationEvent final : public Event, public DeviceOrientationOrMotionEvent {
36 public:
37     static Ref<DeviceOrientationEvent> create(const AtomString& eventType, DeviceOrientationData* orientation)
38     {
39         return adoptRef(*new DeviceOrientationEvent(eventType, orientation));
40     }
41 
42     static Ref<DeviceOrientationEvent> createForBindings()
43     {
44         return adoptRef(*new DeviceOrientationEvent);
45     }
46 
47     virtual ~DeviceOrientationEvent();
48 
49     Optional<double> alpha() const;
50     Optional<double> beta() const;
51     Optional<double> gamma() const;
52 
53 #if PLATFORM(IOS_FAMILY)
54     Optional<double> compassHeading() const;
55     Optional<double> compassAccuracy() const;
56 
57     void initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<double> compassHeading, Optional<double> compassAccuracy);
58 #else
59     Optional<bool> absolute() const;
60 
61     void initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<bool> absolute);
62 #endif
63 
64 private:
65     DeviceOrientationEvent();
66     DeviceOrientationEvent(const AtomString& eventType, DeviceOrientationData*);
67 
68     EventInterface eventInterface() const override;
69 
70     RefPtr<DeviceOrientationData> m_orientation;
71 };
72 
73 } // namespace WebCore
