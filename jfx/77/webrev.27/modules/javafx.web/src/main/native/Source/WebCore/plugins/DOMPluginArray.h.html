23 #include "DOMPlugin.h"

29 namespace WebCore {
30 
31 class PluginData;
32 
33 class DOMPluginArray final : public ScriptWrappable, public RefCounted<DOMPluginArray> {
34     WTF_MAKE_ISO_ALLOCATED(DOMPluginArray);
35 public:
36     static Ref<DOMPluginArray> create(Navigator& navigator) { return adoptRef(*new DOMPluginArray(navigator)); }
37     ~DOMPluginArray();
38 
39     unsigned length() const;
40     RefPtr<DOMPlugin> item(unsigned index);
41     RefPtr<DOMPlugin> namedItem(const AtomString& propertyName);
42     Vector<AtomString> supportedPropertyNames();
43 
44     void refresh(bool reloadPages);
45 
46     Navigator* navigator() { return m_navigator.get(); }
47 
48 private:
49     explicit DOMPluginArray(Navigator&);
50 
51     PluginData* pluginData() const;
52     Frame* frame() const { return m_navigator ? m_navigator->frame() : nullptr; }
53 
54     WeakPtr<Navigator> m_navigator;
55 };
56 
57 } // namespace WebCore
