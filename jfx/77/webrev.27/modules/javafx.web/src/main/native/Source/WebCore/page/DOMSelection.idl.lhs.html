30 // https://www.w3.org/TR/selection-api/#idl-def-Selection
31 [
32     GenerateIsReachable=ImplFrame,

33     InterfaceName=Selection,
34 ] interface DOMSelection {
35     readonly attribute Node? anchorNode;
36     readonly attribute unsigned long anchorOffset;
37     readonly attribute Node? focusNode;
38     readonly attribute unsigned long focusOffset;
39 
40     readonly attribute boolean isCollapsed;
41     readonly attribute unsigned long rangeCount;
42 
43     void collapse(Node? node, optional unsigned long offset = 0);
44     [MayThrowException] void collapseToEnd();
45     [MayThrowException] void collapseToStart();
46 
47     [CEReactions] void deleteFromDocument();
48     boolean containsNode(Node node, optional boolean allowPartial = false);
49     void selectAllChildren(Node node);
50 
51     [MayThrowException] void extend(Node node, optional unsigned long offset = 0);
52 
53     [MayThrowException] Range getRangeAt(unsigned long index);
54     void removeAllRanges();
55     void addRange(Range range);
56 
57     [NotEnumerable] DOMString toString();
58 
59     readonly attribute DOMString type;
60 
61     void setBaseAndExtent(Node? baseNode, unsigned long baseOffset, Node? extentNode, unsigned long extentOffset);
62     void setPosition(Node? node, optional unsigned long offset = 0);
63 
64     void empty();
65 
66     // FIXME: The following operation should be implemented.
67     // void removeRange(Range range);
68 
69     // FIXME: Using "undefined" as default parameter value is wrong.
70     void modify(optional DOMString alter = "undefined", optional DOMString direction = "undefined", optional DOMString granularity = "undefined");
71 
72     readonly attribute Node? baseNode;
73     readonly attribute unsigned long baseOffset;
74     readonly attribute Node? extentNode;
75     readonly attribute unsigned long extentOffset;
76 };
