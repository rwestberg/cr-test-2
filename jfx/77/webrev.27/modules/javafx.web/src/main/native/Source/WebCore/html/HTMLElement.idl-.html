21 [
22     CustomConstructor(),
23     CustomPushEventHandlerScope,
24     ExportMacro=WEBCORE_EXPORT,
25     JSGenerateToNativeObject,
26 ] interface HTMLElement : Element {
27     // Metadata attributes.
28     [CEReactions, Reflect] attribute DOMString title;
29     [CEReactions, Reflect] attribute DOMString lang;
30     [CEReactions] attribute boolean translate;
31     [CEReactions] attribute DOMString dir;
32     readonly attribute DOMStringMap dataset; // FIXME: Should be [SameObject].
33 
34     // User interaction.
35     [CEReactions, Reflect] attribute boolean hidden;
36     void click();
37     [CEReactions] attribute long tabIndex;
38     void focus();
39     void blur();
40     [CEReactions, Reflect] attribute DOMString accessKey;
41     // readonly attribute DOMString accessKeyLabel; // FIXME: Not supported.
42     [CEReactions] attribute boolean draggable;
43     // [CEReactions] attribute HTMLMenuElement? contextMenu; // FIXME: Not supported.
44     [CEReactions] attribute boolean spellcheck;
45     // void forceSpellCheck(); // FIXME: Not supported.
46 
47     [CEReactions] attribute [TreatNullAs=EmptyString] DOMString innerText;
48 
49     // FIXME: Should be in a separate ElementContentEditable interface.
50     // https://html.spec.whatwg.org/#elementcontenteditable
51     [CEReactions] attribute DOMString contentEditable;
52     readonly attribute boolean isContentEditable;
53 
54     // Extensions from CSSOM-view specification (https://drafts.csswg.org/cssom-view/#extensions-to-the-htmlelement-interface).
55     [ImplementedAs=offsetParentForBindings] readonly attribute Element? offsetParent;
56     [ImplementedAs=offsetTopForBindings] readonly attribute double offsetTop; // FIXME: Should be of type long.
57     [ImplementedAs=offsetLeftForBindings] readonly attribute double offsetLeft; // FIXME: Should be of type long.
58     readonly attribute double offsetWidth; // FIXME: Should be of type long.
59     readonly attribute double offsetHeight; // FIXME: Should be of type long.
60 
61     // Non Standard but may get added to the specification (https://github.com/whatwg/html/issues/668).
62     [CEReactions] attribute [TreatNullAs=EmptyString] DOMString outerText;
63 
64     // iOS autocorrect / autocapitalization extensions.
65     [Conditional=IOS_AUTOCORRECT_AND_AUTOCAPITALIZE, CEReactions] attribute boolean autocorrect;
66     [Conditional=IOS_AUTOCORRECT_AND_AUTOCAPITALIZE, CEReactions] attribute [TreatNullAs=EmptyString] DOMString autocapitalize;
67 
68     // FIXME: We are the only browser to support this now that Blink dropped it (http://crbug.com/688943).
69     [CEReactions, Reflect] attribute DOMString webkitdropzone;
70 
71     attribute DOMString inputMode;
72 };
73 
74 HTMLElement implements ElementCSSInlineStyle;
75 HTMLElement implements GlobalEventHandlers;
