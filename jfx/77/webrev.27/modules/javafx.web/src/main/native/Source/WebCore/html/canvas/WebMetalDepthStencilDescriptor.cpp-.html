38 bool WebMetalDepthStencilDescriptor::depthWriteEnabled() const
39 {
40     return m_descriptor.depthWriteEnabled();
41 }
42 
43 void WebMetalDepthStencilDescriptor::setDepthWriteEnabled(bool newDepthWriteEnabled)
44 {
45     m_descriptor.setDepthWriteEnabled(newDepthWriteEnabled);
46 }
47 
48 WebMetalCompareFunction WebMetalDepthStencilDescriptor::depthCompareFunction() const
49 {
50     return m_depthCompareFunction;
51 }
52 
53 void WebMetalDepthStencilDescriptor::setDepthCompareFunction(CompareFunction newDepthCompareFunction)
54 {
55     m_depthCompareFunction = newDepthCompareFunction;
56     m_descriptor.setDepthCompareFunction(toGPULegacyCompareFunction(m_depthCompareFunction));
57 }
