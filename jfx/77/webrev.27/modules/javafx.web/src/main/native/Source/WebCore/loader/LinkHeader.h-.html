27 #pragma once
28 
29 #include <wtf/text/WTFString.h>
30 
31 namespace WebCore {
32 
33 class LinkHeader {
34 public:
35     template <typename CharacterType>
36     LinkHeader(CharacterType*& position, CharacterType* const end);
37 
38     const String& url() const { return m_url; }
39     const String& rel() const { return m_rel; }
40     const String& as() const { return m_as; }
41     const String& mimeType() const { return m_mimeType; }
42     const String& media() const { return m_media; }
43     const String& crossOrigin() const { return m_crossOrigin; }
44     bool valid() const { return m_isValid; }
45 
46     enum LinkParameterName {
47         LinkParameterRel,
48         LinkParameterAnchor,
49         LinkParameterTitle,
50         LinkParameterMedia,
51         LinkParameterType,
52         LinkParameterRev,
53         LinkParameterHreflang,
54         // Beyond this point, only link-extension parameters
55         LinkParameterUnknown,
56         LinkParameterCrossOrigin,
57         LinkParameterAs,
58     };
59 
60 private:
61     void setValue(LinkParameterName, String value);
62 
63     String m_url;
64     String m_rel;
65     String m_as;
66     String m_mimeType;
67     String m_media;
68     String m_crossOrigin;
69     bool m_isValid { true };
70 };
71 
72 class LinkHeaderSet {
73 public:
74     LinkHeaderSet(const String& header);
75 
76     Vector<LinkHeader>::const_iterator begin() const { return m_headerSet.begin(); }
77     Vector<LinkHeader>::const_iterator end() const { return m_headerSet.end(); }
78 
79 private:
80     template <typename CharacterType>
81     void init(CharacterType* headerValue, size_t length);
82 
83     Vector<LinkHeader> m_headerSet;
84 };
85 
86 } // namespace WebCore
