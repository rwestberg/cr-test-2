26 #ifndef CountedUserActivity_h
27 #define CountedUserActivity_h
28 
29 #include "UserActivity.h"
30 
31 namespace WebCore {
32 
33 class CountedUserActivity {
34 public:
35     CountedUserActivity(const char* description)
36         : m_activity(description)
37         , m_count(0)
38     {
39     }
40 
41     void increment()
42     {
43         if (!m_count++)
44             m_activity.start();
45     }
46 
47     void decrement()
48     {
49         if (!--m_count)
50             m_activity.stop();
51     }
52 
53 private:
54     UserActivity m_activity;
55     size_t m_count;
56 };
57 
58 } // namespace WebCore
59 
60 using WebCore::CountedUserActivity;
61 
62 #endif // CountedUserActivity_h
