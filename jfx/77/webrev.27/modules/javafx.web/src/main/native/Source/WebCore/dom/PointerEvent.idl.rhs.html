26 dictionary PointerEventInit : MouseEventInit {
27     long pointerId = 0;
28     double width = 1;
29     double height = 1;
30     float pressure = 0;
31     float tangentialPressure = 0;
32     long tiltX = 0;
33     long tiltY = 0;
34     long twist = 0;
35     DOMString pointerType = "";
36     boolean isPrimary = false;
37 };
38 
39 [
40     Conditional=POINTER_EVENTS,
41     EnabledAtRuntime=PointerEvents,
42     DisabledByQuirk=shouldDisablePointerEvents,
43     Constructor(DOMString type, optional PointerEventInit eventInitDict),
44     Exposed=Window
45 ] interface PointerEvent : MouseEvent {
46     readonly attribute long pointerId;
47     readonly attribute double width;
48     readonly attribute double height;
49     readonly attribute float pressure;
50     readonly attribute float tangentialPressure;
51     readonly attribute long tiltX;
52     readonly attribute long tiltY;
53     readonly attribute long twist;
54     readonly attribute DOMString pointerType;
55     readonly attribute boolean isPrimary;
56 };
