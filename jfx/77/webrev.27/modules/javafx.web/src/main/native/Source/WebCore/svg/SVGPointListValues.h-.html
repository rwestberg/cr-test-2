24 #include "FloatPoint.h"
25 #include "SVGPropertyTraits.h"
26 #include <wtf/Vector.h>
27 
28 namespace WebCore {
29 
30 class SVGPoint;
31 class SVGPointList;
32 
33 class SVGPointListValues final : public Vector<FloatPoint> {
34 public:
35     String valueAsString() const;
36 };
37 
38 template<>
39 struct SVGPropertyTraits<SVGPointListValues> {
40     static SVGPointListValues initialValue() { return { }; }
41     static SVGPointListValues fromString(const String& string)
42     {
43         SVGPointListValues list;
44         pointsListFromSVGData(list, string);
45         return list;
46     }
47     static Optional<SVGPointListValues> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return { }; }
48     static String toString(const SVGPointListValues& list) { return list.valueAsString(); }
49 
50     using ListItemType = FloatPoint;
51     using ListItemTearOff = SVGPoint;
52     using ListPropertyTearOff = SVGPointList;
53 };
54 
55 } // namespace WebCore
