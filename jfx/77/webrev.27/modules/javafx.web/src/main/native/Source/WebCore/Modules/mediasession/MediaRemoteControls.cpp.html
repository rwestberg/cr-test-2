26 #include "config.h"
27 #include "MediaRemoteControls.h"
28 
29 #if ENABLE(MEDIA_SESSION)
30 
31 #include "MediaSession.h"
32 
33 namespace WebCore {
34 
35 WTF_MAKE_ISO_ALLOCATED_IMPL(MediaRemoteControls);
36 
37 MediaRemoteControls::MediaRemoteControls(ScriptExecutionContext& context, MediaSession* session)
38     : m_scriptExecutionContext(context)
39     , m_session(session)
40 {
41 }
42 
43 void MediaRemoteControls::clearSession()
44 {
45     m_session = nullptr;
46 }
47 
48 MediaRemoteControls::~MediaRemoteControls() = default;
49 
50 void MediaRemoteControls::setPreviousTrackEnabled(bool isEnabled)
51 {
52     if (m_previousTrackEnabled == isEnabled)
53         return;
54 
55     m_previousTrackEnabled = isEnabled;
56 
57     if (m_session)
58         m_session->controlIsEnabledDidChange();
59 }
60 
61 void MediaRemoteControls::setNextTrackEnabled(bool isEnabled)
62 {
63     if (m_nextTrackEnabled == isEnabled)
64         return;
65 
66     m_nextTrackEnabled = isEnabled;
67 
68     if (m_session)
69         m_session->controlIsEnabledDidChange();
70 }
71 
72 }
73 
74 #endif /* ENABLE(MEDIA_SESSION) */
