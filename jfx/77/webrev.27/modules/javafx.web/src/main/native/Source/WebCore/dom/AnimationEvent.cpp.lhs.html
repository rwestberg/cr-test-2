26 #include "config.h"
27 #include "AnimationEvent.h"
28 
29 namespace WebCore {
30 
31 AnimationEvent::AnimationEvent(const AtomicString& type, const Init& initializer, IsTrusted isTrusted)
32     : Event(type, initializer, isTrusted)
33     , m_animationName(initializer.animationName)
34     , m_elapsedTime(initializer.elapsedTime)
35 {
36 }
37 
38 AnimationEvent::AnimationEvent(const AtomicString& type, const String& animationName, double elapsedTime)
39     : Event(type, CanBubble::Yes, IsCancelable::No)
40     , m_animationName(animationName)
41     , m_elapsedTime(elapsedTime)
42 {
43 }
44 
45 AnimationEvent::~AnimationEvent() = default;
46 
47 const String& AnimationEvent::animationName() const
48 {
49     return m_animationName;
50 }
51 
52 double AnimationEvent::elapsedTime() const
53 {
54     return m_elapsedTime;
55 }
56 
57 EventInterface AnimationEvent::eventInterface() const
58 {
59     return AnimationEventInterfaceType;
60 }
