40 class TypeReference;

42 class UnsignedIntegerLiteralType : public ResolvableType {

43 public:
44     UnsignedIntegerLiteralType(Lexer::Token&& origin, unsigned value);
45 
46     virtual ~UnsignedIntegerLiteralType();
47 
48     UnsignedIntegerLiteralType(const UnsignedIntegerLiteralType&) = delete;
49     UnsignedIntegerLiteralType(UnsignedIntegerLiteralType&&);
50 
51     UnsignedIntegerLiteralType& operator=(const UnsignedIntegerLiteralType&) = delete;
52     UnsignedIntegerLiteralType& operator=(UnsignedIntegerLiteralType&&);
53 
54     bool isUnsignedIntegerLiteralType() const override { return true; }
55 
56     unsigned value() const { return m_value; }
57 
58     TypeReference& preferredType() { return m_preferredType; }
59 
60     bool canResolve(const Type&) const override;
61     unsigned conversionCost(const UnnamedType&) const override;


62 
63 private:
64     unsigned m_value;
65     // This is a unique_ptr to resolve a circular dependency between
66     // ConstantExpression -> LiteralType -> TypeReference -> TypeArguments -> ConstantExpression
67     UniqueRef<TypeReference> m_preferredType;
68 };
