36 HTMLUListElement::HTMLUListElement(const QualifiedName& tagName, Document& document)
37     : HTMLElement(tagName, document)
38 {
39     ASSERT(hasTagName(ulTag));
40 }
41 
42 Ref<HTMLUListElement> HTMLUListElement::create(Document& document)
43 {
44     return adoptRef(*new HTMLUListElement(ulTag, document));
45 }
46 
47 Ref<HTMLUListElement> HTMLUListElement::create(const QualifiedName& tagName, Document& document)
48 {
49     return adoptRef(*new HTMLUListElement(tagName, document));
50 }
51 
52 bool HTMLUListElement::isPresentationAttribute(const QualifiedName& name) const
53 {
54     if (name == typeAttr)
55         return true;
56     return HTMLElement::isPresentationAttribute(name);
57 }
58 
59 void HTMLUListElement::collectStyleForPresentationAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
60 {
61     if (name == typeAttr)
62         addPropertyToPresentationAttributeStyle(style, CSSPropertyListStyleType, value);
63     else
64         HTMLElement::collectStyleForPresentationAttribute(name, value, style);
65 }
