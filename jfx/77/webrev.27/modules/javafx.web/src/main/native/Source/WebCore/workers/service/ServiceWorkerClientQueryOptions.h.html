26 #pragma once
27 
28 #if ENABLE(SERVICE_WORKER)
29 
30 #include "ServiceWorkerClientType.h"
31 #include <wtf/Optional.h>
32 
33 namespace WebCore {
34 
35 struct ServiceWorkerClientQueryOptions {
36     bool includeUncontrolled { false };
37     ServiceWorkerClientType type { ServiceWorkerClientType::Window };
38 
39     template<class Encoder> void encode(Encoder&) const;
40     template<class Decoder> static Optional<ServiceWorkerClientQueryOptions> decode(Decoder&);
41 };
42 
43 template<class Encoder>
44 void ServiceWorkerClientQueryOptions::encode(Encoder& encoder) const
45 {
46     encoder << includeUncontrolled << type;
47 }
48 
49 template<class Decoder>
50 Optional<ServiceWorkerClientQueryOptions> ServiceWorkerClientQueryOptions::decode(Decoder& decoder)
51 {
52     Optional<bool> includeUncontrolled;
53     decoder >> includeUncontrolled;
54     if (!includeUncontrolled)
55         return WTF::nullopt;
56 
57     Optional<ServiceWorkerClientType> type;
58     decoder >> type;
59     if (!type)
60         return WTF::nullopt;
61 
62     return { { *includeUncontrolled, *type } };
63 }
64 
65 } // namespace WebCore
66 
67 #endif // ENABLE(SERVICE_WORKER)
