37 class ScrollbarThemeComposite : public ScrollbarTheme {
38 public:
39     // Implement ScrollbarTheme interface
40     bool paint(Scrollbar&, GraphicsContext&, const IntRect& damageRect) override;
41     ScrollbarPart hitTest(Scrollbar&, const IntPoint&) override;
42     void invalidatePart(Scrollbar&, ScrollbarPart) override;
43     int thumbPosition(Scrollbar&) override;
44     int thumbLength(Scrollbar&) override;
45     int trackPosition(Scrollbar&) override;
46     int trackLength(Scrollbar&) override;
47     void paintOverhangAreas(ScrollView&, GraphicsContext&, const IntRect& horizontalOverhangArea, const IntRect& verticalOverhangArea, const IntRect& dirtyRect) override;
48 
49     virtual bool hasButtons(Scrollbar&) = 0;
50     virtual bool hasThumb(Scrollbar&) = 0;
51 
52     virtual IntRect backButtonRect(Scrollbar&, ScrollbarPart, bool painting = false) = 0;
53     virtual IntRect forwardButtonRect(Scrollbar&, ScrollbarPart, bool painting = false) = 0;
54     virtual IntRect trackRect(Scrollbar&, bool painting = false) = 0;
55     virtual IntRect thumbRect(Scrollbar&);
56 
57     virtual void splitTrack(Scrollbar&, const IntRect& track, IntRect& startTrack, IntRect& thumb, IntRect& endTrack);
58 
59     virtual int minimumThumbLength(Scrollbar&);
60 
61     virtual void willPaintScrollbar(GraphicsContext&, Scrollbar&) { }
62     virtual void didPaintScrollbar(GraphicsContext&, Scrollbar&) { }
63 
64     virtual void paintScrollbarBackground(GraphicsContext&, Scrollbar&) { }
65     virtual void paintTrackBackground(GraphicsContext&, Scrollbar&, const IntRect&) { }
66     virtual void paintTrackPiece(GraphicsContext&, Scrollbar&, const IntRect&, ScrollbarPart) { }
67     virtual void paintButton(GraphicsContext&, Scrollbar&, const IntRect&, ScrollbarPart) { }
68     virtual void paintThumb(GraphicsContext&, Scrollbar&, const IntRect&) { }
69 
70     virtual IntRect constrainTrackRectToTrackPieces(Scrollbar&, const IntRect& rect) { return rect; }
71 };
