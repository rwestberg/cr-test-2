26 #include "config.h"
27 #include "MediaRemoteControls.h"
28 
29 #if ENABLE(MEDIA_SESSION)
30 
31 #include "MediaSession.h"
32 
33 namespace WebCore {
34 
35 MediaRemoteControls::MediaRemoteControls(ScriptExecutionContext& context, MediaSession* session)
36     : m_scriptExecutionContext(context)
37     , m_session(session)
38 {
39 }
40 
41 void MediaRemoteControls::clearSession()
42 {
43     m_session = nullptr;
44 }
45 
46 MediaRemoteControls::~MediaRemoteControls() = default;
47 
48 void MediaRemoteControls::setPreviousTrackEnabled(bool isEnabled)
49 {
50     if (m_previousTrackEnabled == isEnabled)
51         return;
52 
53     m_previousTrackEnabled = isEnabled;
54 
55     if (m_session)
56         m_session->controlIsEnabledDidChange();
57 }
58 
59 void MediaRemoteControls::setNextTrackEnabled(bool isEnabled)
60 {
61     if (m_nextTrackEnabled == isEnabled)
62         return;
63 
64     m_nextTrackEnabled = isEnabled;
65 
66     if (m_session)
67         m_session->controlIsEnabledDidChange();
68 }
69 
70 }
71 
72 #endif /* ENABLE(MEDIA_SESSION) */
