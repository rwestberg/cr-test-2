34 class KeyedEncoderCF final : public KeyedEncoder {
35 public:
36     KeyedEncoderCF();
37     ~KeyedEncoderCF();
38 
39 private:
40     RefPtr<WebCore::SharedBuffer> finishEncoding() final;
41 
42     void encodeBytes(const String& key, const uint8_t*, size_t) final;
43     void encodeBool(const String& key, bool) final;
44     void encodeUInt32(const String& key, uint32_t) final;
45     void encodeUInt64(const String& key, uint64_t) final;
46     void encodeInt32(const String& key, int32_t) final;
47     void encodeInt64(const String& key, int64_t) final;
48     void encodeFloat(const String& key, float) final;
49     void encodeDouble(const String& key, double) final;
50     void encodeString(const String& key, const String&) final;
51 
52     void beginObject(const String& key) final;

54 
55     void beginArray(const String& key) final;
56     void beginArrayElement() final;
57     void endArrayElement() final;
58     void endArray() final;
59 
60     RetainPtr<CFMutableDictionaryRef> m_rootDictionary;
61 
62     Vector<CFMutableDictionaryRef, 16> m_dictionaryStack;
63     Vector<CFMutableArrayRef, 16> m_arrayStack;
64 };
