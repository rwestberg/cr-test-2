32 class FontCascade;
33 class FloatRect;
34 class GraphicsContext;
35 class InlineTextBox;
36 class RenderObject;
37 class RenderStyle;
38 class RenderText;
39 class ShadowData;

42 class TextDecorationPainter {
43 public:
44     struct Styles;
45     TextDecorationPainter(GraphicsContext&, OptionSet<TextDecoration> decorations, const RenderText&, bool isFirstLine, const FontCascade&, Optional<Styles> = WTF::nullopt);
46 
47     void setInlineTextBox(const InlineTextBox* inlineTextBox) { m_inlineTextBox = inlineTextBox; }
48     void setIsHorizontal(bool isHorizontal) { m_isHorizontal = isHorizontal; }
49     void setWidth(float width) { m_width = width; }
50     void setTextShadow(const ShadowData* textShadow) { m_shadow = textShadow; }
51     void setShadowColorFilter(const FilterOperations* colorFilter) { m_shadowColorFilter = colorFilter; }
52 
53     void paintTextDecoration(const TextRun&, const FloatPoint& textOrigin, const FloatPoint& boxOrigin);
54 
55     struct Styles {
56         bool operator==(const Styles&) const;
57         bool operator!=(const Styles& other) const { return !(*this == other); }
58 
59         Color underlineColor;
60         Color overlineColor;
61         Color linethroughColor;
62         TextDecorationStyle underlineStyle;
63         TextDecorationStyle overlineStyle;
64         TextDecorationStyle linethroughStyle;
65     };
66     static Styles stylesForRenderer(const RenderObject&, OptionSet<TextDecoration> requestedDecorations, bool firstLineStyle = false, PseudoId = PseudoId::None);
67 
68 private:
69     GraphicsContext& m_context;
70     OptionSet<TextDecoration> m_decorations;
71     float m_wavyOffset;
72     float m_width { 0 };
73     FloatPoint m_boxOrigin;
74     bool m_isPrinting;
75     bool m_isHorizontal { true };
76     const ShadowData* m_shadow { nullptr };
77     const FilterOperations* m_shadowColorFilter { nullptr };
78     const InlineTextBox* m_inlineTextBox { nullptr };
79     const FontCascade& m_font;
80 
81     Styles m_styles;
82     const RenderStyle& m_lineStyle;
83 };
