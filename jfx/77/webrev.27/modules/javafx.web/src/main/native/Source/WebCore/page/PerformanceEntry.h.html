32 #pragma once
33 
34 #include "Performance.h"
35 #include <wtf/RefCounted.h>
36 #include <wtf/text/WTFString.h>
37 
38 namespace WebCore {
39 
40 class PerformanceEntry : public RefCounted<PerformanceEntry> {
41 public:
42     virtual ~PerformanceEntry();
43 
44     String name() const { return m_name; }
45     String entryType() const { return m_entryType; }
46     double startTime() const { return m_startTime; }
47     double duration() const { return m_duration; }
48 
49     enum class Type {
50         Navigation = 1 << 0,
51         Mark = 1 << 1,
52         Measure = 1 << 2,
53         Resource = 1 << 3,
54     };
55 
56     Type type() const { return m_type; }
57 
58     static Optional<Type> parseEntryTypeString(const String& entryType);
59 
60     bool isResource() const { return m_type == Type::Resource; }
61     bool isMark() const { return m_type == Type::Mark; }
62     bool isMeasure() const { return m_type == Type::Measure; }
63 
64     static bool startTimeCompareLessThan(const RefPtr<PerformanceEntry>& a, const RefPtr<PerformanceEntry>& b)
65     {
66         return a->startTime() < b->startTime();
67     }
68 
69 protected:
70     PerformanceEntry(Type, const String& name, const String& entryType, double startTime, double finishTime);
71 
72 private:
73     const String m_name;
74     const String m_entryType;
75     const double m_startTime;
76     const double m_duration;
77     const Type m_type;
78 };
79 
80 } // namespace WebCore
