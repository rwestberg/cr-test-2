26 #include "config.h"
27 #include "ClipRect.h"
28 
29 #include "HitTestLocation.h"
30 #include <wtf/text/TextStream.h>
31 
32 namespace WebCore {
33 
34 bool ClipRect::intersects(const HitTestLocation& hitTestLocation) const
35 {
36     if (isInfinite())
37         return true;
38 
39     return hitTestLocation.intersects(m_rect);
40 }
41 
42 TextStream& operator<<(TextStream& ts, const ClipRect& clipRect)
43 {
44     ts << "rect ";
45     if (clipRect.isInfinite())
46         ts << "infinite";
47     else
48         ts << clipRect.rect();
49 
50     if (clipRect.affectedByRadius())
51         ts << " affected by radius";
52     return ts;
53 }
