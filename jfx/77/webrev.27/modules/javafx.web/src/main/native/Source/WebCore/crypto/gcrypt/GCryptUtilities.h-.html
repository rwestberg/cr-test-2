60 template<size_t N>
61 static inline bool matches(const void* lhs, size_t size, const std::array<uint8_t, N>& rhs)
62 {
63     if (size != rhs.size())
64         return false;
65 
66     return !std::memcmp(lhs, rhs.data(), rhs.size());
67 }
