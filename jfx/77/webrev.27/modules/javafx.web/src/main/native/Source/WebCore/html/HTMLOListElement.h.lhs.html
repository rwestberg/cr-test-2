23 #pragma once
24 
25 #include "HTMLElement.h"
26 
27 namespace WebCore {
28 
29 class HTMLOListElement final : public HTMLElement {
30     WTF_MAKE_ISO_ALLOCATED(HTMLOListElement);
31 public:
32     static Ref<HTMLOListElement> create(Document&);
33     static Ref<HTMLOListElement> create(const QualifiedName&, Document&);
34 
35     int startForBindings() const { return m_start.valueOr(1); }
36     WEBCORE_EXPORT void setStartForBindings(int);
37 
38     // FIXME: The reason start() does not trigger layout is because it is called
39     // from rendering code. This is unfortunately one of the few cases where the
40     // render tree is mutated during layout.
41 
42     int start() const { return m_start ? m_start.value() : (m_isReversed ? itemCount() : 1); }
43     bool isReversed() const { return m_isReversed; }
44     void itemCountChanged() { m_itemCount = WTF::nullopt; }
45 
46 private:
47     HTMLOListElement(const QualifiedName&, Document&);
48 
49     WEBCORE_EXPORT unsigned itemCount() const;
50 
51     void parseAttribute(const QualifiedName&, const AtomicString&) final;
52     bool isPresentationAttribute(const QualifiedName&) const final;
53     void collectStyleForPresentationAttribute(const QualifiedName&, const AtomicString&, MutableStyleProperties&) final;
54 
55     Optional<int> m_start;
56     mutable Optional<unsigned> m_itemCount;
57     bool m_isReversed { false };
58 };
59 
60 } //namespace
