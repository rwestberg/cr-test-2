22 [
23     ConstantsScope=SVGTransformValue
24 ] interface SVGTransform { 
25     // Transform Types
26     const unsigned short SVG_TRANSFORM_UNKNOWN = 0;
27     const unsigned short SVG_TRANSFORM_MATRIX = 1;
28     const unsigned short SVG_TRANSFORM_TRANSLATE = 2;
29     const unsigned short SVG_TRANSFORM_SCALE = 3;
30     const unsigned short SVG_TRANSFORM_ROTATE = 4;
31     const unsigned short SVG_TRANSFORM_SKEWX = 5;
32     const unsigned short SVG_TRANSFORM_SKEWY = 6;
33 
34     readonly attribute unsigned short type;
35     readonly attribute SVGMatrix matrix;
36     readonly attribute unrestricted float angle;
37 
38     [MayThrowException] void setMatrix(SVGMatrix matrix);
39     [MayThrowException] void setTranslate(unrestricted float tx, unrestricted float ty);
40     [MayThrowException] void setScale(unrestricted float sx, unrestricted float sy);
41     [MayThrowException] void setRotate(unrestricted float angle, unrestricted float cx, unrestricted float cy);
42     [MayThrowException] void setSkewX(unrestricted float angle);
43     [MayThrowException] void setSkewY(unrestricted float angle);
44 };
