26 #include "config.h"

29 #include <wtf/NeverDestroyed.h>
30 
31 #if ENABLE(WIRELESS_PLAYBACK_TARGET)
32 
33 namespace WebCore {
34 
35 static const AtomString& stringForPlaybackTargetAvailability(bool available)
36 {
37     static NeverDestroyed<AtomString> availableString("available", AtomString::ConstructFromLiteral);
38     static NeverDestroyed<AtomString> notAvailableString("not-available", AtomString::ConstructFromLiteral);
39 
40     return available ? availableString : notAvailableString;
41 }
42 
43 WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, bool available)
44     : Event(eventType, CanBubble::No, IsCancelable::No)
45     , m_availability(stringForPlaybackTargetAvailability(available))
46 {
47 }
48 
49 WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, const Init& initializer, IsTrusted isTrusted)
50     : Event(eventType, initializer, isTrusted)
51     , m_availability(initializer.availability)
52 {
53 }
54 
55 } // namespace WebCore

57 #endif // ENABLE(WIRELESS_PLAYBACK_TARGET)
