 73 LengthBox Theme::controlBorder(ControlPart part, const FontCascade&, const LengthBox& zoomedBox, float) const
 74 {
 75     switch (part) {
 76     case PushButtonPart:
 77     case MenulistPart:
 78     case SearchFieldPart:
 79     case CheckboxPart:
 80     case RadioPart:
 81         return LengthBox(0);
 82     default:
 83         return zoomedBox;
 84     }
 85 }
 86 
 87 LengthBox Theme::controlPadding(ControlPart part, const FontCascade&, const LengthBox& zoomedBox, float) const
 88 {
 89     switch (part) {
 90     case MenulistPart:
 91     case MenulistButtonPart:
 92     case CheckboxPart:
 93     case RadioPart:
 94         return LengthBox(0);
 95     default:
 96         return zoomedBox;
 97     }
 98 }

100 void Theme::drawNamedImage(const String& name, GraphicsContext& context, const FloatRect& rect) const
101 {
102     // We only handle one icon at the moment.
103     if (name != "wireless-playback")
104         return;
105 
106     GraphicsContextStateSaver stateSaver(context);
107     context.setFillColor(Color::black);
108 
109     // Draw a generic Wireless Playback icon.
110 
111     context.scale(rect.size() / 100);
112     context.translate(8, 1);
113 
114     Path outline;
115     outline.moveTo(FloatPoint(59, 58.7));
116     outline.addBezierCurveTo(FloatPoint(58.1, 58.7), FloatPoint(57.2, 58.4), FloatPoint(56.4, 57.7));
117     outline.addLineTo(FloatPoint(42, 45.5));
118     outline.addLineTo(FloatPoint(27.6, 57.8));
119     outline.addBezierCurveTo(FloatPoint(25.9, 59.2), FloatPoint(23.4, 59), FloatPoint(22, 57.3));
120     outline.addBezierCurveTo(FloatPoint(20.6, 55.6), FloatPoint(20.8, 53.1), FloatPoint(22.5, 51.7));
121     outline.addLineTo(FloatPoint(39.5, 37.3));
122     outline.addBezierCurveTo(FloatPoint(41, 36), FloatPoint(43.2, 36), FloatPoint(44.7, 37.3));
123     outline.addLineTo(FloatPoint(61.7, 51.7));
124     outline.addBezierCurveTo(FloatPoint(63.4, 53.1), FloatPoint(63.6, 55.7), FloatPoint(62.2, 57.3));
125     outline.addBezierCurveTo(FloatPoint(61.3, 58.2), FloatPoint(60.1, 58.7), FloatPoint(59, 58.7));
126     outline.addLineTo(FloatPoint(59, 58.7));
127     outline.closeSubpath();
128 
129     outline.moveTo(FloatPoint(42, 98));
130     outline.addBezierCurveTo(FloatPoint(39.8, 98), FloatPoint(38, 96.3), FloatPoint(38, 94.2));
131     outline.addLineTo(FloatPoint(38, 43.6));
132     outline.addBezierCurveTo(FloatPoint(38, 41.5), FloatPoint(39.8, 39.8), FloatPoint(42, 39.8));
133     outline.addBezierCurveTo(FloatPoint(44.2, 39.8), FloatPoint(46, 41.5), FloatPoint(46, 43.6));
134     outline.addLineTo(FloatPoint(46, 94.2));
135     outline.addBezierCurveTo(FloatPoint(46, 96.3), FloatPoint(44.2, 98), FloatPoint(42, 98));
136     outline.addLineTo(FloatPoint(42, 98));
137     outline.closeSubpath();
138 
139     outline.moveTo(FloatPoint(83.6, 41.6));
140     outline.addBezierCurveTo(FloatPoint(83.6, 18.6), FloatPoint(65, 0), FloatPoint(42, 0));
141     outline.addBezierCurveTo(FloatPoint(19, 0), FloatPoint(0.4, 18.6), FloatPoint(0.4, 41.6));
142     outline.addBezierCurveTo(FloatPoint(0.4, 62.2), FloatPoint(15, 79.2), FloatPoint(35, 82.6));
143     outline.addLineTo(FloatPoint(35, 74.5));
144     outline.addBezierCurveTo(FloatPoint(20, 71.2), FloatPoint(8.4, 57.7), FloatPoint(8.4, 41.6));
145     outline.addBezierCurveTo(FloatPoint(8.4, 23.1), FloatPoint(23.5, 8), FloatPoint(42, 8));
146     outline.addBezierCurveTo(FloatPoint(60.5, 8), FloatPoint(75.5, 23.1), FloatPoint(75.5, 41.6));
147     outline.addBezierCurveTo(FloatPoint(75.6, 57.7), FloatPoint(64, 71.2), FloatPoint(49, 74.5));
148     outline.addLineTo(FloatPoint(49, 82.6));
149     outline.addBezierCurveTo(FloatPoint(69, 79.3), FloatPoint(83.6, 62.2), FloatPoint(83.6, 41.6));
150     outline.addLineTo(FloatPoint(83.6, 41.6));
151     outline.closeSubpath();
152 
153     context.fillPath(outline);
154 }
