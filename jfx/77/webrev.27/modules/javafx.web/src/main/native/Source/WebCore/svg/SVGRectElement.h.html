22 #pragma once
23 
24 #include "SVGExternalResourcesRequired.h"
25 #include "SVGGeometryElement.h"
26 #include "SVGNames.h"
27 
28 namespace WebCore {
29 
30 class SVGRectElement final : public SVGGeometryElement, public SVGExternalResourcesRequired {
31     WTF_MAKE_ISO_ALLOCATED(SVGRectElement);
32 public:
33     static Ref<SVGRectElement> create(const QualifiedName&, Document&);
34 
35     const SVGLengthValue& x() const { return m_x->currentValue(); }
36     const SVGLengthValue& y() const { return m_y->currentValue(); }
37     const SVGLengthValue& width() const { return m_width->currentValue(); }
38     const SVGLengthValue& height() const { return m_height->currentValue(); }
39     const SVGLengthValue& rx() const { return m_rx->currentValue(); }
40     const SVGLengthValue& ry() const { return m_ry->currentValue(); }
41 
42     SVGAnimatedLength& xAnimated() { return m_x; }
43     SVGAnimatedLength& yAnimated() { return m_y; }
44     SVGAnimatedLength& widthAnimated() { return m_width; }
45     SVGAnimatedLength& heightAnimated() { return m_height; }
46     SVGAnimatedLength& rxAnimated() { return m_rx; }
47     SVGAnimatedLength& ryAnimated() { return m_ry; }
48 
49 private:
50     SVGRectElement(const QualifiedName&, Document&);
51 
52     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGRectElement, SVGGeometryElement, SVGExternalResourcesRequired>;
53     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
54 
55     void parseAttribute(const QualifiedName&, const AtomString&) final;
56     void svgAttributeChanged(const QualifiedName&) final;
57 
58     bool isValid() const final { return SVGTests::isValid(); }
59     bool selfHasRelativeLengths() const final { return true; }
60 
61     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
62 
63     PropertyRegistry m_propertyRegistry { *this };
64     Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, LengthModeWidth) };
65     Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, LengthModeHeight) };
66     Ref<SVGAnimatedLength> m_width { SVGAnimatedLength::create(this, LengthModeWidth) };
67     Ref<SVGAnimatedLength> m_height { SVGAnimatedLength::create(this, LengthModeHeight) };
68     Ref<SVGAnimatedLength> m_rx { SVGAnimatedLength::create(this, LengthModeWidth) };
69     Ref<SVGAnimatedLength> m_ry { SVGAnimatedLength::create(this, LengthModeHeight) };
70 };
71 
72 } // namespace WebCore
