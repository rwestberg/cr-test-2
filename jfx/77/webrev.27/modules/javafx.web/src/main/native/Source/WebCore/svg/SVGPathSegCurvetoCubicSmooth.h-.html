21 #pragma once

23 #include "SVGPathSegWithContext.h"
24 
25 namespace WebCore {
26 
27 class SVGPathSegCurvetoCubicSmooth : public SVGPathSegWithContext {
28 public:
29     SVGPathSegCurvetoCubicSmooth(const SVGPathElement& element, SVGPathSegRole role, float x, float y, float x2, float y2)
30         : SVGPathSegWithContext(element, role)
31         , m_x(x)
32         , m_y(y)
33         , m_x2(x2)
34         , m_y2(y2)
35     {
36     }
37 
38     float x() const { return m_x; }
39     void setX(float x)
40     {
41         m_x = x;
42         commitChange();
43     }
44 
45     float y() const { return m_y; }
46     void setY(float y)
47     {
48         m_y = y;
49         commitChange();
50     }
51 
52     float x2() const { return m_x2; }
53     void setX2(float x2)
54     {
55         m_x2 = x2;
56         commitChange();
57     }
58 
59     float y2() const { return m_y2; }
60     void setY2(float y2)
61     {
62         m_y2 = y2;
63         commitChange();
64     }
65 
66 private:
67     float m_x;
68     float m_y;
69     float m_x2;
70     float m_y2;
71 };

73 } // namespace WebCore
