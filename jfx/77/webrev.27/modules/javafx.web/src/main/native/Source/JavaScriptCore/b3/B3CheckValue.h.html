28 #if ENABLE(B3_JIT)
29 
30 #include "B3StackmapValue.h"
31 
32 namespace JSC { namespace B3 {
33 
34 class CheckValue : public StackmapValue {
35 public:
36     static bool accepts(Kind kind)
37     {
38         switch (kind.opcode()) {
39         case CheckAdd:
40         case CheckSub:
41         case CheckMul:
42         case Check:
43             return true;
44         default:
45             return false;
46         }
47     }
48 
49     ~CheckValue();
50 
51     void convertToAdd();
52 
53     B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_VARARGS_CHILDREN
54 
55 private:
56     friend class Procedure;
57     friend class Value;
58 
59     // Use this form for CheckAdd, CheckSub, and CheckMul.
60     JS_EXPORT_PRIVATE CheckValue(Kind, Origin, Value* left, Value* right);
61 
62     // Use this form for Check.
63     JS_EXPORT_PRIVATE CheckValue(Kind, Origin, Value* predicate);
64 };
65 
66 } } // namespace JSC::B3
67 
68 #endif // ENABLE(B3_JIT)
