32 namespace JSC {
33 
34 class SourceProvider;
35 class VM;
36 
37 enum class DebuggerPausePositionType { Enter, Leave, Pause };
38 struct DebuggerPausePosition {
39     DebuggerPausePositionType type;
40     JSTextPosition position;
41 };
42 
43 class DebuggerPausePositions {
44 public:
45     DebuggerPausePositions() { }
46     ~DebuggerPausePositions() { }
47 
48     void appendPause(const JSTextPosition& position)
49     {
50         m_positions.append({ DebuggerPausePositionType::Pause, position });
51     }
52 
53     void appendEntry(const JSTextPosition& position)
54     {
55         m_positions.append({ DebuggerPausePositionType::Enter, position });
56     }
57 
58     void appendLeave(const JSTextPosition& position)
59     {
60         m_positions.append({ DebuggerPausePositionType::Leave, position });
61     }
62 
63     Optional<JSTextPosition> breakpointLocationForLineColumn(int line, int column);
64 
65     void sort();
66 
67 private:
68     Vector<DebuggerPausePosition> m_positions;
69 };
70 
71 
72 struct DebuggerParseData {
73     DebuggerParseData() { }
74     ~DebuggerParseData() { }
75 
76     DebuggerPausePositions pausePositions;
77 };
78 
79 bool gatherDebuggerParseDataForSource(VM&, SourceProvider*, DebuggerParseData&);
