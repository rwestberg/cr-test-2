26 #ifndef Zone_h
27 #define Zone_h
28 
29 #include "FixedVector.h"
30 #include "Mutex.h"
31 #include "Range.h"
32 #include <malloc/malloc.h>
33 #include <mutex>
34 
35 namespace bmalloc {
36 
37 class Chunk;
38 
39 class Zone : public malloc_zone_t {
40 public:
41     // Enough capacity to track a 64GB heap, so probably enough for anything.
42     static const size_t capacity = 2048;
43 
44     Zone(std::lock_guard<Mutex>&);
45     Zone(task_t, memory_reader_t, vm_address_t);
46 
47     void addRange(Range);
48     FixedVector<Range, capacity>& ranges() { return m_ranges; }
49 
50 private:
51     // This vector has two purposes:
52     //     (1) It stores the list of Chunks so that we can enumerate
53     //         each Chunk and request that it be scanned if reachable.
54     //     (2) It roots a pointer to each Chunk in a global non-malloc
55     //         VM region, making each Chunk appear reachable, and therefore
56     //         ensuring that the leaks tool will scan it. (The leaks tool
57     //         conservatively scans all writeable VM regions that are not malloc
58     //         regions, and then scans malloc regions using the introspection API.)
59     // This prevents the leaks tool from reporting false positive leaks for
60     // objects pointed to from bmalloc memory -- though it also prevents the
61     // leaks tool from finding any leaks in bmalloc memory.
62     FixedVector<Range, capacity> m_ranges;
63 };
64 
65 inline void Zone::addRange(Range range)
66 {
67     if (m_ranges.size() == m_ranges.capacity())
68         return;
69 
70     m_ranges.push(range);
71 }
72 
73 } // namespace bmalloc
74 
75 #endif // Zone_h
