26 #include "config.h"
27 #include <wtf/win/PathWalker.h>
28 
29 #include <wtf/text/WTFString.h>
30 #include <wtf/text/win/WCharStringExtras.h>
31 
32 namespace WTF {
33 
34 PathWalker::PathWalker(const String& directory, const String& pattern)
35 {
36     String path = directory + "\\" + pattern;
37     m_handle = ::FindFirstFileW(stringToNullTerminatedWChar(path).data(), &m_data);
38 }
39 
40 PathWalker::~PathWalker()
41 {
42     if (!isValid())
43         return;
44     ::FindClose(m_handle);
45 }
46 
47 bool PathWalker::step()
48 {
49     return ::FindNextFileW(m_handle, &m_data);
50 }
