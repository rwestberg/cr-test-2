40     [CallWith=ExecState] void inspect(any objectToInspect, any hints);

43     [CallWith=ExecState] record<DOMString, sequence<ListenerEntry>> getEventListeners(Node? node);
44 
45     DOMString databaseId(Database database);
46     DOMString storageId(Storage storage);
47 };
48

50     JSGenerateToJSObject

54     required boolean passive;
55     required boolean once;
56 };
