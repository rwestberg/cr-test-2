 21 #pragma once
 22 
 23 #include "SVGPropertyTearOff.h"
 24 #include "SVGPropertyTraits.h"

 25 
 26 namespace WebCore {
 27 
 28 class SVGRect : public SVGPropertyTearOff<FloatRect> {




 29 public:
 30     static Ref<SVGRect> create(SVGAnimatedProperty& animatedProperty, SVGPropertyRole role, FloatRect& value)
 31     {
 32         return adoptRef(*new SVGRect(animatedProperty, role, value));
 33     }
 34 
 35     static Ref<SVGRect> create(const FloatRect& initialValue = { })
 36     {
 37         return adoptRef(*new SVGRect(initialValue));
 38     }
 39 
 40     template<typename T> static ExceptionOr<Ref<SVGRect>> create(ExceptionOr<T>&& initialValue)

 41     {
 42         if (initialValue.hasException())
 43             return initialValue.releaseException();
 44         return create(initialValue.releaseReturnValue());
 45     }
 46 
 47     float x()
 48     {
 49         return propertyReference().x();
 50     }
 51 
 52     ExceptionOr<void> setX(float xValue)
 53     {
 54         if (isReadOnly())
 55             return Exception { NoModificationAllowedError };
 56 
 57         propertyReference().setX(xValue);
 58         commitChange();
 59 
 60         return { };
 61     }
 62 
 63     float y()
 64     {
 65         return propertyReference().y();
 66     }
 67 
 68     ExceptionOr<void> setY(float xValue)
 69     {
 70         if (isReadOnly())
 71             return Exception { NoModificationAllowedError };
 72 
 73         propertyReference().setY(xValue);
 74         commitChange();
 75 
 76         return { };
 77     }
 78 
 79     float width()
 80     {
 81         return propertyReference().width();
 82     }
 83 
 84     ExceptionOr<void> setWidth(float widthValue)
 85     {
 86         if (isReadOnly())
 87             return Exception { NoModificationAllowedError };
 88 
 89         propertyReference().setWidth(widthValue);
 90         commitChange();
 91 
 92         return { };
 93     }
 94 
 95     float height()
 96     {
 97         return propertyReference().height();
 98     }
 99 
100     ExceptionOr<void> setHeight(float heightValue)
101     {
102         if (isReadOnly())
103             return Exception { NoModificationAllowedError };
104 
105         propertyReference().setHeight(heightValue);
106         commitChange();
107 
108         return { };
109     }
110 
111 private:
112     SVGRect(SVGAnimatedProperty& animatedProperty, SVGPropertyRole role, FloatRect& value)
113         : SVGPropertyTearOff<FloatRect>(&animatedProperty, role, value)
114     {
115     }
116 
117     explicit SVGRect(const FloatRect& initialValue)
118         : SVGPropertyTearOff<FloatRect>(initialValue)
119     {

120     }
121 };
122 
123 
124 } // namespace WebCore
