26 #ifndef ScrollAnimation_h
27 #define ScrollAnimation_h
28 
29 #include "ScrollTypes.h"
30 
31 namespace WebCore {
32 
33 class FloatPoint;
34 class ScrollableArea;
35 
36 class ScrollAnimation {
37     WTF_MAKE_FAST_ALLOCATED;
38 public:
39     virtual ~ScrollAnimation() { };
40     virtual bool scroll(ScrollbarOrientation, ScrollGranularity, float /* step */, float /* multiplier */) { return true; };
41     virtual void stop() = 0;
42     virtual void updateVisibleLengths() { };
43     virtual void setCurrentPosition(const FloatPoint&) { };
44     virtual void serviceAnimation() { };
45 
46 protected:
47     ScrollAnimation(ScrollableArea& scrollableArea)
48         : m_scrollableArea(scrollableArea)
49     {
50     }
51 
52     ScrollableArea& m_scrollableArea;
53 };
54 
55 } // namespace WebCore
56 
57 #endif // ScrollAnimation_h
