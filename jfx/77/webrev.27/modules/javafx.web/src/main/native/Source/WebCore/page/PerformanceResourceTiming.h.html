49     AtomString initiatorType() const { return m_initiatorType; }

52     double workerStart() const;

55     double fetchStart() const;
56     double domainLookupStart() const;
57     double domainLookupEnd() const;
58     double connectStart() const;
59     double connectEnd() const;
60     double secureConnectionStart() const;
61     double requestStart() const;
62     double responseStart() const;
63     double responseEnd() const;
64     const Vector<Ref<PerformanceServerTiming>>& serverTiming() const { return m_serverTiming; }
65 
66 private:
67     PerformanceResourceTiming(MonotonicTime timeOrigin, ResourceTiming&&);
68     ~PerformanceResourceTiming();

70     double networkLoadTimeToDOMHighResTimeStamp(Seconds) const;

72     AtomString m_initiatorType;
73     MonotonicTime m_timeOrigin;
74     LoadTiming m_loadTiming;
75     NetworkLoadMetrics m_networkLoadMetrics;
76     bool m_shouldReportDetails;
77     Vector<Ref<PerformanceServerTiming>> m_serverTiming;
78 };
