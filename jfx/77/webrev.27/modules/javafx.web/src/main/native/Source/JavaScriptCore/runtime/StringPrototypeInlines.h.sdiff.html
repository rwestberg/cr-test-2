32 template<typename NumberType>
33 ALWAYS_INLINE JSString* stringSlice(ExecState* exec, String&& string, NumberType start, NumberType end)
34 {
35     int32_t length = string.length();
36     NumberType from = start < 0 ? length + start : start;
37     NumberType to = end < 0 ? length + end : end;
38     if (to > from && to > 0 && from < length) {
39         if (from < 0)
40             from = 0;
41         if (to > length)
42             to = length;
43         return jsSubstring(exec, WTFMove(string), static_cast<unsigned>(from), static_cast<unsigned>(to) - static_cast<unsigned>(from));
44     }
45     return jsEmptyString(exec);
46 }
