29 #ifndef ZeroPole_h
30 #define ZeroPole_h
31 
32 namespace WebCore {
33 
34 // ZeroPole is a simple filter with one zero and one pole.
35 
36 class ZeroPole {
37 public:
38     ZeroPole()
39         : m_zero(0)
40         , m_pole(0)
41         , m_lastX(0)
42         , m_lastY(0)
43     {
44     }
45 
46     void process(const float *source, float *destination, unsigned framesToProcess);
47 
48     // Reset filter state.
49     void reset() { m_lastX = 0; m_lastY = 0; }
50 
51     void setZero(float zero) { m_zero = zero; }
52     void setPole(float pole) { m_pole = pole; }
53 
54     float zero() const { return m_zero; }
55     float pole() const { return m_pole; }
56 
57 private:
58     float m_zero;
59     float m_pole;
60     float m_lastX;
61     float m_lastY;
62 };
63 
64 } // namespace WebCore
65 
66 #endif // ZeroPole_h
