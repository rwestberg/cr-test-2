35 namespace WebCore {

37 WTF_MAKE_ISO_ALLOCATED_IMPL(MathMLUnderOverElement);

39 using namespace MathMLNames;
40 
41 inline MathMLUnderOverElement::MathMLUnderOverElement(const QualifiedName& tagName, Document& document)
42     : MathMLScriptsElement(tagName, document)
43 {
44 }
45 
46 Ref<MathMLUnderOverElement> MathMLUnderOverElement::create(const QualifiedName& tagName, Document& document)
47 {
48     return adoptRef(*new MathMLUnderOverElement(tagName, document));
49 }
50 
51 const MathMLElement::BooleanValue& MathMLUnderOverElement::accent()
52 {
53     return cachedBooleanAttribute(accentAttr, m_accent);
54 }
55 
56 const MathMLElement::BooleanValue& MathMLUnderOverElement::accentUnder()
57 {
58     return cachedBooleanAttribute(accentunderAttr, m_accentUnder);
59 }
60 
61 void MathMLUnderOverElement::parseAttribute(const QualifiedName& name, const AtomicString& value)
62 {
63     if (name == accentAttr)
64         m_accent = WTF::nullopt;
65     else if (name == accentunderAttr)
66         m_accentUnder = WTF::nullopt;
67 
68     MathMLElement::parseAttribute(name, value);
69 }
70 
71 RenderPtr<RenderElement> MathMLUnderOverElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
72 {
73     ASSERT(hasTagName(munderTag) || hasTagName(moverTag) || hasTagName(munderoverTag));
74     return createRenderer<RenderMathMLUnderOver>(*this, WTFMove(style));
75 }
