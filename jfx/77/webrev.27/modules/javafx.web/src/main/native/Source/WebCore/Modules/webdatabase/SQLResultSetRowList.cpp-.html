32 namespace WebCore {

34 unsigned SQLResultSetRowList::length() const
35 {
36     if (m_result.isEmpty())
37         return 0;
38 
39     ASSERT(m_result.size() % m_columns.size() == 0);
40 
41     return m_result.size() / m_columns.size();
42 }
43 
44 ExceptionOr<Vector<WTF::KeyValuePair<String, SQLValue>>> SQLResultSetRowList::item(unsigned index) const
45 {
46     if (index >= length())
47         return Exception { IndexSizeError };
48 
49     Vector<WTF::KeyValuePair<String, SQLValue>> result;
50 
51     unsigned numberOfColumns = m_columns.size();
52     unsigned valuesIndex = index * numberOfColumns;
53     for (unsigned i = 0; i < numberOfColumns; i++)
54         result.append({ m_columns[i], m_result[valuesIndex + i] });
55 
56     return WTFMove(result);
57 }
