26 #pragma once
27 
28 #include "WebGLObject.h"
29 
30 namespace WebCore {
31 
32 class GraphicsContext3D;
33 class WebGLContextGroup;
34 class WebGLRenderingContextBase;
35 
36 // WebGLSharedObject the base class for objects that can be shared by multiple WebGLRenderingContexts.
37 class WebGLSharedObject : public WebGLObject {
38 public:
39     virtual ~WebGLSharedObject();
40 
41     WebGLContextGroup* contextGroup() const { return m_contextGroup; }
42 
43     virtual bool isRenderbuffer() const { return false; }
44     virtual bool isTexture() const { return false; }
45 
46     bool validate(const WebGLContextGroup* contextGroup, const WebGLRenderingContextBase&) const override
47     {
48         return contextGroup == m_contextGroup;
49     }
50 
51     void detachContextGroup();
52 
53 protected:
54     WebGLSharedObject(WebGLRenderingContextBase&);
55 
56     bool hasGroupOrContext() const override
57     {
58         return m_contextGroup;
59     }
60 
61     GraphicsContext3D* getAGraphicsContext3D() const override;
62 
63 private:
64     WebGLContextGroup* m_contextGroup;
65 };
66 
67 } // namespace WebCore
