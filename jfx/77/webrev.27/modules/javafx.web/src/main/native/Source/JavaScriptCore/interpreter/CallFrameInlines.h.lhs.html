32 inline bool CallFrame::isStackOverflowFrame() const
33 {
34     if (callee().isWasm())
35         return false;
36     return jsCallee() == jsCallee()->globalObject()->stackOverflowFrameCallee();
37 }
38 
39 inline bool CallFrame::isWasmFrame() const
40 {
41     return callee().isWasm();
42 }
