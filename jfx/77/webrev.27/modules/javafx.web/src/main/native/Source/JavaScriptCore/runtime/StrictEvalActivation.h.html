30 namespace JSC {

32 class StrictEvalActivation final : public JSScope {
33 public:
34     using Base = JSScope;
35 
36     static StrictEvalActivation* create(VM& vm, Structure* structure, JSScope* currentScope)
37     {
38         StrictEvalActivation* scope = new (NotNull, allocateCell<StrictEvalActivation>(vm.heap)) StrictEvalActivation(vm, structure, currentScope);
39         scope->finishCreation(vm);
40         return scope;
41     }
42 
43     static bool deleteProperty(JSCell*, ExecState*, PropertyName);
44 
45     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
46     {
47         return Structure::create(vm, globalObject, prototype, TypeInfo(StrictEvalActivationType, StructureFlags), info());
48     }
49 
50     DECLARE_INFO;
51 
52 private:
53     StrictEvalActivation(VM&, Structure*, JSScope*);
54 };

56 } // namespace JSC
