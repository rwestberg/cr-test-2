36 #include "RTCIceGatheringState.h"

38 #include "ScriptWrappable.h"
39 #include <wtf/RefCounted.h>
40 #include <wtf/RefPtr.h>
41 
42 namespace WebCore {
43 
44 class RTCIceTransport : public RefCounted<RTCIceTransport>, public ScriptWrappable {

45 public:
46     static Ref<RTCIceTransport> create()
47     {
48         return adoptRef(*new RTCIceTransport());
49     }
50     virtual ~RTCIceTransport() = default;
51 
52     RTCIceTransportState state() const { return m_transportState; }
53     void setState(RTCIceTransportState state) { m_transportState = state; }
54 
55     RTCIceGatheringState gatheringState() const { return m_gatheringState; }
56     void setGatheringState(RTCIceGatheringState state) { m_gatheringState = state; }
57 
58 private:
59     RTCIceTransport() { };
60 
61     RTCIceTransportState m_transportState { RTCIceTransportState::New };
62     RTCIceGatheringState m_gatheringState { RTCIceGatheringState::New };
63 };
64 
65 } // namespace WebCore
66 
67 #endif // ENABLE(WEB_RTC)
