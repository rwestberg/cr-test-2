33 void JSPerformanceObserver::visitAdditionalChildren(JSC::SlotVisitor& visitor)
34 {
35     wrapped().callback().visitJSFunction(visitor);
36 }
37 
38 bool JSPerformanceObserverOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, JSC::SlotVisitor&, const char** reason)
39 {
40     if (UNLIKELY(reason))
41         *reason = "Registered PerformanceObserver callback";
42 
43     return JSC::jsCast<JSPerformanceObserver*>(handle.slot()->asCell())->wrapped().isRegistered();
44 }
