27 #pragma once
28 
29 #include "DOMWindow.h"
30 #include "Event.h"
31 #include "Supplementable.h"
32 #include "Timer.h"
33 #include <wtf/HashCountedSet.h>
34 
35 namespace WebCore {
36 
37 class DeviceClient;
38 class Page;
39 
40 class DeviceController : public Supplement<Page> {
41     WTF_MAKE_FAST_ALLOCATED;
42 public:
43     explicit DeviceController(DeviceClient&);
44     virtual ~DeviceController() = default;
45 
46     void addDeviceEventListener(DOMWindow&);
47     void removeDeviceEventListener(DOMWindow&);
48     void removeAllDeviceEventListeners(DOMWindow&);
49     bool hasDeviceEventListener(DOMWindow&) const;
50 
51     void dispatchDeviceEvent(Event&);
52     bool isActive() { return !m_listeners.isEmpty(); }
53     DeviceClient& client() { return m_client; }
54 
55     virtual bool hasLastData() { return false; }
56     virtual RefPtr<Event> getLastEvent() { return nullptr; }
57 
58 protected:
59     void fireDeviceEvent();
60 
61     HashCountedSet<RefPtr<DOMWindow>> m_listeners;
62     HashCountedSet<RefPtr<DOMWindow>> m_lastEventListeners;
63     DeviceClient& m_client;
64     Timer m_timer;
65 };
66 
67 } // namespace WebCore
