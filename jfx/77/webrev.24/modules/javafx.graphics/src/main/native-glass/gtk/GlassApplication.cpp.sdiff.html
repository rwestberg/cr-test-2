507                 case GDK_CLIENT_EVENT:
508                 case GDK_VISIBILITY_NOTIFY:
509                 case GDK_SETTING:
510                 case GDK_OWNER_CHANGE:

512                     break;

514                     break;
515             }

517         }

534         }
