 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 330 void WindowContext::process_focus(GdkEventFocus *event) {
 331     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 332         ungrab_mouse_drag_focus();
 333     }
 334 
 335     if (!event->in && WindowContext::sm_grab_window == this) {
 336         ungrab_focus();
 337     }
 338 
 339     if (xim.enabled && xim.ic) {
 340         if (event->in) {
 341             XSetICFocus(xim.ic);
 342         } else {
 343             XUnsetICFocus(xim.ic);
 344         }
 345     }
 346 
 347     if (jwindow) {
 348         if (!event->in || isEnabled()) {
 349             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 350                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 351                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 352             CHECK_JNI_EXCEPTION(mainEnv)
 353         } else {
 354             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 355             CHECK_JNI_EXCEPTION(mainEnv)
 356         }
 357     }
 358 }

 360 void WindowContext::process_property_notify(GdkEventProperty *event) {
 361     if (event->window == gdk_window) {
 362         if (event->atom == atom_net_wm_state) {
 363             process_net_wm_property();
 364         } else if (event->atom == atom_net_wm_frame_extents) {
 365             if (frame_type != TITLED) {
 366                 return;
 367             }
 368 
 369             int top, left, bottom, right;


 370 
 371             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 372                 if (top + left + bottom + right > 0) {
 373                     geometry.frame_extents_received = true;
 374                     geometry.adjust_w = left + right;
 375                     geometry.adjust_h = top + bottom;
 376                     geometry.view_x = left;
 377                     geometry.view_y = top;
 378 
 379                     save_cached_extents();


 380 
 381                     // set bounds again to set to correct window size that must
 382                     // be the total width and height accounting extents
 383                     // this is ignored if size is "content size" instead of "window size"
 384                     if (geometry.window_size_set) {
 385                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 386                     }
 387 
 388                     // force position notify so java will know about view_y and view_x
 389                     size_position_notify(false, true);
 390                 }
 391             }
 392         }


 393     }
 394 }

 396 void WindowContext::process_configure() {
 397     calculate_adjustments();
 398 
 399     gint x, y, w, h, gtk_w, gtk_h;
 400 
 401     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 402     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 403     w = gtk_w + geometry.adjust_w;
 404     h = gtk_h + geometry.adjust_h;
 405 
 406     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 407     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 408                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 409 
 410     geometry.current_x = x;
 411     geometry.current_y = y;
 412     geometry.current_w = w;
 413     geometry.current_h = h;
 414     geometry.current_cw = gtk_w;
 415     geometry.current_ch = gtk_h;
 416 
 417     size_position_notify(size_changed, pos_changed);
 418 }

 420 void WindowContext::process_destroy() {
 421     if (owner) {
 422         owner->remove_child(this);
 423     }
 424 
 425     if (WindowContext::sm_mouse_drag_window == this) {
 426         ungrab_mouse_drag_focus();
 427     }
 428 
 429     if (WindowContext::sm_grab_window == this) {
 430         ungrab_focus();
 431     }
 432 
 433     std::set<WindowContext *>::iterator it;
 434     for (it = children.begin(); it != children.end(); ++it) {
 435         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 436         // WindowContext::process_destroy() to call remove_child() (because children
 437         // is being iterated here) but also prevents gtk_window_set_transient_for from
 438         // being called - this causes the crash on gnome.
 439         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 440         (*it)->set_owner(NULL);
 441         destroy_and_delete_ctx(*it);
 442     }
 443     children.clear();
 444 
 445     if (jwindow) {
 446         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 447         EXCEPTION_OCCURED(mainEnv);
 448     }
 449 
 450     if (jview) {
 451         mainEnv->DeleteGlobalRef(jview);
 452         jview = NULL;
 453     }
 454 
 455     if (jwindow) {
 456         mainEnv->DeleteGlobalRef(jwindow);
 457         jwindow = NULL;
 458     }
 459 
 460     can_be_deleted = true;
 461 }

 479 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 480     bool press = event->type == GDK_BUTTON_PRESS;
 481     bool release = event->type == GDK_BUTTON_RELEASE;
 482 
 483     guint state = event->state;
 484     guint mask = 0;
 485 
 486     // We need to add/remove current mouse button from the modifier flags
 487     // as X lib state represents the state just prior to the event and
 488     // glass needs the state just after the event
 489     switch (event->button) {
 490         case 1:
 491             mask = GDK_BUTTON1_MASK;
 492             break;
 493         case 2:
 494             mask = GDK_BUTTON2_MASK;
 495             break;
 496         case 3:
 497             mask = GDK_BUTTON3_MASK;
 498             break;
 499         case MOUSE_BACK_BTN:
 500             mask = GDK_BUTTON4_MASK;
 501             break;
 502         case MOUSE_FORWARD_BTN:
 503             mask = GDK_BUTTON5_MASK;
 504             break;
 505     }
 506 
 507     if (press) {
 508         state |= mask;
 509     } else {
 510         state &= ~mask;
 511     }
 512 











 513     // Upper layers expects from us Windows behavior:
 514     // all mouse events should be delivered to window where drag begins
 515     // and no exit/enter event should be reported during this drag.
 516     // We can grab mouse pointer for these needs.
 517     if (press) {
 518         grab_mouse_drag_focus(NULL, true);
 519     } else {
 520         if ((event->state & MOUSE_BUTTONS_MASK)
 521             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 522             ungrab_mouse_drag_focus();
 523         } else if (event->button == 8 || event->button == 9) {
 524             // GDK X backend interprets button press events for buttons 4-7 as
 525             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 526             // be set on the event->state from GDK. Thus we cannot check if all
 527             // buttons have been released in the usual way (as above).
 528             ungrab_mouse_drag_focus();
 529         }
 530     }
 531 
 532     jint button = gtk_button_number_to_mouse_button(event->button);
 533 
 534     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 535         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 536                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 537                                 button,
 538                                 (jint) event->x, (jint) event->y,
 539                                 (jint) event->x_root, (jint) event->y_root,
 540                                 gdk_modifier_mask_to_glass(state),
 541                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 542                                 JNI_FALSE);
 543         CHECK_JNI_EXCEPTION(mainEnv)
 544 
 545         if (jview && event->button == 3 && press) {
 546             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 547                                     (jint) event->x, (jint) event->y,
 548                                     (jint) event->x_root, (jint) event->y_root,
 549                                     JNI_FALSE);
 550             CHECK_JNI_EXCEPTION(mainEnv)
 551         }
 552     }
 553 }

 555 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 556     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 557     jint isDrag = glass_modifier & (
 558             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 559             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 560             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 561             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 563     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 564 
 565     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 566         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 567     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 568         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 569     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 570         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 571     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 572         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 573     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 574         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 575     }
 576 
 577     if (jview) {
 578         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 579                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 580                                 button,
 581                                 (jint) event->x, (jint) event->y,
 582                                 (jint) event->x_root, (jint) event->y_root,
 583                                 glass_modifier,
 584                                 JNI_FALSE,
 585                                 JNI_FALSE);
 586         CHECK_JNI_EXCEPTION(mainEnv)
 587     }
 588 }

 590 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 591     jdouble dx = 0;
 592     jdouble dy = 0;
 593 
 594     // converting direction to change in pixels
 595     switch (event->direction) {
 596 #if GTK_CHECK_VERSION(3, 4, 0)

 657 void WindowContext::process_key(GdkEventKey *event) {
 658     bool press = event->type == GDK_KEY_PRESS;
 659     jint glassKey = get_glass_key(event);
 660     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 661     if (press) {
 662         glassModifier |= glass_key_to_modifier(glassKey);
 663     } else {
 664         glassModifier &= ~glass_key_to_modifier(glassKey);
 665     }
 666     jcharArray jChars = NULL;
 667     jchar key = gdk_keyval_to_unicode(event->keyval);
 668     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 669         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 670     } else {
 671 #ifdef GLASS_GTK2
 672         if (key == 0) {
 673             // Work around "bug" fixed in gtk-3.0:
 674             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 675             switch (event->keyval) {
 676             case 0xFF08 /* Backspace */: key =  '\b';
 677             case 0xFF09 /* Tab       */: key =  '\t';
 678             case 0xFF0A /* Linefeed  */: key =  '\n';
 679             case 0xFF0B /* Vert. Tab */: key =  '\v';
 680             case 0xFF0D /* Return    */: key =  '\r';
 681             case 0xFF1B /* Escape    */: key =  '\033';
 682             case 0xFFFF /* Delete    */: key =  '\177';
 683             }
 684         }
 685 #endif
 686     }
 687 
 688     if (key > 0) {
 689         jChars = mainEnv->NewCharArray(1);
 690         if (jChars) {
 691             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 692             CHECK_JNI_EXCEPTION(mainEnv)
 693         }
 694     } else {
 695         jChars = mainEnv->NewCharArray(0);
 696     }
 697     if (jview) {
 698         if (press) {
 699             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 700                                     com_sun_glass_events_KeyEvent_PRESS,
 701                                     glassKey,
 702                                     jChars,
 703                                     glassModifier);
 704             CHECK_JNI_EXCEPTION(mainEnv)
 705 
 706             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 707                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 708                                         com_sun_glass_events_KeyEvent_TYPED,
 709                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 710                                         jChars,
 711                                         glassModifier);
 712                 CHECK_JNI_EXCEPTION(mainEnv)
 713             }
 714         } else {
 715             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 716                                     com_sun_glass_events_KeyEvent_RELEASE,
 717                                     glassKey,
 718                                     jChars,
 719                                     glassModifier);
 720             CHECK_JNI_EXCEPTION(mainEnv)
 721         }
 722     }
 723 }

 757 void WindowContext::process_net_wm_property() {
 758     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 759 
 760     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 761     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 762     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 763     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 764 
 765     gint length;
 766     glong *atoms = NULL;

 767 
 768     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 769                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 770 
 771         bool is_hidden = false;
 772         bool is_above = false;
 773         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 774             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 775                 is_hidden = true;
 776             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 777                 is_above = true;
 778             }
 779         }
 780 
 781         g_free(atoms);
 782 
 783         if (is_iconified != is_hidden) {
 784             is_iconified = is_hidden;
 785 
 786             notify_state((is_hidden)
 787                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 788                          : com_sun_glass_events_WindowEvent_RESTORE);
 789         }
 790 
 791         notify_on_top(is_above);
 792     }
 793 }

 975 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 976     calculate_adjustments();















 977 
 978     // newW / newH always content sizes compatible with GTK+
 979     // if window has no decoration, adjustments will be ZERO
 980     int newW = w > 0 ? w - geometry.adjust_w : cw;
 981     int newH = h > 0 ? h - geometry.adjust_h : ch;
 982 
 983     gboolean size_changed = FALSE;
 984     gboolean pos_changed = FALSE;

 985 
 986     geometry.window_size_set = (w > 0 || h > 0);


 987 
 988     if (newW > 0 && newH > 0) {
 989         size_changed = TRUE;

























































































 990 
 991         geometry.current_cw = newW;
 992         geometry.current_ch = newH;
 993         geometry.current_w = newW + geometry.adjust_w;
 994         geometry.current_h = newH + geometry.adjust_h;
 995 
 996         if (visible_received) {
 997             // update constraints if not resized by the user interface so it will
 998             // let gtk_window_resize succeed, because it's bound to geometry constraints
 999             apply_geometry();
1000             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1001         } else {
1002             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);





1003         }
1004     }

1005 
1006     if (xSet || ySet) {
1007         int newX = (xSet) ? x : geometry.current_x;
1008         int newY = (ySet) ? y : geometry.current_y;
1009 
1010         if (newX != geometry.current_x || newY != geometry.current_y) {
1011             pos_changed = TRUE;
1012             geometry.current_x = newX;
1013             geometry.current_y = newY;
1014             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1015         }
1016     }

1017 
1018     size_position_notify(size_changed, pos_changed);
1019 }

1213 
1214     gint x, y;
1215     gdk_window_get_origin(gdk_window, &x, &y);
1216 
1217     gint rx, ry;
1218     gdk_window_get_root_origin(gdk_window, &rx, &ry);

1219 
1220     if (rx != x || ry != y) {
1221         // the left extends are correct - the right one is guessed to be the same
1222         geometry.adjust_w = (rx - x) * 2;
1223         // guess that bottom size is the same as left and right
1224         geometry.adjust_h = (ry - y) + (rx - x);


1225 
1226         // those will be correct
1227         geometry.view_x = (rx - x);
1228         geometry.view_y = (ry - y);
1229 
1230         save_cached_extents();
1231     } else {
1232         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1233         geometry.adjust_w = c.adjust_w;
1234         geometry.adjust_h = c.adjust_h;
1235         geometry.view_x = c.view_x;
1236         geometry.view_y = c.view_y;
1237     }
1238 
1239     apply_geometry();
1240 }

1260 void WindowContext::apply_geometry() {
1261     if (!map_received) {
1262         return;








1263     }
1264 
1265     GdkGeometry gdk_geometry;
1266     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;


1267 
1268     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1269         // not resizeable
1270         int w = (geometry.current_w - geometry.adjust_w) > 0
1271                 ? geometry.current_w - geometry.adjust_w
1272                 : geometry.current_cw;

1273 
1274         int h = (geometry.current_h - geometry.adjust_h) > 0
1275                 ? geometry.current_h - geometry.adjust_h
1276                 : geometry.current_ch;













1277 
1278         gdk_geometry.min_width = gdk_geometry.max_width = w;
1279         gdk_geometry.min_height = gdk_geometry.max_height = h;
1280     } else {
1281         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1282                                  ? geometry.minw - geometry.adjust_w : 1;
1283         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1284                                   ? geometry.minh - geometry.adjust_h : 1;
1285 
1286         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1287                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1288         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1289                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1290     }
1291 
1292     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1293                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1294 }

1383     WindowContext *o = owner;

1388             return true;
