  79 void WindowContextBase::notify_state(jint glass_state) {
  80     if (glass_state == com_sun_glass_events_WindowEvent_RESTORE) {
  81         if (is_maximized) {
  82             glass_state = com_sun_glass_events_WindowEvent_MAXIMIZE;
  83         }
  84 
  85         int w, h;
  86         glass_gdk_window_get_size(gdk_window, &w, &h);
  87         if (jview) {
  88             mainEnv->CallVoidMethod(jview,
  89                     jViewNotifyRepaint,
  90                     0, 0, w, h);
  91             CHECK_JNI_EXCEPTION(mainEnv);
  92         }
  93     }
  94 
  95     if (jwindow) {
  96        mainEnv->CallVoidMethod(jwindow,
  97                jGtkWindowNotifyStateChanged,
  98                glass_state);
  99        CHECK_JNI_EXCEPTION(mainEnv);
 100     }

 103 void WindowContextBase::process_state(GdkEventWindowState* event) {
 104     if (event->changed_mask &
 105             (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 106 
 107         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 108             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 109         }
 110         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 111             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 112         }
 113 
 114         jint stateChangeEvent;


 115 
 116         if (is_iconified) {
 117             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 118         } else if (is_maximized) {
 119             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 120         } else {
 121             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 122             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 123                 // in this case - the window manager will not support the programatic
 124                 // request to iconify - so we need to restore it now.
 125                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 126             }
 127         }
 128 
 129         notify_state(stateChangeEvent);
 130     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 131         notify_on_top( event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 132     }
 133 }

 135 void WindowContextBase::process_focus(GdkEventFocus* event) {
 136     if (!event->in && WindowContextBase::sm_mouse_drag_window == this) {
 137         ungrab_mouse_drag_focus();
 138     }
 139     if (!event->in && WindowContextBase::sm_grab_window == this) {

 140         ungrab_focus();
 141     }
 142 
 143     if (xim.enabled && xim.ic) {
 144         if (event->in) {
 145             XSetICFocus(xim.ic);
 146         } else {
 147             XUnsetICFocus(xim.ic);
 148         }
 149     }
 150 
 151     if (jwindow) {
 152         if (!event->in || isEnabled()) {
 153             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 154                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED : com_sun_glass_events_WindowEvent_FOCUS_LOST);

 155             CHECK_JNI_EXCEPTION(mainEnv)
 156         } else {
 157             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 158             CHECK_JNI_EXCEPTION(mainEnv)
 159         }
 160     }
 161 }

 191 void WindowContextBase::process_destroy() {
 192     if (WindowContextBase::sm_mouse_drag_window == this) {




 193         ungrab_mouse_drag_focus();
 194     }
 195 
 196     if (WindowContextBase::sm_grab_window == this) {
 197         ungrab_focus();
 198     }
 199 
 200     std::set<WindowContextTop*>::iterator it;
 201     for (it = children.begin(); it != children.end(); ++it) {
 202         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 203         // WindowContextTop::process_destroy() to call remove_child() (because children
 204         // is being iterated here) but also prevents gtk_window_set_transient_for from
 205         // being called - this causes the crash on gnome.
 206         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 207         (*it)->set_owner(NULL);
 208         destroy_and_delete_ctx(*it);
 209     }
 210     children.clear();
 211 
 212     if (jwindow) {
 213         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 214         EXCEPTION_OCCURED(mainEnv);
 215     }
 216 
 217     if (jview) {
 218         mainEnv->DeleteGlobalRef(jview);
 219         jview = NULL;
 220     }
 221 
 222     if (jwindow) {
 223         mainEnv->DeleteGlobalRef(jwindow);
 224         jwindow = NULL;
 225     }
 226 
 227     can_be_deleted = true;
 228 }

 244 static inline jint gtk_button_number_to_mouse_button(guint button) {
 245     switch (button) {
 246         case 1:
 247             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 248         case 2:
 249             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 250         case 3:
 251             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 252         case MOUSE_BACK_BTN:
 253             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 254         case MOUSE_FORWARD_BTN:
 255             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 256         default:
 257             // Other buttons are not supported by quantum and are not reported by other platforms
 258             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 259     }
 260 }

 262 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 263     bool press = event->type == GDK_BUTTON_PRESS;


 264     guint state = event->state;
 265     guint mask = 0;
 266 
 267     // We need to add/remove current mouse button from the modifier flags
 268     // as X lib state represents the state just prior to the event and
 269     // glass needs the state just after the event
 270     switch (event->button) {
 271         case 1:
 272             mask = GDK_BUTTON1_MASK;
 273             break;
 274         case 2:
 275             mask = GDK_BUTTON2_MASK;
 276             break;
 277         case 3:
 278             mask = GDK_BUTTON3_MASK;
 279             break;
 280         case MOUSE_BACK_BTN:
 281             mask = GDK_BUTTON4_MASK;
 282             break;
 283         case MOUSE_FORWARD_BTN:
 284             mask = GDK_BUTTON5_MASK;
 285             break;
 286     }
 287 
 288     if (press) {
 289         state |= mask;
 290     } else {
 291         state &= ~mask;
 292     }
 293 
 294     if (press) {
 295         GdkDevice* device = event->device;
 296 
 297         if (glass_gdk_device_is_grabbed(device)
 298                 && (glass_gdk_device_get_window_at_position(device, NULL, NULL)
 299                 == NULL)) {
 300             ungrab_focus();
 301             return;
 302         }
 303     }
 304 
 305     // Upper layers expects from us Windows behavior:
 306     // all mouse events should be delivered to window where drag begins
 307     // and no exit/enter event should be reported during this drag.
 308     // We can grab mouse pointer for these needs.
 309     if (press) {
 310         grab_mouse_drag_focus();
 311     } else {
 312         if ((event->state & MOUSE_BUTTONS_MASK)
 313             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 314             ungrab_mouse_drag_focus();
 315         } else if (event->button == 8 || event->button == 9) {
 316             // GDK X backend interprets button press events for buttons 4-7 as
 317             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 318             // be set on the event->state from GDK. Thus we cannot check if all
 319             // buttons have been released in the usual way (as above).
 320             ungrab_mouse_drag_focus();
 321         }
 322     }
 323 
 324     jint button = gtk_button_number_to_mouse_button(event->button);
 325 
 326     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 327         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 328                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 329                 button,
 330                 (jint) event->x, (jint) event->y,
 331                 (jint) event->x_root, (jint) event->y_root,
 332                 gdk_modifier_mask_to_glass(state),
 333                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 334                 JNI_FALSE);
 335         CHECK_JNI_EXCEPTION(mainEnv)
 336 
 337         if (jview && event->button == 3 && press) {
 338             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 339                     (jint)event->x, (jint)event->y,
 340                     (jint)event->x_root, (jint)event->y_root,
 341                     JNI_FALSE);
 342             CHECK_JNI_EXCEPTION(mainEnv)
 343         }
 344     }
 345 }

 347 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 348     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 349     jint isDrag = glass_modifier & (
 350             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 351             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 352             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 353             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 354             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 355     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 356 
 357     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 358         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 359     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 360         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 361     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 362         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 363     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 364         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 365     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 366         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 367     }
 368 
 369     if (jview) {
 370         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 371                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 372                 button,
 373                 (jint) event->x, (jint) event->y,
 374                 (jint) event->x_root, (jint) event->y_root,
 375                 glass_modifier,
 376                 JNI_FALSE,
 377                 JNI_FALSE);
 378         CHECK_JNI_EXCEPTION(mainEnv)
 379     }
 380 }

 382 void WindowContextBase::process_mouse_scroll(GdkEventScroll* event) {
 383     jdouble dx = 0;
 384     jdouble dy = 0;
 385 
 386     // converting direction to change in pixels
 387     switch (event->direction) {
 388 #if GTK_CHECK_VERSION(3, 4, 0)
 389         case GDK_SCROLL_SMOOTH:
 390             //FIXME 3.4 ???
 391             break;
 392 #endif
 393         case GDK_SCROLL_UP:
 394             dy = 1;
 395             break;
 396         case GDK_SCROLL_DOWN:
 397             dy = -1;
 398             break;
 399         case GDK_SCROLL_LEFT:
 400             dx = 1;
 401             break;
 402         case GDK_SCROLL_RIGHT:
 403             dx = -1;
 404             break;
 405     }
 406     if (event->state & GDK_SHIFT_MASK) {
 407         jdouble t = dy;
 408         dy = dx;
 409         dx = t;
 410     }
 411     if (jview) {
 412         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 413                 (jint) event->x, (jint) event->y,
 414                 (jint) event->x_root, (jint) event->y_root,
 415                 dx, dy,
 416                 gdk_modifier_mask_to_glass(event->state),
 417                 (jint) 0, (jint) 0,
 418                 (jint) 0, (jint) 0,
 419                 (jdouble) 40.0, (jdouble) 40.0);
 420         CHECK_JNI_EXCEPTION(mainEnv)
 421     }
 422 
 423 }

 425 void WindowContextBase::process_mouse_cross(GdkEventCrossing* event) {
 426     bool enter = event->type == GDK_ENTER_NOTIFY;

 427     if (jview) {
 428         guint state = event->state;
 429         if (enter) { // workaround for RT-21590
 430             state &= ~MOUSE_BUTTONS_MASK;
 431         }
 432 
 433         if (enter != is_mouse_entered) {
 434             is_mouse_entered = enter;
 435             mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 436                     enter ? com_sun_glass_events_MouseEvent_ENTER : com_sun_glass_events_MouseEvent_EXIT,
 437                     com_sun_glass_events_MouseEvent_BUTTON_NONE,
 438                     (jint) event->x, (jint) event->y,
 439                     (jint) event->x_root, (jint) event->y_root,
 440                     gdk_modifier_mask_to_glass(state),
 441                     JNI_FALSE,
 442                     JNI_FALSE);

 443             CHECK_JNI_EXCEPTION(mainEnv)
 444         }
 445     }

 448 void WindowContextBase::process_key(GdkEventKey* event) {
 449     bool press = event->type == GDK_KEY_PRESS;
 450     jint glassKey = get_glass_key(event);
 451     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 452     if (press) {
 453         glassModifier |= glass_key_to_modifier(glassKey);
 454     } else {
 455         glassModifier &= ~glass_key_to_modifier(glassKey);
 456     }
 457     jcharArray jChars = NULL;
 458     jchar key = gdk_keyval_to_unicode(event->keyval);
 459     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 460         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 461     } else {
 462 #ifdef GLASS_GTK2
 463         if (key == 0) {
 464             // Work around "bug" fixed in gtk-3.0:
 465             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 466             switch (event->keyval) {
 467             case 0xFF08 /* Backspace */: key =  '\b';
 468             case 0xFF09 /* Tab       */: key =  '\t';

 471             case 0xFF0D /* Return    */: key =  '\r';
 472             case 0xFF1B /* Escape    */: key =  '\033';
 473             case 0xFFFF /* Delete    */: key =  '\177';
 474             }
 475         }
 476 #endif
 477     }
 478 
 479     if (key > 0) {
 480         jChars = mainEnv->NewCharArray(1);
 481         if (jChars) {
 482             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 483             CHECK_JNI_EXCEPTION(mainEnv)
 484         }
 485     } else {
 486         jChars = mainEnv->NewCharArray(0);
 487     }
 488     if (jview) {
 489         if (press) {
 490             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 491                     com_sun_glass_events_KeyEvent_PRESS,
 492                     glassKey,
 493                     jChars,
 494                     glassModifier);
 495             CHECK_JNI_EXCEPTION(mainEnv)
 496 
 497             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 498                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 499                         com_sun_glass_events_KeyEvent_TYPED,
 500                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 501                         jChars,
 502                         glassModifier);
 503                 CHECK_JNI_EXCEPTION(mainEnv)
 504             }
 505         } else {
 506             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 507                     com_sun_glass_events_KeyEvent_RELEASE,
 508                     glassKey,
 509                     jChars,
 510                     glassModifier);
 511             CHECK_JNI_EXCEPTION(mainEnv)
 512         }
 513     }
 514 }

 516 void WindowContextBase::paint(void* data, jint width, jint height)
 517 {
 518     if (!is_visible()) {
 519         return;
 520     }
 521 #ifdef GLASS_GTK3
 522     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 523     gdk_window_begin_paint_region(gdk_window, region);
 524 #endif
 525     cairo_t* context;
 526     context = gdk_cairo_create(gdk_window);
 527 
 528     cairo_surface_t* cairo_surface;
 529     cairo_surface = cairo_image_surface_create_for_data(
 530             (unsigned char*)data,
 531             CAIRO_FORMAT_ARGB32,
 532             width, height, width * 4);
 533 
 534     applyShapeMask(data, width, height);
 535 
 536     cairo_set_source_surface(context, cairo_surface, 0, 0);
 537     cairo_set_operator (context, CAIRO_OPERATOR_SOURCE);
 538     cairo_paint(context);
 539 #ifdef GLASS_GTK3
 540     gdk_window_end_paint(gdk_window);
 541     cairo_region_destroy(region);
 542 #endif
 543 
 544     cairo_destroy(context);
 545     cairo_surface_destroy(cairo_surface);
 546 }

 621 bool WindowContextBase::grab_mouse_drag_focus() {
 622     if (glass_gdk_mouse_devices_grab_with_cursor(
 623             gdk_window, gdk_window_get_cursor(gdk_window), FALSE)) {
 624         WindowContextBase::sm_mouse_drag_window = this;
 625         return true;
 626     } else {
 627         return false;

 632     WindowContextBase::sm_mouse_drag_window = NULL;
 633     glass_gdk_mouse_devices_ungrab();
 634     if (WindowContextBase::sm_grab_window) {
 635         WindowContextBase::sm_grab_window->grab_focus();
 636     }

 971     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");

 476 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 477     bool press = event->type == GDK_BUTTON_PRESS;
 478     bool release = event->type == GDK_BUTTON_RELEASE;
 479 
 480     guint state = event->state;
 481     guint mask = 0;
 482 
 483     // We need to add/remove current mouse button from the modifier flags
 484     // as X lib state represents the state just prior to the event and
 485     // glass needs the state just after the event
 486     switch (event->button) {
 487         case 1:
 488             mask = GDK_BUTTON1_MASK;
 489             break;
 490         case 2:
 491             mask = GDK_BUTTON2_MASK;
 492             break;
 493         case 3:
 494             mask = GDK_BUTTON3_MASK;
 495             break;
 496         case MOUSE_BACK_BTN:
 497             mask = GDK_BUTTON4_MASK;
 498             break;
 499         case MOUSE_FORWARD_BTN:
 500             mask = GDK_BUTTON5_MASK;
 501             break;
 502     }
 503 
 504     if (press) {
 505         state |= mask;
 506     } else {
 507         state &= ~mask;
 508     }
 509 











 510     // Upper layers expects from us Windows behavior:
 511     // all mouse events should be delivered to window where drag begins
 512     // and no exit/enter event should be reported during this drag.
 513     // We can grab mouse pointer for these needs.
 514     if (press) {
 515         grab_mouse_drag_focus(NULL, true);
 516     } else {
 517         if ((event->state & MOUSE_BUTTONS_MASK)
 518             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 519             ungrab_mouse_drag_focus();
 520         } else if (event->button == 8 || event->button == 9) {
 521             // GDK X backend interprets button press events for buttons 4-7 as
 522             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 523             // be set on the event->state from GDK. Thus we cannot check if all
 524             // buttons have been released in the usual way (as above).
 525             ungrab_mouse_drag_focus();
 526         }
 527     }
 528 
 529     jint button = gtk_button_number_to_mouse_button(event->button);
 530 
 531     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 532         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 533                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 534                                 button,
 535                                 (jint) event->x, (jint) event->y,
 536                                 (jint) event->x_root, (jint) event->y_root,
 537                                 gdk_modifier_mask_to_glass(state),
 538                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 539                                 JNI_FALSE);
 540         CHECK_JNI_EXCEPTION(mainEnv)
 541 
 542         if (jview && event->button == 3 && press) {
 543             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 544                                     (jint) event->x, (jint) event->y,
 545                                     (jint) event->x_root, (jint) event->y_root,
 546                                     JNI_FALSE);

 654 void WindowContext::process_key(GdkEventKey *event) {
 655     bool press = event->type == GDK_KEY_PRESS;
 656     jint glassKey = get_glass_key(event);
 657     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 658     if (press) {
 659         glassModifier |= glass_key_to_modifier(glassKey);
 660     } else {
 661         glassModifier &= ~glass_key_to_modifier(glassKey);
 662     }
 663     jcharArray jChars = NULL;
 664     jchar key = gdk_keyval_to_unicode(event->keyval);
 665     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 666         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 667     } else {
 668 #ifdef GLASS_GTK2
 669         if (key == 0) {
 670             // Work around "bug" fixed in gtk-3.0:
 671             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 672             switch (event->keyval) {
 673             case 0xFF08 /* Backspace */: key =  '\b';
 674             case 0xFF09 /* Tab       */: key =  '\t';

 677             case 0xFF0D /* Return    */: key =  '\r';
 678             case 0xFF1B /* Escape    */: key =  '\033';
 679             case 0xFFFF /* Delete    */: key =  '\177';
 680             }
 681         }
 682 #endif
 683     }
 684 
 685     if (key > 0) {
 686         jChars = mainEnv->NewCharArray(1);
 687         if (jChars) {
 688             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 689             CHECK_JNI_EXCEPTION(mainEnv)
 690         }
 691     } else {
 692         jChars = mainEnv->NewCharArray(0);
 693     }
 694     if (jview) {
 695         if (press) {
 696             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 697                                     com_sun_glass_events_KeyEvent_PRESS,
 698                                     glassKey,
 699                                     jChars,
 700                                     glassModifier);
 701             CHECK_JNI_EXCEPTION(mainEnv)
 702 
 703             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 704                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 705                                         com_sun_glass_events_KeyEvent_TYPED,
 706                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 707                                         jChars,
 708                                         glassModifier);
 709                 CHECK_JNI_EXCEPTION(mainEnv)
 710             }
 711         } else {
 712             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 713                                     com_sun_glass_events_KeyEvent_RELEASE,
 714                                     glassKey,
 715                                     jChars,
 716                                     glassModifier);
 717             CHECK_JNI_EXCEPTION(mainEnv)
 718         }
 719     }
 720 }

 754 void WindowContext::process_net_wm_property() {
 755     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 756 
 757     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 758     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 759     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 760     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 761 
 762     gint length;
 763     glong *atoms = NULL;

 764 
 765     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 766                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 767 
 768         bool is_hidden = false;
 769         bool is_above = false;
 770         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 771             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 772                 is_hidden = true;
 773             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 774                 is_above = true;
 775             }
 776         }
 777 
 778         g_free(atoms);
 779 
 780         if (is_iconified != is_hidden) {
 781             is_iconified = is_hidden;
 782 
 783             notify_state((is_hidden)
 784                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 785                          : com_sun_glass_events_WindowEvent_RESTORE);
 786         }
 787 
 788         notify_on_top(is_above);
 789     }

1202 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1203 // Only useful if the window manager does not support the frame extents extension - in this case
1204 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1205 // if that's not available.
1206 void WindowContext::calculate_adjustments() {
1207     if (frame_type != TITLED || geometry.frame_extents_received) {
1208         return;

1209     }
1210 
1211     gint x, y;
1212     gdk_window_get_origin(gdk_window, &x, &y);
1213 
1214     gint rx, ry;
1215     gdk_window_get_root_origin(gdk_window, &rx, &ry);

1216 
1217     if (rx != x || ry != y) {
1218         // the left extends are correct - the right one is guessed to be the same
1219         geometry.adjust_w = (rx - x) * 2;
1220         // guess that bottom size is the same as left and right
1221         geometry.adjust_h = (ry - y) + (rx - x);


1222 
1223         // those will be correct
1224         geometry.view_x = (rx - x);
1225         geometry.view_y = (ry - y);
1226 
1227         save_cached_extents();
1228     } else {
1229         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1230         geometry.adjust_w = c.adjust_w;
1231         geometry.adjust_h = c.adjust_h;
1232         geometry.view_x = c.view_x;
1233         geometry.view_y = c.view_y;
1234     }
1235 
1236     apply_geometry();
1237 }

1257 void WindowContext::apply_geometry() {
1258     if (!map_received) {
1259         return;








1260     }
1261 
1262     GdkGeometry gdk_geometry;
1263     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;


1264 
1265     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1266         // not resizeable

1268                 ? geometry.current_w - geometry.adjust_w

1270

1272                 ? geometry.current_h - geometry.adjust_h

1276         gdk_geometry.min_height = gdk_geometry.max_height = h;

1279                                  ? geometry.minw - geometry.adjust_w : 1;
1280         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1281                                   ? geometry.minh - geometry.adjust_h : 1;

1286                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1287     }
