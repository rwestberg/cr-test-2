 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 330 void WindowContext::process_focus(GdkEventFocus *event) {
 331     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 332         ungrab_mouse_drag_focus();
 333     }
 334 
 335     if (!event->in && WindowContext::sm_grab_window == this) {
 336         ungrab_focus();
 337     }
 338 
 339     if (xim.enabled && xim.ic) {
 340         if (event->in) {
 341             XSetICFocus(xim.ic);
 342         } else {
 343             XUnsetICFocus(xim.ic);
 344         }
 345     }
 346 
 347     if (jwindow) {
 348         if (!event->in || isEnabled()) {
 349             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 350                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 351                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 352             CHECK_JNI_EXCEPTION(mainEnv)
 353         } else {
 354             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 355             CHECK_JNI_EXCEPTION(mainEnv)
 356         }
 357     }
 358 }

 360 void WindowContext::process_property_notify(GdkEventProperty *event) {
 361     if (event->window == gdk_window) {
 362         if (event->atom == atom_net_wm_state) {
 363             process_net_wm_property();
 364         } else if (event->atom == atom_net_wm_frame_extents) {
 365             if (frame_type != TITLED) {
 366                 return;
 367             }
 368 
 369             int top, left, bottom, right;
 370 
 371             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 372                 if (top + left + bottom + right > 0) {
 373                     geometry.frame_extents_received = true;
 374                     geometry.adjust_w = left + right;
 375                     geometry.adjust_h = top + bottom;
 376                     geometry.view_x = left;
 377                     geometry.view_y = top;
 378 
 379                     save_cached_extents();
 380 
 381                     // set bounds again to set to correct window size that must
 382                     // be the total width and height accounting extents
 383                     // this is ignored if size is "content size" instead of "window size"
 384                     if (geometry.window_size_set) {
 385                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 386                     }
 387                 }
 388             }
 389         }
 390     }
 391 }

 393 void WindowContext::process_configure() {
 394     calculate_adjustments();

 395 
 396     gint x, y, w, h, gtk_w, gtk_h;


 397 
 398     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 399     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 400     w = gtk_w + geometry.adjust_w;
 401     h = gtk_h + geometry.adjust_h;
 402 
 403     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 404     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 405                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 406 
 407     geometry.current_x = x;
 408     geometry.current_y = y;
 409     geometry.current_w = w;
 410     geometry.current_h = h;
 411     geometry.current_cw = gtk_w;
 412     geometry.current_ch = gtk_h;
 413 
 414     size_position_notify(size_changed, pos_changed);
 415 }

 417 void WindowContext::process_destroy() {
 418     if (owner) {
 419         owner->remove_child(this);
 420     }
 421 
 422     if (WindowContext::sm_mouse_drag_window == this) {
 423         ungrab_mouse_drag_focus();
 424     }
 425 
 426     if (WindowContext::sm_grab_window == this) {
 427         ungrab_focus();
 428     }
 429 
 430     std::set<WindowContext *>::iterator it;
 431     for (it = children.begin(); it != children.end(); ++it) {
 432         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 433         // WindowContext::process_destroy() to call remove_child() (because children
 434         // is being iterated here) but also prevents gtk_window_set_transient_for from
 435         // being called - this causes the crash on gnome.
 436         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 437         (*it)->set_owner(NULL);
 438         destroy_and_delete_ctx(*it);
 439     }
 440     children.clear();
 441 
 442     if (jwindow) {
 443         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 444         EXCEPTION_OCCURED(mainEnv);
 445     }
 446 
 447     if (jview) {
 448         mainEnv->DeleteGlobalRef(jview);
 449         jview = NULL;
 450     }
 451 
 452     if (jwindow) {
 453         mainEnv->DeleteGlobalRef(jwindow);
 454         jwindow = NULL;
 455     }
 456 
 457     can_be_deleted = true;
 458 }

 476 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 477     bool press = event->type == GDK_BUTTON_PRESS;
 478     bool release = event->type == GDK_BUTTON_RELEASE;
 479 
 480     guint state = event->state;
 481     guint mask = 0;
 482 
 483     // We need to add/remove current mouse button from the modifier flags
 484     // as X lib state represents the state just prior to the event and
 485     // glass needs the state just after the event
 486     switch (event->button) {
 487         case 1:
 488             mask = GDK_BUTTON1_MASK;
 489             break;
 490         case 2:
 491             mask = GDK_BUTTON2_MASK;
 492             break;
 493         case 3:
 494             mask = GDK_BUTTON3_MASK;
 495             break;
 496         case MOUSE_BACK_BTN:
 497             mask = GDK_BUTTON4_MASK;
 498             break;
 499         case MOUSE_FORWARD_BTN:
 500             mask = GDK_BUTTON5_MASK;
 501             break;
 502     }
 503 
 504     if (press) {
 505         state |= mask;
 506     } else {
 507         state &= ~mask;
 508     }
 509 











 510     // Upper layers expects from us Windows behavior:
 511     // all mouse events should be delivered to window where drag begins
 512     // and no exit/enter event should be reported during this drag.
 513     // We can grab mouse pointer for these needs.
 514     if (press) {
 515         grab_mouse_drag_focus(NULL, true);
 516     } else {
 517         if ((event->state & MOUSE_BUTTONS_MASK)
 518             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 519             ungrab_mouse_drag_focus();
 520         } else if (event->button == 8 || event->button == 9) {
 521             // GDK X backend interprets button press events for buttons 4-7 as
 522             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 523             // be set on the event->state from GDK. Thus we cannot check if all
 524             // buttons have been released in the usual way (as above).
 525             ungrab_mouse_drag_focus();
 526         }
 527     }
 528 
 529     jint button = gtk_button_number_to_mouse_button(event->button);
 530 
 531     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 532         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 533                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 534                                 button,
 535                                 (jint) event->x, (jint) event->y,
 536                                 (jint) event->x_root, (jint) event->y_root,
 537                                 gdk_modifier_mask_to_glass(state),
 538                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 539                                 JNI_FALSE);
 540         CHECK_JNI_EXCEPTION(mainEnv)
 541 
 542         if (jview && event->button == 3 && press) {
 543             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 544                                     (jint) event->x, (jint) event->y,
 545                                     (jint) event->x_root, (jint) event->y_root,
 546                                     JNI_FALSE);
 547             CHECK_JNI_EXCEPTION(mainEnv)
 548         }
 549     }
 550 }

 552 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 553     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 554     jint isDrag = glass_modifier & (
 555             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 556             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 557             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 558             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 559             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 560     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 561 
 562     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 563         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 564     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 565         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 566     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 567         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 568     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 570     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 572     }
 573 
 574     if (jview) {
 575         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 576                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 577                                 button,
 578                                 (jint) event->x, (jint) event->y,
 579                                 (jint) event->x_root, (jint) event->y_root,
 580                                 glass_modifier,
 581                                 JNI_FALSE,
 582                                 JNI_FALSE);
 583         CHECK_JNI_EXCEPTION(mainEnv)
 584     }
 585 }

 587 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 588     jdouble dx = 0;
 589     jdouble dy = 0;
 590 
 591     // converting direction to change in pixels
 592     switch (event->direction) {
 593 #if GTK_CHECK_VERSION(3, 4, 0)
 594         case GDK_SCROLL_SMOOTH:
 595             //FIXME 3.4 ???
 596             break;
 597 #endif

 654 void WindowContext::process_key(GdkEventKey *event) {
 655     bool press = event->type == GDK_KEY_PRESS;
 656     jint glassKey = get_glass_key(event);
 657     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 658     if (press) {
 659         glassModifier |= glass_key_to_modifier(glassKey);
 660     } else {
 661         glassModifier &= ~glass_key_to_modifier(glassKey);
 662     }
 663     jcharArray jChars = NULL;
 664     jchar key = gdk_keyval_to_unicode(event->keyval);
 665     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 666         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 667     } else {
 668 #ifdef GLASS_GTK2
 669         if (key == 0) {
 670             // Work around "bug" fixed in gtk-3.0:
 671             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 672             switch (event->keyval) {
 673             case 0xFF08 /* Backspace */: key =  '\b';
 674             case 0xFF09 /* Tab       */: key =  '\t';
 675             case 0xFF0A /* Linefeed  */: key =  '\n';
 676             case 0xFF0B /* Vert. Tab */: key =  '\v';
 677             case 0xFF0D /* Return    */: key =  '\r';
 678             case 0xFF1B /* Escape    */: key =  '\033';
 679             case 0xFFFF /* Delete    */: key =  '\177';
 680             }
 681         }
 682 #endif
 683     }
 684 
 685     if (key > 0) {
 686         jChars = mainEnv->NewCharArray(1);
 687         if (jChars) {
 688             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 689             CHECK_JNI_EXCEPTION(mainEnv)
 690         }
 691     } else {
 692         jChars = mainEnv->NewCharArray(0);
 693     }
 694     if (jview) {
 695         if (press) {
 696             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 697                                     com_sun_glass_events_KeyEvent_PRESS,
 698                                     glassKey,
 699                                     jChars,
 700                                     glassModifier);
 701             CHECK_JNI_EXCEPTION(mainEnv)
 702 
 703             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 704                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 705                                         com_sun_glass_events_KeyEvent_TYPED,
 706                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 707                                         jChars,
 708                                         glassModifier);
 709                 CHECK_JNI_EXCEPTION(mainEnv)
 710             }
 711         } else {
 712             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 713                                     com_sun_glass_events_KeyEvent_RELEASE,
 714                                     glassKey,
 715                                     jChars,
 716                                     glassModifier);
 717             CHECK_JNI_EXCEPTION(mainEnv)
 718         }
 719     }
 720 }

 754 void WindowContext::process_net_wm_property() {
 755     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 756 
 757     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 758     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 759     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 760     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 761 
 762     gint length;
 763     glong *atoms = NULL;

 764 
 765     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 766                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 767 
 768         bool is_hidden = false;
 769         bool is_above = false;
 770         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 771             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 772                 is_hidden = true;
 773             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 774                 is_above = true;
 775             }
 776         }
 777 
 778         g_free(atoms);
 779 
 780         if (is_iconified != is_hidden) {
 781             is_iconified = is_hidden;
 782 
 783             notify_state((is_hidden)
 784                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 785                          : com_sun_glass_events_WindowEvent_RESTORE);
 786         }
 787 
 788         notify_on_top(is_above);
 789     }

 972 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 973     calculate_adjustments();

















































































 974 
 975     // newW / newH always content sizes compatible with GTK+
 976     // if window has no decoration, adjustments will be ZERO
 977     int newW = w > 0 ? w - geometry.adjust_w : cw;
 978     int newH = h > 0 ? h - geometry.adjust_h : ch;




 979 
 980     gboolean size_changed = FALSE;
 981     gboolean pos_changed = FALSE;





 982 
 983     geometry.window_size_set = (w > 0 || h > 0);











 984 
 985     if (newW > 0 && newH > 0) {
 986         size_changed = TRUE;





 987 
 988         geometry.current_cw = newW;
 989         geometry.current_ch = newH;
 990         geometry.current_w = newW + geometry.adjust_w;
 991         geometry.current_h = newH + geometry.adjust_h;
 992 
 993         if (visible_received) {
 994             // update constraints if not resized by the user interface so it will
 995             // let gtk_window_resize succeed, because it's bound to geometry constraints
 996             apply_geometry();
 997             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 998         } else {
 999             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);





1000         }
1001     }

1002 
1003     if (xSet || ySet) {
1004         int newX = (xSet) ? x : geometry.current_x;
1005         int newY = (ySet) ? y : geometry.current_y;
1006 
1007         if (newX != geometry.current_x || newY != geometry.current_y) {
1008             pos_changed = TRUE;
1009             geometry.current_x = newX;
1010             geometry.current_y = newY;
1011             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1012         }
1013     }

1014 
1015     size_position_notify(size_changed, pos_changed);
1016 }

1202 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1203 // Only useful if the window manager does not support the frame extents extension - in this case
1204 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder

1208         return;

1209     }
1210 
1211     gint x, y;
1212     gdk_window_get_origin(gdk_window, &x, &y);
1213 
1214     gint rx, ry;
1215     gdk_window_get_root_origin(gdk_window, &rx, &ry);

1216 
1217     if (rx != x || ry != y) {
1218         // the left extends are correct - the right one is guessed to be the same
1219         geometry.adjust_w = (rx - x) * 2;
1220         // guess that bottom size is the same as left and right
1221         geometry.adjust_h = (ry - y) + (rx - x);


1222 
1223         // those will be correct
1224         geometry.view_x = (rx - x);
1225         geometry.view_y = (ry - y);
1226 
1227         save_cached_extents();
1228     } else {
1229         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1230         geometry.adjust_w = c.adjust_w;
1231         geometry.adjust_h = c.adjust_h;
1232         geometry.view_x = c.view_x;
1233         geometry.view_y = c.view_y;
1234     }
1235 
1236     apply_geometry();
1237 }

1246         normal_extents.adjust_h = geometry.adjust_h;

1257 void WindowContext::apply_geometry() {
1258     if (!map_received) {
1259         return;








1260     }
1261 
1262     GdkGeometry gdk_geometry;
1263     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;


1264 
1265     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1266         // not resizeable
1267         int w = (geometry.current_w - geometry.adjust_w) > 0
1268                 ? geometry.current_w - geometry.adjust_w
1269                 : geometry.current_cw;

1270 
1271         int h = (geometry.current_h - geometry.adjust_h) > 0
1272                 ? geometry.current_h - geometry.adjust_h
1273                 : geometry.current_ch;













1274 
1275         gdk_geometry.min_width = gdk_geometry.max_width = w;
1276         gdk_geometry.min_height = gdk_geometry.max_height = h;
1277     } else {
1278         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1279                                  ? geometry.minw - geometry.adjust_w : 1;
1280         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1281                                   ? geometry.minh - geometry.adjust_h : 1;
1282 
1283         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1284                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1285         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1286                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1287     }
1288 
1289     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1290                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1291 }

1379 bool WindowContext::on_top_inherited() {
1380     WindowContext *o = owner;
1381     while (o) {
1382         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1383         if (!topO) break;
1384         if (topO->on_top) {
1385             return true;
1386         }
1387         o = topO->owner;
1388     }
1389     return false;
