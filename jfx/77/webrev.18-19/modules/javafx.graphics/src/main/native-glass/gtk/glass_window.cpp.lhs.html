 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 173 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 174                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 175         screen(_screen),
 176         frame_type(_frame_type),
 177         window_type(type),
 178         owner(_owner),
 179         jview(NULL),
 180         map_received(false),
 181         visible_received(false),
 182         on_top(false),
 183         is_fullscreen(false),
 184         is_iconified(false),
 185         is_maximized(false),
 186         is_mouse_entered(false),
 187         can_be_deleted(false),
 188         events_processing_cnt(0),
 189         win_group(NULL),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (type != POPUP && owner == NULL) {
 197         win_group = gtk_window_group_new();
 198     }
 199 
 200     if (gchar * app_name = get_application_name()) {
 201         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 202         g_free(app_name);
 203     }
 204 
 205     if (owner) {
 206         owner->add_child(this);
 207         if (on_top_inherited()) {
 208             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 209         }
 210     }
 211 
 212     if (type == UTILITY) {
 213         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 214     }
 215 
 216     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 217 
 218     if (xvisualID != 0) {
 219         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 220         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 221     }
 222 
 223     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 224     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 225 
 226     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 227     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 228 
 229     gdk_window = gtk_widget_get_window(gtk_widget);
 230     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 231 
 232     glass_dnd_attach_context(this);
 233 
 234     gdk_windowManagerFunctions = wmf;
 235     if (wmf) {
 236         gdk_window_set_functions(gdk_window, wmf);
 237     }
 238 
 239     if (frame_type != TITLED) {
 240         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 241     }
 242 
 243     connect_signals(gtk_widget, this);
 244 }

 246 void WindowContext::paint(void *data, jint width, jint height) {
 247 #if GTK_CHECK_VERSION(3, 0, 0)
 248     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 249 #if GTK_CHECK_VERSION(3, 22, 0)
 250     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 251     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 252 #else
 253     gdk_window_begin_paint_region(gdk_window, region);
 254     cairo_t* context = gdk_cairo_create(gdk_window);
 255 #endif
 256 #else
 257     cairo_t *context = gdk_cairo_create(gdk_window);
 258 #endif
 259 
 260     if (bg_color.is_set) {
 261         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 262                                 (frame_type == TRANSPARENT) ? 0 : 1);
 263         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 264         cairo_paint(context);
 265     }
 266 
 267     cairo_surface_t *cairo_surface;
 268     cairo_surface = cairo_image_surface_create_for_data(
 269             (unsigned char *) data,
 270             CAIRO_FORMAT_ARGB32,
 271             width, height, width * 4);
 272 
 273     cairo_set_source_surface(context, cairo_surface, 0, 0);
 274 
 275     applyShapeMask(data, width, height);
 276     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 277     cairo_paint(context);
 278 
 279 #if GTK_CHECK_VERSION(3, 0, 0)
 280 #if GTK_CHECK_VERSION(3, 22, 0)
 281     gdk_window_end_draw_frame(gdk_window, dcontext);
 282     cairo_region_destroy(region);
 283 #else
 284     gdk_window_end_paint(gdk_window);
 285     cairo_region_destroy(region);
 286     cairo_destroy(context);
 287 #endif
 288 #else
 289     cairo_destroy(context);
 290 #endif
 291 
 292     cairo_surface_destroy(cairo_surface);
 293 }

 334 void WindowContext::process_focus(GdkEventFocus *event) {
 335     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 336         ungrab_mouse_drag_focus();
 337     }
 338 
 339     if (!event->in) {
 340         ungrab_focus();
 341     }
 342 
 343     if (xim.enabled && xim.ic) {
 344         if (event->in) {
 345             XSetICFocus(xim.ic);
 346         } else {
 347             XUnsetICFocus(xim.ic);
 348         }
 349     }
 350 
 351     if (jwindow) {
 352         if (!event->in || isEnabled()) {
 353             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 354                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 355                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         } else {
 358             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 359             CHECK_JNI_EXCEPTION(mainEnv)
 360         }
 361     }
 362 }

 364 void WindowContext::process_property_notify(GdkEventProperty *event) {
 365     if (event->window == gdk_window) {
 366         if (event->atom == atom_net_wm_state) {
 367             process_net_wm_property();
 368         } else if (event->atom == atom_net_wm_frame_extents) {
 369             if (frame_type != TITLED) {
 370                 return;
 371             }
 372 
 373             int top, left, bottom, right;
 374 
 375             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 376                 if (top + left + bottom + right > 0) {
 377                     geometry.frame_extents_received = true;
 378                     geometry.adjust_w = left + right;
 379                     geometry.adjust_h = top + bottom;
 380                     geometry.view_x = left;
 381                     geometry.view_y = top;
 382 
 383                     save_cached_extents();
 384 
 385                     // set bounds again to set to correct window size that must
 386                     // be the total width and height accounting extents
 387                     // this is ignored if size is "content size" instead of "window size"
 388                     if (geometry.window_size_set) {
 389                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 390                     }
 391                 }
 392             }
 393         }
 394     }
 395 }

 397 void WindowContext::process_configure() {
 398     calculate_adjustments();
 399 
 400     gint x, y, w, h, gtk_w, gtk_h;
 401 
 402     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 403     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 404     w = gtk_w + geometry.adjust_w;
 405     h = gtk_h + geometry.adjust_h;
 406 
 407     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 408     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 409                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 410 
 411     geometry.current_x = x;
 412     geometry.current_y = y;
 413     geometry.current_w = w;
 414     geometry.current_h = h;
 415     geometry.current_cw = gtk_w;
 416     geometry.current_ch = gtk_h;
 417 
 418     size_position_notify(size_changed, pos_changed);
 419 }

 421 void WindowContext::process_destroy() {
 422     if (owner) {
 423         owner->remove_child(this);
 424     }
 425 
 426     if (WindowContext::sm_mouse_drag_window == this) {
 427         ungrab_mouse_drag_focus();
 428     }
 429 
 430     ungrab_focus();


 431 
 432     std::set<WindowContext *>::iterator it;
 433     for (it = children.begin(); it != children.end(); ++it) {
 434         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 435         // WindowContext::process_destroy() to call remove_child() (because children
 436         // is being iterated here) but also prevents gtk_window_set_transient_for from
 437         // being called - this causes the crash on gnome.
 438         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 439         (*it)->set_owner(NULL);
 440         destroy_and_delete_ctx(*it);
 441     }
 442     children.clear();
 443 
 444     if (jwindow) {
 445         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 446         EXCEPTION_OCCURED(mainEnv);
 447     }
 448 
 449     if (jview) {
 450         mainEnv->DeleteGlobalRef(jview);
 451         jview = NULL;
 452     }
 453 
 454     if (jwindow) {
 455         mainEnv->DeleteGlobalRef(jwindow);
 456         jwindow = NULL;
 457     }
 458 
 459     can_be_deleted = true;
 460 }

 478 void WindowContext::process_mouse_button(GdkEventButton *event) {
 479     bool press = event->type == GDK_BUTTON_PRESS;
 480     bool release = event->type == GDK_BUTTON_RELEASE;
 481 
 482     guint state = event->state;
 483     guint mask = 0;
 484 
 485     // We need to add/remove current mouse button from the modifier flags
 486     // as X lib state represents the state just prior to the event and
 487     // glass needs the state just after the event
 488     switch (event->button) {
 489         case 1:
 490             mask = GDK_BUTTON1_MASK;
 491             break;
 492         case 2:
 493             mask = GDK_BUTTON2_MASK;
 494             break;
 495         case 3:
 496             mask = GDK_BUTTON3_MASK;
 497             break;
 498         case MOUSE_BACK_BTN:
 499             mask = GDK_BUTTON4_MASK;
 500             break;
 501         case MOUSE_FORWARD_BTN:
 502             mask = GDK_BUTTON5_MASK;
 503             break;
 504     }
 505 
 506     if (press) {
 507         state |= mask;
 508     } else {
 509         state &= ~mask;
 510     }
 511 
 512     // Upper layers expects from us Windows behavior:
 513     // all mouse events should be delivered to window where drag begins
 514     // and no exit/enter event should be reported during this drag.
 515     // We can grab mouse pointer for these needs.
 516     if (press) {
 517         grab_mouse_drag_focus();
 518     } else {
 519         if ((event->state & MOUSE_BUTTONS_MASK)
 520             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 521             ungrab_mouse_drag_focus();
 522         } else if (event->button == 8 || event->button == 9) {
 523             // GDK X backend interprets button press events for buttons 4-7 as
 524             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 525             // be set on the event->state from GDK. Thus we cannot check if all
 526             // buttons have been released in the usual way (as above).
 527             ungrab_mouse_drag_focus();
 528         }
 529     }
 530 
 531     jint button = gtk_button_number_to_mouse_button(event->button);
 532 
 533     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 534         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 535                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 536                                 button,
 537                                 (jint) event->x, (jint) event->y,
 538                                 (jint) event->x_root, (jint) event->y_root,
 539                                 gdk_modifier_mask_to_glass(state),
 540                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 541                                 JNI_FALSE);
 542         CHECK_JNI_EXCEPTION(mainEnv)
 543 
 544         if (jview && event->button == 3 && press) {
 545             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 546                                     (jint) event->x, (jint) event->y,
 547                                     (jint) event->x_root, (jint) event->y_root,
 548                                     JNI_FALSE);
 549             CHECK_JNI_EXCEPTION(mainEnv)
 550         }
 551     }
 552 }

 554 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 555     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 556     jint isDrag = glass_modifier & (
 557             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 558             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 559             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 560             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 561             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 562     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 563 
 564     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 565         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 566     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 567         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 568     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 570     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 572     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 573         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 574     }
 575 
 576     if (jview) {
 577         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 578                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 579                                 button,
 580                                 (jint) event->x, (jint) event->y,
 581                                 (jint) event->x_root, (jint) event->y_root,
 582                                 glass_modifier,
 583                                 JNI_FALSE,
 584                                 JNI_FALSE);
 585         CHECK_JNI_EXCEPTION(mainEnv)
 586     }
 587 }

 589 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 590     jdouble dx = 0;
 591     jdouble dy = 0;
 592 
 593     // converting direction to change in pixels
 594     switch (event->direction) {
 595 #if GTK_CHECK_VERSION(3, 4, 0)
 596         case GDK_SCROLL_SMOOTH:
 597             //FIXME 3.4 ???
 598             break;
 599 #endif
 600         case GDK_SCROLL_UP:

 623                                 gdk_modifier_mask_to_glass(event->state),

 626                                 (jdouble) 40.0, (jdouble) 40.0);

 656 void WindowContext::process_key(GdkEventKey *event) {
 657     bool press = event->type == GDK_KEY_PRESS;
 658     jint glassKey = get_glass_key(event);
 659     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 660     if (press) {
 661         glassModifier |= glass_key_to_modifier(glassKey);
 662     } else {
 663         glassModifier &= ~glass_key_to_modifier(glassKey);
 664     }
 665     jcharArray jChars = NULL;
 666     jchar key = gdk_keyval_to_unicode(event->keyval);
 667     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 668         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 669     } else {
 670 #ifdef GLASS_GTK2
 671         if (key == 0) {
 672             // Work around "bug" fixed in gtk-3.0:
 673             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 674             switch (event->keyval) {
 675             case 0xFF08 /* Backspace */: key =  '\b';
 676             case 0xFF09 /* Tab       */: key =  '\t';
 677             case 0xFF0A /* Linefeed  */: key =  '\n';
 678             case 0xFF0B /* Vert. Tab */: key =  '\v';
 679             case 0xFF0D /* Return    */: key =  '\r';
 680             case 0xFF1B /* Escape    */: key =  '\033';
 681             case 0xFFFF /* Delete    */: key =  '\177';
 682             }
 683         }
 684 #endif
 685     }
 686 
 687     if (key > 0) {
 688         jChars = mainEnv->NewCharArray(1);
 689         if (jChars) {
 690             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 691             CHECK_JNI_EXCEPTION(mainEnv)
 692         }
 693     } else {
 694         jChars = mainEnv->NewCharArray(0);
 695     }
 696     if (jview) {
 697         if (press) {
 698             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 699                                     com_sun_glass_events_KeyEvent_PRESS,
 700                                     glassKey,
 701                                     jChars,
 702                                     glassModifier);
 703             CHECK_JNI_EXCEPTION(mainEnv)
 704 
 705             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 706                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 707                                         com_sun_glass_events_KeyEvent_TYPED,
 708                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 709                                         jChars,
 710                                         glassModifier);
 711                 CHECK_JNI_EXCEPTION(mainEnv)
 712             }
 713         } else {
 714             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 715                                     com_sun_glass_events_KeyEvent_RELEASE,
 716                                     glassKey,
 717                                     jChars,
 718                                     glassModifier);
 719             CHECK_JNI_EXCEPTION(mainEnv)
 720         }
 721     }
 722 }

 724 void WindowContext::process_state(GdkEventWindowState *event) {
 725     if (event->changed_mask &
 726         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 727 
 728         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 729             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 730         }
 731         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 732             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 733         }
 734 
 735         jint stateChangeEvent;
 736 
 737         if (is_iconified) {
 738             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 739         } else if (is_maximized) {
 740             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 741         } else {
 742             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 743             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 744                 // in this case - the window manager will not support the programatic
 745                 // request to iconify - so we need to restore it now.
 746                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 747             }
 748         }
 749 
 750         notify_state(stateChangeEvent);
 751     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 752         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 753     }
 754 }

 756 void WindowContext::process_net_wm_property() {
 757     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 758 
 759     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 760     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 761     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 762     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 763 
 764     gint length;
 765     glong *atoms = NULL;
 766 
 767     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 768                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 769 
 770         bool is_hidden = false;
 771         bool is_above = false;
 772         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 773             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 774                 is_hidden = true;
 775             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 776                 is_above = true;
 777             }
 778         }
 779 
 780         g_free(atoms);
 781 
 782         if (is_iconified != is_hidden) {
 783             is_iconified = is_hidden;
 784 
 785             notify_state((is_hidden)
 786                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 787                          : com_sun_glass_events_WindowEvent_RESTORE);
 788         }
 789 
 790         notify_on_top(is_above);
 791     }
 792 }

 966 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 967     calculate_adjustments();
 968 
 969     // newW / newH always content sizes compatible with GTK+
 970     // if window has no decoration, adjustments will be ZERO
 971     int newW = w > 0 ? w - geometry.adjust_w : cw;
 972     int newH = h > 0 ? h - geometry.adjust_h : ch;
 973 
 974     gboolean size_changed = FALSE;
 975     gboolean pos_changed = FALSE;
 976 
 977     geometry.window_size_set = (w > 0 || h > 0);
 978 
 979     if (newW > 0 && newH > 0) {
 980         size_changed = TRUE;
 981 
 982         geometry.current_cw = newW;
 983         geometry.current_ch = newH;
 984         geometry.current_w = newW + geometry.adjust_w;
 985         geometry.current_h = newH + geometry.adjust_h;
 986 
 987         if (visible_received) {
 988             // update constraints if not resized by the user interface so it will
 989             // let gtk_window_resize succeed, because it's bound to geometry constraints
 990             apply_geometry();
 991             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 992         } else {
 993             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 994         }
 995     }
 996 
 997     if (xSet || ySet) {
 998         int newX = (xSet) ? x : geometry.current_x;
 999         int newY = (ySet) ? y : geometry.current_y;
1000 
1001         if (newX != geometry.current_x || newY != geometry.current_y) {
1002             pos_changed = TRUE;
1003             geometry.current_x = newX;
1004             geometry.current_y = newY;
1005             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1006         }
1007     }
1008 
1009     size_position_notify(size_changed, pos_changed);
1010 }

1214 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1215 // Only useful if the window manager does not support the frame extents extension - in this case
1216 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1217 // if that's not available.
1218 void WindowContext::calculate_adjustments() {
1219     if (frame_type != TITLED || geometry.frame_extents_received) {
1220         return;
1221     }
1222 
1223     gint x, y;
1224     gdk_window_get_origin(gdk_window, &x, &y);
1225 
1226     gint rx, ry;
1227     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1228 
1229     if (rx != x || ry != y) {
1230         // the left extends are correct - the right one is guessed to be the same
1231         geometry.adjust_w = (rx - x) * 2;
1232         // guess that bottom size is the same as left and right
1233         geometry.adjust_h = (ry - y) + (rx - x);
1234 
1235         // those will be correct
1236         geometry.view_x = (rx - x);
1237         geometry.view_y = (ry - y);
1238 
1239         save_cached_extents();
1240     } else {
1241         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1242         geometry.adjust_w = c.adjust_w;
1243         geometry.adjust_h = c.adjust_h;
1244         geometry.view_x = c.view_x;
1245         geometry.view_y = c.view_y;
1246     }
1247 
1248     apply_geometry();
1249 }

1269 void WindowContext::apply_geometry() {
1270     if (!map_received) {
1271         return;
1272     }
1273 
1274     GdkGeometry gdk_geometry;
1275     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1276 
1277     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1278         // not resizeable
1279         int w = (geometry.current_w - geometry.adjust_w) > 0
1280                 ? geometry.current_w - geometry.adjust_w
1281                 : geometry.current_cw;
1282 
1283         int h = (geometry.current_h - geometry.adjust_h) > 0
1284                 ? geometry.current_h - geometry.adjust_h
1285                 : geometry.current_ch;
1286 
1287         gdk_geometry.min_width = gdk_geometry.max_width = w;
1288         gdk_geometry.min_height = gdk_geometry.max_height = h;
1289     } else {
1290         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1291                                  ? geometry.minw - geometry.adjust_w : 1;
1292         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1293                                   ? geometry.minh - geometry.adjust_h : 1;
1294 
1295         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1296                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1297         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1298                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1299     }
1300 
1301     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1302                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1303 }

1392     WindowContext *o = owner;
