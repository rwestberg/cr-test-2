 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 330 void WindowContext::process_focus(GdkEventFocus *event) {
 331     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 332         ungrab_mouse_drag_focus();
 333     }
 334 
 335     if (!event->in && WindowContext::sm_grab_window == this) {
 336         ungrab_focus();
 337     }
 338 
 339     if (xim.enabled && xim.ic) {
 340         if (event->in) {
 341             XSetICFocus(xim.ic);
 342         } else {
 343             XUnsetICFocus(xim.ic);
 344         }
 345     }
 346 
 347     if (jwindow) {
 348         if (!event->in || isEnabled()) {
 349             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 350                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 351                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 352             CHECK_JNI_EXCEPTION(mainEnv)
 353         } else {
 354             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 355             CHECK_JNI_EXCEPTION(mainEnv)
 356         }
 357     }
 358 }

 360 void WindowContext::process_property_notify(GdkEventProperty *event) {
 361     if (event->window == gdk_window) {
 362         if (event->atom == atom_net_wm_state) {
 363             process_net_wm_property();
 364         } else if (event->atom == atom_net_wm_frame_extents) {
 365             if (frame_type != TITLED) {
 366                 return;
 367             }
 368 
 369             int top, left, bottom, right;
 370 
 371             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 372                 if (top + left + bottom + right > 0) {
 373                     geometry.frame_extents_received = true;
 374                     geometry.adjust_w = left + right;
 375                     geometry.adjust_h = top + bottom;
 376                     geometry.view_x = left;
 377                     geometry.view_y = top;
 378 
 379                     save_cached_extents();
 380 
 381                     // set bounds again to set to correct window size that must
 382                     // be the total width and height accounting extents
 383                     // this is ignored if size is "content size" instead of "window size"
 384                     if (geometry.window_size_set) {
 385                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 386                     }
 387 
 388                     // force position notify so java will know about view_y and view_x
 389                     size_position_notify(false, true);
 390                 }
 391             }
 392         }
 393     }
 394 }

 396 void WindowContext::process_configure() {
 397     calculate_adjustments();
 398 
 399     gint x, y, w, h, gtk_w, gtk_h;
 400 
 401     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 402     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 403 
 404     w = gtk_w + geometry.adjust_w;
 405     h = gtk_h + geometry.adjust_h;
 406 
 407     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 408     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 409                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 410 
 411     geometry.current_x = x;
 412     geometry.current_y = y;
 413     geometry.current_w = w;
 414     geometry.current_h = h;
 415     geometry.current_cw = gtk_w;
 416     geometry.current_ch = gtk_h;
 417 
 418     size_position_notify(size_changed, pos_changed);
 419 }

 421 void WindowContext::process_destroy() {
 422     if (owner) {
 423         owner->remove_child(this);
 424     }
 425 
 426     if (WindowContext::sm_mouse_drag_window == this) {
 427         ungrab_mouse_drag_focus();
 428     }
 429 
 430     if (WindowContext::sm_grab_window == this) {
 431         ungrab_focus();
 432     }
 433 
 434     std::set<WindowContext *>::iterator it;
 435     for (it = children.begin(); it != children.end(); ++it) {
 436         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 437         // WindowContext::process_destroy() to call remove_child() (because children
 438         // is being iterated here) but also prevents gtk_window_set_transient_for from
 439         // being called - this causes the crash on gnome.
 440         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 441         (*it)->set_owner(NULL);
 442         destroy_and_delete_ctx(*it);
 443     }
 444     children.clear();
 445 
 446     if (jwindow) {
 447         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 448         EXCEPTION_OCCURED(mainEnv);
 449     }
 450 
 451     if (jview) {
 452         mainEnv->DeleteGlobalRef(jview);
 453         jview = NULL;
 454     }
 455 
 456     if (jwindow) {
 457         mainEnv->DeleteGlobalRef(jwindow);
 458         jwindow = NULL;
 459     }
 460 
 461     can_be_deleted = true;
 462 }

 464 void WindowContext::process_delete() {

 466         gtk_widget_hide_on_delete(gtk_widget);
 467         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 468         CHECK_JNI_EXCEPTION(mainEnv)
 469     }

 480 void WindowContext::process_mouse_button(GdkEventButton *event) {
 481     bool press = event->type == GDK_BUTTON_PRESS;
 482     bool release = event->type == GDK_BUTTON_RELEASE;
 483 
 484     guint state = event->state;
 485     guint mask = 0;
 486 
 487     // We need to add/remove current mouse button from the modifier flags
 488     // as X lib state represents the state just prior to the event and
 489     // glass needs the state just after the event
 490     switch (event->button) {
 491         case 1:
 492             mask = GDK_BUTTON1_MASK;
 493             break;
 494         case 2:
 495             mask = GDK_BUTTON2_MASK;
 496             break;
 497         case 3:
 498             mask = GDK_BUTTON3_MASK;
 499             break;
 500         case MOUSE_BACK_BTN:
 501             mask = GDK_BUTTON4_MASK;
 502             break;
 503         case MOUSE_FORWARD_BTN:
 504             mask = GDK_BUTTON5_MASK;
 505             break;
 506     }
 507 
 508     if (press) {
 509         state |= mask;
 510     } else {
 511         state &= ~mask;
 512     }
 513 
 514     // Upper layers expects from us Windows behavior:
 515     // all mouse events should be delivered to window where drag begins
 516     // and no exit/enter event should be reported during this drag.
 517     // We can grab mouse pointer for these needs.
 518     if (press) {
 519         grab_mouse_drag_focus(NULL, true);
 520     } else {
 521         if ((event->state & MOUSE_BUTTONS_MASK)
 522             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 523             ungrab_mouse_drag_focus();
 524         } else if (event->button == 8 || event->button == 9) {
 525             // GDK X backend interprets button press events for buttons 4-7 as
 526             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 527             // be set on the event->state from GDK. Thus we cannot check if all
 528             // buttons have been released in the usual way (as above).
 529             ungrab_mouse_drag_focus();
 530         }
 531     }
 532 
 533     jint button = gtk_button_number_to_mouse_button(event->button);
 534 
 535     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 536         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 537                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 538                                 button,
 539                                 (jint) event->x, (jint) event->y,
 540                                 (jint) event->x_root, (jint) event->y_root,
 541                                 gdk_modifier_mask_to_glass(state),
 542                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 543                                 JNI_FALSE);
 544         CHECK_JNI_EXCEPTION(mainEnv)
 545 
 546         if (jview && event->button == 3 && press) {
 547             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 548                                     (jint) event->x, (jint) event->y,
 549                                     (jint) event->x_root, (jint) event->y_root,
 550                                     JNI_FALSE);
 551             CHECK_JNI_EXCEPTION(mainEnv)
 552         }
 553     }
 554 }

 556 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 557     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 558     jint isDrag = glass_modifier & (
 559             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 560             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 561             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 564     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 565 
 566     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 567         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 568     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 570     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 572     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 573         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 574     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 575         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 576     }
 577 
 578     if (jview) {
 579         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 580                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 581                                 button,
 582                                 (jint) event->x, (jint) event->y,
 583                                 (jint) event->x_root, (jint) event->y_root,
 584                                 glass_modifier,
 585                                 JNI_FALSE,
 586                                 JNI_FALSE);
 587         CHECK_JNI_EXCEPTION(mainEnv)
 588     }
 589 }

 591 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 592     jdouble dx = 0;
 593     jdouble dy = 0;
 594 
 595     // converting direction to change in pixels
 596     switch (event->direction) {
 597 #if GTK_CHECK_VERSION(3, 4, 0)
 598         case GDK_SCROLL_SMOOTH:
 599             //FIXME 3.4 ???
 600             break;
 601 #endif
 602         case GDK_SCROLL_UP:
 603             dy = 1;
 604             break;
 605         case GDK_SCROLL_DOWN:
 606             dy = -1;
 607             break;
 608         case GDK_SCROLL_LEFT:
 609             dx = 1;
 610             break;
 611         case GDK_SCROLL_RIGHT:
 612             dx = -1;
 613             break;
 614     }
 615     if (event->state & GDK_SHIFT_MASK) {
 616         jdouble t = dy;
 617         dy = dx;
 618         dx = t;
 619     }
 620     if (jview) {
 621         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 622                                 (jint) event->x, (jint) event->y,
 623                                 (jint) event->x_root, (jint) event->y_root,
 624                                 dx, dy,
 625                                 gdk_modifier_mask_to_glass(event->state),
 626                                 (jint) 0, (jint) 0,
 627                                 (jint) 0, (jint) 0,
 628                                 (jdouble) 40.0, (jdouble) 40.0);
 629         CHECK_JNI_EXCEPTION(mainEnv)
 630     }

 658 void WindowContext::process_key(GdkEventKey *event) {
 659     bool press = event->type == GDK_KEY_PRESS;
 660     jint glassKey = get_glass_key(event);
 661     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 662     if (press) {
 663         glassModifier |= glass_key_to_modifier(glassKey);
 664     } else {
 665         glassModifier &= ~glass_key_to_modifier(glassKey);
 666     }
 667     jcharArray jChars = NULL;
 668     jchar key = gdk_keyval_to_unicode(event->keyval);
 669     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 670         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 671     } else {
 672 #ifdef GLASS_GTK2
 673         if (key == 0) {
 674             // Work around "bug" fixed in gtk-3.0:
 675             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 676             switch (event->keyval) {
 677             case 0xFF08 /* Backspace */: key =  '\b';
 678             case 0xFF09 /* Tab       */: key =  '\t';
 679             case 0xFF0A /* Linefeed  */: key =  '\n';
 680             case 0xFF0B /* Vert. Tab */: key =  '\v';
 681             case 0xFF0D /* Return    */: key =  '\r';
 682             case 0xFF1B /* Escape    */: key =  '\033';
 683             case 0xFFFF /* Delete    */: key =  '\177';
 684             }
 685         }
 686 #endif
 687     }
 688 
 689     if (key > 0) {
 690         jChars = mainEnv->NewCharArray(1);
 691         if (jChars) {
 692             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 693             CHECK_JNI_EXCEPTION(mainEnv)
 694         }
 695     } else {
 696         jChars = mainEnv->NewCharArray(0);
 697     }
 698     if (jview) {
 699         if (press) {
 700             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 701                                     com_sun_glass_events_KeyEvent_PRESS,
 702                                     glassKey,
 703                                     jChars,
 704                                     glassModifier);
 705             CHECK_JNI_EXCEPTION(mainEnv)
 706 
 707             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 708                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 709                                         com_sun_glass_events_KeyEvent_TYPED,
 710                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 711                                         jChars,
 712                                         glassModifier);
 713                 CHECK_JNI_EXCEPTION(mainEnv)
 714             }
 715         } else {
 716             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 717                                     com_sun_glass_events_KeyEvent_RELEASE,
 718                                     glassKey,
 719                                     jChars,
 720                                     glassModifier);
 721             CHECK_JNI_EXCEPTION(mainEnv)
 722         }
 723     }
 724 }

 726 void WindowContext::process_state(GdkEventWindowState *event) {
 727     if (event->changed_mask &
 728         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 729 
 730         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 731             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 732         }
 733         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 734             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 735         }
 736 
 737         jint stateChangeEvent;
 738 
 739         if (is_iconified) {
 740             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 741         } else if (is_maximized) {
 742             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 743         } else {
 744             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 745             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 746                 // in this case - the window manager will not support the programatic
 747                 // request to iconify - so we need to restore it now.
 748                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 749             }
 750         }
 751 
 752         notify_state(stateChangeEvent);
 753     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 754         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 755     }
 756 }

 758 void WindowContext::process_net_wm_property() {
 759     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 760 
 761     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 762     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 763     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 764     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 765 
 766     gint length;
 767     glong *atoms = NULL;
 768 
 769     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 770                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 771 
 772         bool is_hidden = false;
 773         bool is_above = false;
 774         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 775             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 776                 is_hidden = true;
 777             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 778                 is_above = true;
 779             }
 780         }
 781 
 782         g_free(atoms);
 783 
 784         if (is_iconified != is_hidden) {
 785             is_iconified = is_hidden;
 786 
 787             notify_state((is_hidden)
 788                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 789                          : com_sun_glass_events_WindowEvent_RESTORE);
 790         }
 791 
 792         notify_on_top(is_above);
 793     }
 794 }

 977 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 978     calculate_adjustments();
 979 
 980     // newW / newH always content sizes compatible with GTK+
 981     // if window has no decoration, adjustments will be ZERO
 982     int newW = w > 0 ? w - geometry.adjust_w : cw;
 983     int newH = h > 0 ? h - geometry.adjust_h : ch;
 984 
 985     gboolean size_changed = FALSE;
 986     gboolean pos_changed = FALSE;
 987 
 988     geometry.window_size_set = (w > 0 || h > 0);
 989 
 990     if (newW > 0 && newH > 0) {
 991         size_changed = TRUE;
 992 
 993         geometry.current_cw = newW;
 994         geometry.current_ch = newH;
 995         geometry.current_w = newW + geometry.adjust_w;
 996         geometry.current_h = newH + geometry.adjust_h;
 997 
 998         if (visible_received) {
 999             // update constraints if not resized by the user interface so it will
1000             // let gtk_window_resize succeed, because it's bound to geometry constraints
1001             apply_geometry();
1002             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1003         } else {
1004             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1005         }
1006     }
1007 
1008     if (xSet || ySet) {
1009         int newX = (xSet) ? x : geometry.current_x;
1010         int newY = (ySet) ? y : geometry.current_y;
1011 
1012         if (newX != geometry.current_x || newY != geometry.current_y) {
1013             pos_changed = TRUE;
1014             geometry.current_x = newX;
1015             geometry.current_y = newY;
1016             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1017         }
1018     }
1019 
1020     size_position_notify(size_changed, pos_changed);
1021 }

1207 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1208 // Only useful if the window manager does not support the frame extents extension - in this case
1209 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1210 // if that's not available.
1211 void WindowContext::calculate_adjustments() {
1212     if (frame_type != TITLED || geometry.frame_extents_received) {
1213         return;
1214     }
1215 
1216     gint x, y;
1217     gdk_window_get_origin(gdk_window, &x, &y);
1218 
1219     gint rx, ry;
1220     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1221 
1222     if (rx != x || ry != y) {
1223         // the left extends are correct - the right one is guessed to be the same
1224         geometry.adjust_w = (x - rx) * 2;
1225         // guess that bottom size is the same as left and right
1226         geometry.adjust_h = (y - ry) + (x - rx);
1227 
1228         // those will be correct
1229         geometry.view_x = (x - rx);
1230         geometry.view_y = (y - ry);
1231 
1232         save_cached_extents();
1233     } else {
1234         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1235         geometry.adjust_w = c.adjust_w;
1236         geometry.adjust_h = c.adjust_h;
1237         geometry.view_x = c.view_x;
1238         geometry.view_y = c.view_y;
1239     }
1240 
1241     apply_geometry();
1242 }

1244 void WindowContext::save_cached_extents() {
1245     if (frame_type != TITLED) {
1246         return;
1247     }
1248 
1249     if (window_type == NORMAL) {
1250         normal_extents.adjust_w = geometry.adjust_w;
1251         normal_extents.adjust_h = geometry.adjust_h;
1252         normal_extents.view_x = geometry.view_x;
1253         normal_extents.view_y = geometry.view_y;
1254     } else {
1255         utility_extents.adjust_w = geometry.adjust_w;
1256         utility_extents.adjust_h = geometry.adjust_h;
1257         utility_extents.view_x = geometry.view_x;
1258         utility_extents.view_y = geometry.view_y;
1259     }

1262 void WindowContext::apply_geometry() {
1263     if (!map_received) {
1264         return;
1265     }
1266 
1267     GdkGeometry gdk_geometry;
1268     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1269 
1270     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1271         // not resizeable
1272         int w = (geometry.current_w - geometry.adjust_w) > 0
1273                 ? geometry.current_w - geometry.adjust_w
1274                 : geometry.current_cw;
1275 
1276         int h = (geometry.current_h - geometry.adjust_h) > 0
1277                 ? geometry.current_h - geometry.adjust_h
1278                 : geometry.current_ch;
1279 
1280         gdk_geometry.min_width = gdk_geometry.max_width = w;
1281         gdk_geometry.min_height = gdk_geometry.max_height = h;
1282     } else {
1283         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1284                                  ? geometry.minw - geometry.adjust_w : 1;
1285         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1286                                   ? geometry.minh - geometry.adjust_h : 1;
1287 
1288         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1289                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1290         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1291                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1292     }
1293 
1294     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1295                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1296 }

1324 void WindowContext::activate_window() {
1325     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1326     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1327     if (navAtom != None) {
1328         XClientMessageEvent clientMessage;
1329         memset(&clientMessage, 0, sizeof(clientMessage));
1330 
1331         clientMessage.type = ClientMessage;
1332         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1333         clientMessage.message_type = navAtom;
1334         clientMessage.format = 32;
1335         clientMessage.data.l[0] = 1;
1336         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1337         clientMessage.data.l[2] = 0;

1340                    SubstructureRedirectMask | SubstructureNotifyMask,
1341                    (XEvent * ) & clientMessage);

1385 bool WindowContext::on_top_inherited() {
1386     WindowContext *o = owner;
1387     while (o) {
1388         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1389         if (!topO) break;
1390         if (topO->on_top) {
1391             return true;
1392         }
1393         o = topO->owner;
1394     }
1395     return false;
1396 }

1399     if (owner) {
1400         WindowContext *topO = dynamic_cast<WindowContext *>(owner);
1401         return (topO && topO->effective_on_top()) || on_top;
1402     }
1403     return on_top;
