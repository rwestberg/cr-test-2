26 package com.sun.pisces;
27 
28 import java.nio.IntBuffer;
29 
30 public final class JavaSurface extends AbstractSurface {
31 
32     private IntBuffer dataBuffer;
33 
34     private int[] dataInt;
35 
36     public JavaSurface(int[] dataInt, int dataType, int width, int height) {
37         super(width, height);
38         if (dataInt.length / width < height) {
39             throw new IllegalArgumentException("width(=" + width + ") * height(="
40                     + height + ") is greater than dataInt.length(=" + dataInt.length + ")");
41         }
42         this.dataInt = dataInt;
43         this.dataBuffer = IntBuffer.wrap(this.dataInt);
44 
45         initialize(dataType, width, height);






46     }
47 
48     public IntBuffer getDataIntBuffer() {
49         return this.dataBuffer;
50     }
51 
52     private native void initialize(int dataType, int width, int height);
53 }
