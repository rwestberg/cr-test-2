 584     @SuppressWarnings("unchecked")
 585     public static void reverse(ObservableList list) {
 586         Object[] newContent = list.toArray();
 587         for (int i = 0; i < newContent.length / 2; ++i) {
 588             Object tmp = newContent[i];
