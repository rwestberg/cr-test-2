33 static gboolean fxplugins_init (GstPlugin * plugin)
34 {
35     return audiodecoder_plugin_init(plugin) &&
36            videodecoder_plugin_init(plugin) &&
37            mpegts_demuxer_plugin_init(plugin);
38 }
