 175 // Support for forcing alignment of thread objects for biased locking
 176 void* Thread::allocate(size_t size, bool throw_excpt, MEMFLAGS flags) {
 177   if (UseBiasedLocking) {
 178     const size_t alignment = markWord::biased_lock_alignment;
 179     size_t aligned_size = size + (alignment - sizeof(intptr_t));
 180     void* real_malloc_addr = throw_excpt? AllocateHeap(aligned_size, flags, CURRENT_PC)
 181                                           : AllocateHeap(aligned_size, flags, CURRENT_PC,
 182                                                          AllocFailStrategy::RETURN_NULL);
 183     void* aligned_addr     = align_up(real_malloc_addr, alignment);
 184     assert(((uintptr_t) aligned_addr + (uintptr_t) size) <=
 185            ((uintptr_t) real_malloc_addr + (uintptr_t) aligned_size),
 186            "JavaThread alignment code overflowed allocated storage");
 187     if (aligned_addr != real_malloc_addr) {
 188       log_info(biasedlocking)("Aligned thread " INTPTR_FORMAT " to " INTPTR_FORMAT,
 189                               p2i(real_malloc_addr),
 190                               p2i(aligned_addr));
 191     }
 192     ((Thread*) aligned_addr)->_real_malloc_address = real_malloc_addr;
 193     return aligned_addr;
 194   } else {
 195     return throw_excpt? AllocateHeap(size, flags, CURRENT_PC)
 196                        : AllocateHeap(size, flags, CURRENT_PC, AllocFailStrategy::RETURN_NULL);
 197   }
 198 }

 200 void Thread::operator delete(void* p) {
 201   if (UseBiasedLocking) {
 202     FreeHeap(((Thread*) p)->_real_malloc_address);
 203   } else {
 204     FreeHeap(p);
 205   }

 349 void Thread::record_stack_base_and_size() {
 350   // Note: at this point, Thread object is not yet initialized. Do not rely on
 351   // any members being initialized. Do not rely on Thread::current() being set.
 352   // If possible, refrain from doing anything which may crash or assert since
 353   // quite probably those crash dumps will be useless.
 354   set_stack_base(os::current_stack_base());
 355   set_stack_size(os::current_stack_size());
 356 
 357 #ifdef SOLARIS
 358   if (os::is_primordial_thread()) {
 359     os::Solaris::correct_stack_boundaries_for_primordial_thread(this);
 360   }
 361 #endif
 362 
 363   // Set stack limits after thread is initialized.
 364   if (is_Java_thread()) {
 365     ((JavaThread*) this)->set_stack_overflow_limit();
 366     ((JavaThread*) this)->set_reserved_stack_activation(stack_base());
 367   }
 368 }

 376 void Thread::call_run() {
 377   DEBUG_ONLY(_run_state = CALL_RUN;)
 378 
 379   // At this point, Thread object should be fully initialized and
 380   // Thread::current() should be set.
 381 
 382   assert(Thread::current_or_null() != NULL, "current thread is unset");
 383   assert(Thread::current_or_null() == this, "current thread is wrong");
 384 
 385   // Perform common initialization actions
 386 
 387   register_thread_stack_with_NMT();
 388 
 389   JFR_ONLY(Jfr::on_thread_start(this);)
 390 
 391   log_debug(os, thread)("Thread " UINTX_FORMAT " stack dimensions: "
 392     PTR_FORMAT "-" PTR_FORMAT " (" SIZE_FORMAT "k).",
 393     os::current_thread_id(), p2i(stack_base() - stack_size()),
 394     p2i(stack_base()), stack_size()/1024);
 395 
 396   // Perform <ChildClass> initialization actions
 397   DEBUG_ONLY(_run_state = PRE_RUN;)
 398   this->pre_run();
 399 
 400   // Invoke <ChildClass>::run()
 401   DEBUG_ONLY(_run_state = RUN;)
 402   this->run();
 403   // Returned from <ChildClass>::run(). Thread finished.
 404 
 405   // Perform common tear-down actions
 406 
 407   assert(Thread::current_or_null() != NULL, "current thread is unset");
 408   assert(Thread::current_or_null() == this, "current thread is wrong");
 409 
 410   // Perform <ChildClass> tear-down actions
 411   DEBUG_ONLY(_run_state = POST_RUN;)
 412   this->post_run();
 413 
 414   // Note: at this point the thread object may already have deleted itself,
 415   // so from here on do not dereference *this*. Not all thread types currently
 416   // delete themselves when they terminate. But no thread should ever be deleted
 417   // asynchronously with respect to its termination - that is what _run_state can
 418   // be used to check.
 419 
 420   assert(Thread::current_or_null() == NULL, "current thread still present");
 421 }

 423 Thread::~Thread() {
 424 
 425   // Attached threads will remain in PRE_CALL_RUN, as will threads that don't actually
 426   // get started due to errors etc. Any active thread should at least reach post_run
 427   // before it is deleted (usually in post_run()).
 428   assert(_run_state == PRE_CALL_RUN ||
 429          _run_state == POST_RUN, "Active Thread deleted before post_run(): "
 430          "_run_state=%d", (int)_run_state);
 431 
 432   // Notify the barrier set that a thread is being destroyed. Note that a barrier
 433   // set might not be available if we encountered errors during bootstrapping.
 434   BarrierSet* const barrier_set = BarrierSet::barrier_set();
 435   if (barrier_set != NULL) {
 436     barrier_set->on_thread_destroy(this);
 437   }
 438 
 439   // stack_base can be NULL if the thread is never started or exited before
 440   // record_stack_base_and_size called. Although, we would like to ensure
 441   // that all started threads do call record_stack_base_and_size(), there is
 442   // not proper way to enforce that.
 443 #if INCLUDE_NMT
 444   if (_stack_base != NULL) {
 445     MemTracker::release_thread_stack(stack_end(), stack_size());
 446 #ifdef ASSERT
 447     set_stack_base(NULL);
 448 #endif
 449   }
 450 #endif // INCLUDE_NMT
 451 
 452   // deallocate data structures
 453   delete resource_area();
 454   // since the handle marks are using the handle area, we have to deallocated the root
 455   // handle mark before deallocating the thread's handle area,
 456   assert(last_handle_mark() != NULL, "check we have an element");
 457   delete last_handle_mark();
 458   assert(last_handle_mark() == NULL, "check we have reached the end");
 459 
 460   // It's possible we can encounter a null _ParkEvent, etc., in stillborn threads.
 461   // We NULL out the fields for good hygiene.
 462   ParkEvent::Release(_ParkEvent); _ParkEvent   = NULL;
 463   ParkEvent::Release(_MuxEvent); _MuxEvent    = NULL;
 464 
 465   delete handle_area();
 466   delete metadata_handles();
 467 
 468   // SR_handler uses this as a termination indicator -
 469   // needs to happen before os::free_thread()
 470   delete _SR_lock;
 471   _SR_lock = NULL;
 472 
 473   // osthread() can be NULL, if creation of thread failed.
 474   if (osthread() != NULL) os::free_thread(osthread());
 475 
 476   // Clear Thread::current if thread is deleting itself and it has not
 477   // already been done. This must be done before the memory is deallocated.
 478   // Needed to ensure JNI correctly detects non-attached threads.
 479   if (this == Thread::current_or_null()) {
 480     Thread::clear_thread_current();
 481   }
 482 
 483   CHECK_UNHANDLED_OOPS_ONLY(if (CheckUnhandledOops) delete unhandled_oops();)
 484 }

 515 void Thread::start(Thread* thread) {
 516   // Start is different from resume in that its safety is guaranteed by context or
 517   // being called from a Java method synchronized on the Thread object.
 518   if (!DisableStartThread) {
 519     if (thread->is_Java_thread()) {
 520       // Initialize the thread state to RUNNABLE before starting this thread.
 521       // Can not set it after the thread started because we do not know the
 522       // exact thread state at that time. It could be in MONITOR_WAIT or
 523       // in SLEEPING or some other state.
 524       java_lang_Thread::set_thread_status(((JavaThread*)thread)->threadObj(),
 525                                           java_lang_Thread::RUNNABLE);
 526     }
 527     os::start_thread(thread);
 528   }
 529 }

 553 // Check if an external suspend request has completed (or has been
 554 // cancelled). Returns true if the thread is externally suspended and
 555 // false otherwise.
 556 //
 557 // The bits parameter returns information about the code path through
 558 // the routine. Useful for debugging:
 559 //
 560 // set in is_ext_suspend_completed():
 561 // 0x00000001 - routine was entered
 562 // 0x00000010 - routine return false at end
 563 // 0x00000100 - thread exited (return false)
 564 // 0x00000200 - suspend request cancelled (return false)
 565 // 0x00000400 - thread suspended (return true)
 566 // 0x00001000 - thread is in a suspend equivalent state (return true)
 567 // 0x00002000 - thread is native and walkable (return true)
 568 // 0x00004000 - thread is native_trans and walkable (needed retry)
 569 //
 570 // set in wait_for_ext_suspend_completion():
 571 // 0x00010000 - routine was entered
 572 // 0x00020000 - suspend request cancelled before loop (return false)
 573 // 0x00040000 - thread suspended before loop (return true)
 574 // 0x00080000 - suspend request cancelled in loop (return false)
 575 // 0x00100000 - thread suspended in loop (return true)
 576 // 0x00200000 - suspend not completed during retry loop (return false)
 577 
 578 // Helper class for tracing suspend wait debug bits.
 579 //
 580 // 0x00000100 indicates that the target thread exited before it could
 581 // self-suspend which is not a wait failure. 0x00000200, 0x00020000 and
 582 // 0x00080000 each indicate a cancelled suspend request so they don't
 583 // count as wait failures either.
 584 #define DEBUG_FALSE_BITS (0x00000010 | 0x00200000)
 585 
 586 class TraceSuspendDebugBits : public StackObj {
 587  private:
 588   JavaThread * jt;
 589   bool         is_wait;
 590   bool         called_by_wait;  // meaningful when !is_wait
 591   uint32_t *   bits;
 592 
 593  public:
 594   TraceSuspendDebugBits(JavaThread *_jt, bool _is_wait, bool _called_by_wait,
 595                         uint32_t *_bits) {
 596     jt             = _jt;
 597     is_wait        = _is_wait;
 598     called_by_wait = _called_by_wait;
 599     bits           = _bits;
 600   }
 601 
 602   ~TraceSuspendDebugBits() {
 603     if (!is_wait) {
 604 #if 1
 605       // By default, don't trace bits for is_ext_suspend_completed() calls.
 606       // That trace is very chatty.
 607       return;
 608 #else
 609       if (!called_by_wait) {
 610         // If tracing for is_ext_suspend_completed() is enabled, then only
 611         // trace calls to it from wait_for_ext_suspend_completion()
 612         return;
 613       }
 614 #endif
 615     }
 616 
 617     if (AssertOnSuspendWaitFailure || TraceSuspendWaitFailures) {
 618       if (bits != NULL && (*bits & DEBUG_FALSE_BITS) != 0) {
 619         MutexLocker ml(Threads_lock);  // needed for get_thread_name()
 620         ResourceMark rm;
 621 
 622         tty->print_cr(
 623                       "Failed wait_for_ext_suspend_completion(thread=%s, debug_bits=%x)",
 624                       jt->get_thread_name(), *bits);
 625 
 626         guarantee(!AssertOnSuspendWaitFailure, "external suspend wait failed");
 627       }
 628     }
 629   }
 630 };

 634 bool JavaThread::is_ext_suspend_completed(bool called_by_wait, int delay,
 635                                           uint32_t *bits) {
 636   TraceSuspendDebugBits tsdb(this, false /* !is_wait */, called_by_wait, bits);
 637 
 638   bool did_trans_retry = false;  // only do thread_in_native_trans retry once
 639   bool do_trans_retry;           // flag to force the retry
 640 
 641   *bits |= 0x00000001;
 642 
 643   do {
 644     do_trans_retry = false;
 645 
 646     if (is_exiting()) {
 647       // Thread is in the process of exiting. This is always checked
 648       // first to reduce the risk of dereferencing a freed JavaThread.
 649       *bits |= 0x00000100;
 650       return false;
 651     }
 652 
 653     if (!is_external_suspend()) {
 654       // Suspend request is cancelled. This is always checked before
 655       // is_ext_suspended() to reduce the risk of a rogue resume
 656       // confusing the thread that made the suspend request.
 657       *bits |= 0x00000200;
 658       return false;
 659     }
 660 
 661     if (is_ext_suspended()) {
 662       // thread is suspended
 663       *bits |= 0x00000400;
 664       return true;
 665     }
 666 
 667     // Now that we no longer do hard suspends of threads running
 668     // native code, the target thread can be changing thread state
 669     // while we are in this routine:
 670     //
 671     //   _thread_in_native -> _thread_in_native_trans -> _thread_blocked
 672     //
 673     // We save a copy of the thread state as observed at this moment
 674     // and make our decision about suspend completeness based on the
 675     // copy. This closes the race where the thread state is seen as
 676     // _thread_in_native_trans in the if-thread_blocked check, but is
 677     // seen as _thread_blocked in if-thread_in_native_trans check.
 678     JavaThreadState save_state = thread_state();
 679 
 680     if (save_state == _thread_blocked && is_suspend_equivalent()) {
 681       // If the thread's state is _thread_blocked and this blocking
 682       // condition is known to be equivalent to a suspend, then we can
 683       // consider the thread to be externally suspended. This means that
 684       // the code that sets _thread_blocked has been modified to do
 685       // self-suspension if the blocking condition releases. We also
 686       // used to check for CONDVAR_WAIT here, but that is now covered by
 687       // the _thread_blocked with self-suspension check.
 688       //
 689       // Return true since we wouldn't be here unless there was still an
 690       // external suspend request.
 691       *bits |= 0x00001000;
 692       return true;
 693     } else if (save_state == _thread_in_native && frame_anchor()->walkable()) {
 694       // Threads running native code will self-suspend on native==>VM/Java
 695       // transitions. If its stack is walkable (should always be the case
 696       // unless this function is called before the actual java_suspend()
 697       // call), then the wait is done.
 698       *bits |= 0x00002000;
 699       return true;
 700     } else if (!called_by_wait && !did_trans_retry &&
 701                save_state == _thread_in_native_trans &&
 702                frame_anchor()->walkable()) {
 703       // The thread is transitioning from thread_in_native to another
 704       // thread state. check_safepoint_and_suspend_for_native_trans()
 705       // will force the thread to self-suspend. If it hasn't gotten
 706       // there yet we may have caught the thread in-between the native
 707       // code check above and the self-suspend. Lucky us. If we were
 708       // called by wait_for_ext_suspend_completion(), then it
 709       // will be doing the retries so we don't have to.
 710       //
 711       // Since we use the saved thread state in the if-statement above,
 712       // there is a chance that the thread has already transitioned to
 713       // _thread_blocked by the time we get here. In that case, we will
 714       // make a single unnecessary pass through the logic below. This
 715       // doesn't hurt anything since we still do the trans retry.
 716 
 717       *bits |= 0x00004000;
 718 
 719       // Once the thread leaves thread_in_native_trans for another
 720       // thread state, we break out of this retry loop. We shouldn't
 721       // need this flag to prevent us from getting back here, but
 722       // sometimes paranoia is good.
 723       did_trans_retry = true;
 724 
 725       // We wait for the thread to transition to a more usable state.
 726       for (int i = 1; i <= SuspendRetryCount; i++) {
 727         // We used to do an "os::yield_all(i)" call here with the intention
 728         // that yielding would increase on each retry. However, the parameter
 729         // is ignored on Linux which means the yield didn't scale up. Waiting
 730         // on the SR_lock below provides a much more predictable scale up for
 731         // the delay. It also provides a simple/direct point to check for any
 732         // safepoint requests from the VMThread
 733 
 734         // temporarily drops SR_lock while doing wait with safepoint check
 735         // (if we're a JavaThread - the WatcherThread can also call this)
 736         // and increase delay with each retry
 737         if (Thread::current()->is_Java_thread()) {
 738           SR_lock()->wait(i * delay);
 739         } else {
 740           SR_lock()->wait_without_safepoint_check(i * delay);
 741         }
 742 
 743         // check the actual thread state instead of what we saved above
 744         if (thread_state() != _thread_in_native_trans) {
 745           // the thread has transitioned to another thread state so
 746           // try all the checks (except this one) one more time.
 747           do_trans_retry = true;
 748           break;
 749         }
 750       } // end retry loop
 751 
 752 
 753     }
 754   } while (do_trans_retry);
 755 
 756   *bits |= 0x00000010;
 757   return false;
 758 }

 760 // Wait for an external suspend request to complete (or be cancelled).
 761 // Returns true if the thread is externally suspended and false otherwise.
 762 //
 763 bool JavaThread::wait_for_ext_suspend_completion(int retries, int delay,
 764                                                  uint32_t *bits) {
 765   TraceSuspendDebugBits tsdb(this, true /* is_wait */,
 766                              false /* !called_by_wait */, bits);
 767 
 768   // local flag copies to minimize SR_lock hold time
 769   bool is_suspended;
 770   bool pending;
 771   uint32_t reset_bits;
 772 
 773   // set a marker so is_ext_suspend_completed() knows we are the caller
 774   *bits |= 0x00010000;
 775 
 776   // We use reset_bits to reinitialize the bits value at the top of
 777   // each retry loop. This allows the caller to make use of any
 778   // unused bits for their own marking purposes.
 779   reset_bits = *bits;
 780 
 781   {
 782     MutexLocker ml(SR_lock(), Mutex::_no_safepoint_check_flag);
 783     is_suspended = is_ext_suspend_completed(true /* called_by_wait */,
 784                                             delay, bits);
 785     pending = is_external_suspend();
 786   }
 787   // must release SR_lock to allow suspension to complete
 788 
 789   if (!pending) {
 790     // A cancelled suspend request is the only false return from
 791     // is_ext_suspend_completed() that keeps us from entering the
 792     // retry loop.
 793     *bits |= 0x00020000;
 794     return false;
 795   }
 796 
 797   if (is_suspended) {
 798     *bits |= 0x00040000;
 799     return true;
 800   }
 801 
 802   for (int i = 1; i <= retries; i++) {
 803     *bits = reset_bits;  // reinit to only track last retry
 804 
 805     // We used to do an "os::yield_all(i)" call here with the intention
 806     // that yielding would increase on each retry. However, the parameter
 807     // is ignored on Linux which means the yield didn't scale up. Waiting
 808     // on the SR_lock below provides a much more predictable scale up for
 809     // the delay. It also provides a simple/direct point to check for any
 810     // safepoint requests from the VMThread
 811 
 812     {
 813       Thread* t = Thread::current();
 814       MonitorLocker ml(SR_lock(),
 815                        t->is_Java_thread() ? Mutex::_safepoint_check_flag : Mutex::_no_safepoint_check_flag);
 816       // wait with safepoint check (if we're a JavaThread - the WatcherThread
 817       // can also call this)  and increase delay with each retry
 818       ml.wait(i * delay);
 819 
 820       is_suspended = is_ext_suspend_completed(true /* called_by_wait */,
 821                                               delay, bits);
 822 
 823       // It is possible for the external suspend request to be cancelled
 824       // (by a resume) before the actual suspend operation is completed.
 825       // Refresh our local copy to see if we still need to wait.
 826       pending = is_external_suspend();
 827     }
 828 
 829     if (!pending) {
 830       // A cancelled suspend request is the only false return from
 831       // is_ext_suspend_completed() that keeps us from staying in the
 832       // retry loop.
 833       *bits |= 0x00080000;
 834       return false;
 835     }
 836 
 837     if (is_suspended) {
 838       *bits |= 0x00100000;
 839       return true;
 840     }
 841   } // end retry loop
 842 
 843   // thread did not suspend after all our retries
 844   *bits |= 0x00200000;
 845   return false;
 846 }

 848 // Called from API entry points which perform stack walking. If the
 849 // associated JavaThread is the current thread, then wait_for_suspend
 850 // is not used. Otherwise, it determines if we should wait for the
 851 // "other" thread to complete external suspension. (NOTE: in future
 852 // releases the suspension mechanism should be reimplemented so this
 853 // is not necessary.)
 854 //
 855 bool
 856 JavaThread::is_thread_fully_suspended(bool wait_for_suspend, uint32_t *bits) {
 857   if (this != JavaThread::current()) {
 858     // "other" threads require special handling.
 859     if (wait_for_suspend) {
 860       // We are allowed to wait for the external suspend to complete
 861       // so give the other thread a chance to get suspended.
 862       if (!wait_for_ext_suspend_completion(SuspendRetryCount,
 863                                            SuspendRetryDelay, bits)) {
 864         // Didn't make it so let the caller know.
 865         return false;
 866       }
 867     }
 868     // We aren't allowed to wait for the external suspend to complete
 869     // so if the other thread isn't externally suspended we need to
 870     // let the caller know.
 871     else if (!is_ext_suspend_completed_with_lock(bits)) {
 872       return false;
 873     }
 874   }
 875 
 876   return true;
 877 }

1411 int WatcherThread::sleep() const {
1412   // The WatcherThread does not participate in the safepoint protocol
1413   // for the PeriodicTask_lock because it is not a JavaThread.
1414   MonitorLocker ml(PeriodicTask_lock, Mutex::_no_safepoint_check_flag);
1415 
1416   if (_should_terminate) {
1417     // check for termination before we do any housekeeping or wait
1418     return 0;  // we did not sleep.
1419   }
1420 
1421   // remaining will be zero if there are no tasks,
1422   // causing the WatcherThread to sleep until a task is
1423   // enrolled
1424   int remaining = PeriodicTask::time_to_wait();
1425   int time_slept = 0;
1426 
1427   // we expect this to timeout - we only ever get unparked when
1428   // we should terminate or when a new task has been enrolled
1429   OSThreadWaitState osts(this->osthread(), false /* not Object.wait() */);
1430 
1431   jlong time_before_loop = os::javaTimeNanos();
1432 
1433   while (true) {
1434     bool timedout = ml.wait(remaining);
1435     jlong now = os::javaTimeNanos();
1436 
1437     if (remaining == 0) {
1438       // if we didn't have any tasks we could have waited for a long time
1439       // consider the time_slept zero and reset time_before_loop
1440       time_slept = 0;
1441       time_before_loop = now;
1442     } else {
1443       // need to recalculate since we might have new tasks in _tasks
1444       time_slept = (int) ((now - time_before_loop) / 1000000);
1445     }
1446 
1447     // Change to task list or spurious wakeup of some kind
1448     if (timedout || _should_terminate) {
1449       break;
1450     }
1451 
1452     remaining = PeriodicTask::time_to_wait();
1453     if (remaining == 0) {
1454       // Last task was just disenrolled so loop around and wait until
1455       // another task gets enrolled
1456       continue;
1457     }
1458 
1459     remaining -= time_slept;
1460     if (remaining <= 0) {
1461       break;
1462     }
1463   }
1464 
1465   return time_slept;
1466 }

1468 void WatcherThread::run() {
1469   assert(this == watcher_thread(), "just checking");
1470 
1471   this->set_active_handles(JNIHandleBlock::allocate_block());
1472   while (true) {
1473     assert(watcher_thread() == Thread::current(), "thread consistency check");
1474     assert(watcher_thread() == this, "thread consistency check");
1475 
1476     // Calculate how long it'll be until the next PeriodicTask work
1477     // should be done, and sleep that amount of time.
1478     int time_waited = sleep();
1479 
1480     if (VMError::is_error_reported()) {
1481       // A fatal error has happened, the error handler(VMError::report_and_die)
1482       // should abort JVM after creating an error log file. However in some
1483       // rare cases, the error handler itself might deadlock. Here periodically
1484       // check for error reporting timeouts, and if it happens, just proceed to
1485       // abort the VM.
1486 
1487       // This code is in WatcherThread because WatcherThread wakes up
1488       // periodically so the fatal error handler doesn't need to do anything;
1489       // also because the WatcherThread is less likely to crash than other
1490       // threads.
1491 
1492       for (;;) {
1493         // Note: we use naked sleep in this loop because we want to avoid using
1494         // any kind of VM infrastructure which may be broken at this point.
1495         if (VMError::check_timeout()) {
1496           // We hit error reporting timeout. Error reporting was interrupted and
1497           // will be wrapping things up now (closing files etc). Give it some more
1498           // time, then quit the VM.
1499           os::naked_short_sleep(200);
1500           // Print a message to stderr.
1501           fdStream err(defaultStream::output_fd());
1502           err.print_raw_cr("# [ timer expired, abort... ]");
1503           // skip atexit/vm_exit/vm_abort hooks
1504           os::die();
1505         }
1506 
1507         // Wait a second, then recheck for timeout.
1508         os::naked_short_sleep(999);
1509       }
1510     }
1511 
1512     if (_should_terminate) {
1513       // check for termination before posting the next tick
1514       break;
1515     }
1516 
1517     PeriodicTask::real_time_tick(time_waited);
1518   }
1519 
1520   // Signal that it is terminated
1521   {
1522     MutexLocker mu(Terminator_lock, Mutex::_no_safepoint_check_flag);
1523     _watcher_thread = NULL;
1524     Terminator_lock->notify_all();
1525   }
1526 }

1770 bool JavaThread::is_interrupted(bool clear_interrupted) {
1771   debug_only(check_for_dangling_thread_pointer(this);)
1772 
1773   if (threadObj() == NULL) {
1774     // If there is no j.l.Thread then it is impossible to have
1775     // been interrupted. We can find NULL during VM initialization
1776     // or when a JNI thread is still in the process of attaching.
1777     // In such cases this must be the current thread.
1778     assert(this == Thread::current(), "invariant");
1779     return false;
1780   }
1781 
1782   bool interrupted = java_lang_Thread::interrupted(threadObj());
1783 
1784   // NOTE that since there is no "lock" around the interrupt and
1785   // is_interrupted operations, there is the possibility that the
1786   // interrupted flag will be "false" but that the
1787   // low-level events will be in the signaled state. This is
1788   // intentional. The effect of this is that Object.wait() and
1789   // LockSupport.park() will appear to have a spurious wakeup, which
1790   // is allowed and not harmful, and the possibility is so rare that
1791   // it is not worth the added complexity to add yet another lock.
1792   // For the sleep event an explicit reset is performed on entry
1793   // to JavaThread::sleep, so there is no early return. It has also been
1794   // recommended not to put the interrupted flag into the "event"
1795   // structure because it hides the issue.
1796   // Also, because there is no lock, we must only clear the interrupt
1797   // state if we are going to report that we were interrupted; otherwise
1798   // an interrupt that happens just after we read the field would be lost.
1799   if (interrupted && clear_interrupted) {
1800     assert(this == Thread::current(), "only the current thread can clear");
1801     java_lang_Thread::set_interrupted(threadObj(), false);
1802     osthread()->set_interrupted(false);
1803   }

1808 bool JavaThread::reguard_stack(address cur_sp) {
1809   if (_stack_guard_state != stack_guard_yellow_reserved_disabled
1810       && _stack_guard_state != stack_guard_reserved_disabled) {
1811     return true; // Stack already guarded or guard pages not needed.
1812   }

1814   if (register_stack_overflow()) {

1816     // memory stacks, we must check the register stack to see if
1817     // it has overflowed.
1818     return false;
1819   }

1821   // Java code never executes within the yellow zone: the latter is only
1822   // there to provoke an exception during stack banging.  If java code
1823   // is executing there, either StackShadowPages should be larger, or
1824   // some exception code in c1, c2 or the interpreter isn't unwinding
1825   // when it should.

1829     enable_stack_yellow_reserved_zone();

1831       set_reserved_stack_activation(stack_base());
1832     }
1833   } else if (_stack_guard_state == stack_guard_reserved_disabled) {
1834     set_reserved_stack_activation(stack_base());
1835     enable_stack_reserved_zone();
1836   }
1837   return true;

1940 // The main routine called by a new Java thread. This isn't overridden
1941 // by subclasses, instead different subclasses define a different "entry_point"
1942 // which defines the actual logic for that kind of thread.
1943 void JavaThread::run() {
1944   // initialize thread-local alloc buffer related fields
1945   this->initialize_tlab();
1946 
1947   // Used to test validity of stack trace backs.
1948   // This can't be moved into pre_run() else we invalidate
1949   // the requirement that thread_main_inner is lower on
1950   // the stack. Consequently all the initialization logic
1951   // stays here in run() rather than pre_run().
1952   this->record_base_of_stack_pointer();
1953 
1954   this->create_stack_guard_pages();
1955 
1956   this->cache_global_variables();
1957 
1958   // Thread is now sufficiently initialized to be handled by the safepoint code as being
1959   // in the VM. Change thread state from _thread_new to _thread_in_vm
1960   ThreadStateTransition::transition(this, _thread_new, _thread_in_vm);
1961   // Before a thread is on the threads list it is always safe, so after leaving the
1962   // _thread_new we should emit a instruction barrier. The distance to modified code
1963   // from here is probably far enough, but this is consistent and safe.
1964   OrderAccess::cross_modify_fence();
1965 
1966   assert(JavaThread::current() == this, "sanity check");
1967   assert(!Thread::current()->owns_locks(), "sanity check");
1968 
1969   DTRACE_THREAD_PROBE(start, this);
1970 
1971   // This operation might block. We call that after all safepoint checks for a new thread has
1972   // been completed.
1973   this->set_active_handles(JNIHandleBlock::allocate_block());
1974 
1975   if (JvmtiExport::should_post_thread_life()) {
1976     JvmtiExport::post_thread_start(this);
1977 
1978   }
1979 
1980   // We call another function to do the rest so we are sure that the stack addresses used
1981   // from there will be lower than the stack base just computed.
1982   thread_main_inner();
1983 }

2015 static void ensure_join(JavaThread* thread) {
2016   // We do not need to grab the Threads_lock, since we are operating on ourself.
2017   Handle threadObj(thread, thread->threadObj());
2018   assert(threadObj.not_null(), "java thread object must exist");
2019   ObjectLocker lock(threadObj, thread);
2020   // Ignore pending exception (ThreadDeath), since we are exiting anyway
2021   thread->clear_pending_exception();
2022   // Thread is exiting. So set thread_status field in  java.lang.Thread class to TERMINATED.
2023   java_lang_Thread::set_thread_status(threadObj(), java_lang_Thread::TERMINATED);
2024   // Clear the native thread instance - this makes isAlive return false and allows the join()
2025   // to complete once we've done the notify_all below
2026   java_lang_Thread::set_thread(threadObj(), NULL);
2027   lock.notify_all(thread);
2028   // Ignore pending exception (ThreadDeath), since we are exiting anyway
2029   thread->clear_pending_exception();
2030 }

2303 // _thread_in_native_trans state (such as from
2304 // check_special_condition_for_native_trans()).
2305 void JavaThread::check_and_handle_async_exceptions(bool check_unsafe_error) {
2306 
2307   if (has_last_Java_frame() && has_async_condition()) {
2308     // If we are at a polling page safepoint (not a poll return)
2309     // then we must defer async exception because live registers
2310     // will be clobbered by the exception path. Poll return is
2311     // ok because the call we a returning from already collides
2312     // with exception handling registers and so there is no issue.
2313     // (The exception handling path kills call result registers but
2314     //  this is ok since the exception kills the result anyway).
2315 
2316     if (is_at_poll_safepoint()) {
2317       // if the code we are returning to has deoptimized we must defer
2318       // the exception otherwise live registers get clobbered on the
2319       // exception path before deoptimization is able to retrieve them.
2320       //
2321       RegisterMap map(this, false);
2322       frame caller_fr = last_frame().sender(&map);
2323       assert(caller_fr.is_compiled_frame(), "what?");
2324       if (caller_fr.is_deoptimized_frame()) {
2325         log_info(exceptions)("deferred async exception at compiled safepoint");
2326         return;
2327       }
2328     }
2329   }
2330 
2331   JavaThread::AsyncRequests condition = clear_special_runtime_exit_condition();
2332   if (condition == _no_async_condition) {
2333     // Conditions have changed since has_special_runtime_exit_condition()
2334     // was called:
2335     // - if we were here only because of an external suspend request,
2336     //   then that was taken care of above (or cancelled) so we are done
2337     // - if we were here because of another async request, then it has
2338     //   been cleared between the has_special_runtime_exit_condition()
2339     //   and now so again we are done
2340     return;
2341   }
2342 
2343   // Check for pending async. exception
2344   if (_pending_async_exception != NULL) {
2345     // Only overwrite an already pending exception, if it is not a threadDeath.
2346     if (!has_pending_exception() || !pending_exception()->is_a(SystemDictionary::ThreadDeath_klass())) {
2347 
2348       // We cannot call Exceptions::_throw(...) here because we cannot block
2349       set_pending_exception(_pending_async_exception, __FILE__, __LINE__);
2350 
2351       LogTarget(Info, exceptions) lt;
2352       if (lt.is_enabled()) {
2353         ResourceMark rm;
2354         LogStream ls(lt);
2355         ls.print("Async. exception installed at runtime exit (" INTPTR_FORMAT ")", p2i(this));
2356           if (has_last_Java_frame()) {
2357             frame f = last_frame();
2358            ls.print(" (pc: " INTPTR_FORMAT " sp: " INTPTR_FORMAT " )", p2i(f.pc()), p2i(f.sp()));
2359           }
2360         ls.print_cr(" of type: %s", _pending_async_exception->klass()->external_name());
2361       }
2362       _pending_async_exception = NULL;
2363       clear_has_async_exception();
2364     }
2365   }
2366 
2367   if (check_unsafe_error &&
2368       condition == _async_unsafe_access_error && !has_pending_exception()) {
2369     condition = _no_async_condition;  // done
2370     switch (thread_state()) {
2371     case _thread_in_vm: {
2372       JavaThread* THREAD = this;
2373       THROW_MSG(vmSymbols::java_lang_InternalError(), "a fault occurred in an unsafe memory access operation");
2374     }
2375     case _thread_in_native: {
2376       ThreadInVMfromNative tiv(this);
2377       JavaThread* THREAD = this;
2378       THROW_MSG(vmSymbols::java_lang_InternalError(), "a fault occurred in an unsafe memory access operation");
2379     }
2380     case _thread_in_Java: {
2381       ThreadInVMfromJava tiv(this);
2382       JavaThread* THREAD = this;
2383       THROW_MSG(vmSymbols::java_lang_InternalError(), "a fault occurred in a recent unsafe memory access operation in compiled Java code");
2384     }
2385     default:
2386       ShouldNotReachHere();
2387     }
2388   }
2389 
2390   assert(condition == _no_async_condition || has_pending_exception() ||
2391          (!check_unsafe_error && condition == _async_unsafe_access_error),
2392          "must have handled the async condition, if no exception");

2467 // External suspension mechanism.
2468 //
2469 // Tell the VM to suspend a thread when ever it knows that it does not hold on
2470 // to any VM_locks and it is at a transition
2471 // Self-suspension will happen on the transition out of the vm.
2472 // Catch "this" coming in from JNIEnv pointers when the thread has been freed
2473 //
2474 // Guarantees on return:
2475 //   + Target thread will not execute any new bytecode (that's why we need to
2476 //     force a safepoint)
2477 //   + Target thread will not enter any new monitors
2478 //
2479 void JavaThread::java_suspend() {
2480   ThreadsListHandle tlh;
2481   if (!tlh.includes(this) || threadObj() == NULL || is_exiting()) {
2482     return;
2483   }
2484 
2485   { MutexLocker ml(SR_lock(), Mutex::_no_safepoint_check_flag);
2486     if (!is_external_suspend()) {
2487       // a racing resume has cancelled us; bail out now
2488       return;
2489     }
2490 
2491     // suspend is done
2492     uint32_t debug_bits = 0;
2493     // Warning: is_ext_suspend_completed() may temporarily drop the
2494     // SR_lock to allow the thread to reach a stable thread state if
2495     // it is currently in a transient thread state.
2496     if (is_ext_suspend_completed(false /* !called_by_wait */,
2497                                  SuspendRetryDelay, &debug_bits)) {
2498       return;
2499     }
2500   }
2501 
2502   if (Thread::current() == this) {
2503     // Safely self-suspend.
2504     // If we don't do this explicitly it will implicitly happen
2505     // before we transition back to Java, and on some other thread-state
2506     // transition paths, but not as we exit a JVM TI SuspendThread call.
2507     // As SuspendThread(current) must not return (until resumed) we must
2508     // self-suspend here.
2509     ThreadBlockInVM tbivm(this);
2510     java_suspend_self();
2511   } else {
2512     VM_ThreadSuspend vm_suspend;
2513     VMThread::execute(&vm_suspend);
2514   }
2515 }

2517 // Part II of external suspension.
2518 // A JavaThread self suspends when it detects a pending external suspend
2519 // request. This is usually on transitions. It is also done in places
2520 // where continuing to the next transition would surprise the caller,
2521 // e.g., monitor entry.
2522 //
2523 // Returns the number of times that the thread self-suspended.
2524 //
2525 // Note: DO NOT call java_suspend_self() when you just want to block current
2526 //       thread. java_suspend_self() is the second stage of cooperative
2527 //       suspension for external suspend requests and should only be used
2528 //       to complete an external suspend request.
2529 //
2530 int JavaThread::java_suspend_self() {
2531   assert(thread_state() == _thread_blocked, "wrong state for java_suspend_self()");
2532   int ret = 0;
2533 
2534   // we are in the process of exiting so don't suspend
2535   if (is_exiting()) {
2536     clear_external_suspend();
2537     return ret;
2538   }
2539 
2540   assert(_anchor.walkable() ||
2541          (is_Java_thread() && !((JavaThread*)this)->has_last_Java_frame()),
2542          "must have walkable stack");
2543 
2544   MonitorLocker ml(SR_lock(), Mutex::_no_safepoint_check_flag);
2545 
2546   assert(!this->is_ext_suspended(),
2547          "a thread trying to self-suspend should not already be suspended");
2548 
2549   if (this->is_suspend_equivalent()) {
2550     // If we are self-suspending as a result of the lifting of a
2551     // suspend equivalent condition, then the suspend_equivalent
2552     // flag is not cleared until we set the ext_suspended flag so
2553     // that wait_for_ext_suspend_completion() returns consistent
2554     // results.
2555     this->clear_suspend_equivalent();
2556   }
2557 
2558   // A racing resume may have cancelled us before we grabbed SR_lock
2559   // above. Or another external suspend request could be waiting for us
2560   // by the time we return from SR_lock()->wait(). The thread
2561   // that requested the suspension may already be trying to walk our
2562   // stack and if we return now, we can change the stack out from under
2563   // it. This would be a "bad thing (TM)" and cause the stack walker
2564   // to crash. We stay self-suspended until there are no more pending
2565   // external suspend requests.
2566   while (is_external_suspend()) {
2567     ret++;
2568     this->set_ext_suspended();
2569 
2570     // _ext_suspended flag is cleared by java_resume()
2571     while (is_ext_suspended()) {
2572       ml.wait();
2573     }
2574   }
2575   return ret;
2576 }

2578 // Helper routine to set up the correct thread state before calling java_suspend_self.
2579 // This is called when regular thread-state transition helpers can't be used because
2580 // we can be in various states, in particular _thread_in_native_trans.
2581 // Because this thread is external suspended the safepoint code will count it as at
2582 // a safepoint, regardless of what its actual current thread-state is. But
2583 // is_ext_suspend_completed() may be waiting to see a thread transition from
2584 // _thread_in_native_trans to _thread_blocked. So we set the thread state directly
2585 // to _thread_blocked. The problem with setting thread state directly is that a
2586 // safepoint could happen just after java_suspend_self() returns after being resumed,
2587 // and the VM thread will see the _thread_blocked state. So we must check for a safepoint
2588 // after restoring the state to make sure we won't leave while a safepoint is in progress.
2589 // However, not all initial-states are allowed when performing a safepoint check, as we
2590 // should never be blocking at a safepoint whilst in those states. Of these 'bad' states
2591 // only _thread_in_native is possible when executing this code (based on our two callers).
2592 // A thread that is _thread_in_native is already safepoint-safe and so it doesn't matter
2593 // whether the VMThread sees the _thread_blocked state, or the _thread_in_native state,
2594 // and so we don't need the explicit safepoint check.
2595 
2596 void JavaThread::java_suspend_self_with_safepoint_check() {
2597   assert(this == Thread::current(), "invariant");
2598   JavaThreadState state = thread_state();
2599   set_thread_state(_thread_blocked);
2600   java_suspend_self();
2601   set_thread_state_fence(state);
2602   // Since we are not using a regular thread-state transition helper here,
2603   // we must manually emit the instruction barrier after leaving a safe state.
2604   OrderAccess::cross_modify_fence();
2605   if (state != _thread_in_native) {
2606     SafepointMechanism::block_if_requested(this);
2607   }
2608 }

3446 // java.lang.Thread.sleep support
3447 // Returns true if sleep time elapsed as expected, and false
3448 // if the thread was interrupted.
3449 bool JavaThread::sleep(jlong millis) {
3450   assert(this == Thread::current(),  "thread consistency check");
3451 
3452   ParkEvent * const slp = this->_SleepEvent;
3453   // Because there can be races with thread interruption sending an unpark()
3454   // to the event, we explicitly reset it here to avoid an immediate return.
3455   // The actual interrupt state will be checked before we park().
3456   slp->reset();
3457   // Thread interruption establishes a happens-before ordering in the
3458   // Java Memory Model, so we need to ensure we synchronize with the
3459   // interrupt state.
3460   OrderAccess::fence();
3461 
3462   jlong prevtime = os::javaTimeNanos();
3463 
3464   for (;;) {
3465     // interruption has precedence over timing out
3466     if (this->is_interrupted(true)) {
3467       return false;
3468     }
3469 
3470     if (millis <= 0) {
3471       return true;
3472     }
3473 
3474     {
3475       ThreadBlockInVM tbivm(this);
3476       OSThreadWaitState osts(this->osthread(), false /* not Object.wait() */);
3477 
3478       this->set_suspend_equivalent();
3479       // cleared by handle_special_suspend_equivalent_condition() or
3480       // java_suspend_self() via check_and_wait_while_suspended()
3481 
3482       slp->park(millis);
3483 
3484       // were we externally suspended while we were waiting?
3485       this->check_and_wait_while_suspended();
3486     }
3487 
3488     // Update elapsed time tracking
3489     jlong newtime = os::javaTimeNanos();
3490     if (newtime - prevtime < 0) {
3491       // time moving backwards, should only happen if no monotonic clock
3492       // not a guarantee() because JVM should not abort on kernel/glibc bugs
3493       assert(!os::supports_monotonic_clock(),
3494              "unexpected time moving backwards detected in JavaThread::sleep()");
3495     } else {
3496       millis -= (newtime - prevtime) / NANOSECS_PER_MILLISEC;
3497     }
3498     prevtime = newtime;
3499   }

3655 // The system initialization in the library has three phases.
3656 //
3657 // Phase 1: java.lang.System class initialization
3658 //     java.lang.System is a primordial class loaded and initialized
3659 //     by the VM early during startup.  java.lang.System.<clinit>
3660 //     only does registerNatives and keeps the rest of the class
3661 //     initialization work later until thread initialization completes.
3662 //
3663 //     System.initPhase1 initializes the system properties, the static
3664 //     fields in, out, and err. Set up java signal handlers, OS-specific
3665 //     system settings, and thread group of the main thread.
3666 static void call_initPhase1(TRAPS) {
3667   Klass* klass =  SystemDictionary::resolve_or_fail(vmSymbols::java_lang_System(), true, CHECK);
3668   JavaValue result(T_VOID);
3669   JavaCalls::call_static(&result, klass, vmSymbols::initPhase1_name(),
3670                                          vmSymbols::void_method_signature(), CHECK);
3671 }
3672 
3673 // Phase 2. Module system initialization
3674 //     This will initialize the module system.  Only java.base classes
3675 //     can be loaded until phase 2 completes.
3676 //
3677 //     Call System.initPhase2 after the compiler initialization and jsr292
3678 //     classes get initialized because module initialization runs a lot of java
3679 //     code, that for performance reasons, should be compiled.  Also, this will
3680 //     enable the startup code to use lambda and other language features in this
3681 //     phase and onward.
3682 //
3683 //     After phase 2, The VM will begin search classes from -Xbootclasspath/a.
3684 static void call_initPhase2(TRAPS) {
3685   TraceTime timer("Initialize module system", TRACETIME_LOG(Info, startuptime));
3686 
3687   Klass* klass = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_System(), true, CHECK);
3688 
3689   JavaValue result(T_INT);
3690   JavaCallArguments args;
3691   args.push_int(DisplayVMOutputToStderr);
3692   args.push_int(log_is_enabled(Debug, init)); // print stack trace if exception thrown
3693   JavaCalls::call_static(&result, klass, vmSymbols::initPhase2_name(),
3694                                          vmSymbols::boolean_boolean_int_signature(), &args, CHECK);
3695   if (result.get_jint() != JNI_OK) {
3696     vm_exit_during_initialization(); // no message or exception
3697   }
3698 
3699   universe_post_module_init();
3700 }

3702 // Phase 3. final setup - set security manager, system class loader and TCCL
3703 //
3704 //     This will instantiate and set the security manager, set the system class
3705 //     loader as well as the thread context class loader.  The security manager
3706 //     and system class loader may be a custom class loaded from -Xbootclasspath/a,
3707 //     other modules or the application's classpath.
3708 static void call_initPhase3(TRAPS) {
3709   Klass* klass = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_System(), true, CHECK);
3710   JavaValue result(T_VOID);
3711   JavaCalls::call_static(&result, klass, vmSymbols::initPhase3_name(),
3712                                          vmSymbols::void_method_signature(), CHECK);
3713 }

4406 // Threads::destroy_vm() is normally called from jni_DestroyJavaVM() when
4407 // the program falls off the end of main(). Another VM exit path is through
4408 // vm_exit() when the program calls System.exit() to return a value or when
4409 // there is a serious error in VM. The two shutdown paths are not exactly
4410 // the same, but they share Shutdown.shutdown() at Java level and before_exit()
4411 // and VM_Exit op at VM level.
4412 //
4413 // Shutdown sequence:
4414 //   + Shutdown native memory tracking if it is on
4415 //   + Wait until we are the last non-daemon thread to execute
4416 //     <-- every thing is still working at this moment -->
4417 //   + Call java.lang.Shutdown.shutdown(), which will invoke Java level
4418 //        shutdown hooks
4419 //   + Call before_exit(), prepare for VM exit
4420 //      > run VM level shutdown hooks (they are registered through JVM_OnExit(),
4421 //        currently the only user of this mechanism is File.deleteOnExit())
4422 //      > stop StatSampler, watcher thread,
4423 //        post thread end and vm death events to JVMTI,
4424 //        stop signal thread
4425 //   + Call JavaThread::exit(), it will:
4426 //      > release JNI handle blocks, remove stack guard pages
4427 //      > remove this thread from Threads list
4428 //     <-- no more Java code from this thread after this point -->
4429 //   + Stop VM thread, it will bring the remaining VM to a safepoint and stop
4430 //     the compiler threads at safepoint
4431 //     <-- do not use anything that could get blocked by Safepoint -->
4432 //   + Disable tracing at JNI/JVM barriers
4433 //   + Set _vm_exited flag for threads that are still running native code
4434 //   + Call exit_globals()
4435 //      > deletes tty
4436 //      > deletes PerfMemory resources
4437 //   + Delete this thread
4438 //   + Return to caller
4439 
4440 bool Threads::destroy_vm() {
4441   JavaThread* thread = JavaThread::current();
4442 
4443 #ifdef ASSERT
4444   _vm_complete = false;
4445 #endif
4446   // Wait until we are the last non-daemon thread to execute
4447   { MonitorLocker nu(Threads_lock);
4448     while (Threads::number_of_non_daemon_threads() > 1)
4449       // This wait should make safepoint checks, wait without a timeout,
4450       // and wait as a suspend-equivalent condition.
4451       nu.wait(0, Mutex::_as_suspend_equivalent_flag);
4452   }
4453 
4454   EventShutdown e;
4455   if (e.should_commit()) {
4456     e.set_reason("No remaining non-daemon Java threads");
4457     e.commit();
4458   }
4459 
4460   // Hang forever on exit if we are reporting an error.
4461   if (ShowMessageBoxOnError && VMError::is_error_reported()) {
4462     os::infinite_sleep();
4463   }
4464   os::wait_for_keypress_at_exit();
4465 
4466   // run Java level shutdown hooks
4467   thread->invoke_shutdown_hooks();
4468 
4469   before_exit(thread);
4470 
4471   thread->exit(true);
4472 
4473   // Stop VM thread.
4474   {
4475     // 4945125 The vm thread comes to a safepoint during exit.
4476     // GC vm_operations can get caught at the safepoint, and the
4477     // heap is unparseable if they are caught. Grab the Heap_lock
4478     // to prevent this. The GC vm_operations will not be able to
4479     // queue until after the vm thread is dead. After this point,
4480     // we'll never emerge out of the safepoint before the VM exits.
4481 
4482     MutexLocker ml(Heap_lock, Mutex::_no_safepoint_check_flag);
4483 
4484     VMThread::wait_for_vm_thread_exit();
4485     assert(SafepointSynchronize::is_at_safepoint(), "VM thread should exit at Safepoint");
4486     VMThread::destroy();
4487   }
4488 
4489   // Now, all Java threads are gone except daemon threads. Daemon threads
4490   // running Java code or in VM are stopped by the Safepoint. However,
4491   // daemon threads executing native code are still running.  But they
4492   // will be stopped at native=>Java/VM barriers. Note that we can't
4493   // simply kill or suspend them, as it is inherently deadlock-prone.
4494 
4495   VM_Exit::set_vm_exited();
4496 
4497   // Clean up ideal graph printers after the VMThread has started
4498   // the final safepoint which will block all the Compiler threads.
4499   // Note that this Thread has already logically exited so the
4500   // clean_up() function's use of a JavaThreadIteratorWithHandle
4501   // would be a problem except set_vm_exited() has remembered the
4502   // shutdown thread which is granted a policy exception.
4503 #if defined(COMPILER2) && !defined(PRODUCT)
4504   IdealGraphPrinter::clean_up();
4505 #endif
4506 
4507   notify_vm_shutdown();
4508 
4509   // exit_globals() will delete tty
4510   exit_globals();
4511 
4512   // We are after VM_Exit::set_vm_exited() so we can't call
4513   // thread->smr_delete() or we will block on the Threads_lock.
4514   // Deleting the shutdown thread here is safe because another
4515   // JavaThread cannot have an active ThreadsListHandle for
4516   // this JavaThread.
4517   delete thread;
4518 
4519 #if INCLUDE_JVMCI
4520   if (JVMCICounterSize > 0) {
4521     FREE_C_HEAP_ARRAY(jlong, JavaThread::_jvmci_old_thread_counters);
4522   }
4523 #endif
4524 
4525   LogConfiguration::finalize();
4526 
4527   return true;
4528 }

4631     // collisions between a new global token value and a stale token

4643     // avoid the special "never claimed" initial thread value.
4644     _thread_claim_token = 1;
4645   }

4906 // Ad-hoc mutual exclusion primitives: SpinLock and Mux
4907 //
4908 // We employ SpinLocks _only for low-contention, fixed-length
4909 // short-duration critical sections where we're concerned
4910 // about native mutex_t or HotSpot Mutex:: latency.
4911 // The mux construct provides a spin-then-block mutual exclusion
4912 // mechanism.
4913 //
4914 // Testing has shown that contention on the ListLock guarding gFreeList
4915 // is common.  If we implement ListLock as a simple SpinLock it's common
4916 // for the JVM to devolve to yielding with little progress.  This is true
4917 // despite the fact that the critical sections protected by ListLock are
4918 // extremely short.
4919 //
4920 // TODO-FIXME: ListLock should be of type SpinLock.
4921 // We should make this a 1st-class type, integrated into the lock
4922 // hierarchy as leaf-locks.  Critically, the SpinLock structure
4923 // should have sufficient padding to avoid false-sharing and excessive
4924 // cache-coherency traffic.
4925 
4926 
4927 typedef volatile int SpinLockT;
4928 
4929 void Thread::SpinAcquire(volatile int * adr, const char * LockName) {
4930   if (Atomic::cmpxchg(adr, 0, 1) == 0) {
4931     return;   // normal fast-path return
4932   }
4933 
4934   // Slow-path : We've encountered contention -- Spin/Yield/Block strategy.
4935   int ctr = 0;
4936   int Yields = 0;
4937   for (;;) {
4938     while (*adr != 0) {
4939       ++ctr;
4940       if ((ctr & 0xFFF) == 0 || !os::is_MP()) {
4941         if (Yields > 5) {
4942           os::naked_short_sleep(1);
4943         } else {
4944           os::naked_yield();
4945           ++Yields;
4946         }
4947       } else {
4948         SpinPause();
4949       }
4950     }
4951     if (Atomic::cmpxchg(adr, 0, 1) == 0) return;
4952   }
4953 }

4955 void Thread::SpinRelease(volatile int * adr) {
4956   assert(*adr != 0, "invariant");
4957   OrderAccess::fence();      // guarantee at least release consistency.
4958   // Roach-motel semantics.
4959   // It's safe if subsequent LDs and STs float "up" into the critical section,
4960   // but prior LDs and STs within the critical section can't be allowed
4961   // to reorder or float past the ST that releases the lock.
4962   // Loads and stores in the critical section - which appear in program
4963   // order before the store that releases the lock - must also appear
4964   // before the store that releases the lock in memory visibility order.
4965   // Conceptually we need a #loadstore|#storestore "release" MEMBAR before
4966   // the ST of 0 into the lock-word which releases the lock, so fence
4967   // more than covers this on all platforms.
4968   *adr = 0;

4971 // muxAcquire and muxRelease:
4972 //
4973 // *  muxAcquire and muxRelease support a single-word lock-word construct.
4974 //    The LSB of the word is set IFF the lock is held.
4975 //    The remainder of the word points to the head of a singly-linked list
4976 //    of threads blocked on the lock.
4977 //
4978 // *  The current implementation of muxAcquire-muxRelease uses its own
4979 //    dedicated Thread._MuxEvent instance.  If we're interested in
4980 //    minimizing the peak number of extant ParkEvent instances then
4981 //    we could eliminate _MuxEvent and "borrow" _ParkEvent as long
4982 //    as certain invariants were satisfied.  Specifically, care would need
4983 //    to be taken with regards to consuming unpark() "permits".
4984 //    A safe rule of thumb is that a thread would never call muxAcquire()
4985 //    if it's enqueued (cxq, EntryList, WaitList, etc) and will subsequently
4986 //    park().  Otherwise the _ParkEvent park() operation in muxAcquire() could
4987 //    consume an unpark() permit intended for monitorenter, for instance.
4988 //    One way around this would be to widen the restricted-range semaphore
4989 //    implemented in park().  Another alternative would be to provide
4990 //    multiple instances of the PlatformEvent() for each thread.  One
4991 //    instance would be dedicated to muxAcquire-muxRelease, for instance.
4992 //
4993 // *  Usage:
4994 //    -- Only as leaf locks
4995 //    -- for short-term locking only as muxAcquire does not perform
4996 //       thread state transitions.
4997 //
4998 // Alternatives:
4999 // *  We could implement muxAcquire and muxRelease with MCS or CLH locks
5000 //    but with parking or spin-then-park instead of pure spinning.
5001 // *  Use Taura-Oyama-Yonenzawa locks.
5002 // *  It's possible to construct a 1-0 lock if we encode the lockword as
5003 //    (List,LockByte).  Acquire will CAS the full lockword while Release
5004 //    will STB 0 into the LockByte.  The 1-0 scheme admits stranding, so
5005 //    acquiring threads use timers (ParkTimed) to detect and recover from
5006 //    the stranding window.  Thread/Node structures must be aligned on 256-byte
5007 //    boundaries by using placement-new.
5008 // *  Augment MCS with advisory back-link fields maintained with CAS().
5009 //    Pictorially:  LockWord -> T1 <-> T2 <-> T3 <-> ... <-> Tn <-> Owner.
5010 //    The validity of the backlinks must be ratified before we trust the value.
5011 //    If the backlinks are invalid the exiting thread must back-track through the
5012 //    the forward links, which are always trustworthy.
5013 // *  Add a successor indication.  The LockWord is currently encoded as
5014 //    (List, LOCKBIT:1).  We could also add a SUCCBIT or an explicit _succ variable
5015 //    to provide the usual futile-wakeup optimization.
5016 //    See RTStt for details.
5017 //
5018 
5019 
5020 const intptr_t LOCKBIT = 1;
5021 
5022 void Thread::muxAcquire(volatile intptr_t * Lock, const char * LockName) {
5023   intptr_t w = Atomic::cmpxchg(Lock, (intptr_t)0, LOCKBIT);
5024   if (w == 0) return;
5025   if ((w & LOCKBIT) == 0 && Atomic::cmpxchg(Lock, w, w|LOCKBIT) == w) {
5026     return;
5027   }
5028 
5029   ParkEvent * const Self = Thread::current()->_MuxEvent;
5030   assert((intptr_t(Self) & LOCKBIT) == 0, "invariant");
5031   for (;;) {
5032     int its = (os::is_MP() ? 100 : 0) + 1;
5033 
5034     // Optional spin phase: spin-then-park strategy
5035     while (--its >= 0) {
5036       w = *Lock;
5037       if ((w & LOCKBIT) == 0 && Atomic::cmpxchg(Lock, w, w|LOCKBIT) == w) {
5038         return;
5039       }
5040     }
5041 
5042     Self->reset();
5043     Self->OnList = intptr_t(Lock);
5044     // The following fence() isn't _strictly necessary as the subsequent
5045     // CAS() both serializes execution and ratifies the fetched *Lock value.
5046     OrderAccess::fence();
5047     for (;;) {
5048       w = *Lock;
5049       if ((w & LOCKBIT) == 0) {
5050         if (Atomic::cmpxchg(Lock, w, w|LOCKBIT) == w) {
5051           Self->OnList = 0;   // hygiene - allows stronger asserts
5052           return;
5053         }
5054         continue;      // Interference -- *Lock changed -- Just retry
5055       }
5056       assert(w & LOCKBIT, "invariant");
5057       Self->ListNext = (ParkEvent *) (w & ~LOCKBIT);
5058       if (Atomic::cmpxchg(Lock, w, intptr_t(Self)|LOCKBIT) == w) break;
5059     }
5060 
5061     while (Self->OnList != 0) {
5062       Self->park();
5063     }
5064   }
5065 }

5067 // Release() must extract a successor from the list and then wake that thread.
5068 // It can "pop" the front of the list or use a detach-modify-reattach (DMR) scheme
5069 // similar to that used by ParkEvent::Allocate() and ::Release().  DMR-based
5070 // Release() would :
5071 // (A) CAS() or swap() null to *Lock, releasing the lock and detaching the list.
5072 // (B) Extract a successor from the private list "in-hand"
5073 // (C) attempt to CAS() the residual back into *Lock over null.
5074 //     If there were any newly arrived threads and the CAS() would fail.
5075 //     In that case Release() would detach the RATs, re-merge the list in-hand
5076 //     with the RATs and repeat as needed.  Alternately, Release() might
5077 //     detach and extract a successor, but then pass the residual list to the wakee.
5078 //     The wakee would be responsible for reattaching and remerging before it
5079 //     competed for the lock.
5080 //
5081 // Both "pop" and DMR are immune from ABA corruption -- there can be
5082 // multiple concurrent pushers, but only one popper or detacher.
5083 // This implementation pops from the head of the list.  This is unfair,
5084 // but tends to provide excellent throughput as hot threads remain hot.
5085 // (We wake recently run threads first).
5086 //
5087 // All paths through muxRelease() will execute a CAS.
5088 // Release consistency -- We depend on the CAS in muxRelease() to provide full
5089 // bidirectional fence/MEMBAR semantics, ensuring that all prior memory operations
5090 // executed within the critical section are complete and globally visible before the
5091 // store (CAS) to the lock-word that releases the lock becomes globally visible.
5092 void Thread::muxRelease(volatile intptr_t * Lock)  {
5093   for (;;) {
5094     const intptr_t w = Atomic::cmpxchg(Lock, LOCKBIT, (intptr_t)0);
5095     assert(w & LOCKBIT, "invariant");
5096     if (w == LOCKBIT) return;
5097     ParkEvent * const List = (ParkEvent *) (w & ~LOCKBIT);
5098     assert(List != NULL, "invariant");
5099     assert(List->OnList == intptr_t(Lock), "invariant");
5100     ParkEvent * const nxt = List->ListNext;
5101     guarantee((intptr_t(nxt) & LOCKBIT) == 0, "invariant");
5102 
5103     // The following CAS() releases the lock and pops the head element.
5104     // The CAS() also ratifies the previously fetched lock-word value.
5105     if (Atomic::cmpxchg(Lock, w, intptr_t(nxt)) != w) {
5106       continue;
5107     }
5108     List->OnList = 0;
5109     OrderAccess::fence();
5110     List->unpark();
5111     return;
5112   }
5113 }
