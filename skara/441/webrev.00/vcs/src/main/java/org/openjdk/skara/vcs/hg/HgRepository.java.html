  44     private void copyResource(String name, Path p) throws IOException {

  48     private java.lang.Process start(String... cmd) throws IOException {
  49         return start(Arrays.asList(cmd));
  50     }
  51 
  52     private java.lang.Process start(List<String> cmd) throws IOException {
  53         log.fine("Executing " + String.join(" ", cmd));
  54         var pb = new ProcessBuilder(cmd);
  55         pb.directory(dir.toFile());
  56         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  57         pb.environment().put("HGRCPATH", "");
  58         pb.environment().put("HGPLAIN", "");
  59         return pb.start();
  60     }
  61 
  62     private static void stop(java.lang.Process p) throws IOException {
  63         if (p != null && p.isAlive()) {
  64             var stream = p.getInputStream();
  65             var read = 0;
  66             var buf = new byte[128];
  67             while (read != -1) {
  68                 read = stream.read(buf);
  69             }
  70             try {
  71                 p.waitFor();
  72             } catch (InterruptedException e) {
  73                 throw new IOException(e);
  74             }
  75         }
  76     }

  86     private static Execution capture(Path cwd, List<String> cmd) {
  87         return capture(cwd, cmd.toArray(new String[0]));
  88     }
  89     private static Execution capture(Path cwd, String... cmd) {
  90         return Process.capture(cmd)
  91                       .environ("HGRCPATH", "")
  92                       .environ("HGPLAIN", "")
  93                       .workdir(cwd)
  94                       .execute();
  95     }

  97     private static Execution.Result await(Execution e) throws IOException {
  98         var result = e.await();
  99         if (result.status() != 0) {
 100             if (result.exception().isPresent()) {
 101                 throw new IOException("Unexpected exit code\n" + result, result.exception().get());
 102             } else {
 103                 throw new IOException("Unexpected exit code\n" + result);
 104             }
 105         }
 106         return result;
 107     }
 108 
 109     private static void await(java.lang.Process p) throws IOException {
 110         try {
 111             var res = p.waitFor();
 112             if (res != 0) {
 113                 throw new IOException("Unexpected exit code: " + res);
 114             }
 115         } catch (InterruptedException e) {
 116             throw new IOException(e);
 117         }
 118     }

 163     private void checkout(String ref, boolean force) throws IOException {
 164         var cmd = new ArrayList<String>();
 165         cmd.addAll(List.of("hg", "update"));
 166         if (!force) {
 167             cmd.add("--check");
 168         }
 169         cmd.add(ref);
 170         try (var p = capture(cmd)) {
 171             await(p);
 172         }
 173     }

 186     public Optional<Hash> resolve(String ref) throws IOException {
 187         try (var p = capture("hg", "log", "--rev=" + ref, "--template={node}\n")) {
 188             var res = p.await();
 189             if (res.status() == 0 && res.stdout().size() == 1) {
 190                 return Optional.of(new Hash(res.stdout().get(0)));
 191             }
 192             return Optional.empty();
 193         }

 274     @Override
 275     public List<CommitMetadata> commitMetadata(String range, List<Path> paths, boolean reverse) throws IOException {
 276         var ext = Files.createTempFile("ext", ".py");
 277         copyResource(EXT_PY, ext);
 278 
 279         var args = new ArrayList<String>();
 280         args.addAll(List.of("hg", "--config", "extensions.dump=" + ext.toAbsolutePath().toString(), "metadata"));
 281         range = range == null ? "tip:0" : range;
 282         var revset = reverse ? "reverse(" + range + ")" : range;
 283         args.add(revset);
 284         if (paths != null && !paths.isEmpty()) {
 285             args.add(paths.stream().map(Path::toString).collect(Collectors.joining("\t")));
 286         }
 287         var p = start(args);
 288         var reader = new UnixStreamReader(p.getInputStream());
 289         var result = new ArrayList<CommitMetadata>();
 290 
 291         var line = reader.readLine();
 292         while (line != null) {
 293             result.add(HgCommitMetadata.read(reader));
 294             line = reader.readLine();
 295         }
 296 
 297         await(p);
 298         return result;
 299     }

 341     @Override
 342     public boolean isEmpty() throws IOException {
 343         var numBranches = branches().size();
 344         var numTags = tags().size();
 345 
 346         if (numBranches > 0 || numTags > 1) {
 347             return false;
 348         }
 349 
 350         var tip = resolve("tip");
 351         return tip.isEmpty() || tip.get().hex().equals("0".repeat(40));
 352     }

 358                  Files.exists(Path.of(root, ".hg", "store", "lock")));

 361     @Override
 362     public void clean() throws IOException {
 363         try (var p = capture("hg", "merge", "--abort")) {
 364             p.await();
 365         }
 366 
 367         try (var p = capture("hg", "recover")) {
 368             p.await();
 369         }
 370 
 371         try (var p = capture("hg", "status", "--ignored", "--no-status")) {
 372             var root = root().toString();
 373             for (var filename : await(p).stdout()) {
 374                 Files.delete(Path.of(root, filename));
 375             }
 376         }
 377 
 378         try (var p = capture("hg", "status", "--unknown", "--no-status")) {
 379             var root = root().toString();
 380             for (var filename : await(p).stdout()) {
 381                 Files.delete(Path.of(root, filename));
 382             }
 383         }
 384 
 385         try (var p = capture("hg", "revert", "--no-backup", "--all")) {
 386             await(p);
 387         }
 388     }

 410     private Hash fetch(String from, String refspec) throws IOException {
 411         var oldHeads = new HashSet<Hash>(heads());
 412 
 413         var cmd = new ArrayList<String>();
 414         cmd.add("hg");
 415         cmd.add("pull");
 416         if (refspec != null) {
 417             cmd.add("--rev");
 418             cmd.add(refspec);
 419         }
 420         if (from != null) {
 421             cmd.add(from);
 422         }
 423         try (var p = capture(cmd)) {
 424             await(p);
 425         }
 426 
 427         var newHeads = new HashSet<Hash>(heads());
 428         newHeads.removeAll(oldHeads);
 429 
 430         if (newHeads.size() > 1) {
 431             throw new IllegalStateException("fetching multiple heads is not supported");
 432         } else if (newHeads.size() == 0) {
 433             // no new head was fetched, return current head
 434             return head();
 435         }
 436         return newHeads.iterator().next();
 437     }

 439     @Override
 440     public void fetchAll() throws IOException {
 441         var pullPaths = new ArrayList<URI>();
 442         try (var p = capture("hg", "paths")) {
 443             var res = await(p);
 444             for (var line : res.stdout()) {
 445                 var parts = line.split("=");
 446                 var name = parts[0].trim();
 447                 var uri = parts[1].trim();
 448                 if (!name.endsWith("-push")) {
 449                     pullPaths.add(URI.create(uri));
 450                 }
 451             }
 452         }
 453 
 454         for (var uri : pullPaths) {
 455             fetch(uri, null);
 456         }
 457     }

 530     private void export(String revset, Path to) throws IOException {
 531         var cmd = List.of("hg", "export", "--git", "--rev", revset);
 532         log.fine("Executing " + String.join(" ", cmd));
 533         var pb = new ProcessBuilder(cmd);
 534         pb.directory(dir.toFile());
 535         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 536         pb.redirectOutput(to.toFile());
 537         pb.environment().put("HGRCPATH", "");
 538         pb.environment().put("HGPLAIN", "");
 539         var p = pb.start();
 540         try {
 541             await(p);
 542         } catch (Throwable t) {
 543             if (p.isAlive()) {
 544                 try {
 545                     p.waitFor();
 546                 } catch (InterruptedException e) {
 547                     throw new IOException(e);
 548                 }
 549             }
 550 
 551             throw new IOException(t);
 552         }
 553     }

 555     @Override
 556     public void squash(Hash h) throws IOException {
 557         var revset = ".:" + h.hex() + " and not .";
 558         var patch = Files.createTempFile("squash", ".patch");
 559         export(revset, patch);
 560 
 561         try (var p = capture("hg", "--config", "extensions.mq=", "strip", "--rev", revset)) {
 562             await(p);
 563         }
 564 
 565         try (var p = capture("hg", "import", "--no-commit", patch.toString())) {
 566             await(p);
 567         }
 568     }

 576     @Override
 577     public Hash commit(String message, String authorName, String authorEmail, ZonedDateTime authorDate)  throws IOException {
 578         var user = authorEmail == null ? authorName : authorName + " <" + authorEmail + ">";
 579         var cmd = new ArrayList<String>();
 580         cmd.addAll(List.of("hg", "commit", "--message=" + message, "--user=" + user));
 581         if (authorDate != null) {
 582             var formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME;
 583             cmd.add("--date=" + authorDate.format(formatter));
 584         }
 585         try (var p = capture(cmd)) {
 586             await(p);
 587         }
 588         return resolve("tip").orElseThrow(() -> new IOException("Could not resolve 'tip'"));
 589     }

 749     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 750         var ext = Files.createTempFile("ext", ".py");
 751         copyResource(EXT_PY, ext);
 752 
 753         var include = new HashSet<>(paths);
 754 
 755         try (var p = capture("hg", "--config", "extensions.ls-tree=" + ext, "ls-tree", hash.hex())) {
 756             var res = await(p);
 757             var entries = new ArrayList<FileEntry>();
 758             for (var line : res.stdout()) {
 759                 var parts = line.split("\t");
 760                 var metadata = parts[0].split(" ");
 761                 var path = Path.of(parts[1]);
 762                 if (include.isEmpty() || include.contains(path)) {
 763                     var entry = new FileEntry(hash,
 764                                               FileType.fromOctal(metadata[0]),
 765                                               new Hash(metadata[2]),
 766                                               path);
 767                     entries.add(entry);
 768                 }
 769             }
 770             return entries;
 771         }
 772     }

 775     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 776         if (paths.isEmpty()) {
 777             return allFiles(hash, paths);
 778         }
 779 
 780         var entries = new ArrayList<FileEntry>();
 781         var batchSize = 64;
 782         var start = 0;
 783         while (start < paths.size()) {
 784             var end = start + batchSize;
 785             if (end > paths.size()) {
 786                 end = paths.size();
 787             }
 788             entries.addAll(allFiles(hash, paths.subList(start, end)));
 789             start = end;
 790         }
 791         return entries;
 792     }

 980     @Override
 981     public void setPaths(String remote, String pullPath, String pushPath) throws IOException {
 982         var hgrc = Path.of(root().toString(), ".hg", "hgrc");
 983         if (!Files.exists(hgrc)) {
 984             Files.createFile(hgrc);
 985         }
 986 
 987         var lines = Files.readAllLines(hgrc);
 988         var newLines = new ArrayList<String>();
 989 
 990         var isInPathsSection = false;
 991         var hasPathsSection = false;
 992         for (var line : lines) {
 993             var isSectionHeader = line.startsWith("[") && line.endsWith("]");
 994             if (isSectionHeader && !isInPathsSection) {
 995                 isInPathsSection = line.equals("[paths]");
 996                 if (isInPathsSection) {
 997                     newLines.add(line);
 998                     newLines.add(remote + " = " + (pullPath == null ? "" : pullPath));
 999                     newLines.add(remote + "-push = " + (pushPath == null ? "" : pushPath));
1000                     hasPathsSection = true;
1001                     continue;
1002                 }
1003             }
1004 
1005             if (isInPathsSection && line.startsWith(remote)) {
1006                 if (line.startsWith(remote + "-push")) {
1007                     // skip
1008                 } else if (line.startsWith(remote + ":pushurl")) {
1009                     // skip
1010                 } else if (line.startsWith(remote + " ") || line.startsWith(remote + "=")) {
1011                     // skip
1012                 } else {
1013                     newLines.add(line);
1014                 }
1015             } else {
1016                 newLines.add(line);
1017             }
1018         }
1019 
1020         Files.write(hgrc, newLines, StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
1021         if (!hasPathsSection) {
1022             var section = List.of("[paths]",
1023                                   remote + " = " + (pullPath == null ? "" : pullPath),
1024                                   remote + "-push = " + (pushPath == null ? "" : pushPath));
1025             Files.write(hgrc, section, StandardOpenOption.WRITE, StandardOpenOption.APPEND);
1026         }
1027     }

1285         for (var path : urls.keySet()) {
1286             var url = urls.get(path);
1287             var hash = hashes.get(path);

1294     @Override
1295     public Optional<Tag.Annotated> annotate(Tag tag) throws IOException {
1296         var hgtags = root().resolve(".hgtags");
1297         if (!Files.exists(hgtags)) {
1298             return Optional.empty();
1299         }
1300         try (var p = capture("hg", "annotate", hgtags.toString())) {
1301             var reversed = new ArrayList<>(await(p).stdout());
1302             Collections.reverse(reversed);
1303             for (var line : reversed) {
1304                 var parts = line.split(" ");
1305                 var tagName = parts[2];
1306                 if (tagName.equals(tag.name())) {
1307                     var target = new Hash(parts[1]);
1308                     var rev = parts[0].substring(0, parts[0].length() - 1).trim(); // skip last ':' and ev. whitespace
1309                     var hash = resolve(rev).orElseThrow(IOException::new);
1310                     var commit = lookup(hash).orElseThrow(IOException::new);
1311                     var message = String.join("\n", commit.message()) + "\n";
1312                     return Optional.of(new Tag.Annotated(tagName, target, commit.author(), commit.date(), message));
1313                 }
1314             }
1315         }
1316         return Optional.empty();
1317     }
