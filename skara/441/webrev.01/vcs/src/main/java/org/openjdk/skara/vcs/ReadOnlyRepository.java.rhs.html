 23 package org.openjdk.skara.vcs;
 24 
 25 import java.io.IOException;
 26 import java.nio.file.Path;
 27 import java.nio.charset.StandardCharsets;
 28 import java.util.stream.Collectors;
 29 import java.util.*;
 30 
 31 public interface ReadOnlyRepository {
 32     Hash head() throws IOException;
 33     Optional<Branch> currentBranch() throws IOException;
 34     Optional<Bookmark> currentBookmark() throws IOException;
 35     Branch defaultBranch() throws IOException;
 36     List<Branch> branches() throws IOException;
 37     List<Branch> branches(String remote) throws IOException;
 38     Optional<Tag> defaultTag() throws IOException;
 39     List<Tag> tags() throws IOException;
 40     Commits commits() throws IOException;
 41     Commits commits(int n) throws IOException;
 42     Commits commits(boolean reverse) throws IOException;
 43     Commits commits(int n, boolean reverse) throws IOException;
 44     Commits commits(String range) throws IOException;
 45     Commits commits(String range, boolean reverse) throws IOException;
 46     Commits commits(String range, int n) throws IOException;
 47     Commits commits(String range, int n, boolean reverse) throws IOException;
 48     Optional<Commit> lookup(Hash h) throws IOException;
 49     Optional<Commit> lookup(Branch b) throws IOException;
 50     Optional<Commit> lookup(Tag t) throws IOException;
 51     List<CommitMetadata> commitMetadata() throws IOException;
 52     List<CommitMetadata> commitMetadata(boolean reverse) throws IOException;
 53     List<CommitMetadata> commitMetadata(String range) throws IOException;
 54     List<CommitMetadata> commitMetadata(Hash from, Hash to) throws IOException;
 55     List<CommitMetadata> commitMetadata(String range, boolean reverse) throws IOException;
 56     List<CommitMetadata> commitMetadata(Hash from, Hash to, boolean reverse) throws IOException;
 57     List<CommitMetadata> commitMetadata(List<Path> paths) throws IOException;
 58     List<CommitMetadata> commitMetadata(List<Path> paths, boolean reverse) throws IOException;
 59     List<CommitMetadata> commitMetadata(String range, List<Path> paths) throws IOException;
 60     List<CommitMetadata> commitMetadata(Hash from, Hash to, List<Path> paths) throws IOException;
 61     List<CommitMetadata> commitMetadata(String range, List<Path> paths, boolean reverse) throws IOException;
 62     List<CommitMetadata> commitMetadata(Hash from, Hash to, List<Path> paths, boolean reverse) throws IOException;
 63     Path root() throws IOException;
 64     boolean exists() throws IOException;
 65     boolean isHealthy() throws IOException;
 66     boolean isEmpty() throws IOException;
 67     boolean isClean() throws IOException;
 68     Hash mergeBase(Hash first, Hash second) throws IOException;
 69     boolean isAncestor(Hash ancestor, Hash descendant) throws IOException;
 70     Optional<Hash> resolve(String ref) throws IOException;
 71     default Optional<Hash> resolve(Tag t) throws IOException {
 72         return resolve(t.name());
 73     }
 74     default Optional<Hash> resolve(Branch b) throws IOException {
 75         return resolve(b.name());
 76     }
 77     boolean contains(Branch b, Hash h) throws IOException;
 78     Optional<String> username() throws IOException;
 79     Optional<byte[]> show(Path p, Hash h) throws IOException;
 80     default Optional<List<String>> lines(Path p, Hash h) throws IOException {
 81         return show(p, h).map(bytes -> new String(bytes, StandardCharsets.UTF_8).lines().collect(Collectors.toList()));
 82     }
 83 
 84     List<FileEntry> files(Hash h, List<Path> paths) throws IOException;
 85     default List<FileEntry> files(Hash h, Path... paths) throws IOException {
 86         return files(h, Arrays.asList(paths));
 87     }
 88 
 89     void dump(FileEntry entry, Path to) throws IOException;
 90     List<StatusEntry> status(Hash from, Hash to) throws IOException;
 91     List<StatusEntry> status() throws IOException;
 92     Diff diff(Hash base, Hash head) throws IOException;
 93     Diff diff(Hash base, Hash head, List<Path> files) throws IOException;
 94     Diff diff(Hash head) throws IOException;
 95     Diff diff(Hash head, List<Path> files) throws IOException;
 96     List<String> config(String key) throws IOException;
 97     Repository copyTo(Path destination) throws IOException;
 98     String pullPath(String remote) throws IOException;
 99     String pushPath(String remote) throws IOException;
100     boolean isValidRevisionRange(String expression) throws IOException;
101     Optional<String> upstreamFor(Branch branch) throws IOException;
102     List<Reference> remoteBranches(String remote) throws IOException;
103     List<String> remotes() throws IOException;
104     List<Submodule> submodules() throws IOException;
105 
106     static Optional<ReadOnlyRepository> get(Path p) throws IOException {
107         return Repository.get(p).map(r -> r);
108     }
109 
110     static boolean exists(Path p) throws IOException {
111         return Repository.exists(p);
112     }
113 
114     Optional<Tag.Annotated> annotate(Tag tag) throws IOException;
115 }
