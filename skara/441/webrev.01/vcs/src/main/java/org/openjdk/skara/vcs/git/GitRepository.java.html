  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 202     @Override
 203     public List<CommitMetadata> commitMetadata(String range, List<Path> paths, boolean reverse) throws IOException {
 204         var args = new ArrayList<String>();
 205         args.addAll(List.of("git", "rev-list",
 206                                    "--format=" + GitCommitMetadata.FORMAT,
 207                                    "--no-abbrev",
 208                                    "--no-color",
 209                                    range));
 210         if (reverse) {
 211             args.add("--reverse");
 212         }
 213         if (paths != null && !paths.isEmpty()) {
 214             args.add("--");
 215             for (var path : paths) {
 216                 args.add(path.toString());
 217             }
 218         }
 219         var p = start(args);
 220         var reader = new UnixStreamReader(p.getInputStream());
 221         var result = new ArrayList<CommitMetadata>();
 222 
 223         var line = reader.readLine();
 224         while (line != null) {
 225             if (!line.startsWith("commit")) {
 226                 throw new IOException("Unexpected line: " + line);
 227             }
 228 
 229             result.add(GitCommitMetadata.read(reader));
 230             line = reader.readLine();
 231         }
 232 
 233         await(p);
 234         return result;
 235     }

 292     private List<Hash> refs() throws IOException {
 293         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 294             var res = p.await();
 295             if (res.status() == -1) {
 296                 if (res.stdout().size() != 0) {
 297                     throw new IOException("Unexpected output\n" + res);
 298                 }
 299                 return new ArrayList<>();
 300             } else {
 301                 return res.stdout().stream()
 302                           .map(Hash::new)
 303                           .collect(Collectors.toList());
 304             }
 305         }
 306     }

 308     @Override
 309     public boolean isEmpty() throws IOException {
 310         int numLooseObjects = -1;
 311         int numPackedObjects = -1;
 312 
 313         try (var p = capture("git", "count-objects", "-v")) {
 314             var res = await(p);
 315             var stdout = res.stdout();
 316 
 317             for (var line : stdout) {
 318                 if (line.startsWith("count: ")) {
 319                     try {
 320                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 321                     } catch (NumberFormatException e) {
 322                         throw new IOException("Unexpected 'count' value\n" + res, e);
 323                     }
 324 
 325                 } else if (line.startsWith("in-pack: ")) {
 326                     try {
 327                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 328                     } catch (NumberFormatException e) {
 329                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 330                     }
 331                 }
 332             }
 333         }
 334 
 335         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 336     }

 509     @Override
 510     public Path root() throws IOException {
 511         if (cachedRoot != null) {
 512             return cachedRoot;
 513         }
 514 
 515         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 516             var res = p.await();
 517             if (res.status() != 0 || res.stdout().size() != 1) {
 518                 // Perhaps this is a bare repository
 519                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 520                     var res2 = await(p2);
 521                     if (res2.stdout().size() != 1) {
 522                         throw new IOException("Unexpected output\n" + res2);
 523                     }
 524                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 525                     return cachedRoot;
 526                 }
 527             }
 528 
 529             cachedRoot = Path.of(res.stdout().get(0));
 530             return cachedRoot;
 531         }
 532     }

 541     @FunctionalInterface
 542     private static interface Operation {
 543         void execute(List<Path> args) throws IOException;
 544     }
 545 
 546     private void batch(Operation op, List<Path> args) throws IOException {
 547         var batchSize = 64;
 548         var start = 0;
 549         while (start < args.size()) {
 550             var end = start + batchSize;
 551             if (end > args.size()) {
 552                 end = args.size();
 553             }
 554             op.execute(args.subList(start, end));
 555             start = end;
 556         }
 557     }

 622     @Override
 623     public Hash commit(String message,
 624                        String authorName,
 625                        String authorEmail,
 626                        ZonedDateTime authorDate,
 627                        String committerName,
 628                        String committerEmail,
 629                        ZonedDateTime committerDate) throws IOException {
 630         var cmd = Process.capture("git", "commit", "--message=" + message)
 631                          .workdir(dir)
 632                          .environ("GIT_AUTHOR_NAME", authorName)
 633                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 634                          .environ("GIT_COMMITTER_NAME", committerName)
 635                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 636         if (authorDate != null) {
 637             cmd = cmd.environ("GIT_AUTHOR_DATE",
 638                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 639         }
 640         if (committerDate != null) {
 641             cmd = cmd.environ("GIT_COMMITTER_DATE",
 642                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 643         }
 644         try (var p = cmd.execute()) {
 645             await(p);
 646             return head();
 647         }
 648     }

 727     public void rebase(Hash hash, String committerName, String committerEmail) throws IOException {

 803     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 804         var cmd = new ArrayList<String>();
 805         cmd.addAll(List.of("git", "ls-tree", "-r"));
 806         cmd.add(hash.hex());
 807         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 808         try (var p = Process.capture(cmd.toArray(new String[0]))
 809                             .workdir(root())
 810                             .execute()) {
 811             var res = await(p);
 812             var entries = new ArrayList<FileEntry>();
 813             for (var line : res.stdout()) {
 814                 var parts = line.split("\t");
 815                 var metadata = parts[0].split(" ");
 816                 var filename = parts[1];
 817 
 818                 var entry = new FileEntry(hash,
 819                                           FileType.fromOctal(metadata[0]),
 820                                           new Hash(metadata[2]),
 821                                           Path.of(filename));
 822                 entries.add(entry);
 823             }
 824             return entries;
 825         }
 826     }

 828     @Override
 829     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 830         if (paths.isEmpty()) {
 831             return allFiles(hash, paths);
 832         }
 833 
 834         var entries = new ArrayList<FileEntry>();
 835         var batchSize = 64;
 836         var start = 0;
 837         while (start < paths.size()) {
 838             var end = start + batchSize;
 839             if (end > paths.size()) {
 840                 end = paths.size();
 841             }
 842             entries.addAll(allFiles(hash, paths.subList(start, end)));
 843             start = end;
 844         }
 845         return entries;
 846     }

 859     @Override
 860     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 861         var entries = files(hash, path);
 862         if (entries.size() == 0) {
 863             return Optional.empty();
 864         } else if (entries.size() > 1) {
 865             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 866         }
 867 
 868         var entry = entries.get(0);
 869         var type = entry.type();
 870         if (type.isVCSLink()) {
 871             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 872             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 873         } else if (type.isRegular()) {
 874             var tmp = unpackFile(entry.hash().hex());
 875             var content = Files.readAllBytes(tmp);
 876             Files.delete(tmp);
 877             return Optional.of(content);
 878         }
 879 
 880         return Optional.empty();
 881     }

 940         var cmd = new ArrayList<>(List.of("git", "diff", "--patch",
 941                                                          "--find-renames=99%",

1235     @Override
1236     public List<Submodule> submodules() throws IOException {
1237         var gitModules = root().resolve(".gitmodules");
1238         if (!Files.exists(gitModules)) {
1239             return List.of();
1240         }
1241 
1242         var urls = new HashMap<String, String>();
1243         var paths = new HashMap<String, String>();
1244         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1245                                               "--list")) {
1246             for (var line : await(p).stdout()) {
1247                 if (line.startsWith("submodule.")) {
1248                     line = line.substring("submodule.".length());
1249                     var parts = line.split("=");
1250                     var nameAndProperty = parts[0].split("\\.");
1251                     var name = nameAndProperty[0];
1252                     var prop = nameAndProperty[1];
1253                     var value = parts[1];
1254                     if (prop.equals("path")) {
1255                         paths.put(name, value);
1256                     } else if (prop.equals("url")) {
1257                         urls.put(name, value);
1258                     } else {
1259                         throw new IOException("Unexpected submodule property: " + prop);
1260                     }
1261                 }
1262             }
1263         }
1264 
1265         var hashes = new HashMap<String, String>();
1266         try (var p = capture("git", "submodule", "status")) {
1267             for (var line : await(p).stdout()) {
1268                 var parts = line.substring(1).split(" ");
1269                 var hash = parts[0];
1270                 var path = parts[1];
1271                 hashes.put(path, hash);
1272             }
1273         }
1274 
1275         var modules = new ArrayList<Submodule>();
1276         for (var name : paths.keySet()) {
1277             var url = urls.get(name);
1278             var path = paths.get(name);
1279             var hash = hashes.get(path);
1280 
1281             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1282         }
1283 
1284         return modules;
1285     }
