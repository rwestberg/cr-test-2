  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 193     @Override
 194     public List<CommitMetadata> commitMetadata(String range) throws IOException {
 195         var p = start("git", "rev-list", "--format=" + GitCommitMetadata.FORMAT, "--no-abbrev", "--reverse", "--no-color", range);
 196         var reader = new UnixStreamReader(p.getInputStream());
 197         var result = new ArrayList<CommitMetadata>();
 198 
 199         var line = reader.readLine();
 200         while (line != null) {
 201             if (!line.startsWith("commit")) {
 202                 throw new IOException("Unexpected line: " + line);
 203             }
 204 
 205             result.add(GitCommitMetadata.read(reader));
 206             line = reader.readLine();
 207         }
 208 
 209         await(p);
 210         return result;
 211     }

 218     private List<Hash> refs() throws IOException {
 219         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 220             var res = p.await();
 221             if (res.status() == -1) {
 222                 if (res.stdout().size() != 0) {
 223                     throw new IOException("Unexpected output\n" + res);
 224                 }
 225                 return new ArrayList<>();
 226             } else {
 227                 return res.stdout().stream()
 228                           .map(Hash::new)
 229                           .collect(Collectors.toList());
 230             }
 231         }
 232     }

 234     @Override
 235     public boolean isEmpty() throws IOException {
 236         int numLooseObjects = -1;
 237         int numPackedObjects = -1;
 238 
 239         try (var p = capture("git", "count-objects", "-v")) {
 240             var res = await(p);
 241             var stdout = res.stdout();
 242 
 243             for (var line : stdout) {
 244                 if (line.startsWith("count: ")) {
 245                     try {
 246                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 247                     } catch (NumberFormatException e) {
 248                         throw new IOException("Unexpected 'count' value\n" + res, e);
 249                     }
 250 
 251                 } else if (line.startsWith("in-pack: ")) {
 252                     try {
 253                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 254                     } catch (NumberFormatException e) {
 255                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 256                     }
 257                 }
 258             }
 259         }
 260 
 261         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 262     }

 264     @Override
 265     public boolean isHealthy() throws IOException {
 266         var refs = refs();
 267         if (refs.size() == 0) {
 268             return true;
 269         }
 270 
 271         var name = "health-check";
 272         try (var p = capture("git", "branch", name, refs.get(0).hex())) {
 273             if (p.await().status() != 0) {
 274                 return false;
 275             }
 276         }
 277         try (var p = capture("git", "branch", "-D", name)) {
 278             if (p.await().status() != 0) {
 279                 return false;
 280             }
 281         }
 282 
 283         return true;
 284     }

 446     @Override
 447     public Path root() throws IOException {
 448         if (cachedRoot != null) {
 449             return cachedRoot;
 450         }
 451 
 452         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 453             var res = p.await();
 454             if (res.status() != 0 || res.stdout().size() != 1) {
 455                 // Perhaps this is a bare repository
 456                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 457                     var res2 = await(p2);
 458                     if (res2.stdout().size() != 1) {
 459                         throw new IOException("Unexpected output\n" + res2);
 460                     }
 461                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 462                     return cachedRoot;
 463                 }
 464             }
 465 
 466             cachedRoot = Path.of(res.stdout().get(0));
 467             return cachedRoot;
 468         }
 469     }

 478     @FunctionalInterface
 479     private static interface Operation {
 480         void execute(List<Path> args) throws IOException;
 481     }
 482 
 483     private void batch(Operation op, List<Path> args) throws IOException {
 484         var batchSize = 64;
 485         var start = 0;
 486         while (start < args.size()) {
 487             var end = start + batchSize;
 488             if (end > args.size()) {
 489                 end = args.size();
 490             }
 491             op.execute(args.subList(start, end));
 492             start = end;
 493         }
 494     }

 559     @Override
 560     public Hash commit(String message,
 561                        String authorName,
 562                        String authorEmail,
 563                        ZonedDateTime authorDate,
 564                        String committerName,
 565                        String committerEmail,
 566                        ZonedDateTime committerDate) throws IOException {
 567         var cmd = Process.capture("git", "commit", "--message=" + message)
 568                          .workdir(dir)
 569                          .environ("GIT_AUTHOR_NAME", authorName)
 570                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 571                          .environ("GIT_COMMITTER_NAME", committerName)
 572                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 573         if (authorDate != null) {
 574             cmd = cmd.environ("GIT_AUTHOR_DATE",
 575                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 576         }
 577         if (committerDate != null) {
 578             cmd = cmd.environ("GIT_COMMITTER_DATE",
 579                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 580         }
 581         try (var p = cmd.execute()) {
 582             await(p);
 583             return head();
 584         }
 585     }

 725     private String treeEntry(Path path, Hash hash) throws IOException {
 726         try (var p = Process.capture("git", "ls-tree", hash.hex(), path.toString())
 727                             .workdir(root())
 728                             .execute()) {
 729             var res = await(p);
 730             if (res.stdout().size() == 0) {
 731                 return null;
 732             }

 740     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 741         var cmd = new ArrayList<String>();
 742         cmd.addAll(List.of("git", "ls-tree", "-r"));
 743         cmd.add(hash.hex());
 744         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 745         try (var p = Process.capture(cmd.toArray(new String[0]))
 746                             .workdir(root())
 747                             .execute()) {
 748             var res = await(p);
 749             var entries = new ArrayList<FileEntry>();
 750             for (var line : res.stdout()) {
 751                 var parts = line.split("\t");
 752                 var metadata = parts[0].split(" ");
 753                 var filename = parts[1];
 754 
 755                 var entry = new FileEntry(hash,
 756                                           FileType.fromOctal(metadata[0]),
 757                                           new Hash(metadata[2]),
 758                                           Path.of(filename));
 759                 entries.add(entry);
 760             }
 761             return entries;
 762         }
 763     }

 765     @Override
 766     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 767         if (paths.isEmpty()) {
 768             return allFiles(hash, paths);
 769         }
 770 
 771         var entries = new ArrayList<FileEntry>();
 772         var batchSize = 64;
 773         var start = 0;
 774         while (start < paths.size()) {
 775             var end = start + batchSize;
 776             if (end > paths.size()) {
 777                 end = paths.size();
 778             }
 779             entries.addAll(allFiles(hash, paths.subList(start, end)));
 780             start = end;
 781         }
 782         return entries;
 783     }

 796     @Override
 797     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 798         var entries = files(hash, path);
 799         if (entries.size() == 0) {
 800             return Optional.empty();
 801         } else if (entries.size() > 1) {
 802             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 803         }
 804 
 805         var entry = entries.get(0);
 806         var type = entry.type();
 807         if (type.isVCSLink()) {
 808             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 809             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 810         } else if (type.isRegular()) {
 811             var tmp = unpackFile(entry.hash().hex());
 812             var content = Files.readAllBytes(tmp);
 813             Files.delete(tmp);
 814             return Optional.of(content);
 815         }
 816 
 817         return Optional.empty();
 818     }

 858     public Diff diff(Hash from, Hash to, List<Path> files) throws IOException {
 859         var cmd = new ArrayList<>(List.of("git", "diff", "--patch",
 860                                                          "--find-renames=99%",
 861                                                          "--find-copies=99%",
 862                                                          "--find-copies-harder",
 863                                                          "--binary",
 864                                                          "--raw",
 865                                                          "--no-abbrev",
 866                                                          "--unified=0",
 867                                                          "--no-color",
 868                                                          from.hex()));
 869         if (to != null) {
 870             cmd.add(to.hex());
 871         }
 872 
 873         if (files != null && !files.isEmpty()) {
 874             cmd.add("--");
 875             for (var file : files) {
 876                 cmd.add(file.toString());
 877             }
 878         }
 879 
 880         var p = start(cmd);
 881         try {
 882             var patches = UnifiedDiffParser.parseGitRaw(p.getInputStream());
 883             await(p);
 884             return new Diff(from, to, patches);
 885         } catch (Throwable t) {
 886             stop(p);
 887             throw t;
 888         }
 889     }

1141     @Override
1142     public List<Submodule> submodules() throws IOException {
1143         var gitModules = root().resolve(".gitmodules");
1144         if (!Files.exists(gitModules)) {
1145             return List.of();
1146         }
1147 
1148         var urls = new HashMap<String, String>();
1149         var paths = new HashMap<String, String>();
1150         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1151                                               "--list")) {
1152             for (var line : await(p).stdout()) {
1153                 if (line.startsWith("submodule.")) {
1154                     line = line.substring("submodule.".length());
1155                     var parts = line.split("=");
1156                     var nameAndProperty = parts[0].split("\\.");
1157                     var name = nameAndProperty[0];
1158                     var prop = nameAndProperty[1];
1159                     var value = parts[1];
1160                     if (prop.equals("path")) {
1161                         paths.put(name, value);
1162                     } else if (prop.equals("url")) {
1163                         urls.put(name, value);
1164                     } else {
1165                         throw new IOException("Unexpected submodule property: " + prop);
1166                     }
1167                 }
1168             }
1169         }
1170 
1171         var hashes = new HashMap<String, String>();
1172         try (var p = capture("git", "submodule", "status")) {
1173             for (var line : await(p).stdout()) {
1174                 var parts = line.substring(1).split(" ");
1175                 var hash = parts[0];
1176                 var path = parts[1];
1177                 hashes.put(path, hash);
1178             }
1179         }
1180 
1181         var modules = new ArrayList<Submodule>();
1182         for (var name : paths.keySet()) {
1183             var url = urls.get(name);
1184             var path = paths.get(name);
1185             var hash = hashes.get(path);
1186 
1187             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1188         }
1189 
1190         return modules;
1191     }

1195         var ref = "refs/tags/" + tag.name();
1196         var format = "%(refname:short)%0a%(*objectname)%0a%(taggername) %(taggeremail)%0a%(taggerdate:iso-strict)%0a%(contents)";
1197         try (var p = capture("git", "for-each-ref", "--format", format, ref)) {

1200                 var name = lines.get(0);
1201                 var target = new Hash(lines.get(1));

1203

1206                 var message = String.join("\n", lines.subList(4, lines.size()));
1207 
1208                 return Optional.of(new Tag.Annotated(name, target, author, date, message));
