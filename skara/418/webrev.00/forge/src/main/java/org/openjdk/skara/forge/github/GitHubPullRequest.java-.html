 81     @Override
 82     public List<Review> reviews() {
 83         var reviews = request.get("pulls/" + json.get("number").toString() + "/reviews").execute().stream()
 84                              .map(JSONValue::asObject)
 85                              .filter(obj -> !(obj.get("state").asString().equals("COMMENTED") && obj.get("body").asString().isEmpty()))
 86                              .map(obj -> {
 87                                  var reviewer = host.parseUserField(obj);
 88                                  var hash = new Hash(obj.get("commit_id").asString());
 89                                  Review.Verdict verdict;
 90                                  switch (obj.get("state").asString()) {
 91                                      case "APPROVED":
 92                                          verdict = Review.Verdict.APPROVED;
 93                                          break;
 94                                      case "CHANGES_REQUESTED":
 95                                          verdict = Review.Verdict.DISAPPROVED;
 96                                          break;
 97                                      default:
 98                                          verdict = Review.Verdict.NONE;
 99                                          break;
100                                  }
101                                  var id = obj.get("id").asInt();
102                                  var body = obj.get("body").asString();
103                                  var createdAt = ZonedDateTime.parse(obj.get("submitted_at").asString());
104                                  return new Review(createdAt, reviewer, verdict, hash, id, body);
105                              })
106                              .collect(Collectors.toList());
107         return reviews;
108     }

110     @Override
111     public void addReview(Review.Verdict verdict, String body) {
112         var query = JSON.object();
113         switch (verdict) {
114             case APPROVED:
115                 query.put("event", "APPROVE");
116                 break;
117             case DISAPPROVED:
118                 query.put("event", "REQUEST_CHANGES");
119                 break;
120             case NONE:
121                 query.put("event", "COMMENT");
122                 break;
123         }
124         query.put("body", body);
125         request.post("pulls/" + json.get("number").toString() + "/reviews")
126                .body(query)
127                .execute();
128     }

130     private ReviewComment parseReviewComment(ReviewComment parent, JSONObject json) {
131         var author = host.parseUserField(json);
132         var threadId = parent == null ? json.get("id").toString() : parent.threadId();
133 
134         int line = json.get("original_line").asInt();
135         var hash = new Hash(json.get("original_commit_id").asString());
136         var path = json.get("path").asString();
137 
138         if (json.get("side").asString().equals("LEFT")) {
139             var commitInfo = request.get("commits/" + hash).execute();
140 
141             // It's possible that the file in question was renamed / deleted in an earlier commit, would
142             // need to parse all the commits in the PR to be sure. But this should cover most cases.
143             hash = new Hash(commitInfo.get("parents").asArray().get(0).get("sha").asString());
144             for (var file : commitInfo.get("files").asArray()) {
145                 if (file.get("filename").asString().equals(path)) {
146                     if (file.get("status").asString().equals("renamed")) {
147                         path = file.get("previous_filename").asString();
148                     }
149                     break;
150                 }
151             }
152         }
153 
154         var comment = new ReviewComment(parent,
155                                         threadId,
156                                         hash,
157                                         path,
158                                         line,
159                                         json.get("id").toString(),
160                                         json.get("body").asString(),
161                                         author,
162                                         ZonedDateTime.parse(json.get("created_at").asString()),
163                                         ZonedDateTime.parse(json.get("updated_at").asString()));
164         return comment;
165     }

167     @Override
168     public ReviewComment addReviewComment(Hash base, Hash hash, String path, int line, String body) {
169         var query = JSON.object()
170                         .put("body", body)
171                         .put("commit_id", hash.hex())
172                         .put("path", path)
173                         .put("side", "RIGHT")
174                         .put("line", line);
175         var response = request.post("pulls/" + json.get("number").toString() + "/comments")
176                               .body(query)
177                               .execute();
178         return parseReviewComment(null, response.asObject());
179     }

181     @Override
182     public ReviewComment addReviewCommentReply(ReviewComment parent, String body) {
183         var query = JSON.object()
184                         .put("body", body)
185                         .put("in_reply_to", Integer.parseInt(parent.threadId()));
186         var response = request.post("pulls/" + json.get("number").toString() + "/comments")
187                               .body(query)
188                               .execute();
189         return parseReviewComment(parent, response.asObject());
190     }

192     @Override
193     public List<ReviewComment> reviewComments() {
194         var ret = new ArrayList<ReviewComment>();
195         var reviewComments = request.get("pulls/" + json.get("number").toString() + "/comments").execute().stream()
196                                     .map(JSONValue::asObject)
197                                     .collect(Collectors.toList());
198         var idToComment = new HashMap<String, ReviewComment>();
199 
200         for (var reviewComment : reviewComments) {
201             ReviewComment parent = null;
202             if (reviewComment.contains("in_reply_to_id")) {
203                 parent = idToComment.get(reviewComment.get("in_reply_to_id").toString());
204             }
205             var comment = parseReviewComment(parent, reviewComment);
206             idToComment.put(comment.id(), comment);
207             ret.add(comment);
208         }
209 
210         return ret;
211     }

321     @Override
322     public Map<String, Check> checks(Hash hash) {
323         var checks = request.get("commits/" + hash.hex() + "/check-runs").execute();
324 
325         return checks.get("check_runs").stream()
326                 .collect(Collectors.toMap(c -> c.get("name").asString(),
327                         c -> {
328                             var checkBuilder = CheckBuilder.create(c.get("name").asString(), new Hash(c.get("head_sha").asString()));
329                             checkBuilder.startedAt(ZonedDateTime.parse(c.get("started_at").asString()));
330 
331                             var completed = c.get("status").asString().equals("completed");
332                             if (completed) {
333                                 var conclusion = c.get("conclusion").asString();
334                                 var completedAt = ZonedDateTime.parse(c.get("completed_at").asString());
335                                 switch (conclusion) {
336                                     case "cancelled":
337                                         checkBuilder.cancel(completedAt);
338                                         break;
339                                     case "success":
340                                         checkBuilder.complete(true, completedAt);
341                                         break;
342                                     case "failure":
343                                         // fallthrough
344                                     case "neutral":
345                                         checkBuilder.complete(false, completedAt);
346                                         break;
347                                     default:
348                                         throw new IllegalStateException("Unexpected conclusion: " + conclusion);
349                                 }
350                             }
351                             if (c.contains("external_id")) {
352                                 checkBuilder.metadata(c.get("external_id").asString());
353                             }
354                             if (c.contains("output")) {
355                                 var output = c.get("output").asObject();
356                                 if (output.contains("title")) {
357                                     checkBuilder.title(output.get("title").asString());
358                                 }
359                                 if (output.contains("summary")) {
360                                     checkBuilder.summary(output.get("summary").asString());
361                                 }
362                             }
363 
364                             return checkBuilder.build();
365                         }, (a, b) -> b));
366     }

368     @Override
369     public void createCheck(Check check) {
370         // update and create are currently identical operations, both do an HTTP
371         // POST to the /repos/:owner/:repo/check-runs endpoint. There is an additional
372         // endpoint explicitly for updating check-runs, but that is not currently used.
373         updateCheck(check);
374     }

376     @Override
377     public void updateCheck(Check check) {
378         var completedQuery = JSON.object();
379         completedQuery.put("name", check.name());
380         completedQuery.put("head_branch", json.get("head").get("ref"));
381         completedQuery.put("head_sha", check.hash().hex());
382 
383         if (check.title().isPresent() && check.summary().isPresent()) {
384             var outputQuery = JSON.object();
385             outputQuery.put("title", check.title().get());
386             outputQuery.put("summary", check.summary().get());
387 
388             var annotations = JSON.array();
389             for (var annotation : check.annotations().subList(0, Math.min(check.annotations().size(), 50))) {
390                 var annotationQuery = JSON.object();
391                 annotationQuery.put("path", annotation.path());
392                 annotationQuery.put("start_line", annotation.startLine());
393                 annotationQuery.put("end_line", annotation.endLine());
394                 annotation.startColumn().ifPresent(startColumn -> annotationQuery.put("start_column", startColumn));
395                 annotation.endColumn().ifPresent(endColumn -> annotationQuery.put("end_column", endColumn));
396                 switch (annotation.level()) {
397                     case NOTICE:
398                         annotationQuery.put("annotation_level", "notice");
399                         break;
400                     case WARNING:
401                         annotationQuery.put("annotation_level", "warning");
402                         break;
403                     case FAILURE:
404                         annotationQuery.put("annotation_level", "failure");
405                         break;
406                 }
407 
408                 annotationQuery.put("message", annotation.message());
409                 annotation.title().ifPresent(title -> annotationQuery.put("title", title));
410                 annotations.add(annotationQuery);
411             }
412 
413             outputQuery.put("annotations", annotations);
414             completedQuery.put("output", outputQuery);
415         }
416 
417         if (check.status() == CheckStatus.IN_PROGRESS) {
418             completedQuery.put("status", "in_progress");
419         } else {
420             completedQuery.put("status", "completed");
421             completedQuery.put("conclusion", check.status().name().toLowerCase());
422             completedQuery.put("completed_at", check.completedAt().orElse(ZonedDateTime.now(ZoneOffset.UTC))
423                     .format(DateTimeFormatter.ISO_INSTANT));
424         }
425 
426         completedQuery.put("started_at", check.startedAt().format(DateTimeFormatter.ISO_INSTANT));
427         check.metadata().ifPresent(metadata -> completedQuery.put("external_id", metadata));
428 
429         request.post("check-runs").body(completedQuery).execute();
430     }

463     @Override
464     public void removeLabel(String label) {
465         labels = null;
466         request.delete("issues/" + json.get("number").toString() + "/labels/" + label)
467                .onError(r -> {
468                    // The GitHub API explicitly states that 404 is the response for deleting labels currently not set
469                    if (r.statusCode() == 404) {
470                        return JSONValue.fromNull();
471                    }
472                    throw new RuntimeException("Invalid response");
473                })
474                .execute();
475     }
