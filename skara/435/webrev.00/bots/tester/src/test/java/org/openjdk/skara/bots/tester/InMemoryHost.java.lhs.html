23 package org.openjdk.skara.bots.tester;
24 
25 import org.openjdk.skara.forge.*;
26 import org.openjdk.skara.host.*;
27 
28 import java.util.*;
29 
30 class InMemoryHost implements Forge {
31     HostUser currentUserDetails = new HostUser(0, "openjdk", "openjdk [bot]");
32     Map<String, Set<HostUser>> groups;
33 
34     @Override
35     public boolean isValid() {
36         return false;
37     }
38 
39     @Override
40     public Optional<HostedRepository> repository(String name) {
41         return Optional.empty();
42     }
43 
44     @Override
45     public HostUser user(String username) {
46         return null;
47     }
48 
49     @Override
50     public HostUser currentUser() {
51         return currentUserDetails;
52     }
53 
54     @Override
55     public boolean supportsReviewBody() {
56         return false;
57     }
58 
59     @Override
60     public boolean isMemberOf(String groupId, HostUser user) {
61         return groups.get(groupId).contains(user);
62     }
63 }
