23 package org.openjdk.skara.vcs.openjdk;
24 
25 import java.util.regex.Pattern;
26 
27 public class CommitMessageSyntax {
28         private static final String OPENJDK_USERNAME_REGEX = "[-.a-z0-9]+";
29         public static final String EMAIL_ADDR_REGEX = "[a-z0-9._%+-]+@[a-z0-9.-]+\\.[a-z]+";
30         public static final String REAL_NAME_REGEX = "[-_a-zA-Z0-9][-_ a-zA-Z0-9'.]+";
31         private static final String REAL_NAME_AND_EMAIL_ATTR_REGEX = REAL_NAME_REGEX + " +<" + EMAIL_ADDR_REGEX + ">";
32         private static final String ATTR_REGEX = "(?:(?:" + EMAIL_ADDR_REGEX + ")|(?:" + REAL_NAME_AND_EMAIL_ATTR_REGEX + "))";
33 
34         public static final Pattern ISSUE_PATTERN = Pattern.compile("((?:[A-Z][A-Z0-9]+-)?[0-9]+): (\\S.*)$");
35         public static final Pattern SUMMARY_PATTERN = Pattern.compile("Summary: (\\S.*)");
36         public static final Pattern REVIEWED_BY_PATTERN = Pattern.compile("Reviewed-by: ((?:" + OPENJDK_USERNAME_REGEX + ")(?:, " + OPENJDK_USERNAME_REGEX + ")*)$");
37         public static final Pattern CONTRIBUTED_BY_PATTERN = Pattern.compile("Contributed-by: (" + ATTR_REGEX + "(?:, " + ATTR_REGEX + ")*)$");
38         public static final Pattern CO_AUTHOR_PATTERN = Pattern.compile("Co-authored-by: ((?:" + REAL_NAME_AND_EMAIL_ATTR_REGEX + ")(?:, " + REAL_NAME_AND_EMAIL_ATTR_REGEX + ")*)$");
39 }
