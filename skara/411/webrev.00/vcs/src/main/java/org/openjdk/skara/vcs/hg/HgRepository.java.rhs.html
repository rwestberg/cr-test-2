  48     private java.lang.Process start(String... cmd) throws IOException {
  49         return start(Arrays.asList(cmd));
  50     }

  52     private java.lang.Process start(List<String> cmd) throws IOException {
  53         log.fine("Executing " + String.join(" ", cmd));
  54         var pb = new ProcessBuilder(cmd);
  55         pb.directory(dir.toFile());
  56         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  57         pb.environment().put("HGRCPATH", "");
  58         pb.environment().put("HGPLAIN", "");
  59         return pb.start();
  60     }

  62     private static void stop(java.lang.Process p) throws IOException {
  63         if (p != null && p.isAlive()) {
  64             var stream = p.getInputStream();
  65             var read = 0;
  66             var buf = new byte[128];
  67             while (read != -1) {
  68                 read = stream.read(buf);
  69             }
  70             try {
  71                 p.waitFor();
  72             } catch (InterruptedException e) {
  73                 throw new IOException(e);
  74             }
  75         }
  76     }

  86     private static Execution capture(Path cwd, List<String> cmd) {
  87         return capture(cwd, cmd.toArray(new String[0]));

  89     private static Execution capture(Path cwd, String... cmd) {
  90         return Process.capture(cmd)
  91                       .environ("HGRCPATH", "")
  92                       .environ("HGPLAIN", "")
  93                       .workdir(cwd)
  94                       .execute();
  95     }

  97     private static Execution.Result await(Execution e) throws IOException {
  98         var result = e.await();
  99         if (result.status() != 0) {
 100             if (result.exception().isPresent()) {
 101                 throw new IOException("Unexpected exit code\n" + result, result.exception().get());
 102             } else {
 103                 throw new IOException("Unexpected exit code\n" + result);
 104             }
 105         }
 106         return result;
 107     }
 108 
 109     private static void await(java.lang.Process p) throws IOException {
 110         try {
 111             var res = p.waitFor();
 112             if (res != 0) {
 113                 throw new IOException("Unexpected exit code: " + res);
 114             }
 115         } catch (InterruptedException e) {
 116             throw new IOException(e);
 117         }
 118     }

 157     private void checkout(String ref, boolean force) throws IOException {
 158         var cmd = new ArrayList<String>();
 159         cmd.addAll(List.of("hg", "update"));
 160         if (!force) {
 161             cmd.add("--check");
 162         }
 163         cmd.add(ref);
 164         try (var p = capture(cmd)) {
 165             await(p);
 166         }
 167     }

 179     @Override
 180     public Optional<Hash> resolve(String ref) throws IOException {
 181         try (var p = capture("hg", "log", "--rev=" + ref, "--template={node}\n")) {
 182             var res = p.await();
 183             if (res.status() == 0 && res.stdout().size() == 1) {
 184                 return Optional.of(new Hash(res.stdout().get(0)));
 185             }
 186             return Optional.empty();

 253     @Override
 254     public List<CommitMetadata> commitMetadata() throws IOException {
 255         var ext = Files.createTempFile("ext", ".py");
 256         copyResource(EXT_PY, ext);
 257 
 258         var p = start("hg", "--config", "extensions.dump=" + ext.toAbsolutePath().toString(), "metadata");
 259         var reader = new UnixStreamReader(p.getInputStream());
 260         var result = new ArrayList<CommitMetadata>();
 261 
 262         var line = reader.readLine();
 263         while (line != null) {
 264             result.add(HgCommitMetadata.read(reader));
 265             line = reader.readLine();
 266         }
 267 
 268         await(p);
 269         return result;
 270     }

 272     @Override
 273     public boolean isEmpty() throws IOException {
 274         var numBranches = branches().size();
 275         var numTags = tags().size();
 276 
 277         if (numBranches > 0 || numTags > 1) {
 278             return false;
 279         }
 280 
 281         var tip = resolve("tip");
 282         return tip.isEmpty() || tip.get().hex().equals("0".repeat(40));
 283     }

 292     @Override
 293     public void clean() throws IOException {
 294         try (var p = capture("hg", "merge", "--abort")) {
 295             p.await();
 296         }
 297 
 298         try (var p = capture("hg", "recover")) {
 299             p.await();
 300         }
 301 
 302         try (var p = capture("hg", "status", "--ignored", "--no-status")) {
 303             var root = root().toString();
 304             for (var filename : await(p).stdout()) {
 305                 Files.delete(Path.of(root, filename));
 306             }
 307         }
 308 
 309         try (var p = capture("hg", "status", "--unknown", "--no-status")) {
 310             var root = root().toString();
 311             for (var filename : await(p).stdout()) {
 312                 Files.delete(Path.of(root, filename));
 313             }
 314         }
 315 
 316         try (var p = capture("hg", "revert", "--no-backup", "--all")) {
 317             await(p);
 318         }
 319     }

 341     private Hash fetch(String from, String refspec) throws IOException {
 342         var oldHeads = new HashSet<Hash>(heads());
 343 
 344         var cmd = new ArrayList<String>();
 345         cmd.add("hg");
 346         cmd.add("pull");
 347         if (refspec != null) {
 348             cmd.add("--rev");
 349             cmd.add(refspec);
 350         }
 351         if (from != null) {
 352             cmd.add(from);
 353         }
 354         try (var p = capture(cmd)) {
 355             await(p);
 356         }
 357 
 358         var newHeads = new HashSet<Hash>(heads());
 359         newHeads.removeAll(oldHeads);
 360 
 361         if (newHeads.size() > 1) {
 362             throw new IllegalStateException("fetching multiple heads is not supported");
 363         } else if (newHeads.size() == 0) {
 364             // no new head was fetched, return current head
 365             return head();
 366         }
 367         return newHeads.iterator().next();
 368     }

 370     @Override
 371     public void fetchAll() throws IOException {
 372         var pullPaths = new ArrayList<URI>();
 373         try (var p = capture("hg", "paths")) {
 374             var res = await(p);
 375             for (var line : res.stdout()) {
 376                 var parts = line.split("=");
 377                 var name = parts[0].trim();
 378                 var uri = parts[1].trim();
 379                 if (!name.endsWith("-push")) {
 380                     pullPaths.add(URI.create(uri));
 381                 }
 382             }
 383         }
 384 
 385         for (var uri : pullPaths) {
 386             fetch(uri, null);
 387         }
 388     }

 461     private void export(String revset, Path to) throws IOException {
 462         var cmd = List.of("hg", "export", "--git", "--rev", revset);
 463         log.fine("Executing " + String.join(" ", cmd));
 464         var pb = new ProcessBuilder(cmd);
 465         pb.directory(dir.toFile());
 466         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 467         pb.redirectOutput(to.toFile());
 468         pb.environment().put("HGRCPATH", "");
 469         pb.environment().put("HGPLAIN", "");
 470         var p = pb.start();
 471         try {
 472             await(p);
 473         } catch (Throwable t) {
 474             if (p.isAlive()) {
 475                 try {
 476                     p.waitFor();
 477                 } catch (InterruptedException e) {
 478                     throw new IOException(e);
 479                 }
 480             }
 481 
 482             throw new IOException(t);
 483         }
 484     }

 486     @Override
 487     public void squash(Hash h) throws IOException {
 488         var revset = ".:" + h.hex() + " and not .";
 489         var patch = Files.createTempFile("squash", ".patch");
 490         export(revset, patch);
 491 
 492         try (var p = capture("hg", "--config", "extensions.mq=", "strip", "--rev", revset)) {
 493             await(p);
 494         }
 495 
 496         try (var p = capture("hg", "import", "--no-commit", patch.toString())) {
 497             await(p);
 498         }
 499     }

 507     @Override
 508     public Hash commit(String message, String authorName, String authorEmail, ZonedDateTime authorDate)  throws IOException {
 509         var user = authorEmail == null ? authorName : authorName + " <" + authorEmail + ">";
 510         var cmd = new ArrayList<String>();
 511         cmd.addAll(List.of("hg", "commit", "--message=" + message, "--user=" + user));
 512         if (authorDate != null) {
 513             var formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME;
 514             cmd.add("--date=" + authorDate.format(formatter));
 515         }
 516         try (var p = capture(cmd)) {
 517             await(p);
 518         }
 519         return resolve("tip").orElseThrow(() -> new IOException("Could not resolve 'tip'"));
 520     }

 662     @Override
 663     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 664         var output = Files.createTempFile("hg-cat-rev-" + hash.abbreviate(), ".bin");
 665         try (var p = capture("hg", "cat", "--output=" + output, "--rev=" + hash.hex(), path.toString())) {
 666             var res = p.await();
 667             if (res.status() == 0 && Files.exists(output)) {
 668                 var bytes = Files.readAllBytes(output);
 669                 Files.delete(output);
 670                 return Optional.of(bytes);
 671             }
 672 
 673             if (Files.exists(output)) {
 674                 Files.delete(output);
 675             }
 676             return Optional.empty();
 677         }

 680     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 681         var ext = Files.createTempFile("ext", ".py");
 682         copyResource(EXT_PY, ext);
 683 
 684         var include = new HashSet<>(paths);
 685 
 686         try (var p = capture("hg", "--config", "extensions.ls-tree=" + ext, "ls-tree", hash.hex())) {
 687             var res = await(p);
 688             var entries = new ArrayList<FileEntry>();
 689             for (var line : res.stdout()) {
 690                 var parts = line.split("\t");
 691                 var metadata = parts[0].split(" ");
 692                 var path = Path.of(parts[1]);
 693                 if (include.isEmpty() || include.contains(path)) {
 694                     var entry = new FileEntry(hash,
 695                                               FileType.fromOctal(metadata[0]),
 696                                               new Hash(metadata[2]),
 697                                               path);
 698                     entries.add(entry);
 699                 }
 700             }
 701             return entries;
 702         }
 703     }

 705     @Override
 706     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 707         if (paths.isEmpty()) {
 708             return allFiles(hash, paths);
 709         }
 710 
 711         var entries = new ArrayList<FileEntry>();
 712         var batchSize = 64;
 713         var start = 0;
 714         while (start < paths.size()) {
 715             var end = start + batchSize;
 716             if (end > paths.size()) {
 717                 end = paths.size();
 718             }
 719             entries.addAll(allFiles(hash, paths.subList(start, end)));
 720             start = end;
 721         }
 722         return entries;
 723     }

 725     @Override
 726     public List<StatusEntry> status(Hash from, Hash to) throws IOException {
 727         var ext = Files.createTempFile("ext", ".py");
 728         copyResource(EXT_PY, ext);
 729 
 730         try (var p = capture("hg", "--config", "extensions.diff-git-raw=" + ext.toAbsolutePath().toString(),
 731                                                "diff-git-raw", from.hex(), to.hex())) {
 732             var res = await(p);
 733             var entries = new ArrayList<StatusEntry>();

 896     @Override
 897     public void setPaths(String remote, String pullPath, String pushPath) throws IOException {
 898         var hgrc = Path.of(root().toString(), ".hg", "hgrc");
 899         if (!Files.exists(hgrc)) {
 900             Files.createFile(hgrc);
 901         }
 902 
 903         var lines = Files.readAllLines(hgrc);
 904         var newLines = new ArrayList<String>();
 905 
 906         var isInPathsSection = false;
 907         var hasPathsSection = false;
 908         for (var line : lines) {
 909             var isSectionHeader = line.startsWith("[") && line.endsWith("]");
 910             if (isSectionHeader && !isInPathsSection) {
 911                 isInPathsSection = line.equals("[paths]");
 912                 if (isInPathsSection) {
 913                     newLines.add(line);
 914                     newLines.add(remote + " = " + (pullPath == null ? "" : pullPath));
 915                     newLines.add(remote + "-push = " + (pushPath == null ? "" : pushPath));
 916                     hasPathsSection = true;
 917                     continue;
 918                 }
 919             }
 920 
 921             if (isInPathsSection && line.startsWith(remote)) {
 922                 if (line.startsWith(remote + "-push")) {
 923                     // skip
 924                 } else if (line.startsWith(remote + ":pushurl")) {
 925                     // skip
 926                 } else if (line.startsWith(remote + " ") || line.startsWith(remote + "=")) {
 927                     // skip
 928                 } else {
 929                     newLines.add(line);
 930                 }
 931             } else {
 932                 newLines.add(line);
 933             }
 934         }
 935 
 936         Files.write(hgrc, newLines, StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
 937         if (!hasPathsSection) {
 938             var section = List.of("[paths]",
 939                                   remote + " = " + (pullPath == null ? "" : pullPath),
 940                                   remote + "-push = " + (pushPath == null ? "" : pushPath));
 941             Files.write(hgrc, section, StandardOpenOption.WRITE, StandardOpenOption.APPEND);
 942         }
 943     }

1022     private void batch(Operation op, List<Path> args) throws IOException {
1023         var batchSize = 64;
1024         var start = 0;
1025         while (start < args.size()) {
1026             var end = start + batchSize;
1027             if (end > args.size()) {
1028                 end = args.size();
1029             }
1030             op.execute(args.subList(start, end));
1031             start = end;
1032         }
1033     }

1177     public List<Submodule> submodules() throws IOException {
1178         var hgSub = root().resolve(".hgsub");
1179         var hgSubState = root().resolve(".hgsubstate");
1180         if (!(Files.exists(hgSub) && Files.exists(hgSubState))) {
1181             return List.of();
1182         }

1186             var parts = line.split("=");

1188             var url = parts[1].trim();
1189             urls.put(path, url);
1190         }
1191 
1192         var hashes = new HashMap<String, String>();
1193         for (var line : Files.readAllLines(hgSubState)) {
1194             var parts = line.split(" ");
1195             var hash = parts[0];
1196             var path = parts[1];
1197             hashes.put(path, hash);
1198         }
1199 
1200         var modules = new ArrayList<Submodule>();
1201         for (var path : urls.keySet()) {
1202             var url = urls.get(path);
1203             var hash = hashes.get(path);
1204             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1205         }
1206 
1207         return modules;

1211     public Optional<Tag.Annotated> annotate(Tag tag) throws IOException {
1212         var hgtags = root().resolve(".hgtags");
1213         if (!Files.exists(hgtags)) {
1214             return Optional.empty();
1215         }
1216         try (var p = capture("hg", "annotate", hgtags.toString())) {
1217             var reversed = new ArrayList<>(await(p).stdout());
1218             Collections.reverse(reversed);
1219             for (var line : reversed) {
1220                 var parts = line.split(" ");
1221                 var tagName = parts[2];
1222                 if (tagName.equals(tag.name())) {
1223                     var target = new Hash(parts[1]);
1224                     var rev = parts[0].substring(0, parts[0].length() - 1).trim(); // skip last ':' and ev. whitespace
1225                     var hash = resolve(rev).orElseThrow(IOException::new);
1226                     var commit = lookup(hash).orElseThrow(IOException::new);
1227                     var message = String.join("\n", commit.message()) + "\n";
1228                     return Optional.of(new Tag.Annotated(tagName, target, commit.author(), commit.date(), message));
1229                 }
1230             }
1231         }
1232         return Optional.empty();
1233     }
