 23 package org.openjdk.skara.bots.notify;
 24 
 25 import org.openjdk.skara.email.EmailAddress;
 26 import org.openjdk.skara.mailinglist.MailingList;
 27 
 28 import java.util.Map;
 29 import java.util.regex.Pattern;
 30 
 31 public class MailingListUpdaterBuilder {
 32     private MailingList list;
 33     private EmailAddress recipient;
 34     private EmailAddress sender;
 35     private EmailAddress author = null;
 36     private boolean includeBranch = false;
 37     private boolean reportNewTags = true;
 38     private boolean reportNewBranches = true;
 39     private boolean reportNewBuilds = true;
 40     private MailingListUpdater.Mode mode = MailingListUpdater.Mode.ALL;
 41     private Map<String, String> headers = Map.of();
 42     private Pattern allowedAuthorDomains = Pattern.compile(".*");


 43 
 44     public MailingListUpdaterBuilder list(MailingList list) {
 45         this.list = list;
 46         return this;
 47     }
 48 
 49     public MailingListUpdaterBuilder recipient(EmailAddress recipient) {
 50         this.recipient = recipient;
 51         return this;
 52     }
 53 
 54     public MailingListUpdaterBuilder sender(EmailAddress sender) {
 55         this.sender = sender;
 56         return this;
 57     }
 58 
 59     public MailingListUpdaterBuilder author(EmailAddress author) {
 60         this.author = author;
 61         return this;
 62     }
 63 
 64     public MailingListUpdaterBuilder includeBranch(boolean includeBranch) {
 65         this.includeBranch = includeBranch;
 66         return this;
 67     }
 68 
 69     public MailingListUpdaterBuilder reportNewTags(boolean reportNewTags) {
 70         this.reportNewTags = reportNewTags;
 71         return this;
 72     }
 73 
 74     public MailingListUpdaterBuilder reportNewBranches(boolean reportNewBranches) {
 75         this.reportNewBranches = reportNewBranches;
 76         return this;
 77     }
 78 
 79     public MailingListUpdaterBuilder reportNewBuilds(boolean reportNewBuilds) {
 80         this.reportNewBuilds = reportNewBuilds;
 81         return this;
 82     }
 83 
 84     public MailingListUpdaterBuilder mode(MailingListUpdater.Mode mode) {
 85         this.mode = mode;
 86         return this;
 87     }
 88 
 89     public MailingListUpdaterBuilder headers(Map<String, String> headers) {
 90         this.headers = headers;
 91         return this;
 92     }
 93 
 94     public MailingListUpdaterBuilder allowedAuthorDomains(Pattern allowedAuthorDomains) {
 95         this.allowedAuthorDomains = allowedAuthorDomains;
 96         return this;
 97     }
 98 










 99     public MailingListUpdater build() {
100         return new MailingListUpdater(list, recipient, sender, author, includeBranch, reportNewTags, reportNewBranches,
101                                       reportNewBuilds, mode, headers, allowedAuthorDomains);
102     }
103 }
