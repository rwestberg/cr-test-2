 23 package org.openjdk.skara.bots.notify;
 24 
 25 import org.openjdk.skara.email.EmailAddress;
 26 import org.openjdk.skara.mailinglist.MailingList;
 27 
 28 import java.util.Map;
 29 import java.util.regex.Pattern;
 30 
 31 public class MailingListUpdaterBuilder {
 32     private MailingList list;
 33     private EmailAddress recipient;
 34     private EmailAddress sender;
 35     private EmailAddress author = null;
 36     private boolean includeBranch = false;
 37     private boolean reportNewTags = true;
 38     private boolean reportNewBranches = true;
 39     private boolean reportNewBuilds = true;
 40     private MailingListUpdater.Mode mode = MailingListUpdater.Mode.ALL;
 41     private Map<String, String> headers = Map.of();
 42     private Pattern allowedAuthorDomains = Pattern.compile(".*");
 43     private boolean repoInSubject = false;
 44     private Pattern branchInSubject = Pattern.compile("a^"); // Does not match anything
 45 
 46     public MailingListUpdaterBuilder list(MailingList list) {
 47         this.list = list;
 48         return this;
 49     }
 50 
 51     public MailingListUpdaterBuilder recipient(EmailAddress recipient) {
 52         this.recipient = recipient;
 53         return this;
 54     }
 55 
 56     public MailingListUpdaterBuilder sender(EmailAddress sender) {
 57         this.sender = sender;
 58         return this;
 59     }
 60 
 61     public MailingListUpdaterBuilder author(EmailAddress author) {
 62         this.author = author;
 63         return this;
 64     }
 65 
 66     public MailingListUpdaterBuilder includeBranch(boolean includeBranch) {
 67         this.includeBranch = includeBranch;
 68         return this;
 69     }
 70 
 71     public MailingListUpdaterBuilder reportNewTags(boolean reportNewTags) {
 72         this.reportNewTags = reportNewTags;
 73         return this;
 74     }
 75 
 76     public MailingListUpdaterBuilder reportNewBranches(boolean reportNewBranches) {
 77         this.reportNewBranches = reportNewBranches;
 78         return this;
 79     }
 80 
 81     public MailingListUpdaterBuilder reportNewBuilds(boolean reportNewBuilds) {
 82         this.reportNewBuilds = reportNewBuilds;
 83         return this;
 84     }
 85 
 86     public MailingListUpdaterBuilder mode(MailingListUpdater.Mode mode) {
 87         this.mode = mode;
 88         return this;
 89     }
 90 
 91     public MailingListUpdaterBuilder headers(Map<String, String> headers) {
 92         this.headers = headers;
 93         return this;
 94     }
 95 
 96     public MailingListUpdaterBuilder allowedAuthorDomains(Pattern allowedAuthorDomains) {
 97         this.allowedAuthorDomains = allowedAuthorDomains;
 98         return this;
 99     }
100 
101     public MailingListUpdaterBuilder repoInSubject(boolean repoInSubject) {
102         this.repoInSubject = repoInSubject;
103         return this;
104     }
105 
106     public MailingListUpdaterBuilder branchInSubject(Pattern branchInSubject) {
107         this.branchInSubject = branchInSubject;
108         return this;
109     }
110 
111     public MailingListUpdater build() {
112         return new MailingListUpdater(list, recipient, sender, author, includeBranch, reportNewTags, reportNewBranches,
113                                       reportNewBuilds, mode, headers, allowedAuthorDomains, repoInSubject, branchInSubject);
114     }
115 }
