  65     private static String gitConfig(String key) {
  66         try {
  67             var pb = new ProcessBuilder("git", "config", key);
  68             pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
  69             pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  70             var p = pb.start();
  71 
  72             var output = new String(p.getInputStream().readAllBytes(), StandardCharsets.UTF_8);
  73             var res = p.waitFor();
  74             if (res != 0) {
  75                 return null;
  76             }
  77 
  78             return output == null ? null : output.replace("\n", "");
  79         } catch (InterruptedException e) {
  80             return null;
  81         } catch (IOException e) {
  82             return null;
  83         }
  84     }

  90     private static String getOption(String name, String subsection, Arguments arguments) {
  91         if (arguments.contains(name)) {
  92             return arguments.get(name).asString();
  93         }
  94 
  95         if (subsection != null && !subsection.isEmpty()) {
  96             var subsectionSpecific = gitConfig("pr." + subsection + "." + name);
  97             if (subsectionSpecific != null) {
  98                 return subsectionSpecific;
  99             }
 100         }
 101 
 102         return gitConfig("fork." + name);
 103     }

 109     private static boolean getSwitch(String name, String subsection, Arguments arguments) {
 110         if (arguments.contains(name)) {
 111             return true;
 112         }
 113 
 114         if (subsection != null && !subsection.isEmpty()) {
 115             var subsectionSpecific = gitConfig("pr." + subsection + "." + name);
 116             if (subsectionSpecific != null) {
 117                 return subsectionSpecific.toLowerCase().equals("true");
 118             }
 119         }
 120 
 121         var sectionSpecific = gitConfig("fork." + name);
 122         return sectionSpecific != null && sectionSpecific.toLowerCase().equals("true");
 123     }

 160     private static String statusForPullRequest(PullRequest pr) {
 161         var labels = pr.labels();
 162         if (pr.isDraft()) {
 163             return "DRAFT";
 164         } else if (labels.contains("integrated")) {
 165             return "INTEGRATED";
 166         } else if (labels.contains("ready")) {
 167             return "READY";
 168         } else if (labels.contains("rfr")) {
 169             return "RFR";
 170         } else if (labels.contains("outdated")) {
 171             return "OUTDATED";
 172         } else if (labels.contains("oca")) {
 173             return "OCA";
 174         } else {
 175             var checks = pr.checks(pr.headHash());
 176             var jcheck = Optional.ofNullable(checks.get("jcheck"));
 177             if (jcheck.isPresent()) {
 178                 var checkStatus = jcheck.get().status();
 179                 if (checkStatus == CheckStatus.IN_PROGRESS) {
 180                     return "CHECKING";
 181                 } else if (checkStatus == CheckStatus.SUCCESS) {
 182                     return "RFR";
 183                 } else if (checkStatus == CheckStatus.FAILURE) {
 184                     return "FAILURE";
 185                 }
 186             } else {
 187                 return "CHECKING";
 188             }
 189         }
 190 
 191         return "UNKNOWN";
 192     }

 194     private static String statusForCheck(Check check) {
 195         var checkStatus = check.status();
 196         if (checkStatus == CheckStatus.IN_PROGRESS) {
 197             return "RUNNING";
 198         } else if (checkStatus == CheckStatus.SUCCESS) {
 199             return "OK";
 200         } else if (checkStatus == CheckStatus.FAILURE) {
 201             return "FAILED";
 202         } else if (checkStatus == CheckStatus.CANCELLED) {
 203             return "CANCELLED";
 204         }
 205 
 206         return "UNKNOWN";
 207     }

 209     private static List<String> issuesFromPullRequest(PullRequest pr) {
 210         var issueTitleIndex = -1;
 211         var lines = pr.body().split("\n");
 212         for (var i = 0; i < lines.length; i++) {
 213             if (lines[i].startsWith("## Issue")) {
 214                 issueTitleIndex = i;
 215                 break;
 216             }
 217         }
 218 
 219         if (issueTitleIndex == -1) {
 220             return List.of();
 221         }
 222 
 223         var issues = new ArrayList<String>();
 224         for (var i = issueTitleIndex + 1; i < lines.length; i++) {
 225             var m = ISSUE_MARKDOWN_PATTERN.matcher(lines[i]);
 226             if (m.matches()) {
 227                 issues.add(m.group(1));
 228             } else {
 229                 break;
 230             }
 231         }
 232 
 233         return issues;
 234     }

 244     private static Optional<Issue> getIssue(Commit commit, String project) throws IOException {
 245         var message = CommitMessageParsers.v1.parse(commit.message());
 246         var issues = message.issues();
 247         if (issues.isEmpty()) {
 248             return getIssue(message.title(), project);
 249         } else if (issues.size() == 1) {
 250             var issue = issues.get(0);
 251             return getIssue(issue.id(), project);
 252         }
 253         return Optional.empty();
 254     }

 256     private static Optional<Issue> getIssue(Branch b, String project) throws IOException {
 257         return getIssue(b.name(), project);
 258     }

 260     private static Optional<Issue> getIssue(String s, String project) throws IOException {
 261         var m = ISSUE_ID_PATTERN.matcher(s);
 262         if (m.matches()) {
 263             var id = m.group(2);
 264             if (project == null) {
 265                 project = m.group(1);
 266             }
 267             var issueTracker = IssueTracker.from("jira", URI.create("https://bugs.openjdk.java.net"));
 268             return issueTracker.project(project).issue(id);
 269         }
 270 
 271         return Optional.empty();
 272     }

 274     private static void await(Process p, Integer... allowedExitCodes) throws IOException {
 275         var allowed = new HashSet<>(Arrays.asList(allowedExitCodes));
 276         allowed.add(0);
 277         try {
 278             var res = p.waitFor();
 279 
 280             if (!allowed.contains(res)) {
 281                 throw new IOException("Unexpected exit code " + res);
 282             }
 283         } catch (InterruptedException e) {
 284             throw new IOException(e);
 285         }
 286     }

 288     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 289         String editor = null;
 290         var lines = repo.config("core.editor");
 291         if (lines.size() == 1) {
 292             editor = lines.get(0);
 293         }
 294         if (editor == null) {
 295             editor = System.getenv("GIT_EDITOR");
 296         }
 297         if (editor == null) {
 298             editor = System.getenv("EDITOR");
 299         }
 300         if (editor == null) {
 301             editor = System.getenv("VISUAL");
 302         }
 303         if (editor == null) {
 304             editor = "vi";
 305         }
 306 
 307         var pb = new ProcessBuilder(editor, file.toString());
 308         pb.inheritIO();
 309         var p = pb.start();
 310         try {
 311             return p.waitFor() == 0;
 312         } catch (InterruptedException e) {
 313             throw new IOException(e);
 314         }
 315     }

 325     private static HostedRepository getHostedRepositoryFor(URI uri, ReadOnlyRepository repo, Forge host) throws IOException {
 326         HostedRepository targetRepo = null;
 327 
 328         try {
 329             var upstream = Remote.toWebURI(repo.pullPath("upstream"));
 330             targetRepo = host.repository(projectName(upstream)).orElse(null);
 331         } catch (IOException e) {
 332             // do nothing
 333         }
 334 
 335         if (targetRepo == null) {
 336             var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 337                     new IOException("Could not find repository at: " + uri.toString())
 338             );
 339             var parentRepo = remoteRepo.parent();
 340             targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
 341         }
 342 
 343         return targetRepo;
 344     }

 358     private static void show(String ref, Hash hash, Path dir) throws IOException {
 359         var pb = new ProcessBuilder("git", "diff", "--binary",
 360                                                    "--patch",
 361                                                    "--find-renames=50%",
 362                                                    "--find-copies=50%",
 363                                                    "--find-copies-harder",
 364                                                    "--abbrev",
 365                                                    ref + "..." + hash.hex());
 366         if (dir != null) {
 367             pb.directory(dir.toFile());
 368         }
 369         pb.inheritIO();
 370 
 371         // git will return 141 (128 + 13) when it receive SIGPIPE (signal 13) from
 372         // e.g. less when a user exits less when looking at a large diff. Therefore
 373         // must allow 141 as a valid exit code.
 374         await(pb.start(), 141);
 375     }

 389     private static List<String> hgTags() throws IOException, InterruptedException {
 390         var pb = new ProcessBuilder("hg", "tags", "--quiet");
 391         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 392         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 393         var p = pb.start();
 394         var bytes = p.getInputStream().readAllBytes();
 395         var exited = p.waitFor(1, TimeUnit.MINUTES);
 396         var exitValue = p.exitValue();
 397         if (!exited || exitValue != 0) {
 398             throw new IOException("'hg tags' exited with value: " + exitValue);
 399         }
 400 
 401         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
 402     }

 423     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
 424         var patch = Files.createTempFile(hash.hex(), ".patch");
 425         var pb = new ProcessBuilder("git", "diff", "--binary",
 426                                                    "--patch",
 427                                                    "--find-renames=50%",
 428                                                    "--find-copies=50%",
 429                                                    "--find-copies-harder",
 430                                                    "--abbrev",
 431                                                    ref + "..." + hash.hex());
 432         if (dir != null) {
 433             pb.directory(dir.toFile());
 434         }
 435         pb.redirectOutput(patch.toFile());
 436         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 437         await(pb.start());
 438         return patch;
 439     }

 457     public static void main(String[] args) throws IOException, InterruptedException {
 458         var flags = List.of(
 459             Option.shortcut("u")
 460                   .fullname("username")
 461                   .describe("NAME")
 462                   .helptext("Username on host")
 463                   .optional(),
 464             Option.shortcut("r")
 465                   .fullname("remote")
 466                   .describe("NAME")
 467                   .helptext("Name of remote, defaults to 'origin'")
 468                   .optional(),
 469             Option.shortcut("b")
 470                   .fullname("branch")
 471                   .describe("NAME")
 472                   .helptext("Name of target branch, defaults to 'master'")
 473                   .optional(),
 474             Option.shortcut("")
 475                   .fullname("authors")
 476                   .describe("LIST")
 477                   .helptext("Comma separated list of authors")
 478                   .optional(),
 479             Option.shortcut("")
 480                   .fullname("assignees")
 481                   .describe("LIST")
 482                   .helptext("Comma separated list of assignees")
 483                   .optional(),
 484             Option.shortcut("")
 485                   .fullname("labels")
 486                   .describe("LIST")
 487                   .helptext("Comma separated list of labels")
 488                   .optional(),
 489             Option.shortcut("")
 490                   .fullname("issues")
 491                   .describe("LIST")
 492                   .helptext("Comma separated list of issues")
 493                   .optional(),
 494             Option.shortcut("")
 495                   .fullname("columns")
 496                   .describe("id,title,author,assignees,labels")
 497                   .helptext("Comma separated list of columns to show")
 498                   .optional(),
 499             Switch.shortcut("")
 500                   .fullname("no-decoration")
 501                   .helptext("Hide any decorations when listing PRs")
 502                   .optional(),
 503             Switch.shortcut("")
 504                   .fullname("no-draft")
 505                   .helptext("Hide all pull requests in draft state")
 506                   .optional(),
 507             Switch.shortcut("")
 508                   .fullname("ignore-workspace")
 509                   .helptext("Ignore local changes in worktree and staging area when creating pull request")
 510                   .optional(),
 511             Switch.shortcut("")
 512                   .fullname("ignore-local-commits")
 513                   .helptext("Ignore local commits not pushed when creating pull request")
 514                   .optional(),
 515             Switch.shortcut("")
 516                   .fullname("publish")
 517                   .helptext("Publish the local branch before creating the pull request")
 518                   .optional(),
 519             Switch.shortcut("")
 520                   .fullname("jcheck")
 521                   .helptext("Run jcheck before creating the pull request")
 522                   .optional(),
 523             Switch.shortcut("")
 524                   .fullname("no-token")
 525                   .helptext("Do not use a personal access token (PAT). Only works for read-only operations.")
 526                   .optional(),
 527             Switch.shortcut("")
 528                   .fullname("no-checks")
 529                   .helptext("Do not show check status as part of the 'git pr status' output")
 530                   .optional(),
 531             Switch.shortcut("")
 532                   .fullname("mercurial")
 533                   .helptext("Force use of Mercurial (hg)")
 534                   .optional(),
 535             Switch.shortcut("")
 536                   .fullname("verbose")
 537                   .helptext("Turn on verbose output")
 538                   .optional(),
 539             Switch.shortcut("")
 540                   .fullname("debug")
 541                   .helptext("Turn on debugging output")
 542                   .optional(),
 543             Switch.shortcut("")
 544                   .fullname("version")
 545                   .helptext("Print the version of this tool")
 546                   .optional());
 547 
 548         var actions = List.of("list", "fetch", "show", "checkout", "apply", "integrate",
 549                               "approve", "create", "close", "update", "test", "status");
 550         var inputs = List.of(
 551             Input.position(0)
 552                  .describe(String.join("|", actions))
 553                  .singular()
 554                  .optional(),
 555             Input.position(1)
 556                  .describe("ID")
 557                  .singular()
 558                  .optional()
 559         );
 560 
 561         var parser = new ArgumentParser("git-pr", flags, inputs);
 562         var arguments = parser.parse(args);
 563 
 564         if (arguments.contains("version")) {
 565             System.out.println("git-pr version: " + Version.fromManifest().orElse("unknown"));
 566             System.exit(0);
 567         }
 568 
 569         if (arguments.contains("verbose") || arguments.contains("debug")) {
 570             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
 571             Logging.setup(level);
 572         }
 573 
 574         HttpProxy.setup();
 575 
 576         var isMercurial = getSwitch("mercurial", arguments);
 577         var cwd = Path.of("").toAbsolutePath();
 578         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
 579         var remote = getOption("remote", arguments);
 580         if (remote == null) {
 581             remote = isMercurial ? "default" : "origin";
 582         }
 583         var remotePullPath = repo.pullPath(remote);
 584         var username = getOption("username", arguments);
 585         var token = isMercurial ? System.getenv("HG_TOKEN") : System.getenv("GIT_TOKEN");
 586         var uri = Remote.toWebURI(remotePullPath);
 587         var shouldUseToken = !getSwitch("no-token", arguments);
 588         var credentials = !shouldUseToken ?
 589             null :
 590             GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
 591         var forgeURI = URI.create(uri.getScheme() + "://" + uri.getHost());
 592         var forge = credentials == null ?
 593             Forge.from(forgeURI) :
 594             Forge.from(forgeURI, new Credential(credentials.username(), credentials.password()));
 595         if (forge.isEmpty()) {
 596             if (!shouldUseToken) {
 597                 if (arguments.contains("verbose")) {
 598                     System.err.println("");
 599                 }
 600                 System.err.println("warning: using git-pr with --no-token may result in rate limiting from " + forgeURI);
 601                 if (!arguments.contains("verbose")) {
 602                     System.err.println("         Re-run git-pr with --verbose to see if you are being rate limited");
 603                     System.err.println("");
 604                 }
 605             }
 606             exit("error: failed to connect to host: " + forgeURI);
 607         }
 608         var host = forge.get();
 609 
 610         var action = arguments.at(0).isPresent() ? arguments.at(0).asString() : null;
 611         if (action == null) {
 612             var lines = repo.config("pr.default");
 613             if (lines.size() == 1) {
 614                 action = lines.get(0);
 615             }
 616         }
 617 
 618         if (action == null) {
 619             System.err.println("error: you must supply a valid action:");
 620             for (var a : actions) {
 621                 System.err.println("       - " + a);
 622             }
 623             System.err.println("You can also configure a default action by running 'git configure --global pr.default <action>'");
 624             System.exit(1);
 625         }
 626 
 627         if (!shouldUseToken &&
 628             !List.of("list", "fetch", "show", "checkout", "apply").contains(action)) {
 629             System.err.println("error: --no-token can only be used with read-only operations");
 630             System.exit(1);
 631         }
 632 
 633         if (action.equals("create")) {
 634             if (isMercurial) {
 635                 var currentBookmark = repo.currentBookmark();
 636                 if (!currentBookmark.isPresent()) {
 637                     System.err.println("error: no bookmark is active, you must be on an active bookmark");
 638                     System.err.println("");
 639                     System.err.println("To create a bookmark and activate it, run:");
 640                     System.err.println("");
 641                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 642                     System.err.println("");
 643                     System.exit(1);
 644                 }
 645 
 646                 var bookmark = currentBookmark.get();
 647                 if (bookmark.equals(new Bookmark("master"))) {
 648                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
 649                     System.err.println("To create a bookmark and activate it, run:");
 650                     System.err.println("");
 651                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 652                     System.err.println("");
 653                     System.exit(1);
 654                 }
 655 
 656                 var tags = hgTags();
 657                 var upstreams = tags.stream()
 658                                     .filter(t -> t.endsWith(bookmark.name()))
 659                                     .collect(Collectors.toList());
 660                 if (upstreams.isEmpty()) {
 661                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");
