 23 package org.openjdk.skara.vcs;
 24 
 25 import java.io.IOException;
 26 import java.nio.file.Path;
 27 import java.nio.charset.StandardCharsets;
 28 import java.util.stream.Collectors;
 29 import java.util.*;
 30 
 31 public interface ReadOnlyRepository {
 32     Hash head() throws IOException;
 33     Optional<Branch> currentBranch() throws IOException;
 34     Optional<Bookmark> currentBookmark() throws IOException;
 35     Branch defaultBranch() throws IOException;
 36     List<Branch> branches() throws IOException;
 37     List<Branch> branches(String remote) throws IOException;
 38     Optional<Tag> defaultTag() throws IOException;
 39     List<Tag> tags() throws IOException;
 40     Commits commits() throws IOException;
 41     Commits commits(int n) throws IOException;
 42     Commits commits(boolean reverse) throws IOException;
 43     Commits commits(int n, boolean reverse) throws IOException;
 44     Commits commits(String range) throws IOException;
 45     Commits commits(String range, boolean reverse) throws IOException;
 46     Commits commits(String range, int n) throws IOException;
 47     Commits commits(String range, int n, boolean reverse) throws IOException;
 48     Optional<Commit> lookup(Hash h) throws IOException;
 49     Optional<Commit> lookup(Branch b) throws IOException;
 50     Optional<Commit> lookup(Tag t) throws IOException;
 51     List<CommitMetadata> commitMetadata() throws IOException;
 52     List<CommitMetadata> commitMetadata(String range) throws IOException;
 53     Path root() throws IOException;
 54     boolean exists() throws IOException;
 55     boolean isHealthy() throws IOException;
 56     boolean isEmpty() throws IOException;
 57     boolean isClean() throws IOException;
 58     Hash mergeBase(Hash first, Hash second) throws IOException;
 59     boolean isAncestor(Hash ancestor, Hash descendant) throws IOException;
 60     Optional<Hash> resolve(String ref) throws IOException;
 61     default Optional<Hash> resolve(Tag t) throws IOException {
 62         return resolve(t.name());
 63     }
 64     default Optional<Hash> resolve(Branch b) throws IOException {
 65         return resolve(b.name());
 66     }
 67     boolean contains(Branch b, Hash h) throws IOException;
 68     Optional<String> username() throws IOException;
 69     Optional<byte[]> show(Path p, Hash h) throws IOException;
 70     default Optional<List<String>> lines(Path p, Hash h) throws IOException {
 71         return show(p, h).map(bytes -> new String(bytes, StandardCharsets.UTF_8).lines().collect(Collectors.toList()));
 72     }
 73 
 74     List<FileEntry> files(Hash h, List<Path> paths) throws IOException;
 75     default List<FileEntry> files(Hash h, Path... paths) throws IOException {
 76         return files(h, Arrays.asList(paths));
 77     }
 78 
 79     void dump(FileEntry entry, Path to) throws IOException;
 80     List<StatusEntry> status(Hash from, Hash to) throws IOException;
 81     Diff diff(Hash base, Hash head) throws IOException;
 82     Diff diff(Hash base, Hash head, List<Path> files) throws IOException;
 83     Diff diff(Hash head) throws IOException;
 84     Diff diff(Hash head, List<Path> files) throws IOException;
 85     List<String> config(String key) throws IOException;
 86     Repository copyTo(Path destination) throws IOException;
 87     String pullPath(String remote) throws IOException;
 88     String pushPath(String remote) throws IOException;
 89     boolean isValidRevisionRange(String expression) throws IOException;
 90     Optional<String> upstreamFor(Branch branch) throws IOException;
 91     List<Reference> remoteBranches(String remote) throws IOException;
 92     List<String> remotes() throws IOException;
 93     List<Submodule> submodules() throws IOException;
 94 
 95     static Optional<ReadOnlyRepository> get(Path p) throws IOException {
 96         return Repository.get(p).map(r -> r);
 97     }
 98 
 99     static boolean exists(Path p) throws IOException {
100         return Repository.exists(p);
101     }
102 
103     Optional<Tag.Annotated> annotate(Tag tag) throws IOException;
104 }
