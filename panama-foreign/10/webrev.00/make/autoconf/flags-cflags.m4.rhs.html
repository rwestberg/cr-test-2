 35 AC_DEFUN([FLAGS_SETUP_SHARED_LIBS],
 36 [
 37   if test "x$TOOLCHAIN_TYPE" = xgcc; then
 38     C_FLAG_REORDER=''
 39 
 40     # Default works for linux, might work on other platforms as well.
 41     SHARED_LIBRARY_FLAGS='-shared'
 42     SET_EXECUTABLE_ORIGIN='-Wl,-rpath,\$$ORIGIN[$]1'
 43     SET_SHARED_LIBRARY_ORIGIN="-Wl,-z,origin $SET_EXECUTABLE_ORIGIN"
 44     SET_SHARED_LIBRARY_NAME='-Wl,-soname=[$]1'
 45     SET_SHARED_LIBRARY_MAPFILE='-Wl,-version-script=[$]1'
 46 
 47   elif test "x$TOOLCHAIN_TYPE" = xclang; then
 48     C_FLAG_REORDER=''
 49 
 50     if test "x$OPENJDK_TARGET_OS" = xmacosx; then
 51       # Linking is different on MacOSX
 52       SHARED_LIBRARY_FLAGS="-dynamiclib -compatibility_version 1.0.0 -current_version 1.0.0"
 53       SET_EXECUTABLE_ORIGIN='-Wl,-rpath,@loader_path$(or [$]1,/.)'
 54       SET_SHARED_LIBRARY_ORIGIN="$SET_EXECUTABLE_ORIGIN"
 55       SET_SHARED_LIBRARY_NAME='-Wl,-install_name,@rpath/[$]1'
 56       SET_SHARED_LIBRARY_MAPFILE='-Wl,-exported_symbols_list,[$]1'
 57 
 58     else
 59       # Default works for linux, might work on other platforms as well.
 60       SHARED_LIBRARY_FLAGS='-shared'
 61       SET_EXECUTABLE_ORIGIN='-Wl,-rpath,\$$ORIGIN[$]1'
 62       SET_SHARED_LIBRARY_NAME='-Wl,-soname=[$]1'
 63       SET_SHARED_LIBRARY_MAPFILE='-Wl,-version-script=[$]1'
 64 
 65       # arm specific settings
 66       if test "x$OPENJDK_TARGET_CPU" = "xarm"; then
 67         # '-Wl,-z,origin' isn't used on arm.
 68         SET_SHARED_LIBRARY_ORIGIN='-Wl,-rpath,\$$$$ORIGIN[$]1'
 69       else
 70         SET_SHARED_LIBRARY_ORIGIN="-Wl,-z,origin $SET_EXECUTABLE_ORIGIN"
 71       fi
 72     fi
 73 
 74   elif test "x$TOOLCHAIN_TYPE" = xsolstudio; then
 75     C_FLAG_REORDER='-xF'
 76     SHARED_LIBRARY_FLAGS="-G"
 77     SET_EXECUTABLE_ORIGIN='-R\$$ORIGIN[$]1'
 78     SET_SHARED_LIBRARY_ORIGIN="$SET_EXECUTABLE_ORIGIN"
 79     SET_SHARED_LIBRARY_NAME='-h [$]1'
 80     SET_SHARED_LIBRARY_MAPFILE='-M[$]1'
 81 
 82   elif test "x$TOOLCHAIN_TYPE" = xxlc; then
 83     C_FLAG_REORDER=''
 84     SHARED_LIBRARY_FLAGS="-qmkshrobj -bM:SRE -bnoentry"
 85     SET_EXECUTABLE_ORIGIN=""
 86     SET_SHARED_LIBRARY_ORIGIN=''
 87     SET_SHARED_LIBRARY_NAME=''
 88     SET_SHARED_LIBRARY_MAPFILE=''
 89 
 90   elif test "x$TOOLCHAIN_TYPE" = xmicrosoft; then
 91     C_FLAG_REORDER=''
 92     SHARED_LIBRARY_FLAGS="-dll"
 93     SET_EXECUTABLE_ORIGIN=''
 94     SET_SHARED_LIBRARY_ORIGIN=''
 95     SET_SHARED_LIBRARY_NAME=''
 96     SET_SHARED_LIBRARY_MAPFILE='-def:[$]1'
 97   fi
 98 
 99   AC_SUBST(C_FLAG_REORDER)
100   AC_SUBST(SET_EXECUTABLE_ORIGIN)
101   AC_SUBST(SET_SHARED_LIBRARY_ORIGIN)
102   AC_SUBST(SET_SHARED_LIBRARY_NAME)
103   AC_SUBST(SET_SHARED_LIBRARY_MAPFILE)
104   AC_SUBST(SHARED_LIBRARY_FLAGS)
105 ])

107 AC_DEFUN([FLAGS_SETUP_DEBUG_SYMBOLS],
108 [
109   # By default don't set any specific assembler debug
110   # info flags for toolchains unless we know they work.
111   # See JDK-8207057.
112   ASFLAGS_DEBUG_SYMBOLS=""
113   # Debug symbols
114   if test "x$TOOLCHAIN_TYPE" = xgcc; then
115     CFLAGS_DEBUG_SYMBOLS="-g"
116     ASFLAGS_DEBUG_SYMBOLS="-g"
117   elif test "x$TOOLCHAIN_TYPE" = xclang; then
118     CFLAGS_DEBUG_SYMBOLS="-g"
119     ASFLAGS_DEBUG_SYMBOLS="-g"
120   elif test "x$TOOLCHAIN_TYPE" = xsolstudio; then
121     # -g0 enables debug symbols without disabling inlining.
122     CFLAGS_DEBUG_SYMBOLS="-g0 -xs"
123   elif test "x$TOOLCHAIN_TYPE" = xxlc; then
124     CFLAGS_DEBUG_SYMBOLS="-g1"
125   elif test "x$TOOLCHAIN_TYPE" = xmicrosoft; then
126     CFLAGS_DEBUG_SYMBOLS="-Z7"
127   fi
128 
129   AC_SUBST(CFLAGS_DEBUG_SYMBOLS)
130   AC_SUBST(ASFLAGS_DEBUG_SYMBOLS)
131 ])

133 AC_DEFUN([FLAGS_SETUP_WARNINGS],
134 [
135   AC_ARG_ENABLE([warnings-as-errors], [AS_HELP_STRING([--disable-warnings-as-errors],
136       [do not consider native warnings to be an error @<:@enabled@:>@])])
137 
138   # Set default value.
139   if test "x$TOOLCHAIN_TYPE" = xxlc; then
140     WARNINGS_AS_ERRORS=false
141   else
142     WARNINGS_AS_ERRORS=true
143   fi
144 
145   AC_MSG_CHECKING([if native warnings are errors])
146   if test "x$enable_warnings_as_errors" = "xyes"; then
147     AC_MSG_RESULT([yes (explicitly set)])
148     WARNINGS_AS_ERRORS=true
149   elif test "x$enable_warnings_as_errors" = "xno"; then
150     AC_MSG_RESULT([no (explicitly set)])
151     WARNINGS_AS_ERRORS=false
152   elif test "x$enable_warnings_as_errors" = "x"; then
153     AC_MSG_RESULT([${WARNINGS_AS_ERRORS} (default)])
154   else
155     AC_MSG_ERROR([--enable-warnings-as-errors accepts no argument])
156   fi
157 
158   AC_SUBST(WARNINGS_AS_ERRORS)
159 
160   case "${TOOLCHAIN_TYPE}" in
161     microsoft)
162       DISABLE_WARNING_PREFIX="-wd"
163       CFLAGS_WARNINGS_ARE_ERRORS="-WX"
164 
165       WARNINGS_ENABLE_ALL="-W3"
166       DISABLED_WARNINGS="4800"
167       ;;
168 
169     solstudio)
170       DISABLE_WARNING_PREFIX="-erroff="
171       CFLAGS_WARNINGS_ARE_ERRORS="-errwarn=%all"
172 
173       WARNINGS_ENABLE_ALL_CFLAGS="-v -fd -xtransition"
174       WARNINGS_ENABLE_ALL_CXXFLAGS="+w +w2"
175 
176       DISABLED_WARNINGS_C="E_OLD_STYLE_FUNC_DECL E_OLD_STYLE_FUNC_DEF E_SEMANTICS_OF_OP_CHG_IN_ANSI_C E_NO_REPLACEMENT_IN_STRING E_DECLARATION_IN_CODE"
177       DISABLED_WARNINGS_CXX="inllargeuse inllargeint notused wemptydecl notemsource"
178       ;;
179 
180     gcc)
181       DISABLE_WARNING_PREFIX="-Wno-"
182       CFLAGS_WARNINGS_ARE_ERRORS="-Werror"
183 
184       # Additional warnings that are not activated by -Wall and -Wextra
185       WARNINGS_ENABLE_ADDITIONAL="-Wpointer-arith -Wsign-compare \
186           -Wunused-function -Wundef -Wunused-value -Wreturn-type \
187           -Wtrampolines"
188       WARNINGS_ENABLE_ADDITIONAL_CXX="-Woverloaded-virtual -Wreorder"
189       WARNINGS_ENABLE_ALL_CFLAGS="-Wall -Wextra -Wformat=2 $WARNINGS_ENABLE_ADDITIONAL"
190       WARNINGS_ENABLE_ALL_CXXFLAGS="$WARNINGS_ENABLE_ALL_CFLAGS $WARNINGS_ENABLE_ADDITIONAL_CXX"
191 
192       DISABLED_WARNINGS="unused-parameter unused"
193       BUILD_CC_DISABLE_WARNING_PREFIX="-Wno-"
194       ;;
195 
196     clang)
197       DISABLE_WARNING_PREFIX="-Wno-"
198       CFLAGS_WARNINGS_ARE_ERRORS="-Werror"
199 
200       # Additional warnings that are not activated by -Wall and -Wextra
201       WARNINGS_ENABLE_ADDITIONAL="-Wpointer-arith -Wsign-compare -Wreorder \
202           -Wunused-function -Wundef -Wunused-value -Woverloaded-virtual"
203       WARNINGS_ENABLE_ALL="-Wall -Wextra -Wformat=2 $WARNINGS_ENABLE_ADDITIONAL"
204 
205       DISABLED_WARNINGS="unknown-warning-option unused-parameter unused"
206 
207       if test "x$OPENJDK_TARGET_OS" = xmacosx; then
208         # missing-method-return-type triggers in JavaNativeFoundation framework
209         DISABLED_WARNINGS="$DISABLED_WARNINGS missing-method-return-type"
210       fi
211 
212       ;;
213 
214     xlc)
215       DISABLE_WARNING_PREFIX="-Wno-"
216       CFLAGS_WARNINGS_ARE_ERRORS="-qhalt=w"
217 
218       # Possibly a better subset than "all" is "lan:trx:ret:zea:cmp:ret"
219       WARNINGS_ENABLE_ALL="-qinfo=all -qformat=all"
220       DISABLED_WARNINGS=""
221       ;;
222   esac
223   AC_SUBST(DISABLE_WARNING_PREFIX)
224   AC_SUBST(BUILD_CC_DISABLE_WARNING_PREFIX)
225   AC_SUBST(CFLAGS_WARNINGS_ARE_ERRORS)
226   AC_SUBST(DISABLED_WARNINGS)
227   AC_SUBST(DISABLED_WARNINGS_C)
228   AC_SUBST(DISABLED_WARNINGS_CXX)
229 ])

231 AC_DEFUN([FLAGS_SETUP_QUALITY_CHECKS],
232 [
233   # bounds, memory and behavior checking options
234   if test "x$TOOLCHAIN_TYPE" = xgcc; then
235     case $DEBUG_LEVEL in
236     release )
237       # no adjustment
238       ;;
239     fastdebug )
240       # no adjustment
241       ;;
242     slowdebug )
243       # FIXME: By adding this to C(XX)FLAGS_DEBUG_OPTIONS/JVM_CFLAGS_SYMBOLS it
244       # get's added conditionally on whether we produce debug symbols or not.
245       # This is most likely not really correct.
246 
247       # Add runtime stack smashing and undefined behavior checks.
248       CFLAGS_DEBUG_OPTIONS="-fstack-protector-all --param ssp-buffer-size=1"
249       CXXFLAGS_DEBUG_OPTIONS="-fstack-protector-all --param ssp-buffer-size=1"
250 
251       JVM_CFLAGS_SYMBOLS="$JVM_CFLAGS_SYMBOLS -fstack-protector-all --param ssp-buffer-size=1"
252       ;;
253     esac
254   fi
255 ])

257 AC_DEFUN([FLAGS_SETUP_OPTIMIZATION],
258 [
259   if test "x$TOOLCHAIN_TYPE" = xsolstudio; then
260     CC_HIGHEST="-fns -fsimple -fsingle -xbuiltin=%all -xdepend -xrestrict -xlibmil"
261 
262     C_O_FLAG_HIGHEST_JVM="-xO4"
263     C_O_FLAG_DEBUG_JVM=""
264     C_O_FLAG_SIZE=""
265     C_O_FLAG_DEBUG=""
266     C_O_FLAG_NONE=""
267     if test "x$OPENJDK_TARGET_CPU_ARCH" = "xx86"; then
268       C_O_FLAG_HIGHEST="-xO4 -Wu,-O4~yz $CC_HIGHEST"
269       C_O_FLAG_HI="-xO4 -Wu,-O4~yz"
270       C_O_FLAG_NORM="-xO2 -Wu,-O2~yz"
271     elif test "x$OPENJDK_TARGET_CPU_ARCH" = "xsparc"; then
272       C_O_FLAG_HIGHEST="-xO4 -Wc,-Qrm-s -Wc,-Qiselect-T0 \
273           -xprefetch=auto,explicit $CC_HIGHEST"
274       C_O_FLAG_HI="-xO4 -Wc,-Qrm-s -Wc,-Qiselect-T0"
275       C_O_FLAG_NORM="-xO2 -Wc,-Qrm-s -Wc,-Qiselect-T0"
276     fi
277   elif test "x$TOOLCHAIN_TYPE" = xgcc; then
278     C_O_FLAG_HIGHEST_JVM="-O3"
279     C_O_FLAG_HIGHEST="-O3"
280     C_O_FLAG_HI="-O3"
281     C_O_FLAG_NORM="-O2"
282     C_O_FLAG_SIZE="-Os"
283     C_O_FLAG_DEBUG="-O0"
284     C_O_FLAG_DEBUG_JVM="-O0"
285     C_O_FLAG_NONE="-O0"
286     # -D_FORTIFY_SOURCE=2 hardening option needs optimization (at least -O1) enabled
287     # set for lower O-levels -U_FORTIFY_SOURCE to overwrite previous settings
288     if test "x$OPENJDK_TARGET_OS" = xlinux -a "x$DEBUG_LEVEL" = "xfastdebug"; then
289       ENABLE_FORTIFY_CFLAGS="-D_FORTIFY_SOURCE=2"
290       DISABLE_FORTIFY_CFLAGS="-U_FORTIFY_SOURCE"
291       C_O_FLAG_HIGHEST_JVM="${C_O_FLAG_HIGHEST_JVM} ${ENABLE_FORTIFY_CFLAGS}"
292       C_O_FLAG_HIGHEST="${C_O_FLAG_HIGHEST} ${ENABLE_FORTIFY_CFLAGS}"
293       C_O_FLAG_HI="${C_O_FLAG_HI} ${ENABLE_FORTIFY_CFLAGS}"
294       C_O_FLAG_NORM="${C_O_FLAG_NORM} ${ENABLE_FORTIFY_CFLAGS}"
295       C_O_FLAG_SIZE="${C_O_FLAG_SIZE} ${DISABLE_FORTIFY_CFLAGS}"
296       C_O_FLAG_DEBUG="${C_O_FLAG_DEBUG} ${DISABLE_FORTIFY_CFLAGS}"
297       C_O_FLAG_DEBUG_JVM="${C_O_FLAG_DEBUG_JVM} ${DISABLE_FORTIFY_CFLAGS}"
298       C_O_FLAG_NONE="${C_O_FLAG_NONE} ${DISABLE_FORTIFY_CFLAGS}"
299     fi
300   elif test "x$TOOLCHAIN_TYPE" = xclang; then
301     if test "x$OPENJDK_TARGET_OS" = xmacosx; then
302       # On MacOSX we optimize for size, something
303       # we should do for all platforms?
304       C_O_FLAG_HIGHEST_JVM="-Os"
305       C_O_FLAG_HIGHEST="-Os"
306       C_O_FLAG_HI="-Os"
307       C_O_FLAG_NORM="-Os"
308       C_O_FLAG_DEBUG_JVM=""
309     else
310       C_O_FLAG_HIGHEST_JVM="-O3"
311       C_O_FLAG_HIGHEST="-O3"
312       C_O_FLAG_HI="-O3"
313       C_O_FLAG_NORM="-O2"
314       C_O_FLAG_DEBUG_JVM="-O0"
315     fi
316     C_O_FLAG_SIZE="-Os"
317     C_O_FLAG_DEBUG="-O0"
318     C_O_FLAG_NONE="-O0"
319   elif test "x$TOOLCHAIN_TYPE" = xxlc; then
320     C_O_FLAG_HIGHEST_JVM="-O3 -qhot=level=1 -qinline -qinlglue"
321     C_O_FLAG_HIGHEST="-O3 -qhot=level=1 -qinline -qinlglue"
322     C_O_FLAG_HI="-O3 -qinline -qinlglue"
323     C_O_FLAG_NORM="-O2"
324     C_O_FLAG_DEBUG="-qnoopt"
325     # FIXME: Value below not verified.
326     C_O_FLAG_DEBUG_JVM=""
327     C_O_FLAG_NONE="-qnoopt"
328   elif test "x$TOOLCHAIN_TYPE" = xmicrosoft; then
329     C_O_FLAG_HIGHEST_JVM="-O2 -Oy-"
330     C_O_FLAG_HIGHEST="-O2"
331     C_O_FLAG_HI="-O1"
332     C_O_FLAG_NORM="-O1"
333     C_O_FLAG_DEBUG="-Od"
334     C_O_FLAG_DEBUG_JVM=""
335     C_O_FLAG_NONE="-Od"
336     C_O_FLAG_SIZE="-Os"
337   fi
338 
339   # Now copy to C++ flags
340   CXX_O_FLAG_HIGHEST_JVM="$C_O_FLAG_HIGHEST_JVM"
341   CXX_O_FLAG_HIGHEST="$C_O_FLAG_HIGHEST"
342   CXX_O_FLAG_HI="$C_O_FLAG_HI"
343   CXX_O_FLAG_NORM="$C_O_FLAG_NORM"
344   CXX_O_FLAG_DEBUG="$C_O_FLAG_DEBUG"
345   CXX_O_FLAG_DEBUG_JVM="$C_O_FLAG_DEBUG_JVM"
346   CXX_O_FLAG_NONE="$C_O_FLAG_NONE"
347   CXX_O_FLAG_SIZE="$C_O_FLAG_SIZE"
348 
349   if test "x$TOOLCHAIN_TYPE" = xsolstudio; then
350     # In solstudio, also add this to C (but not C++) flags...
351     C_O_FLAG_HIGHEST="$C_O_FLAG_HIGHEST -xalias_level=basic"
352   fi
353 
354   # Adjust optimization flags according to debug level.
355   case $DEBUG_LEVEL in
356     release )
357       # no adjustment
358       ;;
359     fastdebug )
360       # Not quite so much optimization
361       C_O_FLAG_HI="$C_O_FLAG_NORM"
362       CXX_O_FLAG_HI="$CXX_O_FLAG_NORM"
363       ;;
364     slowdebug )
365       # Disable optimization
366       C_O_FLAG_HIGHEST_JVM="$C_O_FLAG_DEBUG_JVM"
367       C_O_FLAG_HIGHEST="$C_O_FLAG_DEBUG"
368       C_O_FLAG_HI="$C_O_FLAG_DEBUG"
369       C_O_FLAG_NORM="$C_O_FLAG_DEBUG"
370       C_O_FLAG_SIZE="$C_O_FLAG_DEBUG"
371       CXX_O_FLAG_HIGHEST_JVM="$CXX_O_FLAG_DEBUG_JVM"
372       CXX_O_FLAG_HIGHEST="$CXX_O_FLAG_DEBUG"
373       CXX_O_FLAG_HI="$CXX_O_FLAG_DEBUG"
374       CXX_O_FLAG_NORM="$CXX_O_FLAG_DEBUG"
375       CXX_O_FLAG_SIZE="$CXX_O_FLAG_DEBUG"
376       ;;
377   esac
378 
379   AC_SUBST(C_O_FLAG_HIGHEST_JVM)
380   AC_SUBST(C_O_FLAG_HIGHEST)
381   AC_SUBST(C_O_FLAG_HI)
382   AC_SUBST(C_O_FLAG_NORM)
383   AC_SUBST(C_O_FLAG_NONE)
384   AC_SUBST(C_O_FLAG_SIZE)
385   AC_SUBST(CXX_O_FLAG_HIGHEST_JVM)
386   AC_SUBST(CXX_O_FLAG_HIGHEST)
387   AC_SUBST(CXX_O_FLAG_HI)
388   AC_SUBST(CXX_O_FLAG_NORM)
389   AC_SUBST(CXX_O_FLAG_NONE)
390   AC_SUBST(CXX_O_FLAG_SIZE)
391 ])

393 AC_DEFUN([FLAGS_SETUP_CFLAGS],
394 [
395   ### CFLAGS
396 
397   FLAGS_SETUP_CFLAGS_HELPER
398 
399   FLAGS_OS=$OPENJDK_TARGET_OS
400   FLAGS_OS_TYPE=$OPENJDK_TARGET_OS_TYPE
401   FLAGS_CPU=$OPENJDK_TARGET_CPU
402   FLAGS_CPU_ARCH=$OPENJDK_TARGET_CPU_ARCH
403   FLAGS_CPU_BITS=$OPENJDK_TARGET_CPU_BITS
404   FLAGS_CPU_ENDIAN=$OPENJDK_TARGET_CPU_ENDIAN
405   FLAGS_CPU_LEGACY=$OPENJDK_TARGET_CPU_LEGACY
406   FLAGS_CPU_LEGACY_LIB=$OPENJDK_TARGET_CPU_LEGACY_LIB
407 
408   FLAGS_SETUP_CFLAGS_CPU_DEP([TARGET])
409 
410   # Repeat the check for the BUILD_CC and BUILD_CXX. Need to also reset CFLAGS
411   # since any target specific flags will likely not work with the build compiler.
412   CC_OLD="$CC"
413   CXX_OLD="$CXX"
414   CFLAGS_OLD="$CFLAGS"
415   CXXFLAGS_OLD="$CXXFLAGS"
416   CC="$BUILD_CC"
417   CXX="$BUILD_CXX"
418   CFLAGS=""
419   CXXFLAGS=""
420 
421   FLAGS_OS=$OPENJDK_BUILD_OS
422   FLAGS_OS_TYPE=$OPENJDK_BUILD_OS_TYPE
423   FLAGS_CPU=$OPENJDK_BUILD_CPU
424   FLAGS_CPU_ARCH=$OPENJDK_BUILD_CPU_ARCH
425   FLAGS_CPU_BITS=$OPENJDK_BUILD_CPU_BITS
426   FLAGS_CPU_ENDIAN=$OPENJDK_BUILD_CPU_ENDIAN
427   FLAGS_CPU_LEGACY=$OPENJDK_BUILD_CPU_LEGACY
428   FLAGS_CPU_LEGACY_LIB=$OPENJDK_BUILD_CPU_LEGACY_LIB
429 
430   FLAGS_SETUP_CFLAGS_CPU_DEP([BUILD], [OPENJDK_BUILD_], [BUILD_])
431 
432   CC="$CC_OLD"
433   CXX="$CXX_OLD"
434   CFLAGS="$CFLAGS_OLD"
435   CXXFLAGS="$CXXFLAGS_OLD"
436 ])

439 # platform independent
440 AC_DEFUN([FLAGS_SETUP_CFLAGS_HELPER],
441 [
442   #### OS DEFINES, these should be independent on toolchain
443   if test "x$OPENJDK_TARGET_OS" = xlinux; then
444     CFLAGS_OS_DEF_JVM="-DLINUX"
445     CFLAGS_OS_DEF_JDK="-D_GNU_SOURCE -D_REENTRANT -D_LARGEFILE64_SOURCE"
446   elif test "x$OPENJDK_TARGET_OS" = xsolaris; then
447     CFLAGS_OS_DEF_JVM="-DSOLARIS"
448     CFLAGS_OS_DEF_JDK="-D__solaris__"
449   elif test "x$OPENJDK_TARGET_OS" = xmacosx; then
450     CFLAGS_OS_DEF_JVM="-D_ALLBSD_SOURCE -D_DARWIN_C_SOURCE -D_XOPEN_SOURCE"
451     CFLAGS_OS_DEF_JDK="-D_ALLBSD_SOURCE -D_DARWIN_UNLIMITED_SELECT"
452   elif test "x$OPENJDK_TARGET_OS" = xaix; then
453     CFLAGS_OS_DEF_JVM="-DAIX"
454   elif test "x$OPENJDK_TARGET_OS" = xbsd; then
455     CFLAGS_OS_DEF_JDK="-D_ALLBSD_SOURCE"
456   elif test "x$OPENJDK_TARGET_OS" = xwindows; then
457     CFLAGS_OS_DEF_JVM="-D_WINDOWS -DWIN32 -D_JNI_IMPLEMENTATION_"
458   fi
459 
460   CFLAGS_OS_DEF_JDK="$CFLAGS_OS_DEF_JDK -D$OPENJDK_TARGET_OS_UPPERCASE"
461 
462   #### GLOBAL DEFINES
463   # Set some common defines. These works for all compilers, but assume
464   # -D is universally accepted.
465 
466   # Always enable optional macros for VM.
467   ALWAYS_CFLAGS_JVM="-D__STDC_FORMAT_MACROS -D__STDC_LIMIT_MACROS -D__STDC_CONSTANT_MACROS"
468 
469   # Setup some hard coded includes
470   ALWAYS_CFLAGS_JDK=" \
471       -I\$(SUPPORT_OUTPUTDIR)/modules_include/java.base \
472       -I\$(SUPPORT_OUTPUTDIR)/modules_include/java.base/\$(OPENJDK_TARGET_OS_INCLUDE_SUBDIR) \
473       -I${TOPDIR}/src/java.base/share/native/libjava \
474       -I${TOPDIR}/src/java.base/$OPENJDK_TARGET_OS_TYPE/native/libjava \
475       -I${TOPDIR}/src/hotspot/share/include \
476       -I${TOPDIR}/src/hotspot/os/${HOTSPOT_TARGET_OS_TYPE}/include"
477 
478   ###############################################################################
479 
480   # Adjust flags according to debug level.
481   # Setup debug/release defines
482   if test "x$DEBUG_LEVEL" = xrelease; then
483     DEBUG_CFLAGS_JDK="-DNDEBUG"
484     if test "x$OPENJDK_TARGET_OS" = xsolaris; then
485       DEBUG_CFLAGS_JDK="$DEBUG_CFLAGS_JDK -DTRIMMED"
486     fi
487   else
488     DEBUG_CFLAGS_JDK="-DDEBUG"
489 
490     if test "x$TOOLCHAIN_TYPE" = xxlc; then
491       # We need '-qminimaltoc' or '-qpic=large -bbigtoc' if the TOC overflows.
492       # Hotspot now overflows its 64K TOC (currently only for debug),
493       # so for debug we build with '-qpic=large -bbigtoc'.
494       DEBUG_CFLAGS_JVM="-qpic=large"
495     fi
496   fi
497 
498   if test "x$DEBUG_LEVEL" != xrelease; then
499     DEBUG_OPTIONS_FLAGS_JDK="$CFLAGS_DEBUG_OPTIONS"
500     DEBUG_SYMBOLS_CFLAGS_JDK="$CFLAGS_DEBUG_SYMBOLS"
501   fi
502 
503   #### TOOLCHAIN DEFINES
504 
505   if test "x$TOOLCHAIN_TYPE" = xgcc; then
506     ALWAYS_DEFINES_JVM="-D_GNU_SOURCE -D_REENTRANT"
507   elif test "x$TOOLCHAIN_TYPE" = xclang; then
508     ALWAYS_DEFINES_JVM="-D_GNU_SOURCE"
509   elif test "x$TOOLCHAIN_TYPE" = xsolstudio; then
510     ALWAYS_DEFINES_JVM="-DSPARC_WORKS -D_Crun_inline_placement"
511     ALWAYS_DEFINES_JDK="-DTRACING -DMACRO_MEMSYS_OPS -DBREAKPTS"
512     ALWAYS_DEFINES_JDK_CXXONLY="-DCC_NOEX"
513   elif test "x$TOOLCHAIN_TYPE" = xxlc; then
514     ALWAYS_DEFINES_JVM="-D_REENTRANT"
