24 /*
25  * @test
26  * @bug 7162488
27  * @summary VM should print unrecognized -XX option
28  * @library /test/lib
29  * @run driver TestUnrecognizedVmOption
30  */
31 import jdk.test.lib.process.OutputAnalyzer;
32 import jdk.test.lib.process.ProcessTools;
33 
34 public class TestUnrecognizedVmOption {
35     static final String OPTION="this_is_not_an_option";
36 
37     public static void main(String[] args) throws Exception {
38         ProcessBuilder pb =
39             ProcessTools.createJavaProcessBuilder(true, "-showversion", "-XX:" + OPTION);
40         new OutputAnalyzer(pb.start())
41             .shouldNotHaveExitValue(0)
42             .shouldContain("Unrecognized VM option")
43             .shouldContain(OPTION);
44     }
45 }
