 74     /**
 75      * Adds content for the StringContent object.  The method escapes
 76      * HTML characters for the string content that is added.
 77      *
 78      * @param strContent string content to be added
 79      */
 80     @Override
 81     public void add(CharSequence strContent) {
 82         Entity.escapeHtmlChars(strContent, stringContent);
 83     }
 84 
 85     /**
 86      * {@inheritDoc}
 87      */
 88     @Override
 89     public boolean isEmpty() {
 90         return (stringContent.length() == 0);
 91     }
 92 
 93     @Override
 94     public int charCount() {
 95         return RawHtml.charCount(stringContent.toString());
 96     }
 97 
 98     /**
 99      * {@inheritDoc}
100      */
101     @Override
102     public String toString() {
103         return stringContent.toString();
104     }
105 
106     /**
107      * {@inheritDoc}
108      */
109     @Override
110     public boolean write(Writer out, boolean atNewline) throws IOException {
111         String s = stringContent.toString();
112         out.write(s);
113         return s.endsWith(DocletConstants.NL);
114     }
