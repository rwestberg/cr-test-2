26 package jdk.javadoc.internal.tool;
27 
28 import java.io.PrintWriter;
29 import java.util.spi.ToolProvider;
30 
31 /**
32  * An implementation of the {@link java.util.spi.ToolProvider ToolProvider} SPI,
33  * providing access to JDK documentation tool, javadoc.
34  *
35  * @since 9
36  */
37 // This is currently a stand-alone top-level class so that it can easily be excluded
38 // from interims builds of javadoc, used while building JDK.
39 public class JavadocToolProvider implements ToolProvider {
40 
41     @Override
42     public String name() {
43         return "javadoc";
44     }
45 
46     @Override
47     public int run(PrintWriter out, PrintWriter err, String... args) {
48         return Main.execute(args, out, err);
49     }
50 }
