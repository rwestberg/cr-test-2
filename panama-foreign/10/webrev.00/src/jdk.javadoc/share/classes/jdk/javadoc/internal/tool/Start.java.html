136     public Start(Context context) {
137         this.docletClass = null;
138         this.context = Objects.requireNonNull(context);
139         this.apiMode = true;
140         this.docletName = null;
141         this.classLoader = null;
142         this.locale = Locale.getDefault();
143 
144         Log log = context.get(Log.logKey);
145         if (log instanceof Messager) {
146             messager = (Messager) log;
147         } else {
148             PrintWriter out = context.get(Log.errKey);
149             messager = (out == null)
150                     ? new Messager(context, ProgramName)
151                     : new Messager(context, ProgramName, out, out);
152         }
153 
154         options = getToolOptions();
155     }

186     private void showUsage(String headerKey, ToolOption.Kind kind, String footerKey) {
187         messager.notice(headerKey);
188         showToolOptions(kind);
189 
190         // let doclet print usage information
191         if (docletClass != null) {
192             String name = doclet.getName();
193             messager.notice("main.doclet.usage.header", name);
194             showDocletOptions(kind == ToolOption.Kind.EXTENDED
195                     ? Option.Kind.EXTENDED
196                     : Option.Kind.STANDARD);
197         }
198         if (footerKey != null)
199             messager.notice(footerKey);
200     }

224     private void showToolOptions(ToolOption.Kind kind) {
225         Comparator<ToolOption> comp = new Comparator<ToolOption>() {
226             final Collator collator = Collator.getInstance(Locale.US);
227             { collator.setStrength(Collator.PRIMARY); }
228 
229             @Override
230             public int compare(ToolOption o1, ToolOption o2) {
231                 return collator.compare(o1.primaryName, o2.primaryName);
232             }
233         };
234 
235         options.getSupportedOptions().stream()
236                     .filter(opt -> opt.kind == kind)
237                     .sorted(comp)
238                     .forEach(this::showToolOption);
239     }

241     private void showToolOption(ToolOption option) {
242         List<String> names = option.getNames();
243         String primaryName = option.primaryName;
244         String parameters;
245         if (option.hasArg || primaryName.endsWith(":")) {
246             String sep = primaryName.equals(ToolOptions.J) || primaryName.endsWith(":") ? "" : " ";
247             parameters = sep + option.getParameters(messager);
248         } else {
249             parameters = "";
250         }
251         String description = option.getDescription(messager);
252         showOption(names, parameters, description);
253     }

255     private void showDocletOptions(Option.Kind kind) {
256         Comparator<Doclet.Option> comp = new Comparator<Doclet.Option>() {
257             final Collator collator = Collator.getInstance(Locale.US);
258             { collator.setStrength(Collator.PRIMARY); }
259 
260             @Override
261             public int compare(Doclet.Option o1, Doclet.Option o2) {
262                 return collator.compare(o1.getNames().get(0), o2.getNames().get(0));
263             }
264         };
265 
266         doclet.getSupportedOptions().stream()
267                 .filter(opt -> opt.getKind() == kind)
268                 .sorted(comp)
269                 .forEach(this::showDocletOption);
270     }

272     private void showDocletOption(Doclet.Option option) {
273         List<String> names = option.getNames();
274         String parameters;
275         String primaryName = names.get(0);
276         if (option.getArgumentCount() > 0 || primaryName.endsWith(":")) {
277             String sep = primaryName.endsWith(":") ? "" : " ";
278             parameters = sep + option.getParameters();
279         } else {
280             parameters = "";
281         }
282         String description = option.getDescription();
283         showOption(names, parameters, description);
284     }

286     // The following constants are intended to format the output to
287     // be similar to that of the java launcher: i.e. "java -help".
288 
289     /** The indent for the option synopsis. */
290     private static final String SMALL_INDENT = "    ";
291     /** The automatic indent for the description. */
292     private static final String LARGE_INDENT = "                  ";
293     /** The space allowed for the synopsis, if the description is to be shown on the same line. */
294     private static final int DEFAULT_SYNOPSIS_WIDTH = 13;
295     /** The nominal maximum line length, when seeing if text will fit on a line. */
296     private static final int DEFAULT_MAX_LINE_LENGTH = 80;
297     /** The format for a single-line help entry. */
298     private static final String COMPACT_FORMAT = SMALL_INDENT + "%-" + DEFAULT_SYNOPSIS_WIDTH + "s %s";
299 
300     void showOption(List<String> names, String parameters, String description) {
301         String synopses = names.stream()
302                 .map(s -> s + parameters)
303                 .collect(Collectors.joining(", "));
304         // If option synopses and description fit on a single line of reasonable length,
305         // display using COMPACT_FORMAT
306         if (synopses.length() < DEFAULT_SYNOPSIS_WIDTH
307                 && !description.contains("\n")
308                 && (SMALL_INDENT.length() + DEFAULT_SYNOPSIS_WIDTH + 1 + description.length() <= DEFAULT_MAX_LINE_LENGTH)) {
309             messager.printNotice(String.format(COMPACT_FORMAT, synopses, description));
310             return;
311         }
312 
313         // If option synopses fit on a single line of reasonable length, show that;
314         // otherwise, show 1 per line
315         if (synopses.length() <= DEFAULT_MAX_LINE_LENGTH) {
316             messager.printNotice(SMALL_INDENT + synopses);
317         } else {
318             for (String name: names) {
319                 messager.printNotice(SMALL_INDENT + name + parameters);
320             }
321         }
322 
323         // Finally, show the description
324         messager.printNotice(LARGE_INDENT + description.replace("\n", "\n" + LARGE_INDENT));
325     }

361     private Result begin(List<String> options, Iterable<? extends JavaFileObject> fileObjects) {
362         fileManager = context.get(JavaFileManager.class);
363         if (fileManager == null) {
364             JavacFileManager.preRegister(context);
365             fileManager = context.get(JavaFileManager.class);
366             if (fileManager instanceof BaseFileManager) {
367                 ((BaseFileManager) fileManager).autoClose = true;
368             }
369         }
370 
371         // Perform an initial scan of the options to determine the doclet to be used (if any),
372         // so that it may participate in the main round of option processing.
373         try {
374             doclet = preprocess(options);
375         } catch (ToolException te) {
376             if (!te.result.isOK()) {
377                 if (te.message != null) {
378                     messager.printError(te.message);
379                 }
380                 Throwable t = te.getCause();
381                 dumpStack(t == null ? te : t);
382             }
383             return te.result;
384         } catch (OptionException oe) {
385             if (oe.message != null) {
386                 messager.printError(oe.message);
387             }
388             oe.m.run();
389             Throwable t = oe.getCause();
390             dumpStack(t == null ? oe : t);
391             return oe.result;
392         }
393 
394         Result result = OK;
395         try {
396             result = parseAndExecute(options, fileObjects);
397         } catch (com.sun.tools.javac.main.Option.InvalidValueException e) {
398             messager.printError(e.getMessage());
399             Throwable t = e.getCause();
400             dumpStack(t == null ? e : t);
401             return ERROR;
402         } catch (OptionException toe) {
403             if (toe.message != null)
404                 messager.printError(toe.message);
405 
406             toe.m.run();
407             Throwable t = toe.getCause();
408             dumpStack(t == null ? toe : t);
409             return toe.result;
410         } catch (ToolException exc) {
411             if (exc.message != null) {
412                 messager.printError(exc.message);
413             }
414             Throwable t = exc.getCause();
415             if (result == ABNORMAL) {
416                 reportInternalError(t == null ? exc : t);
417             } else {
418                 dumpStack(t == null ? exc : t);
419             }
420             return exc.result;
421         } catch (OutOfMemoryError ee) {
422             error("main.out.of.memory");
423             result = SYSERR;
424             dumpStack(ee);
425         } catch (ClientCodeException e) {
426             // simply rethrow these exceptions, to be caught and handled by JavadocTaskImpl
427             throw e;
428         } catch (Error | Exception ee) {
429             error("main.fatal.error", ee);
430             reportInternalError(ee);
431             result = ABNORMAL;
432         } finally {
433             if (fileManager != null
434                     && fileManager instanceof BaseFileManager
435                     && ((BaseFileManager) fileManager).autoClose) {
436                 try {
437                     fileManager.close();
438                 } catch (IOException ignore) {}
439             }
440             if (this.options.rejectWarnings() && messager.hasWarnings()) {
441                 error("main.warnings.Werror");
442             }
443             boolean haveErrors = messager.hasErrors();
444             if (!result.isOK() && !haveErrors) {
445                 // the doclet failed, but nothing reported, flag it!.
446                 error("main.unknown.error");
447             }
448             if (haveErrors && result.isOK()) {
449                 result = ERROR;
450             }
451             messager.printErrorWarningCounts();
452             messager.flush();
453         }
454         return result;
455     }

475     private Result parseAndExecute(List<String> argList, Iterable<? extends JavaFileObject> fileObjects)
476             throws ToolException, OptionException, com.sun.tools.javac.main.Option.InvalidValueException {
477         long tm = System.currentTimeMillis();
478 
479         List<String> javaNames = new ArrayList<>();
480 
481         // Make sure no obsolete source/target messages are reported
482         try {
483             options.processCompilerOption(com.sun.tools.javac.main.Option.XLINT_CUSTOM, "-Xlint:-options");
484         } catch (com.sun.tools.javac.main.Option.InvalidValueException ignore) {
485         }
486 
487         Arguments arguments = Arguments.instance(context);
488         arguments.init(ProgramName);
489         arguments.allowEmpty();
490 
491         doclet.init(locale, messager);
492         parseArgs(argList, javaNames);
493 
494         if (!arguments.handleReleaseOptions(extra -> true)) {
495             // Arguments does not always increase the error count in the
496             // case of errors, so increment the error count only if it has
497             // not been updated previously, preventing complaints by callers
498             if (!messager.hasErrors() && !messager.hasWarnings())
499                 messager.nerrors++;
500             return CMDERR;
501         }
502 
503         if (!arguments.validate()) {
504             // Arguments does not always increase the error count in the
505             // case of errors, so increment the error count only if it has
506             // not been updated previously, preventing complaints by callers
507             if (!messager.hasErrors() && !messager.hasWarnings())
508                 messager.nerrors++;
509             return CMDERR;
510         }
511 
512         if (fileManager instanceof BaseFileManager) {
513             ((BaseFileManager) fileManager).handleOptions(options.fileManagerOptions());
514         }
515 
516         String mr = com.sun.tools.javac.main.Option.MULTIRELEASE.primaryName;
517         if (fileManager.isSupportedOption(mr) == 1) {
518             Target target = Target.instance(context);
519             List<String> list = List.of(target.multiReleaseValue());
520             fileManager.handleOption(mr, list.iterator());
521         }
522         options.compilerOptions().notifyListeners();
523 
524         if (options.modules().isEmpty()) {
525             if (options.subpackages().isEmpty()) {
526                 if (javaNames.isEmpty() && isEmpty(fileObjects)) {
527                     String text = messager.getText("main.No_modules_packages_or_classes_specified");
528                     throw new ToolException(CMDERR, text);
529                 }
530             }
531         }
532 
533         JavadocTool comp = JavadocTool.make0(context);
534         if (comp == null) return ABNORMAL;
535 
536         DocletEnvironment docEnv = comp.getEnvironment(options, javaNames, fileObjects);
537 
538         // release resources
539         comp = null;
540 
541         if (options.breakIterator() || !locale.getLanguage().equals(Locale.ENGLISH.getLanguage())) {
542             JavacTrees trees = JavacTrees.instance(context);
543             trees.setBreakIterator(BreakIterator.getSentenceInstance(locale));
544         }
545         // pass off control to the doclet
546         Result returnStatus = docEnv != null && doclet.run(docEnv)
547                 ? OK
548                 : ERROR;
549 
550         // We're done.
551         if (options.verbose()) {
552             tm = System.currentTimeMillis() - tm;
553             messager.notice("main.done_in", Long.toString(tm));
554         }
555 
556         return returnStatus;
557     }

559     boolean matches(List<String> names, String arg) {
560         for (String name : names) {
561             if (StringUtils.toLowerCase(name).equals(StringUtils.toLowerCase(arg)))
562                 return true;
563         }
564         return false;
565     }

567     boolean matches(Doclet.Option option, String arg) {
568         if (matches(option.getNames(), arg))
569              return true;
570         int sep = arg.indexOf(':');
571         String targ = arg.substring(0, sep + 1);
572         return matches(option.getNames(), targ);
573     }

575     private Set<? extends Doclet.Option> docletOptions = null;
576     int handleDocletOption(int idx, List<String> args, boolean isToolOption)
577             throws OptionException {
578         if (docletOptions == null) {
579             docletOptions = doclet.getSupportedOptions();
580         }
581         String arg = args.get(idx);
582         String argBase, argVal;
583         if (arg.startsWith("--") && arg.contains("=")) {
584             int sep = arg.indexOf("=");
585             argBase = arg.substring(0, sep);
586             argVal = arg.substring(sep + 1);
587         } else {
588             argBase = arg;
589             argVal = null;
590         }
591         String text = null;
592         for (Doclet.Option opt : docletOptions) {
593             if (matches(opt, argBase)) {
594                 if (argVal != null) {
595                     switch (opt.getArgumentCount()) {
596                         case 0:
597                             text = messager.getText("main.unnecessary_arg_provided", argBase);
598                             throw new OptionException(ERROR, this::showUsage, text);
599                         case 1:
600                             opt.process(arg, Arrays.asList(argVal));
601                             break;
602                         default:
603                             text = messager.getText("main.only_one_argument_with_equals", argBase);
604                             throw new OptionException(ERROR, this::showUsage, text);
605                     }
606                 } else {
607                     if (args.size() - idx -1 < opt.getArgumentCount()) {
608                         text = messager.getText("main.requires_argument", arg);
609                         throw new OptionException(ERROR, this::showUsage, text);
610                     }
611                     opt.process(arg, args.subList(idx + 1, args.size()));
612                     idx += opt.getArgumentCount();
613                 }
614                 return idx;
615             }
616         }
617         // check if arg is accepted by the tool before emitting error
618         if (!isToolOption) {
619             text = messager.getText("main.invalid_flag", arg);
620             throw new OptionException(ERROR, this::showUsage, text);
621         }
622         return idx;
623     }

625     /**
626      * Performs an initial pass over the options, primarily to determine
627      * the doclet to be used (if any), so that it may participate in the
628      * main round of option decoding. This avoids having to specify that
629      * the options to specify the doclet should appear before any options
630      * that are handled by the doclet.
631      *
632      * The downside of this initial phase is that we have to skip over
633      * unknown options, and assume that we can reliably detect the options
634      * we need to handle.
635      *
636      * @param argv the arguments to be processed
637      * @return the doclet
638      * @throws ToolException if an error occurs initializing the doclet
639      * @throws OptionException if an error occurs while processing an option
640      */
641     private Doclet preprocess(List<String> argv)
642             throws ToolException, OptionException {
643         // doclet specifying arguments
644         String userDocletPath = null;
645         String userDocletName = null;
646 
647         // Step 1: loop through the args, set locale early on, if found.
648         for (int i = 0 ; i < argv.size() ; i++) {
649             String arg = argv.get(i);
650             if (arg.equals(ToolOptions.DUMP_ON_ERROR)) {
651                 // although this option is not needed in order to initialize the doclet,
652                 // it is helpful if it is set before trying to initialize the doclet
653                 options.setDumpOnError(true);
654             } else if (arg.equals(ToolOptions.LOCALE)) {
655                 checkOneArg(argv, i++);
656                 String lname = argv.get(i);
657                 locale = getLocale(lname);
658             } else if (arg.equals(ToolOptions.DOCLET)) {
659                 checkOneArg(argv, i++);
660                 if (userDocletName != null) {
661                     if (apiMode) {
662                         throw new IllegalArgumentException("More than one doclet specified (" +
663                                 userDocletName + " and " + argv.get(i) + ").");
664                     }
665                     String text = messager.getText("main.more_than_one_doclet_specified_0_and_1",
666                             userDocletName, argv.get(i));
667                     throw new ToolException(CMDERR, text);
668                 }
669                 if (docletName != null) {
670                     if (apiMode) {
671                         throw new IllegalArgumentException("More than one doclet specified (" +
672                                 docletName + " and " + argv.get(i) + ").");
673                     }
674                     String text = messager.getText("main.more_than_one_doclet_specified_0_and_1",
675                             docletName, argv.get(i));
676                     throw new ToolException(CMDERR, text);
677                 }
678                 userDocletName = argv.get(i);
679             } else if (arg.equals(ToolOptions.DOCLET_PATH)) {
680                 checkOneArg(argv, i++);
681                 if (userDocletPath == null) {
682                     userDocletPath = argv.get(i);
683                 } else {
684                     userDocletPath += File.pathSeparator + argv.get(i);
685                 }
686             }
687         }
688 
689         // Step 3: doclet name specified ? if so find a ClassLoader,
690         // and load it.
691         if (docletClass == null) {
692             if (userDocletName != null) {
693                 ClassLoader cl = classLoader;
694                 if (cl == null) {
695                     if (!fileManager.hasLocation(DOCLET_PATH)) {
696                         List<File> paths = new ArrayList<>();
697                         if (userDocletPath != null) {
698                             for (String pathname : userDocletPath.split(File.pathSeparator)) {
699                                 paths.add(new File(pathname));
700                             }
701                         }
702                         try {
703                             ((StandardJavaFileManager)fileManager).setLocation(DOCLET_PATH, paths);
704                         } catch (IOException ioe) {
705                             if (apiMode) {
706                                 throw new IllegalArgumentException("Could not set location for " +
707                                         userDocletPath, ioe);
708                             }
709                             String text = messager.getText("main.doclet_could_not_set_location",
710                                     userDocletPath);
711                             throw new ToolException(CMDERR, text, ioe);
712                         }
713                     }
714                     cl = fileManager.getClassLoader(DOCLET_PATH);
715                     if (cl == null) {
716                         // despite doclet specified on cmdline no classloader found!
717                         if (apiMode) {
718                             throw new IllegalArgumentException("Could not obtain classloader to load "
719 
720                                     + userDocletPath);
721                         }
722                         String text = messager.getText("main.doclet_no_classloader_found",
723                                 userDocletName);
724                         throw new ToolException(CMDERR, text);
725                     }
726                 }
727                 docletClass = loadDocletClass(userDocletName, cl);
728             } else if (docletName != null){
729                 docletClass = loadDocletClass(docletName, getClass().getClassLoader());
730             } else {
731                 docletClass = StandardDoclet.class;
732             }
733         }
734 
735         if (Doclet.class.isAssignableFrom(docletClass)) {
736             messager.setLocale(locale);
737             try {
738                 Object o = docletClass.getConstructor().newInstance();
739                 doclet = (Doclet) o;
740             } catch (ReflectiveOperationException exc) {
741                 if (apiMode) {
742                     throw new ClientCodeException(exc);
743                 }
744                 String text = messager.getText("main.could_not_instantiate_class", docletClass.getName());
745                 throw new ToolException(ERROR, text);
746             }
747         } else {
748             String text = messager.getText("main.not_a_doclet", docletClass.getName());
749             throw new ToolException(ERROR, text);
750         }
751         return doclet;
752     }
753 
754     private Class<?> loadDocletClass(String docletName, ClassLoader classLoader) throws ToolException {
755         try {
756             return classLoader == null ? Class.forName(docletName) : classLoader.loadClass(docletName);
757         } catch (ClassNotFoundException cnfe) {
758             if (apiMode) {
759                 throw new IllegalArgumentException("Cannot find doclet class " + docletName);
760             }
761             String text = messager.getText("main.doclet_class_not_found", docletName);
762             throw new ToolException(CMDERR, text, cnfe);
763         }
764     }

766     private void parseArgs(List<String> args, List<String> javaNames) throws ToolException,
767             OptionException, com.sun.tools.javac.main.Option.InvalidValueException {
768         for (int i = 0 ; i < args.size() ; i++) {
769             String arg = args.get(i);
770             ToolOption o = options.getOption(arg);
771             if (o != null) {
772                 // handle a doclet argument that may be needed however
773                 // don't increment the index, and allow the tool to consume args
774                 handleDocletOption(i, args, true);
775                 if (o.hasArg) {
776                     if (arg.startsWith("--") && arg.contains("=")) {
777                         o.process(arg.substring(arg.indexOf('=') + 1));
778                     } else {
779                         checkOneArg(args, i++);
780                         o.process(args.get(i));
781                     }
782                 } else if (o.hasSuffix) {
783                     o.process(arg);
784                 } else {
785                     o.process();
786                 }
787             } else if (arg.startsWith("-XD")) {
788                 // hidden javac options
789                 String s = arg.substring("-XD".length());
790                 int eq = s.indexOf('=');
791                 String key = (eq < 0) ? s : s.substring(0, eq);
792                 String value = (eq < 0) ? s : s.substring(eq+1);
793                 options.compilerOptions().put(key, value);
794             } else if (arg.startsWith("-")) {
795                 i = handleDocletOption(i, args, false);
796             } else {
797                 javaNames.add(arg);
798             }
799         }
800     }

806     /**
807      * Check the one arg option.
808      * Error and exit if one argument is not provided.
809      */
810     private void checkOneArg(List<String> args, int index) throws OptionException {
811         if ((index + 1) >= args.size() || args.get(index + 1).startsWith("-d")) {
812             String text = messager.getText("main.requires_argument", args.get(index));
813             throw new OptionException(CMDERR, this::showUsage, text);
814         }

825     /**
826      * Get the locale if specified on the command line
827      * else return null and if locale option is not used
828      * then return default locale.
829      */
830     private Locale getLocale(String localeName) throws ToolException {
831         try {
832             // Tolerate, at least for a while, the older syntax accepted by javadoc,
833             // using _ as the separator
834             localeName = localeName.replace("_", "-");
835             Locale l =  new Locale.Builder().setLanguageTag(localeName).build();
836             // Ensure that a non-empty language is available for the <HTML lang=...> element
837             return (l.getLanguage().isEmpty()) ? Locale.ENGLISH : l;
838         } catch (IllformedLocaleException e) {
839             String text = messager.getText("main.malformed_locale_name", localeName);
840             throw new ToolException(CMDERR, text);
841         }
842     }

844     /**
845      * Search the locale for specified language, specified country and
846      * specified variant.
847      */
848     private Locale searchLocale(String language, String country,
849                                 String variant) {
850         for (Locale loc : Locale.getAvailableLocales()) {
851             if (loc.getLanguage().equals(language) &&
852                 (country == null || loc.getCountry().equals(country)) &&
853                 (variant == null || loc.getVariant().equals(variant))) {
854                 return loc;
855             }
856         }
857         return null;
858     }
