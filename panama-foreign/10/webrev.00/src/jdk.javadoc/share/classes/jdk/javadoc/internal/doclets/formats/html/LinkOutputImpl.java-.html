26 package jdk.javadoc.internal.doclets.formats.html;
27 
28 import jdk.javadoc.internal.doclets.toolkit.util.links.LinkOutput;
29 
30 /**
31  * Stores output of a link.
32  *
33  *  <p><b>This is NOT part of any supported API.
34  *  If you write code that depends on this, you do so at your own risk.
35  *  This code and its internal interfaces are subject to change or
36  *  deletion without notice.</b>
37  */
38 public class LinkOutputImpl implements LinkOutput {
39 
40     /**
41      * The output of the link.
42      */
43     public StringBuilder output;
44 
45     /**
46      * Construct a new LinkOutputImpl.
47      */
48     public LinkOutputImpl() {
49         output = new StringBuilder();
50     }
51 
52     /**
53      * {@inheritDoc}
54      */
55     public void append(Object o) {
56         output.append(o instanceof String ?
57             (String) o : ((LinkOutputImpl)o).toString());
58     }
59 
60     /**
61      * {@inheritDoc}
62      */
63     public void insert(int offset, Object o) {
64         output.insert(offset, o.toString());
65     }
66 
67     /**
68      * {@inheritDoc}
69      */
70     public String toString() {
71         return output.toString();
72     }
73 
74 }
