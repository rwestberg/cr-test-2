26 package jdk.javadoc.internal.doclets.toolkit;
27 
28 import javax.lang.model.element.Element;
29 import javax.lang.model.element.PackageElement;
30 import javax.tools.FileObject;
31 
32 /**
33  * This is a pseudo element wrapper for the overview element, essentially to
34  * associate overview documentation's DocCommentTree to this element.
35  *
36  *  <p><b>This is NOT part of any supported API.
37  *  If you write code that depends on this, you do so at your own risk.
38  *  This code and its internal interfaces are subject to change or
39  *  deletion without notice.</b>
40  */
41 public class OverviewElement implements DocletElement {
42 
43     private final PackageElement pkg;
44     private final FileObject fo;
45 
46     public OverviewElement(PackageElement pkg, FileObject fo) {
47         this.pkg = pkg;
48         this.fo = fo;
49     }
50 
51     @Override
52     public PackageElement getPackageElement() {
53         return pkg;
54     }
55 
56     @Override
57     public FileObject getFileObject() {
58         return fo;
59     }
60 
61     @Override
62     public Kind getSubKind() {
63         return Kind.OVERVIEW;
64     }
65 }
