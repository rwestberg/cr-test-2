26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;
32 import jdk.javadoc.doclet.Taglet.Location;
33 import jdk.javadoc.internal.doclets.toolkit.Content;
34 
35 /**
36  * An abstract class that implements the {@link Taglet} interface and
37  * serves as a base for JavaFX property getter and setter taglets.
38  *
39  *  <p><b>This is NOT part of any supported API.
40  *  If you write code that depends on this, you do so at your own risk.
41  *  This code and its internal interfaces are subject to change or
42  *  deletion without notice.</b>
43  *
44  */
45 public abstract class BasePropertyTaglet extends BaseTaglet {
46 
47     public BasePropertyTaglet(String name) {
48         super(name, false, EnumSet.of(Location.METHOD));
49     }
50 
51     /**
52      * This method returns the text to be put in the resulting javadoc before
53      * the property name.
54      *
55      * @param tagletWriter the taglet writer for output
56      * @return the string to be put in the resulting javadoc.
57      */
58     abstract String getText(TagletWriter tagletWriter);
59 
60     /**
61      * Given the <code>Tag</code> representation of this custom
62      * tag, return its string representation, which is output
63      * to the generated page.
64      * @param element
65      * @param tag the <code>Tag</code> representation of this custom tag.
66      * @param tagletWriter the taglet writer for output.
67      * @return the TagletOutput representation of this <code>Tag</code>.
68      */
69     public Content getTagletOutput(Element element, DocTree tag, TagletWriter tagletWriter) {
70         return tagletWriter.propertyTagOutput(element, tag, getText(tagletWriter));
71     }
72 }
