 26 package jdk.javadoc.internal.doclets.toolkit.builders;
 27 
 28 import javax.lang.model.element.TypeElement;
 29 
 30 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeOptionalMemberWriter;
 31 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeRequiredMemberWriter;
 32 import jdk.javadoc.internal.doclets.toolkit.Content;
 33 import jdk.javadoc.internal.doclets.toolkit.DocletException;
 34 
 35 import static jdk.javadoc.internal.doclets.toolkit.util.VisibleMemberTable.Kind.*;
 36 
 37 /**
 38  * Builds documentation for optional annotation type members.
 39  *
 40  *  <p><b>This is NOT part of any supported API.
 41  *  If you write code that depends on this, you do so at your own risk.
 42  *  This code and its internal interfaces are subject to change or
 43  *  deletion without notice.</b>
 44  */
 45 public class AnnotationTypeOptionalMemberBuilder extends AnnotationTypeRequiredMemberBuilder {
 46 
 47     /**
 48      * Construct a new AnnotationTypeMemberBuilder.
 49      *
 50      * @param context  the build context.
 51      * @param typeElement the class whose members are being documented.
 52      * @param writer the doclet specific writer.
 53      */
 54     private AnnotationTypeOptionalMemberBuilder(Context context,
 55             TypeElement typeElement,
 56             AnnotationTypeOptionalMemberWriter writer) {
 57         super(context, typeElement, writer, ANNOTATION_TYPE_MEMBER_OPTIONAL);
 58     }
 59 
 60 
 61     /**
 62      * Construct a new AnnotationTypeMemberBuilder.
 63      *
 64      * @param context  the build context.
 65      * @param typeElement the class whose members are being documented.
 66      * @param writer the doclet specific writer.
 67      * @return the new AnnotationTypeMemberBuilder
 68      */
 69     public static AnnotationTypeOptionalMemberBuilder getInstance(
 70             Context context, TypeElement typeElement,
 71             AnnotationTypeOptionalMemberWriter writer) {
 72         return new AnnotationTypeOptionalMemberBuilder(context,
 73                 typeElement, writer);
 74     }
 75 
 76     @Override
 77     public void build(Content contentTree) throws DocletException {
 78         buildAnnotationTypeOptionalMember(contentTree);
 79     }
 80 
 81     /**
 82      * Build the annotation type optional member documentation.
 83      *
 84      * @param memberDetailsTree the content tree to which the documentation will be added
 85      * @throws DocletException if there is a problem while building the documentation
 86      */
 87     protected void buildAnnotationTypeOptionalMember(Content memberDetailsTree)
 88                 throws DocletException {
 89         buildAnnotationTypeMember(memberDetailsTree);
 90     }
 91 
 92     @Override
 93     protected void buildAnnotationTypeMemberChildren(Content annotationDocTree) {
 94         super.buildAnnotationTypeMemberChildren(annotationDocTree);
 95         buildDefaultValueInfo(annotationDocTree);
 96     }
 97 
 98     /**
 99      * Build the default value for this optional member.
100      *
101      * @param annotationDocTree the content tree to which the documentation will be added
102      */
103     protected void buildDefaultValueInfo(Content annotationDocTree) {
104         ((AnnotationTypeOptionalMemberWriter) writer).addDefaultValueInfo(currentMember,
105                 annotationDocTree);
106     }
107 
108     @Override
109     public AnnotationTypeRequiredMemberWriter getWriter() {
110         return writer;
111     }
112 }
