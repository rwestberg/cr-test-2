 51 /**
 52  * This class contains misc helper functions to support creation of memory segments.
 53  */
 54 public final class Utils {
 55 
 56     private static Unsafe unsafe = Unsafe.getUnsafe();
 57 
 58     private static final MethodHandle ADDRESS_FILTER;
 59 
 60     static {
 61         try {
 62             ADDRESS_FILTER = MethodHandles.lookup().findStatic(Utils.class, "filterAddress",
 63                     MethodType.methodType(MemoryAddressProxy.class, MemoryAddress.class));
 64         } catch (Throwable ex) {
 65             throw new ExceptionInInitializerError(ex);
 66         }
 67     }
 68 
 69     // The maximum alignment supported by malloc - typically 16 on 64-bit platforms.
 70     private final static long MAX_ALIGN = 16;
 71 
 72     private static final JavaNioAccess javaNioAccess = SharedSecrets.getJavaNioAccess();
 73 
 74     private static final boolean skipZeroMemory = GetBooleanAction.privilegedGetProperty("jdk.internal.foreign.skipZeroMemory");
 75 
 76     public static long alignUp(long n, long alignment) {
 77         return (n + alignment - 1) & -alignment;
 78     }
 79 
 80     public static long bitsToBytesOrThrow(long bits, Supplier<RuntimeException> exFactory) {
 81         if (bits % 8 == 0) {
 82             return bits / 8;
 83         } else {
 84             throw exFactory.get();
 85         }
 86     }
 87 
 88     // segment factories
 89 
 90     public static MemorySegment makeNativeSegment(long bytesSize, long alignmentBytes) {
 91         long alignedSize = bytesSize;
 92 
 93         if (alignmentBytes > MAX_ALIGN) {
 94             alignedSize = bytesSize + (alignmentBytes - 1);
 95         }
 96 
 97         long buf = unsafe.allocateMemory(alignedSize);
 98         if (!skipZeroMemory) {
 99             unsafe.setMemory(buf, alignedSize, (byte)0);
100         }
101         long alignedBuf = Utils.alignUp(buf, alignmentBytes);
102         MemoryScope scope = new MemoryScope(null, () -> unsafe.freeMemory(buf));
103         MemorySegment segment = new MemorySegmentImpl(buf, null, alignedSize, 0, Thread.currentThread(), scope);
104         if (alignedBuf != buf) {
105             long delta = alignedBuf - buf;
106             segment = segment.asSlice(delta, bytesSize);
107         }
108         return segment;
109     }

111     public static MemorySegment makeArraySegment(byte[] arr) {
112         return makeArraySegment(arr, arr.length, Unsafe.ARRAY_BYTE_BASE_OFFSET, Unsafe.ARRAY_BYTE_INDEX_SCALE);
113     }

117     }

121     }

123     public static MemorySegment makeArraySegment(int[] arr) {
124         return makeArraySegment(arr, arr.length, Unsafe.ARRAY_INT_BASE_OFFSET, Unsafe.ARRAY_INT_INDEX_SCALE);
125     }

128         return makeArraySegment(arr, arr.length, Unsafe.ARRAY_FLOAT_BASE_OFFSET, Unsafe.ARRAY_FLOAT_INDEX_SCALE);

132         return makeArraySegment(arr, arr.length, Unsafe.ARRAY_LONG_BASE_OFFSET, Unsafe.ARRAY_LONG_INDEX_SCALE);

138 
139     private static MemorySegment makeArraySegment(Object arr, int size, int base, int scale) {
140         MemoryScope scope = new MemoryScope(null, null);
141         return new MemorySegmentImpl(base, arr, size * scale, 0, Thread.currentThread(), scope);
142     }
143 
144     public static MemorySegment makeBufferSegment(ByteBuffer bb) {
145         long bbAddress = javaNioAccess.getBufferAddress(bb);
146         Object base = javaNioAccess.getBufferBase(bb);
147 
148         int pos = bb.position();
149         int limit = bb.limit();
150 
151         MemoryScope bufferScope = new MemoryScope(bb, null);
152         return new MemorySegmentImpl(bbAddress + pos, base, limit - pos, 0, Thread.currentThread(), bufferScope);
153     }
154 
155     // create and map a file into a fresh segment
156     public static MemorySegment makeMappedSegment(Path path, long bytesSize, FileChannel.MapMode mapMode) throws IOException {
157         if (bytesSize <= 0) throw new IllegalArgumentException("Requested bytes size must be > 0.");
158         try (FileChannelImpl channelImpl = (FileChannelImpl)FileChannel.open(path, openOptions(mapMode))) {
159             UnmapperProxy unmapperProxy = channelImpl.mapInternal(mapMode, 0L, bytesSize);
160             MemoryScope scope = new MemoryScope(null, unmapperProxy::unmap);
161             return new MemorySegmentImpl(unmapperProxy.address(), null, bytesSize, 0, Thread.currentThread(), scope);
162         }
163     }
164 
165     private static OpenOption[] openOptions(FileChannel.MapMode mapMode) {
166         if (mapMode == FileChannel.MapMode.READ_ONLY) {
167             return new OpenOption[] { StandardOpenOption.READ };
168         } else if (mapMode == FileChannel.MapMode.READ_WRITE || mapMode == FileChannel.MapMode.PRIVATE) {
169             return new OpenOption[] { StandardOpenOption.READ, StandardOpenOption.WRITE };
170         } else {
171             throw new UnsupportedOperationException("Unsupported map mode: " + mapMode);
172         }
173     }
174 
175     public static VarHandle fixUpVarHandle(VarHandle handle) {
176         // This adaptation is required, otherwise the memory access var handle will have type MemoryAddressProxy,
177         // and not MemoryAddress (which the user expects), which causes performance issues with asType() adaptations.
178         return MethodHandles.filterCoordinates(handle, 0, ADDRESS_FILTER);
179     }
180 
181     private static MemoryAddressProxy filterAddress(MemoryAddress addr) {
182         return (MemoryAddressImpl)addr;
183     }
