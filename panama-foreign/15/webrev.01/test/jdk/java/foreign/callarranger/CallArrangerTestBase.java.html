24 import jdk.incubator.foreign.FunctionDescriptor;
25 import jdk.incubator.foreign.MemoryLayout;
26 import jdk.internal.foreign.abi.Binding;
27 import jdk.internal.foreign.abi.CallingSequence;
28 
29 import java.util.ArrayList;
30 import java.util.Arrays;
31 import java.util.List;
32 
33 import static org.testng.Assert.assertEquals;
34 
35 public class CallArrangerTestBase {
36 
37     public static void checkArgumentBindings(CallingSequence callingSequence, Binding[][] argumentBindings) {
38         assertEquals(callingSequence.argumentCount(), argumentBindings.length);
39 
40         for (int i = 0; i < callingSequence.argumentCount(); i++) {
41             List<Binding> actual = callingSequence.argumentBindings(i);
42             Binding[] expected = argumentBindings[i];
43             assertEquals(actual, Arrays.asList(expected));
44         }
45     }
46 
47     public static void checkReturnBindings(CallingSequence callingSequence, Binding[] returnBindings) {
48         assertEquals(callingSequence.returnBindings(), Arrays.asList(returnBindings));
49     }
50 }
