105         private Object invoke(Object[] args) throws Throwable {
106             // one trailing Object[]
107             int nNamedArgs = function.argumentLayouts().size();
108             assert(args.length == nNamedArgs + 1);
109             // The last argument is the array of vararg collector
110             Object[] unnamedArgs = (Object[]) args[args.length - 1];
111 
112             int argsCount = nNamedArgs + unnamedArgs.length;
113             Class<?>[] argTypes = new Class<?>[argsCount];
114             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
115 
116             int pos = 0;
117             for (pos = 0; pos < nNamedArgs; pos++) {
118                 argTypes[pos] = varargs.parameterType(pos);
119                 argLayouts[pos] = function.argumentLayouts().get(pos);
120             }
121 
122             assert pos == nNamedArgs;
123             for (Object o: unnamedArgs) {
124                 argTypes[pos] = normalize(o.getClass());
125                 argLayouts[pos] = variadicLayout(argTypes[pos]);
126                 pos++;
127             }
128             assert pos == argsCount;
129 
130             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
131             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
132                     FunctionDescriptor.ofVoid(false, argLayouts) :
133                     FunctionDescriptor.of(function.returnLayout().get(), false, argLayouts);
134             MethodHandle mh = SystemABI.getInstance().downcallHandle(symbol, mt, f);
135             // flatten argument list so that it can be passed to an asSpreader MH
136             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
137             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
138             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
139 
140             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
141         }

143         private static Class<?> unboxIfNeeded(Class<?> clazz) {

167         private Class<?> normalize(Class<?> c) {
168             c = unboxIfNeeded(c);
169             if (c.isPrimitive()) {
170                 return c;
171             }
172             if (MemoryAddress.class.isAssignableFrom(c)) {
173                 return MemoryAddress.class;
174             }
175             if (MemorySegment.class.isAssignableFrom(c)) {
176                 return MemorySegment.class;
177             }
178             throw new IllegalArgumentException("Invalid type for ABI: " + c.getTypeName());
179         }

181         private MemoryLayout variadicLayout(Class<?> c) {
182             if (c == char.class || c == byte.class || c == short.class || c == int.class || c == long.class) {
183                 //it is ok to approximate with a machine word here; numerics arguments in a prototype-less
184                 //function call are always rounded up to a register size anyway.
185                 return C_LONGLONG;
186             } else if (c == float.class || c == double.class) {
187                 return C_DOUBLE;
188             } else if (MemoryAddress.class.isAssignableFrom(c)) {
189                 return C_POINTER;
190             } else {
191                 throw new IllegalArgumentException("Unhandled variadic argument class: " + c);
192             }
