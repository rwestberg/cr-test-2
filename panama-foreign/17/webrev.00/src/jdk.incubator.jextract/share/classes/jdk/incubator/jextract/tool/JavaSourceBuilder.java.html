 45 /**
 46  * A helper class to generate header interface class in source form.
 47  * After aggregating various constituents of a .java source, build
 48  * method is called to get overall generated source string.
 49  */
 50 class JavaSourceBuilder {
 51     // buffer
 52     protected StringBuffer sb;
 53     // current line alignment (number of 4-spaces)
 54     protected int align;
 55 
 56     JavaSourceBuilder(int align) {
 57         this.align = align;
 58         this.sb = new StringBuffer();
 59     }
 60 
 61     JavaSourceBuilder() {
 62         this(0);
 63     }
 64 
 65     protected int align() {
 66         return align;
 67     }
 68 
 69     final String PUB_CLS_MODS = "public final ";
 70     final String PUB_MODS = "public static final ";
 71     final String PRI_MODS = "private static final ";
 72 
 73     protected void addPackagePrefix(String pkgName) {
 74         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 75         sb.append("// Generated by jextract\n\n");
 76         if (!pkgName.isEmpty()) {
 77             sb.append("package ");
 78             sb.append(pkgName);
 79             sb.append(";\n\n");
 80         }
 81         addImportSection();
 82     }
 83 
 84     protected void addImportSection() {
 85         sb.append("import java.lang.invoke.MethodHandle;\n");
 86         sb.append("import java.lang.invoke.VarHandle;\n");
 87         sb.append("import jdk.incubator.foreign.*;\n");
 88         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 89     }
 90 
 91     protected void classBegin(String name) {
 92         indent();
 93         sb.append(PUB_CLS_MODS + "class ");
 94         sb.append(name);
 95         sb.append(" {\n\n");
 96     }
 97 
 98     protected void classEnd() {
 99         indent();
100         sb.append("}\n\n");
101     }
102 
103     protected void addLibraries(String[] libraryNames) {
104         incrAlign();
105         indent();
106         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
107         sb.append(stringArray(libraryNames) + ");\n");
108         decrAlign();
109     }
110 
111     private String stringArray(String[] elements) {
112         return Stream.of(elements)
113                 .map(n -> "\"" + n + "\"")
114                 .collect(Collectors.joining(",", "new String[] {", "}"));
115     }
116 
117     protected void addLayout(String elementName, MemoryLayout layout) {
118         incrAlign();
119         indent();
120         sb.append(PUB_MODS + "MemoryLayout " + elementName + "$LAYOUT = ");
121         addLayout(layout);
122         sb.append(";\n");
123         decrAlign();
124     }
125 
126     private void addLayout(MemoryLayout l) {
127         if (l instanceof ValueLayout) {
128             boolean found = false;
129             for (Field fs : MemoryLayouts.SysV.class.getDeclaredFields()) {
130                 try {
131                     MemoryLayout constant = (MemoryLayout)fs.get(null);
132                     if (l.name().isPresent()) {
133                         constant = constant.withName(l.name().get());
134                     }
135                     if (constant.equals(l)) {
136                         found = true;
137                         sb.append("MemoryLayouts.SysV." + fs.getName());
138                         break;
139                     }
140                 } catch (ReflectiveOperationException ex) {
141                     throw new AssertionError(ex);
142                 }
143             }
144             if (!found) {
145                 throw new AssertionError("Should not get here: " + l);
146             }
147         } else if (l instanceof SequenceLayout) {
148             sb.append("MemoryLayout.ofSequence(");
149             if (((SequenceLayout) l).elementCount().isPresent()) {
150                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
151             }
152             addLayout(((SequenceLayout) l).elementLayout());
153             sb.append(")");
154         } else if (l instanceof GroupLayout) {
155             if (((GroupLayout) l).isStruct()) {
156                 sb.append("MemoryLayout.ofStruct(\n");
157             } else {
158                 sb.append("MemoryLayout.ofUnion(\n");
159             }
160             incrAlign();
161             String delim = "";
162             for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
163                 sb.append(delim);
164                 indent();
165                 addLayout(e);
166                 delim = ",\n";
167             }
168             sb.append("\n");
169             decrAlign();
170             indent();
171             sb.append(")");
172         } else {
173             //padding
174             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
175         }
176         if (l.name().isPresent()) {
177             sb.append(".withName(\"" +  l.name().get() + "\")");
178         }
179     }

181     protected void addVarHandle(String name, Class<?> type, String parentName) {
182         incrAlign();
183         indent();
184         String vhName = parentName != null ?
185                 parentName + "$" + name :
186                 name;
187         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
188         if (parentName != null) {
189             addHandlePath(type, parentName, name);
190         } else {
191             addHandlePath(type, name);
192         }
193         sb.append(";\n");
194         decrAlign();
195     }
196 
197     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
198         String ty = type.getName();
199         if (ty.contains("MemoryAddress")) {
200             ty = "long";
201         }
202         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
203         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
204         sb.append(")");
205     }
206 
207     protected void addHandlePath(Class<?> type, String varName) {
208         String ty = type.getName();
209         if (ty.contains("MemoryAddress")) {
210             ty = "long";
211         }
212         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
213     }

215     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
216         incrAlign();
217         indent();
218         sb.append(PUB_MODS + "MethodHandle " + funcTree.name() + " = ");
219         sb.append("RuntimeHelper.downcallHandle(\n");
220         incrAlign();
221         indent();
222         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
223         sb.append(",\n");
224         indent();
225         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
226         indent();
227         addFunction(desc);
228         sb.append(",\n");
229         indent();
230         sb.append(funcTree.type().varargs());
231         decrAlign();
232         sb.append("\n");
233         indent();
234         sb.append(");\n");
235         decrAlign();
236     }
237 
238     protected void addAddressLookup(String name) {
239         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
240     }
241 
242     private void addFunction(FunctionDescriptor f) {
243         if (f.returnLayout().isPresent()) {
244             sb.append("FunctionDescriptor.of(");
245             addLayout(f.returnLayout().get());
246             sb.append(", ");
247         } else {
248             sb.append("FunctionDescriptor.ofVoid(");
249         }
250         if (!f.argumentLayouts().isEmpty()) {
251             sb.append("\n");
252             incrAlign();
253             String delim = "";
254             for (MemoryLayout e : f.argumentLayouts()) {
255                 sb.append(delim);
256                 indent();
257                 addLayout(e);
258                 delim = ",\n";
259             }
260             sb.append("\n");
261             decrAlign();
262             indent();
263         }
264         sb.append(")");
265     }
266 
267     protected void addAddress(String name) {
268         incrAlign();
269         indent();
270         sb.append(PUB_MODS + "MemoryAddress " + name + "$ADDR" + " = ");
271         addAddressLookup(name);
272         sb.append(";\n");
273         decrAlign();
274     }

276     protected void addConstant(String name, Class<?> type, Object value) {
277         incrAlign();
278         indent();
279         if (type == MemoryAddress.class || type == MemorySegment.class) {
280             //todo, skip for now (address constants and string constants)
281         } else {
282             sb.append(PUB_MODS + type.getName() + " " + name);
283             sb.append(" = ");
284             if (type == float.class) {
285                 sb.append(value);
286                 sb.append("f");
287             } else if (type == long.class) {
288                 sb.append(value);
289                 sb.append("L");
290             } else if (type == double.class) {
291                 sb.append(value);
292                 sb.append("d");
293             } else {
294                 sb.append("(" + type.getName() + ")");
295                 sb.append(value + "L");
296             }
297             sb.append(";\n");
298         }
299 
300         decrAlign();
301     }

303     static int funcIntfCounter = 0;
304 
305     protected void addUpcallFactory(FunctionDescriptor desc) {
306         String fnName = "FI" + funcIntfCounter++;
307         incrAlign();
308         indent();
309         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
310         addFunction(desc);
311         sb.append(";\n");
312         indent();
313         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
314         incrAlign();
315         indent();
316         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
317         decrAlign();
318         indent();
319         sb.append("}\n");
320         decrAlign();
321     }

323     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
324         incrAlign();
325         indent();
326         sb.append(PUB_MODS + mtype.returnType().getName() + " " + f.name() + " (");
327         String delim = "";
328         List<String> pNames = new ArrayList<>();
329         final int numParams = f.parameters().size();
330         for (int i = 0 ; i < numParams; i++) {
331             String pName = f.parameters().get(i).name();
332             if (pName.isEmpty()) {
333                 pName = "x" + i;
334             }
335             pNames.add(pName);
336             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
337             delim = ", ";
338         }
339         if (f.type().varargs()) {
340             String lastArg = "x" + numParams;
341             if (numParams > 0) {
342                 sb.append(", ");
343             }
344             sb.append("Object... " + lastArg);
345             pNames.add(lastArg);
346         }
347         sb.append(") {\n");
348         incrAlign();
349         indent();
350         sb.append("try {\n");
351         incrAlign();
352         indent();
353         if (!mtype.returnType().equals(void.class)) {
354             sb.append("return (" + mtype.returnType().getName() + ")");
355         }
356         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
357         decrAlign();
358         indent();
359         sb.append("} catch (Throwable ex) {\n");
360         incrAlign();
361         indent();
362         sb.append("throw new AssertionError(ex);\n");
363         decrAlign();
364         indent();
365         sb.append("}\n");
366         decrAlign();
367         indent();
368         sb.append("}\n");
369         decrAlign();
370     }
371 
372     void addDescriptor(String name, FunctionDescriptor desc) {
373         incrAlign();
374         indent();
375         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
376         addFunction(desc);
377         sb.append(";\n");
378         decrAlign();
379         indent();
380     }

382     void addFunctionalInterface(String name, MethodType mtype) {
383         incrAlign();
384         indent();
385         sb.append("public interface " + name + " {\n");
386         incrAlign();
387         indent();
388         sb.append(mtype.returnType().getName() + " apply(");
389         String delim = "";
390         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
391             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
392             delim = ", ";
393         }
394         sb.append(");\n");
395         decrAlign();
396         indent();
397         sb.append("}\n");
398         decrAlign();
399         indent();
400     }

402     protected void addFunctionalFactory(String name, MethodType mtype) {
403         incrAlign();
404         indent();
405         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
406         incrAlign();
407         indent();
408         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
409                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
410         decrAlign();
411         indent();
412         sb.append("}\n");
413         decrAlign();
414     }

416     void addGetter(String name, Class<?> type, Declaration parent) {
417         incrAlign();
418         indent();
419         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
420         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
421         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
422         incrAlign();
423         indent();
424         String vhParam = parent != null ?
425                 "seg.baseAddress()" : name + "$ADDR";
426         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
427         decrAlign();
428         indent();
429         sb.append("}\n");
430         decrAlign();
431     }
432 
433     void addSetter(String name, Class<?> type, Declaration parent) {
434         incrAlign();
435         indent();
436         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
437         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
438         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
439         incrAlign();
440         indent();
441         String vhParam = parent != null ?
442                 "seg.baseAddress()" : name + "$ADDR";
443         sb.append(vhName + ".set(" + vhParam + ", x);\n");
444         decrAlign();
445         indent();
446         sb.append("}\n");
447         decrAlign();
448     }
449 
450     protected String build() {
451         String res = sb.toString();
452         this.sb = null;
453         return res.toString();
454     }
455 
456     protected void indent() {
457         for (int i = 0; i < align; i++) {
458             sb.append("    ");
459         }
460     }
461 
462     protected void incrAlign() {
463         align++;
464     }
465 
466     protected void decrAlign() {
467         align--;
468     }
