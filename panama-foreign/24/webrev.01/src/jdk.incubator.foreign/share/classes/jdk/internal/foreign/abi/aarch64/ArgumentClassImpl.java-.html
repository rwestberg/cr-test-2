24 package jdk.internal.foreign.abi.aarch64;
25 
26 import jdk.internal.foreign.abi.ArgumentClass;
27 
28 public enum ArgumentClassImpl implements ArgumentClass {
29     POINTER, INTEGER, VECTOR, MEMORY;
30 
31     public boolean isIntegral() {
32         return this == INTEGER || this == POINTER;
33     }
34 
35     @Override
36     public boolean isPointer() {
37         return this == POINTER;
38     }
39 
40     @Override
41     public boolean isIndirect() {
42         return this == MEMORY;
43     }
44 }
