23 package jdk.internal.foreign.abi.x64;
24 
25 import jdk.internal.foreign.abi.ArgumentClass;
26 
27 public enum ArgumentClassImpl implements ArgumentClass {
28     POINTER, INTEGER, SSE, SSEUP, X87, X87UP, COMPLEX_X87, NO_CLASS, MEMORY;
29 
30     public ArgumentClassImpl merge(ArgumentClassImpl other) {
31         if (this == other) {
32             return this;
33         }
34 
35         if (other == NO_CLASS) {
36             return this;
37         }
38         if (this == NO_CLASS) {
39             return other;
40         }
41 
42         if (this == MEMORY || other == MEMORY) {
43             return MEMORY;
44         }
45 
46         if (this == POINTER || other == POINTER) {
47             return POINTER;
48         }
49 
50         if (this == INTEGER || other == INTEGER) {
51             return INTEGER;
52         }
53 
54         if (this == X87 || this == X87UP || this == COMPLEX_X87) {
55             return MEMORY;
56         }
57         if (other == X87 || other == X87UP || other == COMPLEX_X87) {
58             return MEMORY;
59         }
60 
61         return SSE;
62     }
63 
64     public boolean isIntegral() {
65         return this == INTEGER || this == POINTER;
66     }
67 
68     @Override
69     public boolean isPointer() {
70         return this == POINTER;
71     }
72 
73     @Override
74     public boolean isIndirect() {
75         return this == MEMORY;
76     }
77 }
